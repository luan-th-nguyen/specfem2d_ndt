33 
34  subroutine calerf(ARG,RESULT,jintval)
35 
36 !------------------------------------------------------------------
37 !
38 ! This routine can be freely obtained from Netlib
39 ! at http://www.netlib.org/specfun/erf
40 !
41 ! Most Netlib software packages have no restrictions on their use
42 ! but Netlib recommends that you check with the authors to be sure.
43 ! See http://www.netlib.org/misc/faq.html#2.3 for details.
44 !
45 !------------------------------------------------------------------
46 !
47 ! This packet evaluates erf(x) for a real argument x.
48 ! It contains one function type subprogram: ERF,
49 ! and one subroutine type subprogram, CALERF. The calling
50 ! statements for the primary entries are:
51 !
52 ! Y = ERF(X)
53 !
54 ! The routine CALERF is intended for internal packet use only,
55 ! all computations within the packet being concentrated in this
56 ! routine. The function subprograms invoke CALERF with the
57 ! statement
58 !
59 ! call CALERF(ARG,RESULT,jintval)
60 !
61 ! where the parameter usage is as follows
62 !
63 ! function Parameters for CALERF
64 ! call ARG Result jintval
65 !
66 ! ERF(ARG) ANY REAL ARGUMENT ERF(ARG) 0
67 !
68 ! The main computation evaluates near-minimax approximations
69 ! from "Rational Chebyshev approximations for the error function"
70 ! by William J. Cody, Math. Comp., 1969, PP. 631-638. This
71 ! transportable program uses rational functions that theoretically
72 ! approximate erf(x) and erfc(x) to at least 18 significant
73 ! decimal digits. The accuracy achieved depends on the arithmetic
74 ! system, the compiler, the intrinsic functions, and proper
75 ! selection of the machine-dependent constants.
76 !
77 !*******************************************************************
78 !*******************************************************************
79 !
80 ! Explanation of machine-dependent constants
81 !
82 ! XMIN = the smallest positive floating-point number.
83 ! XINF = the largest positive finite floating-point number.
84 ! XNEG = the largest negative argument acceptable to ERFCX;
85 ! the negative of the solution to the equation
86 ! 2*exp(x*x) = XINF.
87 ! XSMALL = argument below which erf(x) may be represented by
88 ! 2*x/sqrt(pi) and above which x*x will not underflow.
89 ! A conservative value is the largest machine number X
90 ! such that 1.0 + X = 1.0 to machine precision.
91 ! XBIG = largest argument acceptable to ERFC; solution to
92 ! the equation: W(x) * (1-0.5/x**2) = XMIN, where
93 ! W(x) = exp(-x*x)/[x*sqrt(pi)].
94 ! XHUGE = argument above which 1.0 - 1/(2*x*x) = 1.0 to
95 ! machine precision. A conservative value is
96 ! 1/[2*sqrt(XSMALL)]
97 ! XMAX = largest acceptable argument to ERFCX; the minimum
98 ! of XINF and 1/[sqrt(pi)*XMIN].
99 !
100 ! Approximate IEEE double precision values are defined below.
101 !
102 !*******************************************************************
103 !*******************************************************************
104 !
105 ! Error returns
106 !
107 ! The program returns ERFC = 0 for ARG >= XBIG;
108 !
109 ! Author: William J. Cody
110 ! Mathematics and Computer Science Division
111 ! Argonne National Laboratory
112 ! Argonne, IL 60439, USA
113 !
114 ! Latest modification: March 19, 1990
115 !
116 ! Converted to Fortran90 and slightly modified by
117 ! Dimitri Komatitsch, University of Pau, France, November 2007.
118 !
119 !------------------------------------------------------------------
120 
121  implicit none
122 
123  integer I,jintval
124  double precision A,ARG,B,C,D,DEL,FOUR,HALF,P,ONE,Q,RESULT,SIXTEEN,SQRPI, &
125  TWO,THRESHOLD,X,XBIG,XDEN,XHUGE,XINF,XMAX,XNEG,XNUM,XSMALL, &
126  Y,YSQ,ZERO
127  dimension a(5),b(4),c(9),d(8),p(6),q(5)
128 
129 !------------------------------------------------------------------
130 ! Mathematical constants
131 !------------------------------------------------------------------
132  data four,one,half,two,zero/4.0d0,1.0d0,0.5d0,2.0d0,0.0d0/, &
133  sqrpi/5.6418958354775628695d-1/,threshold/0.46875d0/, &
134  sixteen/16.0d0/
135 
136 !------------------------------------------------------------------
137 ! Machine-dependent constants
138 !------------------------------------------------------------------
139  data xinf,xneg,xsmall/1.79d308,-26.628d0,1.11d-16/, &
140  xbig,xhuge,xmax/26.543d0,6.71d7,2.53d307/
141 
142 !------------------------------------------------------------------
143 ! Coefficients for approximation to erf in first interval
144 !------------------------------------------------------------------
145  data a/3.16112374387056560d00,1.13864154151050156d02, &
146  3.77485237685302021d02,3.20937758913846947d03, &
147  1.85777706184603153d-1/
148  data b/2.36012909523441209d01,2.44024637934444173d02, &
149  1.28261652607737228d03,2.84423683343917062d03/
150 
151 !------------------------------------------------------------------
152 ! Coefficients for approximation to erfc in second interval
153 !------------------------------------------------------------------
154  data c/5.64188496988670089d-1,8.88314979438837594d0, &
155  6.61191906371416295d01,2.98635138197400131d02, &
156  8.81952221241769090d02,1.71204761263407058d03, &
157  2.05107837782607147d03,1.23033935479799725d03, &
158  2.15311535474403846d-8/
159  data d/1.57449261107098347d01,1.17693950891312499d02, &
160  5.37181101862009858d02,1.62138957456669019d03, &
161  3.29079923573345963d03,4.36261909014324716d03, &
162  3.43936767414372164d03,1.23033935480374942d03/
163 
164 !------------------------------------------------------------------
165 ! Coefficients for approximation to erfc in third interval
166 !------------------------------------------------------------------
167  data p/3.05326634961232344d-1,3.60344899949804439d-1, &
168  1.25781726111229246d-1,1.60837851487422766d-2, &
169  6.58749161529837803d-4,1.63153871373020978d-2/
170  data q/2.56852019228982242d00,1.87295284992346047d00, &
171  5.27905102951428412d-1,6.05183413124413191d-2, &
172  2.33520497626869185d-3/
173 
174  x = arg
175  y = abs(x)
176  if (y <= threshold) then
177 
178 !------------------------------------------------------------------
179 ! Evaluate erf for |X| <= 0.46875
180 !------------------------------------------------------------------
181  ysq = zero
182  if (y > xsmall) ysq = y * y
183  xnum = a(5)*ysq
184  xden = ysq
185 
186  do i = 1, 3
187  xnum = (xnum + a(i)) * ysq
188  xden = (xden + b(i)) * ysq
189  enddo
190 
191  result = x * (xnum + a(4)) / (xden + b(4))
192  if (jintval /= 0) result = one - result
193  if (jintval == 2) result = exp(ysq) * result
194  goto 800
195 
196 !------------------------------------------------------------------
197 ! Evaluate erfc for 0.46875 <= |X| <= 4.0
198 !------------------------------------------------------------------
199  else if (y <= four) then
200  xnum = c(9)*y
201  xden = y
202 
203  do i = 1, 7
204  xnum = (xnum + c(i)) * y
205  xden = (xden + d(i)) * y
206  enddo
207 
208  result = (xnum + c(8)) / (xden + d(8))
209  if (jintval /= 2) then
210  ysq = aint(y*sixteen)/sixteen
211  del = (y-ysq)*(y+ysq)
212  result = exp(-ysq*ysq) * exp(-del) * result
213  endif
214 
215 !------------------------------------------------------------------
216 ! Evaluate erfc for |X| > 4.0
217 !------------------------------------------------------------------
218  else
219  result = zero
220  if (y >= xbig) then
221  if (jintval /= 2 .or. y >= xmax) goto 300
222  if (y >= xhuge) then
223  result = sqrpi / y
224  goto 300
225  endif
226  endif
227  ysq = one / (y * y)
228  xnum = p(6)*ysq
229  xden = ysq
230 
231  do i = 1, 4
232  xnum = (xnum + p(i)) * ysq
233  xden = (xden + q(i)) * ysq
234  enddo
235 
236  result = ysq *(xnum + p(5)) / (xden + q(5))
237  result = (sqrpi - result) / y
238  if (jintval /= 2) then
239  ysq = aint(y*sixteen)/sixteen
240  del = (y-ysq)*(y+ysq)
241  result = exp(-ysq*ysq) * exp(-del) * result
242  endif
243  endif
244 
245 !------------------------------------------------------------------
246 ! Fix up for negative argument, erf, etc.
247 !------------------------------------------------------------------
248  300 if (jintval == 0) then
249  result = (half - result) + half
250  if (x < zero) result = -result
251  else if (jintval == 1) then
252  if (x < zero) result = two - result
253  else
254  if (x < zero) then
255  if (x < xneg) then
256  result = xinf
257  else
258  ysq = aint(x*sixteen)/sixteen
259  del = (x-ysq)*(x+ysq)
260  y = exp(ysq*ysq) * exp(del)
261  result = (y+y) - result
262  endif
263  endif
264  endif
265 
266  800 return
267 
268  end subroutine calerf
269 
270 !--------------------------------------------------------------------
271 
272  double precision function netlib_specfun_erf(X)
273 
274 ! This subprogram computes approximate values for erf(x).
275 ! (see comments heading CALERF).
276 !
277 ! Author/date: William J. Cody, January 8, 1985
278 
279  implicit none
280 
281  integer jintval
282  double precision X, RESULT
283 
284  jintval = 0
285  call calerf(x,result,jintval)
286  netlib_specfun_erf = result
287 
288  end function netlib_specfun_erf
