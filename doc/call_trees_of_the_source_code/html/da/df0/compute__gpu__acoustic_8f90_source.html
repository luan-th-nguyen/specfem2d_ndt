34  subroutine compute_forces_acoustic_gpu()
35 
36  use specfem_par, only: nproc,ninterface,max_nibool_interfaces_ext_mesh,nibool_interfaces_ext_mesh, &
37  my_neighbours,ninterface_acoustic,inum_interfaces_acoustic, &
38  nelem_acoustic_surface,num_fluid_solid_edges, &
39  stacey_absorbing_conditions,any_elastic,any_poroelastic,simulation_type
40 
41  use specfem_par, only: nspec_outer_acoustic, nspec_inner_acoustic
42 
43  use specfem_par_gpu, only: mesh_pointer,deltatover2f,b_deltatover2f, &
44  buffer_send_scalar_gpu,buffer_recv_scalar_gpu, &
45  b_buffer_send_scalar_gpu,b_buffer_recv_scalar_gpu, &
46  request_send_recv_scalar_gpu,b_request_send_recv_scalar_gpu
47 
48  implicit none
49 
50  ! local parameters
51  integer:: iphase
52 
53  if (nelem_acoustic_surface > 0) then
54  ! enforces free surface (zeroes potentials at free surface)
55  call acoustic_enforce_free_surf_cuda(mesh_pointer)
56  endif
57 
58  ! distinguishes two runs: for elements on MPI interfaces, and elements within the partitions
59  do iphase = 1,2
60 
61  ! acoustic pressure term
62  ! includes code for SIMULATION_TYPE==3
63  call compute_forces_acoustic_cuda(mesh_pointer, iphase, &
64  nspec_outer_acoustic, nspec_inner_acoustic)
65 
66  ! computes additional contributions
67  if (iphase == 1) then
68  ! Stacey absorbing boundary conditions
69  if (stacey_absorbing_conditions) then
70  call compute_stacey_acoustic_gpu(iphase)
71  endif
72 
73  ! elastic coupling
74  if (any_elastic) then
75  if (num_fluid_solid_edges > 0) then
76  ! on GPU
77  call compute_coupling_ac_el_cuda(mesh_pointer,iphase, &
78  num_fluid_solid_edges)
79  endif
80  endif
81 
82  ! poroelastic coupling
83  if (any_poroelastic) then
84  stop 'poroelastic not implemented yet in GPU mode'
85  endif
86 
87  ! sources
88  call compute_add_sources_acoustic_gpu(iphase)
89  endif
90 
91  ! assemble all the contributions between slices using MPI
92  if (ninterface_acoustic > 0) then
93 
94  if (iphase == 1) then
95  ! sends potential_dot_dot_acoustic values to corresponding MPI interface neighbors (non-blocking)
96  call transfer_boun_pot_from_device(mesh_pointer, &
97  buffer_send_scalar_gpu, &
98  1) ! -- 1 == fwd accel
99 
100  call assemble_mpi_scalar_send_cuda(nproc, &
101  buffer_send_scalar_gpu,buffer_recv_scalar_gpu, &
102  ninterface,max_nibool_interfaces_ext_mesh, &
103  nibool_interfaces_ext_mesh, &
104  my_neighbours, &
105  request_send_recv_scalar_gpu,ninterface_acoustic,inum_interfaces_acoustic)
106 
107  ! adjoint simulations
108  if (simulation_type == 3) then
109  call transfer_boun_pot_from_device(mesh_pointer, &
110  b_buffer_send_scalar_gpu, &
111  3) ! -- 3 == adjoint b_accel
112 
113  call assemble_mpi_scalar_send_cuda(nproc, &
114  b_buffer_send_scalar_gpu,b_buffer_recv_scalar_gpu, &
115  ninterface,max_nibool_interfaces_ext_mesh, &
116  nibool_interfaces_ext_mesh, &
117  my_neighbours, &
118  b_request_send_recv_scalar_gpu,ninterface_acoustic,inum_interfaces_acoustic)
119  endif
120 
121  else
122  ! waits for send/receive requests to be completed and assembles values
123  call assemble_mpi_scalar_write_cuda(nproc, &
124  mesh_pointer, &
125  buffer_recv_scalar_gpu, &
126  ninterface, &
127  max_nibool_interfaces_ext_mesh, &
128  request_send_recv_scalar_gpu, &
129  1,ninterface_acoustic,inum_interfaces_acoustic)
130 
131 
132 
133  ! adjoint simulations
134  if (simulation_type == 3) then
135  call assemble_mpi_scalar_write_cuda(nproc, &
136  mesh_pointer, &
137  b_buffer_recv_scalar_gpu, &
138  ninterface, &
139  max_nibool_interfaces_ext_mesh, &
140  b_request_send_recv_scalar_gpu, &
141  3,ninterface_acoustic,inum_interfaces_acoustic)
142  endif
143  endif
144 
145  endif !interface_acoustic
146 
147 
148  enddo
149 
150 
151  ! divides pressure with mass matrix
152  ! corrector:
153  ! updates the chi_dot term which requires chi_dot_dot(t+delta)
154  call kernel_3_acoustic_cuda(mesh_pointer,deltatover2f,b_deltatover2f)
155 
156  ! enforces free surface (zeroes potentials at free surface)
157  call acoustic_enforce_free_surf_cuda(mesh_pointer)
158 
159  end subroutine compute_forces_acoustic_gpu
160

164 
165 ! for acoustic solver on GPU
166 
167  subroutine compute_stacey_acoustic_gpu(iphase)
168 
169  use constants, only: custom_real,ngllx
170 
171  use specfem_par, only: nspec_bottom,nspec_left,nspec_top,nspec_right,b_absorb_acoustic_left,b_absorb_acoustic_right, &
172  b_absorb_acoustic_bottom, b_absorb_acoustic_top,it,nstep,simulation_type,save_forward, &
173  nelemabs
174 
175  use specfem_par_gpu, only: mesh_pointer
176 
177  implicit none
178 
179  ! communication overlap
180  integer,intent(in) :: iphase
181 
182  ! adjoint simulations
183  real(kind=CUSTOM_REAL),dimension(NGLLX,nspec_bottom) :: b_absorb_potential_bottom_slice
184  real(kind=CUSTOM_REAL),dimension(NGLLX,nspec_left) :: b_absorb_potential_left_slice
185  real(kind=CUSTOM_REAL),dimension(NGLLX,nspec_right) :: b_absorb_potential_right_slice
186  real(kind=CUSTOM_REAL),dimension(NGLLX,nspec_top) :: b_absorb_potential_top_slice
187 
188  ! checks if anything to do
189  if (nelemabs == 0) return
190 
191  if (simulation_type == 3) then
192  ! gets absorbing contributions buffers
193  b_absorb_potential_left_slice(:,:) = b_absorb_acoustic_left(:,:,nstep-it+1)
194  b_absorb_potential_right_slice(:,:) = b_absorb_acoustic_right(:,:,nstep-it+1)
195  b_absorb_potential_top_slice(:,:) = b_absorb_acoustic_top(:,:,nstep-it+1)
196  b_absorb_potential_bottom_slice(:,:) = b_absorb_acoustic_bottom(:,:,nstep-it+1)
197  endif
198 
199  ! absorbs absorbing-boundary surface using Sommerfeld condition (vanishing field in the outer-space)
200  call compute_stacey_acoustic_cuda(mesh_pointer,iphase,b_absorb_potential_left_slice,b_absorb_potential_right_slice, &
201  b_absorb_potential_top_slice,b_absorb_potential_bottom_slice)
202 
203  ! adjoint simulations: stores absorbed wavefield part
204  if (simulation_type == 1 .and. save_forward) then
205  ! writes out absorbing boundary value only when second phase is running
206  b_absorb_acoustic_bottom(:,:,it) = b_absorb_potential_bottom_slice(:,:)
207  b_absorb_acoustic_right(:,:,it) = b_absorb_potential_right_slice(:,:)
208  b_absorb_acoustic_top(:,:,it) = b_absorb_potential_top_slice(:,:)
209  b_absorb_acoustic_left(:,:,it) = b_absorb_potential_left_slice(:,:)
210  endif
211 
212  end subroutine compute_stacey_acoustic_gpu
213

218  subroutine compute_add_sources_acoustic_gpu(iphase)
219 
220  use specfem_par, only: it,simulation_type,nstep, nadj_rec_local
221 
222  use specfem_par_gpu, only: mesh_pointer
223 
224  implicit none
225 
226  ! communication overlap
227  integer,intent(in) :: iphase
228 
229  ! forward simulations
230  if (simulation_type == 1) call compute_add_sources_ac_cuda(mesh_pointer,iphase,it)
231 
232  ! adjoint simulations
233  if (simulation_type == 3 .and. nadj_rec_local > 0 .and. it < nstep) then
234  call add_sources_ac_sim_2_or_3_cuda(mesh_pointer,iphase, nstep -it + 1, nadj_rec_local,nstep)
235  endif
236 
237  ! adjoint simulations
238  if (simulation_type == 3) call compute_add_sources_ac_s3_cuda(mesh_pointer,iphase,nstep -it + 1)
239 
240  end subroutine compute_add_sources_acoustic_gpu
