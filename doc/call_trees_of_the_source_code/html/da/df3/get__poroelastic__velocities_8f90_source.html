34 !----
35 !---- subroutine to compute poroelastic velocities cpI, cpII, & cs as a function of the dominant frequency
36 !----
37 
38  subroutine get_poroelastic_velocities(cpIsquare,cpIIsquare,cssquare, &
39  H_biot,C_biot,M_biot,mu_fr,phi, &
40  tort,rho_s,rho_f,eta_f,perm_xx, &
41  fi,f0,Q0,w_c,ATTENUATION_PORO_FLUID_PART)
42 
43  use constants, only: pi
44 
45  implicit none
46 
47  double precision,intent(out) :: cpIsquare,cpIIsquare,cssquare
48 
49  double precision,intent(in) :: H_biot,C_biot,M_biot
50  double precision,intent(in) :: eta_f,rho_f,rho_s,perm_xx
51  double precision,intent(in) :: mu_fr,phi,tort
52 
53  double precision,intent(in) :: fi,f0,Q0
54  double precision,intent(out) :: w_c
55 
56  logical,intent(in) :: ATTENUATION_PORO_FLUID_PART
57 
58  ! local parameters
59  double precision :: rho_bar
60  double precision :: a_r,a_i,b_r,b_i,cc,alpha,aa1,aa2
61  double precision :: xx,yy, gXI, gYI,gXII,gYII,f_c
62  double precision :: taus,taue,bbr,bbi
63  double precision :: wi,w0il
64  double precision :: gA,gB,sa,sb,xxs,yys
65  double precision :: att_I,att_II
66 
67  rho_bar = (1.d0 - phi)*rho_s + phi*rho_f
68 
69  w_c = eta_f*phi/(tort*rho_f*perm_xx)
70 
71  f_c = w_c/(2*pi)
72 
73  wi = 2.d0 * pi * fi
74  w0il = 2.d0 * pi * f0
75 
76  alpha = 10.d0**dlog10(wi)
77 
78  taue = (sqrt(q0*q0+1) + 1)/(w0il*q0)
79  taus = (sqrt(q0*q0+1) - 1)/(w0il*q0)
80 
81  if (attenuation_poro_fluid_part) then
82  ! high frequency, with memory variables
83  bbr = eta_f/perm_xx * (1.d0+alpha*alpha*taus*taue)/(1.d0 + alpha*alpha*taus*taus)
84  bbi = eta_f/perm_xx * alpha*(taue-taus)/(1.d0 + alpha*alpha*taus*taus)
85  else
86  ! low frequency
87  bbr = eta_f/perm_xx
88  bbi = 0.d0
89  endif
90 
91  ! cs
92  ga = (rho_f*tort*rho_bar-phi*rho_f**2)**2/(phi*rho_bar)**2 - (bbr**2-bbi**2)/alpha**2*&
93  (phi*rho_f/(rho_bar*tort) -1.d0) - bbi/alpha*phi*rho_f/(rho_bar*tort)*&
94  (rho_f*tort*rho_bar-phi*rho_f**2)/(phi*rho_bar)
95  gb = -2.d0*bbr*bbi/alpha**2*(phi*rho_f/(rho_bar*tort) -1.d0) + bbr/alpha*phi*rho_f/&
96  (rho_bar*tort)*(rho_f*tort*rho_bar-phi*rho_f**2)/(phi*rho_bar)
97  !
98  sa = (rho_f*tort*rho_bar-phi*rho_f**2)**2/(phi*rho_bar)**2 + (bbr**2-bbi**2)/alpha**2
99  sb = 2.d0*bbr*bbi/alpha**2
100  !
101  xxs = sa*ga + sb*gb
102  yys = ga*sb - sa*gb
103 
104  cssquare = mu_fr/(rho_bar-phi*rho_f/tort) * 2.d0*(ga**2+gb**2)/(sqrt(xxs**2+yys**2)+xxs)
105 
106 
107  ! cpI & cpII
108  a_r = rho_bar - phi*rho_f/tort - phi*rho_bar/(tort*rho_f)*bbi/alpha
109  a_i = phi*rho_bar/(tort*rho_f)*bbr
110  b_r = h_biot + m_biot*phi*rho_bar/(tort*rho_f) - 2.d0*phi*c_biot/tort - &
111  phi*h_biot/(tort*rho_f)*bbi/alpha
112  b_i = phi*h_biot/(tort*rho_f)*bbr
113  cc = phi/(tort*rho_f)*(h_biot*m_biot - c_biot*c_biot)
114  !
115  xx = b_r*b_r - b_i*b_i/(alpha*alpha) - 4.d0*a_r*cc
116  yy = 2.d0*b_r*b_i/alpha - 4.d0*a_i/alpha*cc
117  !
118  gxi = a_r*(b_r + sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx)) + &
119  a_i/alpha*(b_i/alpha + sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))
120 
121  gyi = a_i/alpha*(b_r + sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx)) - &
122  a_r*(b_i/alpha + sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))
123  gyi = -gyi
124 
125  gxii = a_r*(b_r - sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx)) + &
126  a_i/alpha*(b_i/alpha - sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))
127 
128  gyii = a_i/alpha*(b_r - sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx)) - &
129  a_r*(b_i/alpha - sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))
130  gyii = -gyii
131  !
132  !
133  !
134  cpisquare = ((b_r + sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx))**2 + &
135  (b_i/alpha + sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))**2)/&
136  (sqrt(gxi**2+gyi**2) + gxi)
137 
138  cpiisquare = ((b_r - sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx))**2 + &
139  (b_i/alpha - sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))**2)/&
140  (sqrt(gxii**2+gyii**2) + gxii)
141 
142  ! attenuation factors
143  att_i = -alpha*sign(1.d0,yy)*sqrt(sqrt(gxi**2+gyi**2)-gxi) / &
144  sqrt((b_r + sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx))**2+&
145  (b_i/alpha + sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))**2)
146  att_ii = -alpha*sign(1.d0,yy)*sqrt(sqrt(gxii**2+gyii**2)-gxii) / &
147  sqrt((b_r - sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)+xx))**2+&
148  (b_i/alpha - sign(1.d0,yy)*sqrt(0.5)*sqrt(sqrt(xx**2+yy**2)-xx))**2)
149 
150  ! inverse quality factors
151  aa1 = -gyi/gxi
152  aa2 = -gyii/gxii
153 
154  end subroutine get_poroelastic_velocities

161  subroutine get_poroelastic_material(ispec,phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar)
162 
163  use constants, only: four_thirds
164 
165  use specfem_par, only: porosity,tortuosity,poroelastcoef,density,kmato ! AXISYM
166 
167  implicit none
168 
169  integer,intent(in) :: ispec
170 
171  double precision,intent(out) :: phi,tort
172  double precision,intent(out) :: kappa_s,kappa_f,kappa_fr,mu_s,mu_fr,rho_s,rho_f,rho_bar,eta_f
173 
174  ! local parameters
175  integer :: material
176 
177  ! gets associated material
178  material = kmato(ispec)
179 
180  ! gets material properties
181  phi = porosity(material)
182  tort = tortuosity(material)
183 
184  ! solid properties
185  mu_s = poroelastcoef(2,1,material)
186  !if (AXISYM) then ! ABAB !! Warning !! This is false for plane strain (look for: bulk modulus plane strain) CHECK Kappa
187  kappa_s = poroelastcoef(3,1,material) - four_thirds * mu_s
188  !else
189  !kappa_s = poroelastcoef(3,1,material) - mu_s
190  !endif
191  rho_s = density(1,material)
192 
193  ! fluid properties
194  kappa_f = poroelastcoef(1,2,material)
195  rho_f = density(2,material)
196  eta_f = poroelastcoef(2,2,material)
197 
198  ! frame properties
199  mu_fr = poroelastcoef(2,3,material)
200  !if (AXISYM) then ! ABAB !! Warning !! This is false for plane strain (look for: bulk modulus plane strain) CHECK Kappa
201  kappa_fr = poroelastcoef(3,3,material) - four_thirds * mu_fr
202  !else
203  ! kappa_fr = poroelastcoef(3,3,material) - mu_fr
204  !endif
205  ! rho bar
206  rho_bar = (1.d0 - phi) * rho_s + phi * rho_f
207 
208  end subroutine get_poroelastic_material

214  subroutine get_poroelastic_biot_coeff(phi,kappa_s,kappa_f,kappa_fr,mu_fr,D_biot,H_biot,C_biot,M_biot)
215 
216  ! use specfem_par, only: AXISYM
217  use constants, only: four_thirds
218 
219  implicit none
220 
221  double precision,intent(in) :: phi,kappa_s,kappa_f,kappa_fr,mu_fr
222  double precision,intent(out) :: D_biot,H_biot,C_biot,M_biot
223 
224  ! Biot coefficients for the input phi
225  d_biot = kappa_s*(1.d0 + phi*(kappa_s/kappa_f - 1.d0))
226  !if (AXISYM) then ! ABAB !! Warning !! This is possibly false for plane strain (look for: bulk modulus plane strain) CHECK Kappa
227  h_biot = (kappa_s - kappa_fr)*(kappa_s - kappa_fr)/(d_biot - kappa_fr) + kappa_fr + four_thirds*mu_fr
228  !else
229  ! H_biot = (kappa_s - kappa_fr)*(kappa_s - kappa_fr)/(D_biot - kappa_fr) + kappa_fr + mu_fr
230  !endif
231  c_biot = kappa_s*(kappa_s - kappa_fr)/(d_biot - kappa_fr)
232  m_biot = kappa_s*kappa_s/(d_biot - kappa_fr)
233 
234  end subroutine get_poroelastic_biot_coeff
