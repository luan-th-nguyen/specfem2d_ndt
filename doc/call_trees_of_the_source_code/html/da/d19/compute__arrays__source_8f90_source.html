34  subroutine compute_arrays_source(ispec_selected_source,xi_source,gamma_source,sourcearray, &
35  Mxx,Mzz,Mxz,xix,xiz,gammax,gammaz,xigll,zigll,nspec)
36 
37  use constants, only: custom_real,ngllx,ngllz,nglj,ndim,zero
38 
39  use specfem_par, only: axisym,is_on_the_axis,xiglj
40 
41  implicit none
42 
43  integer,intent(in) :: ispec_selected_source
44  integer,intent(in) :: nspec
45 
46  double precision,intent(in) :: xi_source,gamma_source
47  double precision,intent(in) :: Mxx,Mzz,Mxz
48 
49  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec) :: xix,xiz,gammax,gammaz
50 
51  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ) :: sourcearray
52 
53  double precision :: xixd,xizd,gammaxd,gammazd
54 
55 ! Gauss-Lobatto-Legendre points of integration and weights
56  double precision, dimension(NGLLX) :: xigll
57  double precision, dimension(NGLLZ) :: zigll
58 
59 ! source arrays
60  double precision, dimension(NGLLX) :: hxis,hpxis
61  double precision, dimension(NGLLZ) :: hgammas,hpgammas
62 
63  integer :: k,m
64 
65  double precision :: hlagrange
66  double precision :: dsrc_dx,dsrc_dz
67  double precision :: dxis_dx,dgammas_dx
68  double precision :: dxis_dz,dgammas_dz
69 
70 ! compute Lagrange polynomials at the source location
71 ! the source does not necessarily correspond to a Gauss-Lobatto point
72  if (axisym) then
73  if (is_on_the_axis(ispec_selected_source)) then ! TODO verify if we have to add : .and. myrank == islice_selected_source(i)
74  call lagrange_any(xi_source,nglj,xiglj,hxis,hpxis)
75  else
76  call lagrange_any(xi_source,ngllx,xigll,hxis,hpxis)
77  endif
78  else
79  call lagrange_any(xi_source,ngllx,xigll,hxis,hpxis)
80  endif
81  call lagrange_any(gamma_source,ngllz,zigll,hgammas,hpgammas)
82 
83  dxis_dx = zero
84  dxis_dz = zero
85  dgammas_dx = zero
86  dgammas_dz = zero
87 
88  do m = 1,ngllz
89  do k = 1,ngllx
90 
91  xixd = xix(k,m,ispec_selected_source)
92  xizd = xiz(k,m,ispec_selected_source)
93  gammaxd = gammax(k,m,ispec_selected_source)
94  gammazd = gammaz(k,m,ispec_selected_source)
95 
96  hlagrange = hxis(k) * hgammas(m)
97 
98  dxis_dx = dxis_dx + hlagrange * xixd
99  dxis_dz = dxis_dz + hlagrange * xizd
100  dgammas_dx = dgammas_dx + hlagrange * gammaxd
101  dgammas_dz = dgammas_dz + hlagrange * gammazd
102 
103  enddo
104  enddo
105 
106 ! calculate source array
107  sourcearray(:,:,:) = zero
108 
109  do m = 1,ngllz
110  do k = 1,ngllx
111 
112  dsrc_dx = (hpxis(k)*dxis_dx)*hgammas(m) + hxis(k)*(hpgammas(m)*dgammas_dx)
113  dsrc_dz = (hpxis(k)*dxis_dz)*hgammas(m) + hxis(k)*(hpgammas(m)*dgammas_dz)
114 
115  sourcearray(1,k,m) = sourcearray(1,k,m) + real(mxx*dsrc_dx + mxz*dsrc_dx,kind=custom_real)
116  sourcearray(2,k,m) = sourcearray(2,k,m) + real(mxz*dsrc_dx + mzz*dsrc_dz,kind=custom_real)
117 
118  enddo
119  enddo
120 
121  end subroutine compute_arrays_source

127  subroutine read_adj_source(irec_local,adj_source_file)
128 
129 ! reads in adjoint source file
130 
131  use constants, only: iin,max_string_len,custom_real,ndim
132 
133  use specfem_par, only: myrank,nstep,p_sv,seismotype,source_adjoint
134 
135  implicit none
136 
137  integer,intent(in) :: irec_local
138 
139  character(len=MAX_STRING_LEN),intent(in) :: adj_source_file
140 
141  ! local parameters
142  integer :: icomp, itime
143  integer :: ier
144  double precision :: junk
145  double precision,dimension(NSTEP,3) :: adj_src_s
146 
147  character(len=3) :: comp(3)
148  character(len=MAX_STRING_LEN) :: filename
149 
150  ! initializes temporary array
151  ! note: we need to explicitly initialize the full array,
152  ! otherwise it can lead to a floating overflow problem (with intel compiler 15.x)
153  ! when copying the temporary array into the actual storage array adj_sourcearray
154  adj_src_s(:,:) = 0.d0
155 
156  if (seismotype == 1 .or. seismotype == 2 .or. seismotype == 3) then
157  ! displacement/velocity/acceleration
158  comp = (/"BXX","BXY","BXZ"/)
159 
160  ! reads corresponding components
161  do icomp = 1,3
162  ! skips unnecessary components
163  if (p_sv) then
164  ! P_SV-case: skips BXY component
165  if (icomp == 2) cycle
166  else
167  ! SH-case: skips BXX and BXZ components
168  if (icomp == 1 .or. icomp == 3) cycle
169  endif
170 
171  ! reads in ascii adjoint source files **.adj
172  filename = 'SEM/'//trim(adj_source_file) // '.'// comp(icomp) // '.adj'
173  open(unit = iin, file = trim(filename), iostat = ier)
174  if (ier /= 0) then
175  print *,'Error: could not find adjoint source file ',trim(filename)
176  print *,'Please check if file exists...'
177  call exit_mpi(myrank,'file '//trim(filename)//' does not exist')
178  endif
179 
180  do itime = 1, nstep
181  read(iin,*) junk, adj_src_s(itime,icomp)
182  enddo
183  close(iin)
184  enddo
185 
186  else if (seismotype == 4) then
187  ! pressure
188  ! reads in ascii adjoint source files **.PRE.adj
189  filename = 'SEM/'//trim(adj_source_file) // '.PRE.adj'
190  open(unit = iin, file = trim(filename), iostat = ier)
191  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' does not exist')
192 
193  ! reads only single component
194  do itime = 1, nstep
195  read(iin,*) junk, adj_src_s(itime,1)
196  enddo
197  close(iin)
198 
199  else if (seismotype == 6) then
200  ! potential
201  ! reads in ascii adjoint source files **.POT.adj
202  filename = 'SEM/'//trim(adj_source_file) // '.POT.adj'
203  open(unit = iin, file = trim(filename), iostat = ier)
204  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' does not exist')
205 
206  ! reads only single component
207  do itime = 1, nstep
208  read(iin,*) junk, adj_src_s(itime,1)
209  enddo
210  close(iin)
211 
212  endif
213 
214  if (p_sv) then
215  ! P_SV-case
216  source_adjoint(irec_local,:,1) = real(adj_src_s(:,1),kind=custom_real)
217  source_adjoint(irec_local,:,2) = real(adj_src_s(:,3),kind=custom_real)
218  else
219  ! SH-case
220  source_adjoint(irec_local,:,1) = real(adj_src_s(:,2),kind=custom_real)
221  endif
222 
223  end subroutine read_adj_source

229  subroutine read_adj_source_su(seismotype)
230 
231 ! reads in all adjoint sources in SU-file format
232 
233  use constants, only: custom_real,max_string_len,ndim
234 
235  use specfem_par, only: myrank, nstep, nrec, &
236  islice_selected_rec, p_sv,source_adjoint
237 
238  implicit none
239 
240  integer,intent(in) :: seismotype
241

258  else

261  ! P_SV-case
262  write(filename, "('./SEM/Ux_file_single.su.adj')")
263  open(111,file=trim(filename),access='direct',recl=240+4*nstep,iostat = ier)
264  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' does not exist')
265 
266  write(filename, "('./SEM/Uz_file_single.su.adj')")
267  open(113,file=trim(filename),access='direct',recl=240+4*nstep,iostat = ier)
268  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' does not exist')
269  else
270  ! SH-case
271  write(filename, "('./SEM/Uy_file_single.su.adj')")
272  open(112,file=trim(filename),access='direct',recl=240+4*nstep,iostat = ier)
273  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' does not exist')
274  endif
275 
276  endif
277 
278  ! allocates temporary array
279  allocate(adj_src_s(nstep,ndim))
280 
281  irec_local = 0
282  do irec = 1, nrec
283 
284  ! only process/slice holding receiver
285  if (myrank == islice_selected_rec(irec)) then
286  irec_local = irec_local + 1
287 
288  adj_src_s(:,:) = 0.0
289 
290  if (seismotype == 4 .or. seismotype == 6) then
291  ! pressure/potential
292  ! single component
293  read(111,rec=irec,iostat=ier) r4head, adj_src_s(:,1)
294  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' read error')
295  else
296  ! displacement/velocity/acceleration
297  if (p_sv) then
298  ! P_SV-case
299  read(111,rec=irec,iostat=ier) r4head, adj_src_s(:,1)
300  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' read error')
301 
302  read(113,rec=irec,iostat=ier) r4head, adj_src_s(:,2)
303  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' read error')
304  else
305  ! SH-case
306  read(112,rec=irec,iostat=ier) r4head, adj_src_s(:,1)
307  if (ier /= 0) call exit_mpi(myrank,'file '//trim(filename)//' read error')
308  endif
309  endif
310 
311  header2 = int(r4head(29), kind=2)
312 
313  if (p_sv) then
314  ! P_SV-case
315  source_adjoint(irec_local,:,1) = real(adj_src_s(:,1),kind=custom_real)
316  source_adjoint(irec_local,:,2) = real(adj_src_s(:,2),kind=custom_real)
317  else
318  ! SH-case
319  source_adjoint(irec_local,:,1) = real(adj_src_s(:,1),kind=custom_real)
320  endif
321 
322  endif ! if (myrank == islice_selected_rec(irec))
323  enddo ! irec
324 
325  ! closes files
326  if (seismotype == 4) then
327  close(111)
328  else
329  if (p_sv) then
330  ! P_SV-case
331  close(111)
332  close(113)
333  else
334  ! SH-case
335  close(112)
336  endif
337  endif
338 
339  ! frees memory
340  deallocate(adj_src_s)
341 
342  end subroutine read_adj_source_su
