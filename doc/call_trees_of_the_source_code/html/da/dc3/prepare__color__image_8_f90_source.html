35  subroutine prepare_color_image_init()
36 
37  use constants, only: nx_nz_image_max,ngllx
38 
39  use specfem_par, only: coord,npgeo,myrank
40 
41  use specfem_par_movie, only: nx_image_color,nz_image_color, &
42  xmin_color_image,xmax_color_image, &
43  zmin_color_image,zmax_color_image,factor_subsample_image
44  implicit none
45 
46  ! local parameters
47  integer :: npgeo_glob
48  double precision :: xmin_color_image_loc, xmax_color_image_loc, &
49  zmin_color_image_loc,zmax_color_image_loc
50 
51  ! horizontal size of the image
52  xmin_color_image_loc = minval(coord(1,:))
53  xmax_color_image_loc = maxval(coord(1,:))
54 
55  ! vertical size of the image, slightly increase it to go beyond maximum topography
56  zmin_color_image_loc = minval(coord(2,:))
57  zmax_color_image_loc = maxval(coord(2,:))
58 
59 ! global values
60  xmin_color_image = xmin_color_image_loc
61  xmax_color_image = xmax_color_image_loc
62  zmin_color_image = zmin_color_image_loc
63  zmax_color_image = zmax_color_image_loc
64  npgeo_glob = npgeo
65 
66  call min_all_all_dp(xmin_color_image_loc, xmin_color_image)
67  call max_all_all_dp(xmax_color_image_loc, xmax_color_image)
68  call min_all_all_dp(zmin_color_image_loc, zmin_color_image)
69  call max_all_all_dp(zmax_color_image_loc, zmax_color_image)
70 
71  ! collects total on all processes
72  call sum_all_all_i(npgeo, npgeo_glob)
73 
74  zmax_color_image = zmin_color_image + 1.05d0 * (zmax_color_image - zmin_color_image)
75 
76  ! compute number of pixels in the horizontal direction based on typical number
77  ! of spectral elements in a given direction (may give bad results for very elongated models)
78  nx_image_color = nint(sqrt(dble(npgeo_glob))) * (ngllx-1) + 1
79 
80  ! compute number of pixels in the vertical direction based on ratio of sizes
81  nz_image_color = nint(nx_image_color * (zmax_color_image - zmin_color_image) &
82  / (xmax_color_image - xmin_color_image))
83 
84  ! convert pixel sizes to even numbers because easier to reduce size,
85  ! create MPEG movies in postprocessing
86  nx_image_color = 2 * nint((nx_image_color / 2 + 1) / factor_subsample_image)
87  nz_image_color = 2 * nint((nz_image_color / 2 + 1) / factor_subsample_image)
88 
89  ! check that image size is not too big
90 ! because from http://www.jpegcameras.com/libjpeg/libjpeg-2.html
91 ! we know that the size limit of the image in each dimension is 65535:
92 ! "JPEG supports image dimensions of 1 to 64K pixels in either direction".
93  if (nx_image_color < 4) call exit_mpi(myrank,'output image too small: NX_IMAGE_color < 4.')
94  if (nz_image_color < 4) call exit_mpi(myrank,'output image too small: NZ_IMAGE_color < 4.')
95 
96  if (nx_image_color > 65534) &
97  call exit_mpi(myrank,'output image too big: NX_IMAGE_color > 65534; increase factor_subsample_image in DATA/Par_file.')
98  if (nz_image_color > 65534) &
99  call exit_mpi(myrank,'output image too big: NZ_IMAGE_color > 65534; increase factor_subsample_image in DATA/Par_file.')
100 
101  if (nx_image_color > nx_nz_image_max) then
102  print *,'NX_IMAGE_color,NX_NZ_IMAGE_MAX = ',nx_image_color,nx_nz_image_max
103  call exit_mpi(myrank, &
104  'output image too big: NX_IMAGE_color > NX_NZ_IMAGE_MAX; increase factor_subsample_image or change NX_NZ_IMAGE_MAX.')
105  endif
106  if (nz_image_color > nx_nz_image_max) then
107  print *,'NZ_IMAGE_color,NX_NZ_IMAGE_MAX = ',nz_image_color,nx_nz_image_max
108  call exit_mpi(myrank, &
109  'output image too big: NZ_IMAGE_color > NX_NZ_IMAGE_MAX; increase factor_subsample_image or change NX_NZ_IMAGE_MAX.')
110  endif
111 
112  end subroutine prepare_color_image_init

119  subroutine prepare_color_image_pixels()
120 
121  use constants, only: imain,hugeval,ngllx,ngllz
122 
123  use specfem_par, only: myrank,coord,coorg,nspec,knods,ibool, &
124  nsources,nrec,x_source,z_source,st_xval,st_zval
125 
126 
127  use specfem_par_movie, only: nx_image_color,nz_image_color, &
128  xmin_color_image,xmax_color_image, &
129  zmin_color_image,zmax_color_image, &
130  nb_pixel_loc,iglob_image_color, &
131  draw_sources_and_receivers, &
132  ix_image_color_source,iy_image_color_source,ix_image_color_receiver,iy_image_color_receiver
133 
134  implicit none
135 
136  ! local parameters
137  double precision :: size_pixel_horizontal,size_pixel_vertical
138  double precision, dimension(2,4) :: elmnt_coords
139  double precision :: i_coord, j_coord
140  double precision :: dist_pixel, dist_min_pixel
141  integer :: min_i, min_j, max_i, max_j
142  integer :: ispec,i,j,k,l,iglob,ier,pixel_k,pixel_l
143  integer :: nb_pixel_total
144  logical :: pixel_is_in
145 
146  ! create all the pixels
147  if (myrank == 0) then
148  write(imain,*) ' locating all the pixels of color images'
149  call flush_imain()
150  endif
151 
152  size_pixel_horizontal = (xmax_color_image - xmin_color_image) / dble(nx_image_color-1)
153  size_pixel_vertical = (zmax_color_image - zmin_color_image) / dble(nz_image_color-1)
154 
155  ! initializes
156  iglob_image_color(:,:) = -1
157 
158  ! checking which pixels are inside each element
159  nb_pixel_loc = 0
160  do ispec = 1, nspec
161 
162  do k = 1, 4
163  elmnt_coords(1,k) = coorg(1,knods(k,ispec))
164  elmnt_coords(2,k) = coorg(2,knods(k,ispec))
165  enddo
166 
167  ! avoid working on the whole pixel grid
168  min_i = floor(minval((elmnt_coords(1,:) - xmin_color_image))/size_pixel_horizontal) + 1
169  max_i = ceiling(maxval((elmnt_coords(1,:) - xmin_color_image))/size_pixel_horizontal) + 1
170  min_j = floor(minval((elmnt_coords(2,:) - zmin_color_image))/size_pixel_vertical) + 1
171  max_j = ceiling(maxval((elmnt_coords(2,:) - zmin_color_image))/size_pixel_vertical) + 1
172 
173  ! avoid edge effects
174  if (min_i < 1) min_i = 1
175  if (min_j < 1) min_j = 1
176 
177  if (max_i > nx_image_color) max_i = nx_image_color
178  if (max_j > nz_image_color) max_j = nz_image_color
179 
180  do j = min_j, max_j
181  do i = min_i, max_i
182  i_coord = (i-1)*size_pixel_horizontal + xmin_color_image
183  j_coord = (j-1)*size_pixel_vertical + zmin_color_image
184 
185  ! checking if the pixel is inside the element (must be a convex quadrilateral)
186  call is_in_convex_quadrilateral( elmnt_coords, i_coord, j_coord, pixel_is_in)
187 
188  ! if inside, getting the nearest point inside the element!
189  if (pixel_is_in) then
190  dist_min_pixel = hugeval
191  do k = 1, ngllx
192  do l = 1, ngllz
193  iglob = ibool(k,l,ispec)
194  dist_pixel = (coord(1,iglob)-i_coord)**2 + (coord(2,iglob)-j_coord)**2
195  if (dist_pixel < dist_min_pixel) then
196  dist_min_pixel = dist_pixel
197  pixel_l = l
198  pixel_k = k
199  endif
200  enddo
201  enddo
202  ! checks if pixel found
203  if (dist_min_pixel >= hugeval) &
204  call exit_mpi(myrank,'Error in detecting pixel for color image')
205 
206  ! sets closest GLL point for pixel location
207  if (iglob_image_color(i,j) == -1) then
208  ! sets new pixel
209  iglob = ibool(pixel_k,pixel_l,ispec)
210  iglob_image_color(i,j) = iglob
211  nb_pixel_loc = nb_pixel_loc + 1
212  endif
213  endif
214  enddo
215  enddo
216  enddo
217 
218  ! user output
219  call sum_all_i(nb_pixel_loc,nb_pixel_total)
220  if (myrank == 0) then
221  write(imain,*) ' total number of image pixels = ',nb_pixel_total
222  call flush_imain()
223  endif
224 
225 
226 !
227 !---- find pixel position of the sources and receivers
228 !
229  if (draw_sources_and_receivers .and. myrank == 0) then
230 
231  ! find pixel position of the sources with orange crosses
232  allocate(ix_image_color_source(nsources), &
233  iy_image_color_source(nsources),stat=ier)
234  if (ier /= 0) stop 'Error allocating image source arrays'
235 
236  do i = 1,nsources
237  ix_image_color_source(i) = int((x_source(i) - xmin_color_image) / size_pixel_horizontal) + 1
238  iy_image_color_source(i) = int((z_source(i) - zmin_color_image) / size_pixel_vertical) + 1
239 
240  ! avoid edge effects
241  if (ix_image_color_source(i) < 1) ix_image_color_source(i) = 1
242  if (iy_image_color_source(i) < 1) iy_image_color_source(i) = 1
243 
244  if (ix_image_color_source(i) > nx_image_color) ix_image_color_source(i) = nx_image_color
245  if (iy_image_color_source(i) > nz_image_color) iy_image_color_source(i) = nz_image_color
246  enddo
247 
248  ! find pixel position of the receivers with green squares
249  allocate(ix_image_color_receiver(nrec), &
250  iy_image_color_receiver(nrec),stat=ier)
251  if (ier /= 0) stop 'Error allocating image receiver arrays'
252 
253  do i = 1,nrec
254  ix_image_color_receiver(i) = int((st_xval(i) - xmin_color_image) / size_pixel_horizontal) + 1
255  iy_image_color_receiver(i) = int((st_zval(i) - zmin_color_image) / size_pixel_vertical) + 1
256 
257  ! avoid edge effects
258  if (ix_image_color_receiver(i) < 1) ix_image_color_receiver(i) = 1
259  if (iy_image_color_receiver(i) < 1) iy_image_color_receiver(i) = 1
260 
261  if (ix_image_color_receiver(i) > nx_image_color) ix_image_color_receiver(i) = nx_image_color
262  if (iy_image_color_receiver(i) > nz_image_color) iy_image_color_receiver(i) = nz_image_color
263  enddo
264 
265  endif
266 
267  end subroutine prepare_color_image_pixels

275  subroutine prepare_color_image_vp()
276 
277 ! stores P-velocity model in image_color_vp_display
278 
279 #ifdef USE_MPI
280  use mpi
281 #endif
282 
283  use constants, only: ngllx,ngllz,half,two,imain
284 
285  use specfem_par, only: nglob,nspec,ispec_is_elastic,ispec_is_poroelastic,ibool,kmato, &
286  density,poroelastcoef,nproc,myrank,assign_external_model,vpext
287 
288  use specfem_par_movie, only: image_color_vp_display,iglob_image_color, &
289  nx_image_color,nz_image_color,nb_pixel_loc,num_pixel_loc,draw_water_in_blue
290 
291  implicit none
292 
293  ! local parameters
294  double precision :: vp_of_the_model
295  double precision, dimension(:), allocatable :: vp_display
296  double precision :: rhol,mul_relaxed,lambdal_relaxed
297 
298  double precision :: phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar
299  double precision :: D_biot,H_biot,C_biot,M_biot
300 
301  double precision :: afactor,bfactor,cfactor
302  double precision :: cpIsquare
303 
304  integer :: i,j,k,ispec
305 #ifdef USE_MPI
306  double precision, dimension(:), allocatable :: data_pixel_recv
307  double precision, dimension(:), allocatable :: data_pixel_send
308  integer, dimension(:,:), allocatable :: num_pixel_recv
309  integer, dimension(:), allocatable :: tmp_nb_pixel_per_proc
310  integer :: iproc
311 #else
312  integer :: dummy
313 #endif
314 
315  ! user output
316  if (myrank == 0) then
317  write(imain,*) ' coloring image background based on vp'
318  call flush_imain()
319  endif
320 
321  ! to display the P-velocity model in background on color images
322  allocate(vp_display(nglob))
323 
324  do ispec = 1,nspec
325 
326  if (ispec_is_poroelastic(ispec)) then
327  !get parameters of current spectral element
328  call get_poroelastic_material(ispec,phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar)
329 
330  ! Biot coefficients for the input phi
331  call get_poroelastic_biot_coeff(phi,kappa_s,kappa_f,kappa_fr,mu_fr,d_biot,h_biot,c_biot,m_biot)
332 
333  ! Approximated velocities (no viscous dissipation)
334  afactor = rho_bar - phi/tort*rho_f
335  bfactor = h_biot + phi*rho_bar/(tort*rho_f)*m_biot - two*phi/tort*c_biot
336  cfactor = phi/(tort*rho_f)*(h_biot*m_biot - c_biot*c_biot)
337 
338  cpisquare = (bfactor + sqrt(bfactor*bfactor - 4.d0*afactor*cfactor))/(2.d0*afactor)
339 
340  do j = 1,ngllz
341  do i = 1,ngllx
342  vp_display(ibool(i,j,ispec)) = sqrt(cpisquare)
343  enddo
344  enddo
345 
346  else
347  ! get relaxed elastic parameters of current spectral element
348  rhol = density(1,kmato(ispec))
349  lambdal_relaxed = poroelastcoef(1,1,kmato(ispec))
350  mul_relaxed = poroelastcoef(2,1,kmato(ispec))
351  do j = 1,ngllz
352  do i = 1,ngllx
353  !--- if external medium, get elastic parameters of current grid point
354  if (assign_external_model) then
355  vp_display(ibool(i,j,ispec)) = vpext(i,j,ispec)
356  else
357  vp_display(ibool(i,j,ispec)) = sqrt((lambdal_relaxed + 2.d0*mul_relaxed) / rhol)
358  endif
359  enddo
360  enddo
361  endif ! of if (ispec_is_poroelastic(ispec)) then
362 
363 ! now display acoustic layers as constant blue, because they likely correspond to water in the case of ocean acoustics
364 ! or in the case of offshore oil industry experiments.
365 ! (simply turn DRAW_WATER_IN_BLUE off in DATA/Par_file if you want to suppress this (for instance when running
366 ! a purely acoustic simulation with different acoustic media for the oil industry, one then wants to see the different
367 ! acoustic wave speeds displayed as a grey scale).
368 ! For now, in this routine, use -1 as a flag to label such acoustic points
369  if (draw_water_in_blue .and. .not. ispec_is_elastic(ispec) .and. .not. ispec_is_poroelastic(ispec)) then
370  do j = 1,ngllz
371  do i = 1,ngllx
372 
373  !--- if external medium, get elastic parameters of current grid point
374  if (assign_external_model) then
375  vp_of_the_model = vpext(i,j,ispec)
376  else
377  vp_of_the_model = sqrt((lambdal_relaxed + 2.d0*mul_relaxed) / rhol)
378  endif
379 
380 ! test that water is indeed water and not an acoustic version of a sediment for instance
381 ! thus check that Vp is the typical Vp of water
382  if (abs(vp_of_the_model - 1480.d0) <= 50.d0) vp_display(ibool(i,j,ispec)) = -1
383 
384  enddo
385  enddo
386  endif
387 
388  enddo
389 
390  ! getting velocity for each local pixels
391  image_color_vp_display(:,:) = 0.d0
392 
393  do k = 1, nb_pixel_loc
394  j = ceiling(real(num_pixel_loc(k)) / real(nx_image_color))
395  i = num_pixel_loc(k) - (j-1)*nx_image_color
396 
397 ! avoid edge effects
398  if (i < 1) i = 1
399  if (i > nx_image_color) i = nx_image_color
400 
401  if (j < 1) j = 1
402  if (j > nz_image_color) j = nz_image_color
403 
404  if (iglob_image_color(i,j) /= -1) image_color_vp_display(i,j) = vp_display(iglob_image_color(i,j))
405  enddo
406 
407 ! assembling array image_color_vp_display on process zero for color output
408 #ifdef USE_MPI
409 
410  allocate(tmp_nb_pixel_per_proc(0:nproc-1))
411  tmp_nb_pixel_per_proc(:) = 0
412  call gather_all_singlei(nb_pixel_loc,tmp_nb_pixel_per_proc,nproc)
413 
414  if (myrank == 0) then
415  allocate(num_pixel_recv(maxval(tmp_nb_pixel_per_proc(:)),nproc))
416  allocate(data_pixel_recv(maxval(tmp_nb_pixel_per_proc(:))))
417  endif
418  allocate(data_pixel_send(nb_pixel_loc))
419 
420  if (nproc > 1) then
421  if (myrank == 0) then
422  do iproc = 1, nproc-1
423 
424  call recv_i(num_pixel_recv(1,iproc+1), tmp_nb_pixel_per_proc(iproc), iproc, 42)
425  call recv_dp(data_pixel_recv(1),tmp_nb_pixel_per_proc(iproc), iproc, 43)
426 
427  do k = 1, tmp_nb_pixel_per_proc(iproc)
428  j = ceiling(real(num_pixel_recv(k,iproc+1)) / real(nx_image_color))
429  i = num_pixel_recv(k,iproc+1) - (j-1)*nx_image_color
430 
431 ! avoid edge effects
432  if (i < 1) i = 1
433  if (i > nx_image_color) i = nx_image_color
434 
435  if (j < 1) j = 1
436  if (j > nz_image_color) j = nz_image_color
437 
438  image_color_vp_display(i,j) = data_pixel_recv(k)
439  enddo
440  enddo
441 
442  else
443  do k = 1, nb_pixel_loc
444  j = ceiling(real(num_pixel_loc(k)) / real(nx_image_color))
445  i = num_pixel_loc(k) - (j-1)*nx_image_color
446 
447 ! avoid edge effects
448  if (i < 1) i = 1
449  if (i > nx_image_color) i = nx_image_color
450 
451  if (j < 1) j = 1
452  if (j > nz_image_color) j = nz_image_color
453 
454  if (iglob_image_color(i,j) /= -1) data_pixel_send(k) = vp_display(iglob_image_color(i,j))
455  enddo
456 
457  call send_i(num_pixel_loc(1), nb_pixel_loc, 0, 42)
458  call send_dp(data_pixel_send(1), nb_pixel_loc, 0, 43)
459 
460  endif
461  endif
462 
463  deallocate(tmp_nb_pixel_per_proc)
464  deallocate(data_pixel_send)
465  if (myrank == 0) then
466  deallocate(num_pixel_recv)
467  deallocate(data_pixel_recv)
468  endif
469 #else
470  ! to avoid compiler warnings
471  dummy = myrank
472  dummy = nproc
473 #endif
474 
475  end subroutine prepare_color_image_vp
