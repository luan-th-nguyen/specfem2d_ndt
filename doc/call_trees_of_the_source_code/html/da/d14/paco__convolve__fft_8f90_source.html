44  subroutine paco_convolve_fft(Field,label,NSTEP,dt,NFREC,output_field,tp,ts)
45 
46  implicit none
47 
48  integer :: NFREC,N,NSTEP
49 
50  complex(selected_real_kind(15,300)), dimension(NFREC+1) :: Field
51 
52  complex(selected_real_kind(15,300)) :: CR(2*nfrec)
53 
54  double precision, dimension(NSTEP) :: output_field
55 
56  integer :: J,label
57 
58  double precision :: AN,FUN,RAIZ,dt,tp,ts
59 
60  double precision, external :: RIC, deRIC, de2RIC
61 
62  n=2*nfrec
63 
64  an = n
65 
66 !
67 ! label=1, champ U en entree, convolution par un ricker pour U
68 ! label=2, champ U en entree, convolution par la derivee de ricker pour V
69 ! label=3, champ U en entree, convolution par la derivee seconde de ricker pour A
70 ! label=4, champ T en entree, convolution par un ricker
71 !
72 ! flag = 0 on a besoin de U, V et A (pas T)
73 ! flag /= 0 on a besoin de T et V (pas U ni A)
74 !
75 ! NSTEP==1, FLAG==0 (flags: interior=0, left= 1, right= 2, bottom=3)
76 !
77 
78  do j = 1,n
79  if (label == 1 .or. label == 4) fun=ric(j,tp,ts,dt)
80  if (label == 2) fun=deric(j,tp,ts,dt)
81  if (label == 3) fun=de2ric(j,tp,ts,dt)
82  cr(j)=cmplx(fun,0.0d0)
83  enddo
84 
85  CALL fourier_transform(n,cr,-1.0d0)
86 
87  raiz = sqrt(an)
88 
89  CALL sinter(field,output_field,nstep,cr,raiz,nfrec,label,dt)
90 
91  end subroutine paco_convolve_fft

93  subroutine sinter(V,output_field,NSTEP,CR,RAIZ,NFREC,label,dt)
94 
95  implicit none
96 
97  integer NSTEP, j,jn,N,label,nfrec,mult,delay
98 
99  double precision :: RAIZ
100 
101  complex(selected_real_kind(15,300)) :: VC
102 
103  double precision VT(2*nfrec)
104 
105  double precision :: filt,dt
106 
107  double precision, dimension(NSTEP) :: output_field
108 
109  complex(selected_real_kind(15,300)), dimension(NFREC+1) :: V
110 
111  complex(selected_real_kind(15,300)) :: CY(2*nfrec),CR(2*nfrec)
112 
113  n=2*nfrec
114 
115  cy(1) = cr(1) * v(1) * raiz * dt
116 
117  do j = 2,n/2+1
118  filt = 1.0d0
119  vc = v(j)
120  cy(j)= cr(j)*vc * raiz * dt/ filt
121  jn = n-j+2
122  cy(jn)=conjg(cy(j))
123  enddo
124 
125  CALL fourier_transform(n,cy,1.0d0)
126 
127  if (label == 1 .or. label == 3 .or. (label == 2 .and. nstep == 1)) then
128 ! coefficients to take time steps needed (t=0: first time step)
129  mult=1
130  delay=0
131  else if (label == 2 .and. nstep > 1) then
132 ! coefficients to take time steps needed (t=i*deltat+1/2: one step on two starting at 1/2)
133  mult=2
134  delay=0
135  else if (label == 4) then
136 ! coefficients to take time steps needed (t=i*deltat+1: one step on two starting at 1)
137  mult=2
138  delay=1
139  endif
140 
141  do j = 1,nstep
142  cy(mult*j+delay)=cy(mult*j+delay)/raiz/dt
143  vt(mult*j+delay)=REAL(cy(mult*j+delay))
144  output_field(j)=vt(mult*j+delay)
145  enddo
146 
147  end subroutine sinter

150 ! Ricker time function
151 !
152  function ric(J,tp,ts,dt)
153 
154  use constants, only: pi
155 
156  implicit none
157 
158  double precision :: A,RIC,tp,ts,dt
159 
160  integer j
161 
162  a=pi*(dt*(j-1)-ts)/tp
163  a=a*a
164  ric=0.0d0
165  if (a > 30.0d0) return
166  ric=(a-0.5)*exp(-a)
167 
168  end function ric

170 !
171 ! first time derivative of Ricker time function
172 !
173  function deric(J,tp,ts,dt)
174 
175  use constants, only: pi
176 
177  implicit none

181 
182  a=pi*(dt*(j-1)-ts)/tp
183  a=a*a

187  deric=a_dot*(1.5-a)*exp(-a)

191 !
192 ! second time derivative of Ricker time function
193 !
194  function de2ric(J,tp,ts,dt)
195 
196  use constants, only: pi
197 
198  implicit none
199 
200  double precision :: A,A_dot,A_dot_dot,de2RIC,tp,ts,dt
201  integer j
202 
203  a=pi*(dt*(j-1)-ts)/tp
204  a=a*a
205  a_dot=2*(pi/tp)**2*(dt*(j-1)-ts)
206  a_dot_dot=2*(pi/tp)**2
207  de2ric=0.0d0
208  if (a > 30.0d0) return
209  de2ric=(a_dot_dot*(1.5-a)-a_dot*a_dot-a_dot*(1.5-a)*a_dot)*exp(-a)
210 
211  end function de2ric

214 ! Fourier transform
215  subroutine fourier_transform(LX,CX,SIGNI)
216 
217  use constants, only: pi
218 
219  implicit none
220 
221  integer LX,i,j,l,istep,m
222 
223  double precision SC
224 
225  complex(selected_real_kind(15,300)) :: CX(lx),CARG,CW,CTEMP
226 
227  double precision SIGNI
228 
229  j=1
230  sc=sqrt(1.0d0/lx)
231  do i = 1,lx
232  if (i <= j) then
233  ctemp=cx(j)*sc
234  cx(j)=cx(i)*sc
235  cx(i)=ctemp
236  endif
237  m=lx/2
238  do while (m >= 1 .and. m < j)
239  j=j-m
240  m=m/2
241  enddo
242  j=j+m
243  enddo
244  l=1
245 
246  do while(l < lx)
247  istep=2*l
248  DO m= 1,l
249  carg=(0.0d0,1.0d0)*(pi*signi*(m-1))/l
250  cw=exp(carg)
251  DO i=m,lx,istep
252  ctemp=cw*cx(i+l)
253  cx(i+l)=cx(i)-ctemp
254  cx(i)=cx(i)+ctemp
255  enddo
256  enddo
257 
258  l=istep
259  enddo
260 
261  end subroutine fourier_transform
