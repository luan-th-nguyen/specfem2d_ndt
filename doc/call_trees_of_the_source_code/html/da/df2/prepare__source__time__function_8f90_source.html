35  subroutine prepare_source_time_function()
36 
37  ! prepares source_time_function array

116  do i_source = 1,nsources

124  ! if (myrank == islice_selected_source(i_source)) then
125  ! if (AXISYM) then
126  ! if (.not. already_done) then
127  ! if ( assign_external_model ) then
128  ! cp = vpext(0,0,ispec_selected_source(i_source))
129  ! TODO (above): We must interpolate to find the exact cp value at source location
130  ! else
131  ! rho = density(1,kmato(ispec_selected_source(i_source)))
132  ! cp = sqrt(poroelastcoef(3,1,kmato(ispec_selected_source(i_source)))/rho)
133  ! endif
134  ! factor(i_source) = - factor(i_source)*2.0d0*cp**2*0.45d-5 !0.225d-5

136  ! already_done = .true.
137  ! endif
138  ! endif
139  ! endif
140 
141  num_file = 800 + i_source
142 
143  ! note: t0 is the simulation start time, tshift_src is the time shift of the source
144  ! relative to this start time
145 
146  ! checks if trick for better pressure can be applied
147  if (use_trick_for_better_pressure) then
148  trick_ok = (time_function_type(i_source) < 4) .or. (time_function_type(i_source) == 7) .or. &
149  (time_function_type(i_source) == 9) .or. (time_function_type(i_source) == 10)

153  endif
154  endif

156  do i_stage = 1,stage_time_scheme
157 
158  ! loop on all the time steps
159  do it = 1,nstep
160  ! compute current time
161  select case(time_stepping_scheme)
162  case (1)
163  ! Newmark
164  timeval = dble(it-1)*deltat
165  case (2)
166  ! LDDRK: Low-Dissipation and low-dispersion Runge-Kutta
167  timeval = dble(it-1)*deltat + dble(c_lddrk(i_stage))*deltat
168  case (3)
169  ! RK: Runge-Kutta
170  timeval = dble(it-1)*deltat + c_rk(i_stage)*deltat
171  case default
172  call exit_mpi(myrank,'Error invalid time stepping scheme chosen, please check...')
173  end select
174 
175  t_used = timeval - t0 - tshift_src(i_source)
176 
177  ! only process/partition containing source must set STF
178  if (myrank == islice_selected_source(i_source) .or. source_is_moving) then
179 
180  ! determines source_time_function value for different source types
181  select case (time_function_type(i_source))
182  case (1)
183  ! Ricker type: second derivative
184  if (use_trick_for_better_pressure) then
185  ! use a trick to increase accuracy of pressure seismograms in fluid (acoustic) elements:
186  ! use the second derivative of the source for the source time function instead of the source itself,
187  ! and then record -potential_acoustic() as pressure seismograms instead of -potential_dot_dot_acoustic();
188  ! this is mathematically equivalent, but numerically significantly more accurate because in the explicit
189  ! Newmark time scheme acceleration is accurate at zeroth order while displacement is accurate at second order,
190  ! thus in fluid elements potential_dot_dot_acoustic() is accurate at zeroth order while potential_acoustic()
191  ! is accurate at second order and thus contains significantly less numerical noise.
192  ! Second derivative of Ricker source time function :
193  source_time_function(i_source,it,i_stage) = - factor(i_source) * &
194  comp_source_time_function_d2rck(t_used,f0_source(i_source))
195  else
196  ! Ricker (second derivative of a Gaussian) source time function
197  source_time_function(i_source,it,i_stage) = - factor(i_source) * &
198  comp_source_time_function_rickr(t_used,f0_source(i_source))
199  endif
200 
201  case (2)
202  ! Ricker type: first derivative
203 
204  ! for the source time function
205  aval = pi*pi*f0_source(i_source)*f0_source(i_source)
206 
207  if (use_trick_for_better_pressure) then
208  ! use a trick to increase accuracy of pressure seismograms in fluid (acoustic) elements:
209  ! use the second derivative of the source for the source time function instead of the source itself,
210  ! and then record -potential_acoustic() as pressure seismograms instead of -potential_dot_dot_acoustic();
211  ! this is mathematically equivalent, but numerically significantly more accurate because in the explicit
212  ! Newmark time scheme acceleration is accurate at zeroth order while displacement is accurate at second order,
213  ! thus in fluid elements potential_dot_dot_acoustic() is accurate at zeroth order while potential_acoustic()
214  ! is accurate at second order and thus contains significantly less numerical noise.
215  ! Third derivative of Gaussian source time function :
216  source_time_function(i_source,it,i_stage) = factor(i_source) * &
217  comp_source_time_function_d3gaussb(t_used,f0_source(i_source))
218  else
219  ! First derivative of a Gaussian source time function
220  source_time_function(i_source,it,i_stage) = factor(i_source) * &
221  comp_source_time_function_dgaussb(t_used,f0_source(i_source))
222  endif
223 
224  case (3,4)
225  ! Gaussian/Dirac type
226 
227  ! for the source time function
228  aval = pi*pi*f0_source(i_source)*f0_source(i_source)
229 
230  if (use_trick_for_better_pressure) then
231  ! use a trick to increase accuracy of pressure seismograms in fluid (acoustic) elements:
232  ! use the second derivative of the source for the source time function instead of the source itself,
233  ! and then record -potential_acoustic() as pressure seismograms instead of -potential_dot_dot_acoustic();
234  ! this is mathematically equivalent, but numerically significantly more accurate because in the explicit
235  ! Newmark time scheme acceleration is accurate at zeroth order while displacement is accurate at second order,
236  ! thus in fluid elements potential_dot_dot_acoustic() is accurate at zeroth order while potential_acoustic()
237  ! is accurate at second order and thus contains significantly less numerical noise.
238  ! Second derivative of Gaussian :
239  source_time_function(i_source,it,i_stage) = factor(i_source) * &
240  comp_source_time_function_d2gaussb(t_used,f0_source(i_source))
241  else
242  ! Gaussian or Dirac (we use a very thin Gaussian instead) source time function
243  source_time_function(i_source,it,i_stage) = factor(i_source) * &
244  comp_source_time_function_gaussb(t_used,f0_source(i_source))
245  endif
246 
247  case (5)
248  ! Heaviside source time function (we use a very thin error function instead)
249  hdur = 1.d0 / f0_source(i_source)
250  hdur_gauss = hdur * 5.d0 / 3.d0
251 
252  ! convert the half duration for triangle STF to the one for Gaussian STF
253  hdur_gauss = hdur_gauss / source_decay_mimic_triangle
254 
255  ! quasi-Heaviside
256  source_time_function(i_source,it,i_stage) = factor(i_source) * comp_source_time_function_heavi(t_used,hdur_gauss)
257 
258  case (6)
259  ! ocean acoustics type I
260  dect = t0 + tshift_src(i_source)
261  tc = 4.d0 / f0_source(i_source) + dect
262  omega_coa = two * pi * f0_source(i_source)
263 
264  if (timeval > dect .and. timeval < tc) then
265  ! source time function from Computational Ocean Acoustics
266  omegat = omega_coa * ( timeval - dect )
267  source_time_function(i_source,it,i_stage) = factor(i_source) * half * &
268  sin( omegat ) * ( one - cos( quarter * omegat ) )
269  !source_time_function(i_source,it,i_stage) = - factor(i_source) * HALF / omega_coa / omega_coa * &
270  ! ( sin(omegat) - 8.d0 / 9.d0 * sin(3.d0/ 4.d0 * omegat) - 8.d0 / 25.d0 * sin(5.d0 / 4.d0 * omegat) )
271  else
272  source_time_function(i_source,it,i_stage) = zero
273  endif
274 
275  case (7)
276  ! ocean acoustics type II
277  dect = t0 + tshift_src(i_source)
278  tc = 4.d0 / f0_source(i_source) + dect
279  omega_coa = two*pi*f0_source(i_source)
280  if (use_trick_for_better_pressure) then !TODO
281  ! use a trick to increase accuracy of pressure seismograms in fluid (acoustic) elements:
282  ! use the second derivative of the source for the source time function instead of the source itself,
283  ! and then record -potential_acoustic() as pressure seismograms instead of -potential_dot_dot_acoustic();
284  ! this is mathematically equivalent, but numerically significantly more accurate because in the explicit
285  ! Newmark time scheme acceleration is accurate at zeroth order while displacement is accurate at second order,
286  ! thus in fluid elements potential_dot_dot_acoustic() is accurate at zeroth order while potential_acoustic()
287  ! is accurate at second order and thus contains significantly less numerical noise.
288  ! Second derivative of source 7 :
289  if (timeval > dect .and. timeval < tc) then ! t_used > 0 t_used < Nc/f0_source(i_source)) then
290  source_time_function(i_source,it,i_stage) = factor(i_source) * &
291  0.5d0*(one-cos(omega_coa*t_used/4.0d0))*sin(omega_coa*t_used)
292  else
293  source_time_function(i_source,it,i_stage) = zero
294  endif
295  else
296  !Tc = 1.d0 / f0_source(i_source) + DecT ! For source 1 OASES
297  !if (timeval > DecT .and. timeval < Tc) then ! t_used > 0 t_used < Nc/f0_source(i_source)) then
298  ! source_time_function(i_source,it,i_stage) = factor(i_source) * ( & ! Source 1 OASES
299  ! 0.75d0 - cos(omega_coa*t_used) + 0.25d0*cos(TWO*omega_coa*t_used))
300  !else
301  ! source_time_function(i_source,it,i_stage) = ZERO
302  !endif
303  if (timeval > dect .and. timeval < tc) then
304  ! source time function from Computational Ocean Acoustics
305  omegat = omega_coa * ( timeval - dect )
306  !source_time_function(i_source,it,i_stage) = factor(i_source) * HALF / omega_coa / omega_coa * &
307  ! ( sin(omegat) - 8.d0 / 9.d0 * sin(3.d0/ 4.d0 * omegat) - &
308  ! 8.d0 / 25.d0 * sin(5.d0 / 4.d0 * omegat) -1./15.*( timeval - DecT ) + 1./15.*4./f0_source(i_source))
309  source_time_function(i_source,it,i_stage) = factor(i_source) * half / omega_coa / omega_coa * &
310  ( - sin(omegat) + 8.d0 / 9.d0 * sin(3.d0 / 4.d0 * omegat) + &
311  8.d0 / 25.d0 * sin(5.d0 / 4.d0 * omegat) - 1.d0 / 15.d0 * omegat )
312  else if (timeval > dect) then
313  source_time_function(i_source,it,i_stage) = &
314  - factor(i_source) * half / omega_coa / 15.d0 * (4.d0 / f0_source(i_source))
315  else
316  source_time_function(i_source,it,i_stage) = zero
317  endif
318 ! if (timeval > DecT .and. timeval < Tc) then
319 ! ! source time function from Computational Ocean Acoustics
320 ! omegat = omega_coa * ( timeval - DecT )
321 ! !source_time_function(i_source,it,i_stage) = factor(i_source) * HALF / omega_coa / omega_coa * &
322 ! ! ( sin(omegat) - 8.d0 / 9.d0 * sin(3.d0/ 4.d0 * omegat) - &
323 ! ! 8.d0 / 25.d0 * sin(5.d0 / 4.d0 * omegat) -1./15.*( timeval - DecT ) + 1./15.*4./f0_source(i_source))
324 ! source_time_function(i_source,it,i_stage) = factor(i_source) * HALF / omega_coa / omega_coa * &
325 ! ( - sin(omegat) + 8.d0 / 9.d0 * sin(3.d0 / 4.d0 * omegat) + &
326 ! 8.d0 / 25.d0 * sin(5.d0 / 4.d0 * omegat) - 1.d0 / 15.d0 * omegat )
327 ! else if (timeval > DecT) then
328 ! source_time_function(i_source,it,i_stage) = &
329 ! - factor(i_source) * HALF / omega_coa / 15.d0 * (4.d0 / f0_source(i_source))
330 ! else
331 ! source_time_function(i_source,it,i_stage) = ZERO
332 ! endif
333  endif

335  case (8)
336  ! external type
337  ! opens external file to read in source time function
338  if (it == 1) then
339  coeff = factor(i_source)
340  error_msg = error_msg1//trim(name_of_source_file(i_source))

352  ! burst type
353  dect = t0 + tshift_src(i_source)

359  if (use_trick_for_better_pressure) then

388  ! source_time_function(i_source,it,i_stage) = ZERO
389  !endif
390  ! Double integral of burst
391  !if (timeval > DecT .and. timeval < Tc) then ! t_used > 0 t_used < Nc/f0_source(i_source)) then
392  ! source_time_function(i_source,it,i_stage) = - factor(i_source) * ( &
393  ! -sin(TWO*f0_source(i_source)*Pi*t_used)/(8.0d0*f0_source(i_source)**TWO*Pi**2) + &

395  ! (Nc**2*sin((TWO*f0_source(i_source)*(Nc+1)*PI*t_used)/Nc))/(16.0d0*f0_source(i_source)**2*(Nc+1)**2*Pi**2) )
396  !else
397  ! source_time_function(i_source,it,i_stage) = ZERO

399  else

401  source_time_function(i_source,it,i_stage) = - factor(i_source) * &
402  0.5d0*(one-cos(omega_coa*t_used/nc))*sin(omega_coa*t_used)
403  !else if (timeval > DecT) then
404  ! source_time_function(i_source,it,i_stage) = ZERO
405  else
406  source_time_function(i_source,it,i_stage) = zero
407  endif
408  endif
409 
410  case (10)
411  ! Sinus source time function
412  omega_coa = two*pi*f0_source(i_source)
413  if (use_trick_for_better_pressure) then
414  ! use a trick to increase accuracy of pressure seismograms in fluid (acoustic) elements:
415  ! use the second derivative of the source for the source time function instead of the source itself,
416  ! and then record -potential_acoustic() as pressure seismograms instead of -potential_dot_dot_acoustic();
417  ! this is mathematically equivalent, but numerically significantly more accurate because in the explicit
418  ! Newmark time scheme acceleration is accurate at zeroth order while displacement is accurate at second order,

424  else
425  ! First derivative of a Gaussian source time function
426  source_time_function(i_source,it,i_stage) = factor(i_source) * sin(omega_coa*t_used)
427  endif
428 
429  case (11)

431  hdur = 1.0 / 35.0

434 
435  case default

438  end select
439 
440  ! outputs (first source) source time function

443 
444  ! note: earliest start time of the simulation is: (it-1)*deltat - t0 - tshift_src(i_source)

446 
447  endif
448  endif

450  enddo
