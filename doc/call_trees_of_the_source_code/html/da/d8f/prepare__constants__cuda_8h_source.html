92 void setConst_wxgll(realw* array,Mesh* mp)
93 {
94  cudaError_t err = cudaMemcpyToSymbol(d_wxgll, array, NGLLX*sizeof(realw));
95  if (err != cudaSuccess)
96  {
97  fprintf(stderr, "Error in setConst_wxgll: %s\n", cudaGetErrorString(err));
98  exit(1);
99  }
100 #ifdef USE_OLDER_CUDA4_GPU
101  err = cudaGetSymbolAddress((void**)&(mp->d_wxgll),"d_wxgll");
102 #else
103  err = cudaGetSymbolAddress((void**)&(mp->d_wxgll),d_wxgll);
104 #endif
105  if (err != cudaSuccess) {
106  fprintf(stderr, "Error with d_wxgll: %s\n", cudaGetErrorString(err));
107  exit(1);
108  }
109 }
