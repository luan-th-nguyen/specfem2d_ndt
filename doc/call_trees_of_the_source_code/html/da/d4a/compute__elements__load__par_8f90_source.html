29 
30  use part_unstruct_par, only: myrank,nelmnts,nb_edges,abs_surface,nelemabs,glob2loc_elmnts,abs_surface_merge,nelemabs_merge, &
31  nxread,nzread
32  use shared_parameters, only: attenuation_viscoelastic,nelem_pml_thickness,pml_boundary_conditions,read_external_mesh, &
33  num_material,phi_read,qkappa,qmu,absorbbottom,absorbtop,absorbleft,absorbright
34  use constants, only: imain,tinyval

36  integer, dimension(:), allocatable :: elmnts_load
37  integer, dimension(:), allocatable :: adjwgt
38 
39  logical, dimension(:), allocatable :: is_elastic,is_acoustic,is_viscoelastic,is_pml
40 
41  !! acoustic-elastic-poroelastic as well as CPML load balancing:
42  !! we define here the relative cost of all types of spectral elements used in the code.
43  integer, parameter :: elastic_load = 113
44  integer, parameter :: acoustic_load = 46
45  integer, parameter :: viscoelastic_load = 280
46  integer, parameter :: elastic_load_pml = 1049
47  integer, parameter :: acoustic_load_pml = 790
48  integer, parameter :: viscoelastic_load_pml = 1306

55 
56  subroutine compute_elements_load()
57  !----------------------------------------------------------------------------------------------
58  ! compute elements load for efficient partitioning
59  ! (a PML element take more time to calculate than a fluid element for ex)
60  ! Fill load_elmnts array contains the element weight to be considered in mesh decomposition
61  ! in order to obtain a good load inbalance
62  !----------------------------------------------------------------------------------------------
63 
64  implicit none
65 
66  ! local parameters
67  integer :: ier,ispec,nelem_elastic,nelem_acoustic,nelem_viscoelastic,nelem_elastic_pml,nelem_acoustic_pml,nelem_viscoelastic_pml
68 
69  ! Allocations WARNING indices start at zero!
70  allocate(elmnts_load(0:nelmnts-1), &
71  adjwgt(0:nb_edges-1),stat=ier)
72  if (ier /= 0) stop 'Error allocating arrays for weights'
73  allocate(is_acoustic(0:nelmnts-1), &
74  is_elastic(0:nelmnts-1), &
75  is_viscoelastic(0:nelmnts-1),stat=ier)
76  if (ier /= 0) stop 'Error allocating arrays for is_acoustic, is_elastic and is_viscoelastic'
77 
78  ! Initialize counters and arrays
79  nelem_elastic = 0
80  nelem_acoustic = 0
81  nelem_viscoelastic = 0
82  nelem_elastic_pml = 0
83  nelem_acoustic_pml = 0
84  nelem_viscoelastic_pml = 0
85  elmnts_load(:) = elastic_load
86  adjwgt(:) = elastic_load
87  is_acoustic(:) = .false.
88  is_elastic(:) = .false.
89  is_viscoelastic(:) = .false.
90 
91  ! Loop on the elements to determine which element is elastic, acoustic or viscoelastic
92  call determine_elements_type()
93 
94  ! Determine which element is PML and which is not when the mesh has been made by internal mesher.
95  ! For external meshes the detection is made directly when reading the absorbing elements file in
96  ! read_external_mesh_file.F90
97  if (pml_boundary_conditions .and. (.not. read_external_mesh)) then
98  call locate_pml_elements_internal()
99  endif
100 
101  ! Loop on the elements to fill the array elmnts_load and count the elements (for user output)
102  do ispec = 0,nelmnts-1
103  if (is_elastic(ispec)) then
104  if (is_pml(ispec)) then
105  elmnts_load(ispec) = elastic_load_pml
106  nelem_elastic_pml = nelem_elastic_pml + 1
107  else
108  nelem_elastic = nelem_elastic + 1
109  !elmnts_load(ispec) = ELASTIC_LOAD ! Useless: it has been initialized as elastic
110  endif
111  else if (is_viscoelastic(ispec)) then
112  if (is_pml(ispec)) then
113  elmnts_load(ispec) = viscoelastic_load_pml
114  nelem_viscoelastic_pml = nelem_viscoelastic_pml + 1
115  else
116  elmnts_load(ispec) = viscoelastic_load
117  nelem_viscoelastic = nelem_viscoelastic + 1
118  endif
119  else if (is_acoustic(ispec)) then
120  if (is_pml(ispec)) then
121  elmnts_load(ispec) = acoustic_load_pml
122  nelem_acoustic_pml = nelem_acoustic_pml + 1
123  else
124  elmnts_load(ispec) = acoustic_load
125  nelem_acoustic = nelem_acoustic + 1
126  endif
127  endif
128  enddo
129 
130  ! User output
131  if (myrank == 0) then
132  write(imain,*)
133  write(imain,*) '************ Computing elements load ************'
134  write(imain,*) 'Number of elastic elements :',nelem_elastic
135  write(imain,*) 'Number of acoustic elements :',nelem_acoustic
136  write(imain,*) 'Number of viscoelastic elements :',nelem_viscoelastic
137  write(imain,*) 'Number of elastic PML elements :',nelem_elastic_pml
138  write(imain,*) 'Number of acoustic PML elements :',nelem_acoustic_pml
139  write(imain,*) 'Number of viscoelastic PML elements :',nelem_viscoelastic_pml
140  write(imain,*) '*************************************************'
141  write(imain,*)
142  call flush_imain()
143  endif
144 
145  end subroutine compute_elements_load
146 
147 !
148 !---------------------------------------------------------------------------------------

150 
151  subroutine determine_elements_type()
152  ! Loop on the elements to determine which element is elastic, acoustic or viscoelastic
153 
154  implicit none
155 
156  integer :: ispec
157 
158  do ispec = 0,nelmnts-1
159  if (phi_read(num_material(ispec+1)) < tinyval) then
160  is_acoustic(ispec) = .false.
161  is_elastic(ispec) = .true.
162  else if (phi_read(num_material(ispec+1)) >= 1.d0) then
163  is_acoustic(ispec) = .true.
164  is_elastic(ispec) = .false.
165  else
166  is_acoustic(ispec) = .false.
167  is_elastic(ispec) = .false.
168  endif
169  if (attenuation_viscoelastic) then
170  if (((abs(qkappa(num_material(ispec+1)) - 9999.0d0)) > tinyval) .or. &
171  ((abs(qmu(num_material(ispec+1)) - 9999.0d0)) > tinyval)) then
172  is_viscoelastic(ispec) = .true.
173  is_elastic(ispec) = .false.
174  is_acoustic(ispec) = .false.
175  endif
176  endif
177  enddo
178 
179  end subroutine determine_elements_type

185  subroutine locate_pml_elements_internal()
186  !----------------------------------------------------------------------------------------------
187  ! Determine which element is PML and which is not when the mesh has been made by the internal
188  ! mesher.
189  ! The elements are numbered like this :
190  !
191  ! nxread
192  ! <----------------------------------------------------->
193  ! ______________________________________________________
194  ! | | | | | nxread*nzread | ^
195  ! |__________|__________|____|___________|_______________| |
196  ! | ... | ... |....| ... | ... | |
197  ! |__________|__________|____|___________|_______________| |
198  ! | nxread+1 | nxread+2 |....|2*nxread-1 | 2*nxread | | nzread
199  ! |__________|__________|____|___________|_______________| |
200  ! | 1 | 2 |....| nxread-1 | nxread | |
201  ! |__________|__________|____|___________|_______________| v
202  !
203  ! !WARNING! is_pml array starts from 0
204  !
205  !----------------------------------------------------------------------------------------------
206 
207  implicit none
208 
209  ! local parameters
210  integer :: i,j
211 
212  ! PML Left
213  if (absorbleft) then
214  do i = 1,nelem_pml_thickness
215  do j = 1,nzread
216  is_pml((j-1)*nxread + (i-1)) = .true.
217  enddo
218  enddo
219  endif
220  ! PML Right
221  if (absorbright) then
222  do i = nxread-nelem_pml_thickness+1,nxread
223  do j = 1,nzread
224  is_pml((j-1)*nxread + (i-1)) = .true.
225  enddo
226  enddo
227  endif
228  ! PML Down
229  if (absorbbottom) then
230  do i = 1,nxread
231  do j = 1,nelem_pml_thickness
232  is_pml((j-1)*nxread + (i-1)) = .true.
233  enddo
234  enddo
235  endif
236  ! PML Up
237  if (absorbtop) then
238  do i = 1,nxread
239  do j = nzread-nelem_pml_thickness+1,nzread
240  is_pml((j-1)*nxread + (i-1)) = .true.
241  enddo
242  enddo
243  endif
244 
245  end subroutine locate_pml_elements_internal
