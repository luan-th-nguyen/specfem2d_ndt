34 subroutine read_model_nspec()
35 
36 ! reads in nspec from database
37 
38  use tomography_par, only: max_string_len,iin,nspec,myrank
39 
40  implicit none
41 
42  ! local parameters
43  integer :: ier
44  character(len=MAX_STRING_LEN) :: prname
45  character(len=MAX_STRING_LEN) :: dummy
46  integer :: idummy
47  logical :: ldummy
48 
49  ! opens database file
50  write(prname,"('./OUTPUT_FILES/Database',i5.5,'.bin')") myrank
51  open(iin,file=trim(prname),status='old',action='read',form='unformatted',iostat=ier)
52  if (ier /= 0) then
53  print *,'Error opening: ',trim(prname)
54  call exit_mpi(myrank,'file not found')
55  endif
56 
57  ! reads lines until nspec
58  read(iin) dummy ! simulation_title
59  read(iin) idummy,idummy,ldummy,ldummy ! SIMULATION_TYPE, NOISE_TOMOGRAPHY, SAVE_FORWARD, UNDO_ATTENUATION
60  read(iin) nspec
61 
62  close(iin)
63 
64  ! user output
65  if (myrank == 0) then
66  print *,'number of spectral elements: ',nspec
67  print *
68  endif
69 
70 end subroutine read_model_nspec
