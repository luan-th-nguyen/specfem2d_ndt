33 
34  subroutine read_source_file(NSOURCES)
35 
36  ! reads in source file DATA/SOURCE
37 
38  use constants, only: imain,ignore_junk,nlines_per_source,tinyval,pi,mygroup,in_data_files
39  use source_file_par
40  use shared_parameters, only: dt,number_of_simultaneous_runs
41 
42  implicit none
43 
44  integer,intent(in) :: NSOURCES
45 
46  ! local parameters
47  integer :: ier,icounter,i_source,num_sources
48  character(len=256) string_read
49  character(len=MAX_STRING_LEN) :: source_filename,path_to_add
50  integer, parameter :: IIN_SOURCE = 22
51 
52  ! allocates memory arrays
53  allocate(source_surf(nsources), &
54  xs(nsources), &
55  zs(nsources), &
56  source_type(nsources), &
57  time_function_type(nsources), &
58  name_of_source_file(nsources), &
59  burst_band_width(nsources), &
60  f0_source(nsources), &
61  tshift_src(nsources), &
62  anglesource(nsources), &
63  mxx(nsources), &
64  mxz(nsources), &
65  mzz(nsources), &
66  factor(nsources),stat=ier)
67  if (ier /= 0) stop 'Error allocating source arrays'
68 
69  ! initializes
70  xs(:) = 0.d0
71  zs(:) = 0.d0
72 
73  source_type(:) = 0
74  time_function_type(:) = 0
75 
76  f0_source(:) = 0.d0
77  tshift_src(:) = 0.d0
78  anglesource(:) = 0.d0
79  mxx(:) = 0.d0
80  mxz(:) = 0.d0
81  mzz(:) = 0.d0
82 
83  source_filename = trim(in_data_files)//'SOURCE'
84 
85  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
86  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
87  source_filename = path_to_add(1:len_trim(path_to_add))//source_filename(1:len_trim(source_filename))
88  endif
89 
90  ! counts lines
91  open(unit=iin_source,file=trim(source_filename),status='old',action='read',iostat=ier)
92  if (ier /= 0) stop 'Error opening source file, please make sure file exists...'
93 
94  ! counts number of lines
95  icounter = 0
96  do while(ier == 0)
97  read(iin_source,"(a)",iostat=ier) string_read
98 
99  if (ier == 0) then
100  ! suppress trailing carriage return (ASCII code 13) if any (e.g. if input text file coming from Windows/DOS)
101  if (index(string_read,achar(13)) > 0) string_read = string_read(1:index(string_read,achar(13))-1)
102 
103  ! suppress leading and trailing white spaces, if any
104  string_read = adjustl(string_read)
105  string_read = string_read(1:len_trim(string_read))
106 
107  ! if the line is not empty and is not a comment, count it
108  if (len_trim(string_read) > 0 .and. (index(string_read,'#') == 0 .or. index(string_read,'#') > 1)) icounter = icounter + 1
109  endif
110  enddo
111  close(iin_source)
112 
113  ! checks counter
114  if (mod(icounter,nlines_per_source) /= 0) &
115  stop 'total number of non blank and non comment lines in SOURCE file should be a multiple of NLINES_PER_SOURCE'
116 
117  ! total number of sources
118  num_sources = icounter / nlines_per_source
119 
120  ! checks number of sources
121  if (num_sources < 1) stop 'need at least one source in SOURCE file'
122  if (num_sources /= nsources) then
123  print *,'Error invalid num_sources :',num_sources
124  print *,'NSOURCES :',nsources
125  stop 'Error: Total number of sources in DATA/SOURCE is different from that declared in the Par_file, please check...'
126  endif
127 
128  ! reads in source parameters
129  open(unit=iin_source,file=trim(source_filename),status='old',action='read',iostat=ier)
130  if (ier /= 0) stop 'Error opening source file, please make sure file exists...'
131 
132  ! reads in all source informations
133  do i_source= 1,nsources
134 
135  ! source set to surface
136  call read_value_logical(iin_source,ignore_junk,source_surf(i_source))
137 
138  ! x/z location
139  call read_value_double_precision(iin_source,ignore_junk,xs(i_source))
140  call read_value_double_precision(iin_source,ignore_junk,zs(i_source))
141 
142  ! source and source time function type
143  call read_value_integer(iin_source,ignore_junk,source_type(i_source))
144  call read_value_integer(iin_source,ignore_junk,time_function_type(i_source))
145 
146  ! external source time function file (sft type == 8)
147  name_of_source_file(i_source) = ''
148  call read_value_string(iin_source,ignore_junk,name_of_source_file(i_source))
149 
150  ! burst (stf type == 9)
151  call read_value_double_precision(iin_source,ignore_junk,burst_band_width(i_source))
152 
153  ! dominant frequency
154  call read_value_double_precision(iin_source,ignore_junk,f0_source(i_source))
155 
156  ! time shift
157  call read_value_double_precision(iin_source,ignore_junk,tshift_src(i_source))
158 
159  ! force source angle
160  call read_value_double_precision(iin_source,ignore_junk,anglesource(i_source))
161 
162  ! moment tensor
163  call read_value_double_precision(iin_source,ignore_junk,mxx(i_source))
164  call read_value_double_precision(iin_source,ignore_junk,mzz(i_source))
165  call read_value_double_precision(iin_source,ignore_junk,mxz(i_source))
166 
167  ! amplification factor
168  call read_value_double_precision(iin_source,ignore_junk,factor(i_source))
169 
170 
171  ! Dirac/Heaviside
172  ! if Dirac source time function, use a very thin Gaussian instead
173  ! if Heaviside source time function, use a very thin error function instead
174  if (time_function_type(i_source) == 4 .or. time_function_type(i_source) == 5) then
175  f0_source(i_source) = 1.d0 / (10.d0 * dt)
176  endif
177 
178  ! checks source frequency
179  if (abs(f0_source(i_source)) < tinyval) then
180  call exit_mpi(0,'Error source frequency is zero')
181  endif
182 
183  ! convert angle from degrees to radians
184  anglesource(i_source) = anglesource(i_source) * pi / 180.d0
185 
186  ! user output
187  ! note: we will further process source info in solver,
188  ! here we just read in the given specifics and show them
189  write(imain,*) 'Source', i_source
190  write(imain,*) ' Position xs, zs = ',xs(i_source),zs(i_source)
191  write(imain,*)
192 
193  ! source type
194  write(imain,*) ' Source type (1=force, 2=moment tensor, 3=Rayleigh wave, 4=plane incident P, &
195  &5=plane incident S): ',source_type(i_source)
196  select case (source_type(i_source))
197  case (1)
198  ! force
199  write(imain,*) ' Force source:'
200  write(imain,*) ' Angle of the source (deg) = ',anglesource(i_source)
201  write(imain,*) ' Multiplying factor = ',factor(i_source)
202  case (2)
203  ! moment tensor
204  write(imain,*) ' Moment tensor source:'
205  write(imain,*) ' Mxx of the source = ',mxx(i_source)
206  write(imain,*) ' Mzz of the source = ',mzz(i_source)
207  write(imain,*) ' Mxz of the source = ',mxz(i_source)
208  case (3)
209  ! Rayleigh wave
210  write(imain,*) ' Rayleigh wave source:'
211  case (4)
212  ! plane P wave without converted/refracted phases
213  write(imain,*) ' Plane P-wave source without converted/refracted phases:'
214  write(imain,*) ' Angle of the incident wave (deg) = ',anglesource(i_source)
215  case (5)
216  ! plane S wave without converted/refracted phases
217  write(imain,*) ' Plane S-wave source without converted/refracted phases:'
218  write(imain,*) ' Angle of the incident wave (deg) = ',anglesource(i_source)
219  case default
220  ! not supported yet
221  stop 'Error invalid source type! must be 1, 2, 3, 4 or 5, exiting...'
222  end select
223  write(imain,*)
224 
225  ! STF
226  write(imain,*) ' Time function type (1=Ricker, 2=First derivative, 3=Gaussian, 4=Dirac, 5=Heaviside, &
227  &8=Read from file, 9=burst):',time_function_type(i_source)
228  select case (time_function_type(i_source))
229  case (1)
230  write(imain,*) ' Ricker wavelet (second-derivative):'
231  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
232  case (2)
233  write(imain,*) ' Ricker wavelet (first-derivative):'
234  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
235  case (3)
236  write(imain,*) ' Gaussian:'
237  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
238  case (4)
239  write(imain,*) ' Dirac:'
240  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
241  case (5)
242  write(imain,*) ' Heaviside:'
243  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
244  case (6)
245  write(imain,*) ' Ocean acoustics (type I):'
246  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
247  case (7)
248  write(imain,*) ' Ocean acoustics (type II):'
249  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
250  case (8)
251  write(imain,*) ' External source time function file:'
252  write(imain,*) ' Source read from file:',trim(name_of_source_file(i_source))
253  case (9)
254  write(imain,*) ' Burst wavelet:'
255  write(imain,*) ' Burst band width: ',burst_band_width(i_source)
256  case (10)
257  write(imain,*) ' Sinus source time function:'
258  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
259  case (11)
260  write(imain,*) ' Ormsby source time function:'
261  write(imain,*) ' Frequency, delay = ',f0_source(i_source),tshift_src(i_source)
262  case default
263  stop 'Error invalid source time function type! must be between 1 and 9, exiting...'
264  end select
265  write(imain,*)
266 
267  enddo ! do i_source= 1,NSOURCES
268  close(iin_source)
269 
270  end subroutine read_source_file
