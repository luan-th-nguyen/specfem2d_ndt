49  subroutine paco_beyond_critical(anglesource,f0,QD,source_type,left_bound,right_bound, &
50  bot_bound,nleft,nright,nbot,x_source,cploc,csloc)
51 
52  use constants, only: pi
53 
54  use specfem_par, only: coord,nglob,deltat,nstep,attenuation_viscoelastic, &
55  displ_elastic,veloc_elastic,accel_elastic, &
56  v0x_left,v0z_left,v0x_right,v0z_right,v0x_bot,v0z_bot, &
57  t0x_left,t0z_left,t0x_right,t0z_right,t0x_bot,t0z_bot
58 
59  implicit none
60 
61  double precision :: f0,dt,TP,anglesource,QD,delta_in_period
62  double precision,intent(in) :: cploc,csloc
63 
64  integer :: npt,source_type,nleft,nright,nbot
65 
66  integer, dimension(nleft) :: left_bound
67  integer, dimension(nright) :: right_bound
68  integer, dimension(nbot) :: bot_bound
69 
70  integer, dimension(:),allocatable :: local_pt
71 
72  double precision, dimension(:), allocatable :: temp_field
73 
74  integer :: J, indice, NSTEP_local, FLAG, N, NFREC, NFREC1
75 
76  double precision :: ANU,BEALF,ALFBE,RLM,VNX,VNZ,A1,B1,TOTO,FJ,AKA,AQA,GAMR
77 
78 ! location of the point
79  double precision :: X, Z, xmin, xmax, zmin, zmax
80  integer :: inode
81 
82  complex(selected_real_kind(15,300)) :: CAKA,CAQA,UI,UR
83  complex(selected_real_kind(15,300)) :: UX,UZ,SX,SZ,SXZ,A2,B2,AL,AK,AM
84 
85  complex(selected_real_kind(15,300)) :: TX,TZ
86 
87  complex(selected_real_kind(15,300)), dimension(:),allocatable::Field_Ux,Field_Uz,Field_Tx,Field_Tz
88 
89  double precision :: TS,offset,x_source
90 
91 ! size of the model
92  xmin=minval(coord(1,:))
93  xmax=maxval(coord(1,:))
94  zmin=minval(coord(2,:))
95  zmax=maxval(coord(2,:))
96 
97  ts=1.2d0/f0
98 
99 ! dominant period of the Ricker
100  tp=1.d0/f0
101 
102 ! offset to move the initial location of the source in the horizontal direction of the mesh
103  offset = x_source
104 
105 ! find optimal period
106 ! if period is too small, you should see several initial plane wave on your initial field
107  delta_in_period=2.d0
108  do while(delta_in_period < 1.5*abs(xmax-xmin)/csloc)
109  delta_in_period=2.d0*delta_in_period
110  enddo
111 
112 ! test Deltat compatibility
113  dt=256.d0
114  do while(dt > deltat)
115  dt=dt/2.d0
116  enddo
117  if (abs(dt-deltat) > 1.0d-13) then
118  print *, "you must take a deltat that is a power of two (power can be negative)"
119  print *, "for example you can take", dt
120  stop "cannot go further, restart with new deltat"
121  endif
122 
123  dt=deltat/2.d0
124 
125  n=2
126  do while(n < 2*nstep+1)
127  n=2*n
128  enddo
129 
130  do while(dt < (delta_in_period/n))
131  n=2*n
132  enddo
133 
134  print *,'N found to perform the frequency calculation:',n
135  print *,'number of discrete frequencies = ',n/2
136  print *,'delta in period (seconds) = ',delta_in_period
137  print *,'delta in frequency (Hz) = ',1.d0/delta_in_period
138  print *,'dt (here we need deltat/2) = ', dt
139 
140  nfrec=n/2
141  nfrec1=nfrec+1
142 
143 
144 !
145 ! FDT: FUNCION DE TRANSFERENCIA
146 !
147 
148 ! calculation of Poisson's ratio
149  anu = (cploc*cploc-2.d0*csloc*csloc)/(2.d0*(cploc*cploc-csloc*csloc))
150  print *,"Poisson's ratio = ",anu
151 
152  ui=(0.0d0, 1.0d0)
153  ur=(1.0d0, 0.0d0)
154 
155 ! convert angle to radians
156  gamr = anglesource
157 
158  bealf=sqrt((1.0d0-2.0d0*anu)/(2.0d0*(1.0d0-anu)))
159  alfbe=1.0d0/bealf
160  rlm=alfbe**2-2.0d0
161 
162 ! flags: interior=0, left= 1, right= 2, bottom=3
163  do flag=0,3
164 
165  if (flag == 0) then
166  print *,"calculation of the initial field for every point of the mesh"
167  npt=nglob
168  allocate(local_pt(npt))
169  do inode= 1,npt
170  local_pt(inode)=inode
171  enddo
172  nstep_local=1
173  else if (flag == 1) then
174  print *,"calculation of every time step on the left absorbing boundary"
175  npt=nleft
176  allocate(local_pt(npt))
177  local_pt=left_bound
178  nstep_local=nstep
179  else if (flag == 2) then
180  print *,"calculation of every time step on the right absorbing boundary"
181  npt=nright
182  allocate(local_pt(npt))
183  local_pt=right_bound
184  nstep_local=nstep
185  else if (flag == 3) then
186  print *,"calculation of every time step on the bottom absorbing boundary"
187  npt=nbot
188  allocate(local_pt(npt))
189  local_pt=bot_bound
190  nstep_local=nstep
191  endif
192 
193 ! to distinguish all model case and boundary case
194  allocate(temp_field(nstep_local))
195 
196  allocate(field_ux(nfrec1))
197  allocate(field_uz(nfrec1))
198  allocate(field_tx(nfrec1))
199  allocate(field_tz(nfrec1))
200 
201 
202  if (mod(n,2) /= 0) stop 'N must be a multiple of 2'
203 
204 ! normal vector to the edge at this grid point
205 ! therefore corners between two grid edges must be computed twice
206 ! because the normal will change
207  if (flag == 1) then
208  vnz = 0.d0
209  vnx = 1.d0
210  else if (flag == 2) then
211  vnz = 0.d0
212  vnx = 1.d0
213  else if (flag == 3) then
214  vnz = 1.d0
215  vnx = 0.d0
216  else
217  vnz = 0.d0
218  vnx = 0.d0
219  endif
220 
221 
222  do indice= 1,npt
223 
224  if (flag == 0) then
225  inode=indice
226  x=coord(1,indice) - offset
227 ! specfem coordinate axes are implemented from bottom to top whereas for this code
228 ! we need from top to bottom
229  z=zmax-coord(2,indice)
230  else
231  inode=local_pt(indice)
232  x=coord(1,inode) - offset
233 ! specfem coordinate axes are implemented from bottom to top whereas for this code
234 ! we need from top to bottom
235  z=zmax-coord(2,inode)
236  endif
237 
238  if (mod(indice,500) == 0) then
239  print *,indice,"points have been computed out of ",npt
240  endif
241 
242 !
243 ! first handle the particular case of zero frequency
244 !
245  toto=0.01d0
246  if (source_type == 1) CALL ondasp(gamr,0.01d0*bealf,a1,b1,a2,b2,al,ak,am,anu,bealf)
247  if (source_type == 2) CALL ondass(gamr,toto,0.01d0*bealf,a1,b1,a2,b2,al,ak,am,anu,bealf)
248  if (source_type == 3) CALL ondasr(0.01d0*bealf,a1,b1,a2,b2,al,ak,am,anu,bealf)
249 
250 
251  toto=0.0d0
252  CALL desfxy(toto,toto,source_type,ux,uz,sx,sz,sxz,a1,b1,a2,b2,al,ak,am,rlm)
253 
254 ! write the frequency seismograms
255  tx = sx *vnx+sxz*vnz
256  tz = sxz*vnx+sz *vnz
257 
258  field_ux(1)=ux
259  field_uz(1)=uz
260  if (flag /= 0) then
261  field_tx(1)=tx
262  field_tz(1)=tz
263  endif
264 
265 !
266 ! then loop on all the other discrete frequencies
267 !
268  do j = 1,n/2
269 
270 ! compute the value of the frequency (= index * delta in frequency = index * 1/delta in period)
271  fj = dble(j) * 1.d0 / delta_in_period
272 
273 ! pulsation (= 2 * PI * frequency)
274  aka=2.0d0*pi*fj
275 
276  aqa=aka*bealf
277 
278 ! exclude attenuation completely if needed
279  if (attenuation_viscoelastic) then
280  caka=cmplx(aka,-aka/(2.0d0*qd))
281  caqa=cmplx(aqa,-aqa/(2.0d0*qd))
282  else
283  caka=cmplx(aka,0)
284  caqa=cmplx(aqa,0)
285  endif
286 
287  if (source_type == 1) CALL ondasp(gamr,aqa,a1,b1,a2,b2,al,ak,am,anu,bealf)
288  if (source_type == 2) CALL ondass(gamr,aka,aqa,a1,b1,a2,b2,al,ak,am,anu,bealf)
289  if (source_type == 3) CALL ondasr(aqa,a1,b1,a2,b2,al,ak,am,anu,bealf)
290 
291  CALL desfxy(x,z,source_type,ux,uz,sx,sz,sxz,a1,b1,a2,b2,al,ak,am,rlm)
292 
293 ! write the frequency seismograms
294  tx = sx *vnx+sxz*vnz
295  tz = sxz*vnx+sz *vnz
296 
297  field_ux(j+1)=ux
298  field_uz(j+1)=uz
299  if (flag /= 0) then
300  field_tx(j+1)=tx
301  field_tz(j+1)=tz
302  endif
303 
304  enddo
305 
306 ! to convert frequency field in time field
307 ! (number at the end are unit numbers for writing in the good file,
308 ! in the case of the traction we fill only one file per call)
309 
310 ! global model case for initial field
311  if (flag == 0) then
312  call paco_convolve_fft(field_ux,1,nstep_local,dt,nfrec,temp_field,tp,ts)
313  displ_elastic(1,indice)=temp_field(1)
314  call paco_convolve_fft(field_uz,1,nstep_local,dt,nfrec,temp_field,tp,ts)
315  displ_elastic(2,indice)=temp_field(1)
316  call paco_convolve_fft(field_ux,2,nstep_local,dt,nfrec,temp_field,tp,ts)
317  veloc_elastic(1,indice)=temp_field(1)
318  call paco_convolve_fft(field_uz,2,nstep_local,dt,nfrec,temp_field,tp,ts)
319  veloc_elastic(2,indice)=temp_field(1)
320  call paco_convolve_fft(field_ux,3,nstep_local,dt,nfrec,temp_field,tp,ts)
321  accel_elastic(1,indice)=temp_field(1)
322  call paco_convolve_fft(field_uz,3,nstep_local,dt,nfrec,temp_field,tp,ts)
323  accel_elastic(2,indice)=temp_field(1)
324 
325 ! absorbing boundaries
326 
327 ! left case
328  else if (flag == 1) then
329  call paco_convolve_fft(field_ux,2,nstep_local,dt,nfrec,temp_field,tp,ts)
330  v0x_left(indice,:)=temp_field(:)
331  call paco_convolve_fft(field_uz,2,nstep_local,dt,nfrec,temp_field,tp,ts)
332  v0z_left(indice,:)=temp_field(:)
333  call paco_convolve_fft(field_tx,4,nstep_local,dt,nfrec,temp_field,tp,ts)
334  t0x_left(indice,:)=temp_field(:)
335  call paco_convolve_fft(field_tz,4,nstep_local,dt,nfrec,temp_field,tp,ts)
336  t0z_left(indice,:)=temp_field(:)
337 
338 ! right case
339  else if (flag == 2) then
340  call paco_convolve_fft(field_ux,2,nstep_local,dt,nfrec,temp_field,tp,ts)
341  v0x_right(indice,:)=temp_field(:)
342  call paco_convolve_fft(field_uz,2,nstep_local,dt,nfrec,temp_field,tp,ts)
343  v0z_right(indice,:)=temp_field(:)
344  call paco_convolve_fft(field_tx,4,nstep_local,dt,nfrec,temp_field,tp,ts)
345  t0x_right(indice,:)=temp_field(:)
346  call paco_convolve_fft(field_tz,4,nstep_local,dt,nfrec,temp_field,tp,ts)
347  t0z_right(indice,:)=temp_field(:)
348 
349 ! bottom case
350  else if (flag == 3) then
351  call paco_convolve_fft(field_ux,2,nstep_local,dt,nfrec,temp_field,tp,ts)
352  v0x_bot(indice,:)=temp_field(:)
353  call paco_convolve_fft(field_uz,2,nstep_local,dt,nfrec,temp_field,tp,ts)
354  v0z_bot(indice,:)=temp_field(:)
355  call paco_convolve_fft(field_tx,4,nstep_local,dt,nfrec,temp_field,tp,ts)
356  t0x_bot(indice,:)=temp_field(:)
357  call paco_convolve_fft(field_tz,4,nstep_local,dt,nfrec,temp_field,tp,ts)
358  t0z_bot(indice,:)=temp_field(:)
359  endif
360  enddo
361 
362  deallocate(temp_field)
363  deallocate(local_pt)
364 
365  deallocate(field_ux)
366  deallocate(field_uz)
367  deallocate(field_tx)
368  deallocate(field_tz)
369 
370  enddo
371 
372  end subroutine paco_beyond_critical

376  subroutine desfxy(X,Z,ICAS,UX,UZ,SX,SZ,SXZ,A1,B1,A2,B2,AL,AK,AM,RLM)
377 
378  implicit none
379 
380  double precision A1,B1,RLM,X,Z
381  integer ICAS
382  complex(selected_real_kind(15,300)) :: UX,UZ,SX,SZ,SXZ,A2,B2,AL,AK,AM
383  complex(selected_real_kind(15,300)) :: UI,FAC
384  complex(selected_real_kind(15,300)) :: AUX1,AUX2,FI1,FI2,PS1,PS2
385 
386  ui=(0.0d0,1.0d0)
387  if (a1 /= 0.0d0) then
388  aux1=a1*exp(ui*(am*z-al*x)) ! campo P incidente
389  else
390  aux1=cmplx(0.0d0)
391  endif
392  if (a2 /= 0.0d0) then
393  aux2=a2*exp(-ui*(am*z+al*x)) *1.0d0 ! campo P reflejado
394  else
395  aux2=cmplx(0.0d0)
396  endif
397  fi1=aux1+aux2
398  fi2=aux1-aux2
399  if (b1 /= 0.0d0) then
400  aux1=b1*exp(ui*(ak*z-al*x)) ! campo S incidente
401  else
402  aux1=cmplx(0.0d0)
403  endif
404  if (b2 /= 0.0d0) then
405  aux2=b2*exp(-ui*(ak*z+al*x)) *1.0d0 ! campo S reflejado
406  else
407  aux2=cmplx(0.0d0)
408  endif
409  ps1=aux1+aux2
410  ps2=aux1-aux2
411 
412 !
413 ! FAC ES PARA TENER CONSISTENCIA CON AKI & RICHARDS (1980)
414 !
415  fac=ui
416  if (icas == 2) fac=-ui
417 
418  ux=(-ui*al*fi1+ui*ak*ps2)*fac
419 
420  uz=(ui*am*fi2+ui*al*ps1)*fac
421 ! Paco's convention for vertical coordinate axis is inverted
422  uz = - uz
423 
424  aux1=al*al+am*am
425  sx=(-rlm*aux1*fi1-2.0d0*al*(al*fi1-ak*ps2))*fac
426  sz=(-rlm*aux1*fi1-2.0d0*(am*am*fi1+ak*al*ps2))*fac
427 
428  sxz=(2.0d0*am*al*fi2+(al*al-ak*ak)*ps1)*fac
429 ! Paco's convention for vertical coordinate axis is inverted
430  sxz = - sxz
431 
432  end subroutine desfxy
433 
434  subroutine fafb(CA,CB,FA,FB)
435 
436  implicit none
437 
438  double precision CA,CB,A,B
439  complex(selected_real_kind(15,300)) :: FA,FB,ZER,UI
440 
441  zer=(0.0d0,0.0d0)
442  ui=(0.0d0,1.0d0)
443  a=ca*ca-1.0d0
444  b=cb*cb-1.0d0
445 
446  if (ca < 1.0d0) then
447  fa=-ui*sqrt(-a)
448  else
449  fa=sqrt(a)+zer
450  endif
451 
452  if (cb < 1.0d0) then
453  fb=-ui*sqrt(-b)
454  else
455  fb=cmplx(sqrt(b),0.0d0)
456  endif
457 
458  end subroutine fafb
459 
460  subroutine a2b2(FA,FB,A2,B2)
461 
462  implicit none
463 
464  complex(selected_real_kind(15,300)) :: FA,FB,A2,B2,DEN,AUX
465 
466  aux=fb*fb-1.0d0
467  den=4.0d0*fa*fb+aux*aux
468  a2=(4.0d0*fa*fb-aux*aux)/den
469  b2=4.0d0*fa*aux/den
470 
471  end subroutine a2b2
472 
473 ! calculation of P waves
474  subroutine ondasp(GP,AQB,A1,B1,A2,B2,AL,AK,AM,ANU,BEALF)
475 
476  implicit none
477 
478  double precision A1,B1,ANU,CA,CB,GP,AQB,BEALF
479  complex(selected_real_kind(15,300)) :: A2,B2,FA,FB,ZER,AL,AK,AM
480 
481  zer=(0.0d0,0.0d0)
482  bealf=sqrt((1.0d0-2.0d0*anu)/2.0d0/(1.0d0-anu))
483  a1=1.0d0/aqb
484  b1=0.0d0
485 
486  if (gp == 0.0d0) then
487  al=zer
488  ak=zer
489  am=aqb+zer
490  a2=(-1.0d0+zer)/aqb
491  b2=zer
492  return
493  endif
494 
495  ca=1.0d0/sin(gp)
496  cb=ca/bealf
497  al=aqb/ca+zer
498  CALL fafb(ca,cb,fa,fb)
499  ak=al*fb
500  am=al*fa
501  CALL a2b2(fa,fb,a2,b2)
502  a2=a2/aqb
503  b2=b2/aqb
504 
505  end subroutine ondasp
506 
507 ! calculation of S waves
508  subroutine ondass(GS,AKB,AQB,A1,B1,A2,B2,AL,AK,AM,ANU,BEALF)
509 
510  implicit none
511 
512  double precision A1,B1,ANU,CA,CB,GS,AQB,BEALF,AKB
513  complex(selected_real_kind(15,300)) :: A2,B2,FA,FB,ZER,AL,AK,AM
514 
515  zer=(0.0d0,0.0d0)
516  bealf=sqrt((1.0d0-2.0d0*anu)/2.0d0/(1.0d0-anu))
517  a1=0.0d0
518  b1=1.0d0/akb
519 
520  if (gs == 0.0d0) then
521  al=zer
522  ak=akb+zer
523  am=zer
524  a2=zer
525  b2=(-1.0d0+zer)/akb
526  return
527  endif
528 
529  cb=1.0d0/sin(gs)
530  ca=cb*bealf
531 
532 !
533 ! case of the critical angle
534 !
535  if (ca == 1.d0) then
536  al=aqb+zer
537  am=zer
538  CALL fafb(ca,cb,fa,fb)
539  ak=al*fb
540  b2=-b1
541  a2=-4.0d0*cos(gs)*b1/(1./bealf-2.*bealf)
542 
543 ! case of an angle that is not critical
544  ELSE
545  al=aqb/ca+zer
546  CALL fafb(ca,cb,fa,fb)
547  ak=al*fb
548  am=al*fa
549  CALL a2b2(fa,fb,b2,a2)
550  a2=-a2*fb/fa
551  a2=a2/akb
552  b2=b2/akb
553  endif
554 
555  end subroutine ondass
556 
557 ! calculation of Rayleigh waves
558  subroutine ondasr(AQB,A1,B1,A2,B2,AL,AK,AM,ANU,BEALF)
559 
560  implicit none
561 
562  double precision A1,B1,ANU,CA,CB,AQB,BEALF,ba2
563  complex(selected_real_kind(15,300)) :: A2,B2,FA,FB,ZER,AL,AK,AM
564 
565  double precision, external :: crb
566 
567  zer=(0.0d0,0.0d0)
568  a1=0.0d0
569  b1=0.0d0
570  b2=1.0d0+zer
571  bealf=sqrt((1.0d0-2.0d0*anu)/2.0d0/(1.0d0-anu))
572  ba2=bealf*bealf
573  cb=crb(bealf)
574  ca=cb*bealf
575  al=aqb/ca+zer
576 
577  CALL fafb(ca,cb,fa,fb)
578 
579  ak=al*fb
580  am=al*fa
581  a2=2.0d0*fb/(fb*fb-1.0d0)*b2
582  b2=b2/(al*a2+ak)
583  a2=a2*b2
584 
585  end subroutine ondasr

588 !---------
589 !
590 
591  function crb(BEALF)
592 
593  use constants, only: pi
594 
595  implicit none
596 
597  double precision U3,BA2,P,Q,FIND,F1,F2,F12,FACT,CRB,BEALF
598
