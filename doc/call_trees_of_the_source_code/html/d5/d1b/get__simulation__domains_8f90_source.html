34  subroutine get_simulation_domains()
35 
36  use constants, only: tinyval
37 
38  use specfem_par, only: any_acoustic,any_gravitoacoustic,any_elastic,any_poroelastic, &
39  ispec_is_anisotropic,ispec_is_acoustic,ispec_is_elastic,ispec_is_poroelastic,ispec_is_gravitoacoustic, &
40  nspec,porosity,anisotropy,kmato
41 
42  implicit none
43 
44  ! local parameters
45  integer :: ispec
46 
47  ! initializes
48  any_acoustic = .false.
49  any_gravitoacoustic = .false.
50  any_elastic = .false.
51  any_poroelastic = .false.
52 
53  ispec_is_acoustic(:) = .false.
54  ispec_is_gravitoacoustic(:) = .false.
55  ispec_is_anisotropic(:) = .false.
56  ispec_is_elastic(:) = .false.
57  ispec_is_poroelastic(:) = .false.
58 
59  ! loops over all elements
60  do ispec = 1,nspec
61 
62  ! checks domain properties
63  if (nint(porosity(kmato(ispec))) == 1) then
64  ! assume acoustic domain
65  ! if gravitoacoustic -> set by read_external_model
66  ispec_is_acoustic(ispec) = .true.
67  any_acoustic = .true.
68 
69  else if (porosity(kmato(ispec)) < tinyval) then
70  ! assume elastic domain
71  ispec_is_elastic(ispec) = .true.
72  any_elastic = .true.
73  if (any(anisotropy(:,kmato(ispec)) /= 0)) then
74  ispec_is_anisotropic(ispec) = .true.
75  endif
76 
77  else
78  ! assume poroelastic domain
79  ispec_is_poroelastic(ispec) = .true.
80  any_poroelastic = .true.
81  endif
82 
83  enddo ! of do ispec = 1,nspec
84 
85  ! safety checks
86  call get_simulation_domain_check()
87 
88  ! sets domain numbers
89  call get_simulation_domain_counts()
90 
91  end subroutine get_simulation_domains

97  subroutine get_simulation_domains_from_external_models()
98 
99  use constants, only: tinyval,ngllx,ngllz,custom_real
100 
101  use specfem_par, only: any_acoustic,any_gravitoacoustic,any_elastic,any_poroelastic, &
102  ispec_is_anisotropic,ispec_is_acoustic,ispec_is_elastic,ispec_is_poroelastic,ispec_is_gravitoacoustic, &
103  nspec,myrank,p_sv,model
104 
105  ! external model parameters
106  use specfem_par, only: vsext,gravityext,c11ext,c13ext,c15ext,c33ext,c35ext,c55ext
107 
108  implicit none
109 
110  ! local parameters
111  integer :: ispec,i,j
112  real(kind=CUSTOM_REAL) :: previous_vsext
113 
114  ! re-assigns flags
115  ! initializes
116  any_acoustic = .false.
117  any_gravitoacoustic = .false.
118  any_elastic = .false.
119  any_poroelastic = .false.
120 
121  ispec_is_acoustic(:) = .false.
122  ispec_is_gravitoacoustic(:) = .false.
123  ispec_is_anisotropic(:) = .false.
124  ispec_is_elastic(:) = .false.
125  ispec_is_poroelastic(:) = .false.
126 
127  do ispec = 1,nspec
128  ! value at corner
129  previous_vsext = vsext(1,1,ispec)
130 
131  do j = 1,ngllz
132  do i = 1,ngllx
133  ! checks velocities inside element
134  !print *,"vsext(i,j,ispec)",vsext(i,j,ispec)
135  !print *,"gravityext(i,j,ispec)",gravityext(i,j,ispec)
136  if (p_sv .and. (.not. (i == 1 .and. j == 1)) .and. &
137  ((vsext(i,j,ispec) >= tinyval .and. previous_vsext < tinyval) .or. &
138  (vsext(i,j,ispec) < tinyval .and. previous_vsext >= tinyval))) &
139  call exit_mpi(myrank,'external velocity model cannot be both fluid and solid inside the same spectral element')
140 
141  if (vsext(i,j,ispec) < tinyval) then
142  ! acoustic
143  ispec_is_acoustic(ispec) = .true.
144  else
145  ! elastic
146  ispec_is_elastic(ispec) = .true.
147  endif
148 
149  if ( trim(model) == 'external' .or. trim(model) == 'tomo' .or. trim(model) == 'binary_voigt' ) then
150 
151  ! sets element type
152  if (c11ext(i,j,ispec) > tinyval .or. c13ext(i,j,ispec) > tinyval .or. c15ext(i,j,ispec) > tinyval .or. &
153  c33ext(i,j,ispec) > tinyval .or. c35ext(i,j,ispec) > tinyval .or. c55ext(i,j,ispec) > tinyval) then
154  ! anisotropic elastic
155  ispec_is_anisotropic(ispec) = .true.
156  ispec_is_elastic(ispec) = .true.
157  ispec_is_acoustic(ispec) = .false.
158  else if ((vsext(i,j,ispec) < tinyval) .and. (gravityext(i,j,ispec) >= tinyval)) then
159  ! gravito-acoustic
160  ispec_is_gravitoacoustic(ispec) = .true.
161  ispec_is_acoustic(ispec) = .false.
162  endif
163 
164  endif
165 
166  ! sets new GLL point value to compare against
167  previous_vsext = vsext(i,j,ispec)
168  enddo
169  enddo
170  enddo ! ispec
171 
172  ! sets domain numbers
173  call get_simulation_domain_counts()
174 
175  ! safety check
176  call get_simulation_domain_check()
177 
178  end subroutine get_simulation_domains_from_external_models

185  subroutine get_simulation_domain_check()
186 
187  use specfem_par
188 
189  implicit none
190  ! local parameters
191  integer :: ispec
192 
193  ! checks simulation domain flags
194  if (any(ispec_is_acoustic(:)) .neqv. any_acoustic) stop 'Error any_acoustic invalid'
195  if (any(ispec_is_elastic(:)) .neqv. any_elastic) stop 'Error any_elastic invalid'
196  if (any(ispec_is_poroelastic(:)) .neqv. any_poroelastic) stop 'Error any_poroelastic invalid'
197  if (any(ispec_is_gravitoacoustic(:)) .neqv. any_gravitoacoustic) stop 'Error any_gravitoacoustic invalid'
198 
199  ! safety checks
200  if (.not. p_sv .and. .not. any_elastic) then
201  print *, '*************** WARNING ***************'
202  print *, 'Surface (membrane) waves calculation needs an elastic medium'
203  print *, '*************** WARNING ***************'
204  stop 'Please set P_SV flag to .true. for acoustic simulations'
205  endif
206 
207  if (pml_boundary_conditions .and. any_poroelastic) then
208  stop 'PML boundary conditions not implemented for poroelastic simulations yet'
209  endif
210 
211  if (pml_boundary_conditions .and. any_elastic .and. (.not. p_sv)) then
212  stop 'PML boundary conditions not implemented for SH simulations yet'
213  endif
214 
215  ! checks material domains
216  do ispec = 1,nspec
217  ! checks if at least one domain is set
218  if ((.not. ispec_is_acoustic(ispec)) .and. (.not. ispec_is_elastic(ispec)) &
219  .and. (.not. ispec_is_poroelastic(ispec)) .and. (.not. ispec_is_gravitoacoustic(ispec))) then
220  print *,'Error material domain not assigned to element:',ispec
221  print *,'acoustic : ',ispec_is_acoustic(ispec)
222  print *,'elastic : ',ispec_is_elastic(ispec)
223  print *,'poroelastic : ',ispec_is_poroelastic(ispec)
224  print *,'gravitoacoustic: ',ispec_is_gravitoacoustic(ispec)
225  stop 'Error material domain index element'
226  endif
227 
228  ! checks if domain is unique
229  if ((ispec_is_acoustic(ispec) .and. ispec_is_elastic(ispec)) .or. &
230  (ispec_is_acoustic(ispec) .and. ispec_is_poroelastic(ispec)) .or. &
231  (ispec_is_acoustic(ispec) .and. ispec_is_gravitoacoustic(ispec)) .or. &
232  (ispec_is_elastic(ispec) .and. ispec_is_poroelastic(ispec)) .or. &
233  (ispec_is_elastic(ispec) .and. ispec_is_gravitoacoustic(ispec)) .or. &
234  (ispec_is_poroelastic(ispec) .and. ispec_is_gravitoacoustic(ispec))) then
235  print *,'Error material domain assigned twice to element:',ispec
236  print *,'acoustic : ',ispec_is_acoustic(ispec)
237  print *,'elastic : ',ispec_is_elastic(ispec)
238  print *,'poroelastic : ',ispec_is_poroelastic(ispec)
239  print *,'gravitoacoustic: ',ispec_is_gravitoacoustic(ispec)
240  stop 'Error material domain index element'
241  endif
242  enddo
243 
244  end subroutine get_simulation_domain_check

251  subroutine get_simulation_domain_counts()
252 
253 ! counts number of elements in different domains

258

263 
264  ! elastic

266  nspec_elastic = count(ispec_is_elastic(:))
267  if (nspec_elastic > 0 ) any_elastic = .true.
268 
269  ! poroelastic
270  ! number of elastic elements in this partition
271  nspec_poroelastic = count(ispec_is_poroelastic(:))
272  if (nspec_poroelastic > 0 ) any_poroelastic = .true.
273 
274  ! gravito-acoustic
275  ! number of elastic elements in this partition
276  nspec_gravitoacoustic = count(ispec_is_gravitoacoustic(:))
277  if (nspec_gravitoacoustic > 0 ) any_gravitoacoustic = .true.
278 
279  end subroutine get_simulation_domain_counts
