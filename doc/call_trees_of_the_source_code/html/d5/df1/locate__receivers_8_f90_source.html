35 !---- locate_receivers finds the correct position of the receivers
36 !----
37 
38  subroutine locate_receivers(ibool,coord,nspec,nglob,xigll,zigll, &
39  nrec,nrecloc,recloc,islice_selected_rec,NPROC,myrank, &
40  st_xval,st_zval,ispec_selected_rec, &
41  xi_receiver,gamma_receiver,station_name,network_name, &
42  x_source,z_source, &
43  coorg,knods,ngnod,npgeo, &
44  x_final_receiver, z_final_receiver)
45 
46  use constants, only: ndim,ngllx,ngllz,max_length_station_name,max_length_network_name, &
47  imain,hugeval,tinyval,num_iter,mygroup,max_string_len,in_data_files
48 
49  use specfem_par, only: axisym,is_on_the_axis,xiglj,ispec_is_acoustic,use_trick_for_better_pressure,number_of_simultaneous_runs
50 
51  implicit none
52 
53  integer :: nrec,nspec,nglob,ngnod,npgeo
54  integer, intent(in) :: NPROC, myrank
55 
56  integer :: knods(ngnod,nspec)
57  double precision :: coorg(ndim,npgeo)
58 
59  integer, dimension(NGLLX,NGLLZ,nspec) :: ibool
60 
61 ! array containing coordinates of the points
62  double precision :: coord(ndim,nglob)
63 
64  integer :: irec,i,j,ispec,iglob,iter_loop,ix_initial_guess,iz_initial_guess
65 
66  double precision :: x_source,z_source,dist_squared,stele,stbur
67  double precision, dimension(nrec) :: distance_receiver
68  double precision :: xi,gamma,dx,dz,dxi,dgamma
69 
70 ! Gauss-Lobatto-Legendre points of integration
71  double precision :: xigll(ngllx)
72  double precision :: zigll(ngllz)
73 
74  double precision :: x,z,xix,xiz,gammax,gammaz,jacobian
75 
76 ! use dynamic allocation
77  double precision :: distmin_squared
78  double precision, dimension(:), allocatable :: final_distance
79 
80 ! receiver information
81  integer :: nrecloc
82  integer, dimension(nrec) :: ispec_selected_rec, recloc
83  double precision, dimension(nrec) :: xi_receiver,gamma_receiver
84 
85 ! station information for writing the seismograms
86  character(len=MAX_LENGTH_STATION_NAME), dimension(nrec) :: station_name
87  character(len=MAX_LENGTH_NETWORK_NAME), dimension(nrec) :: network_name
88 
89  double precision, dimension(nrec) :: st_xval,st_zval
90 
91 ! tangential detection
92  double precision, dimension(nrec) :: x_final_receiver, z_final_receiver
93 
94  double precision, dimension(nrec,NPROC) :: gather_final_distance
95  double precision, dimension(nrec,NPROC) :: gather_xi_receiver, gather_gamma_receiver
96 
97  integer, dimension(nrec), intent(inout) :: islice_selected_rec
98  integer, dimension(:,:), allocatable :: gather_ispec_selected_rec
99  integer :: ier
100  character(len=MAX_STRING_LEN) :: stations_filename,path_to_add
101 
102  stations_filename = trim(in_data_files)//'STATIONS'
103 
104  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
105  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
106  stations_filename = path_to_add(1:len_trim(path_to_add))//stations_filename(1:len_trim(stations_filename))
107  endif
108 
109  ! user output
110  if (myrank == 0) then
111  write(imain,*)
112  write(imain,*) '********************'
113  write(imain,*) ' locating receivers'
114  write(imain,*) '********************'
115  write(imain,*)
116  write(imain,*) 'reading receiver information from the '//trim(stations_filename)//' file'
117  write(imain,*)
118  call flush_imain()
119  endif
120 
121  open(unit= 1,file=trim(stations_filename),status='old',action='read')
122 
123 ! allocate memory for arrays using number of stations
124  allocate(final_distance(nrec))
125 
126 ! loop on all the stations
127  do irec= 1,nrec
128 
129  ! set distance to huge initial value
130  distmin_squared = hugeval
131 
132  read(1,*) station_name(irec),network_name(irec),st_xval(irec),st_zval(irec),stele,stbur
133 
134  ! check that station is not buried, burial is not implemented in current code
135  if (abs(stbur) > tinyval) call exit_mpi(myrank,'stations with non-zero burial not implemented yet')
136 
137  ! compute distance between source and receiver
138  distance_receiver(irec) = sqrt((st_zval(irec)-z_source)**2 + (st_xval(irec)-x_source)**2)
139 
140  do ispec= 1,nspec
141 
142  ! loop only on points inside the element
143  ! exclude edges to ensure this point is not shared with other elements
144  do j = 2,ngllz-1
145  do i = 2,ngllx-1
146 
147  iglob = ibool(i,j,ispec)
148 
149  ! we compare squared distances instead of distances themselves to significantly speed up calculations
150  dist_squared = (st_xval(irec)-dble(coord(1,iglob)))**2 + (st_zval(irec)-dble(coord(2,iglob)))**2
151 
152  ! keep this point if it is closer to the receiver
153  if (dist_squared < distmin_squared) then
154  distmin_squared = dist_squared
155  ispec_selected_rec(irec) = ispec
156  ix_initial_guess = i
157  iz_initial_guess = j
158  endif
159 
160  enddo
161  enddo
162 
163  ! end of loop on all the spectral elements
164  enddo
165 
166  ! ****************************************
167  ! find the best (xi,gamma) for each receiver
168  ! ****************************************
169  ! use initial guess in xi and gamma
170 
171  if (axisym) then
172  if (is_on_the_axis(ispec_selected_rec(irec))) then
173  xi = xiglj(ix_initial_guess)
174  else
175  xi = xigll(ix_initial_guess)
176  endif
177  else
178  xi = xigll(ix_initial_guess)
179  endif
180  gamma = zigll(iz_initial_guess)
181 
182  ! iterate to solve the non linear system
183  do iter_loop = 1,num_iter
184  ! compute coordinates of the new point and derivatives dxi/dx, dxi/dz
185  call recompute_jacobian(xi,gamma,x,z,xix,xiz,gammax,gammaz,jacobian, &
186  coorg,knods,ispec_selected_rec(irec),ngnod,nspec,npgeo, &
187  .true.)
188 
189  ! compute distance to target location
190  dx = - (x - st_xval(irec))
191  dz = - (z - st_zval(irec))
192 
193  ! compute increments
194  dxi = xix*dx + xiz*dz
195  dgamma = gammax*dx + gammaz*dz
196 
197  ! update values
198  xi = xi + dxi
199  gamma = gamma + dgamma
200 
201  ! impose that we stay in that element
202  ! (useful if user gives a receiver outside the mesh for instance)
203  ! we can go slightly outside the [1,1] segment since with finite elements
204  ! the polynomial solution is defined everywhere
205  ! this can be useful for convergence of itertive scheme with distorted elements
206  if (xi > 1.10d0) xi = 1.10d0
207  if (xi < -1.10d0) xi = -1.10d0
208  if (gamma > 1.10d0) gamma = 1.10d0
209  if (gamma < -1.10d0) gamma = -1.10d0
210 
211  ! end of non linear iterations
212  enddo
213 
214  ! compute final coordinates of point found
215  call recompute_jacobian(xi,gamma,x,z,xix,xiz,gammax,gammaz,jacobian, &
216  coorg,knods,ispec_selected_rec(irec),ngnod,nspec,npgeo, &
217  .true.)
218 
219  ! store xi,gamma of point found
220  xi_receiver(irec) = xi
221  gamma_receiver(irec) = gamma
222 
223  ! compute final distance between asked and found
224  final_distance(irec) = sqrt((st_xval(irec)-x)**2 + (st_zval(irec)-z)**2)
225 
226  x_final_receiver(irec) = x
227  z_final_receiver(irec) = z
228 
229  enddo
230 
231  ! close receiver file
232  close(1)
233 
234  ! select one mesh slice for each receiver
235  allocate(gather_ispec_selected_rec(nrec,nproc),stat=ier)
236  if (ier /= 0) call exit_mpi(myrank,'Error allocating gather array')
237 
238  ! gathers infos onto master process
239  call gather_all_dp(final_distance(1),nrec,gather_final_distance(1,1),nrec,nproc)
240  call gather_all_dp(xi_receiver(1),nrec,gather_xi_receiver(1,1),nrec,nproc)
241  call gather_all_dp(gamma_receiver(1),nrec,gather_gamma_receiver(1,1),nrec,nproc)
242 
243  call gather_all_i(ispec_selected_rec(1),nrec,gather_ispec_selected_rec(1,1),nrec, nproc)
244 
245  if (myrank == 0) then
246  ! selects best slice which minimum distance to receiver location
247  do irec = 1, nrec
248  islice_selected_rec(irec:irec) = minloc(gather_final_distance(irec,:)) - 1
249  enddo
250  endif
251  call bcast_all_i(islice_selected_rec(1),nrec)
252 
253  if (use_trick_for_better_pressure) then
254  do irec= 1,nrec
255  if (myrank == islice_selected_rec(irec)) then
256  if (.not. ispec_is_acoustic(ispec_selected_rec(irec))) then
257  call exit_mpi(myrank,'USE_TRICK_FOR_BETTER_PRESSURE : receivers must be in acoustic elements')
258  endif
259  endif
260  enddo
261  endif
262 
263  ! counts local receivers in this slice
264  nrecloc = 0
265  do irec = 1, nrec
266  if (myrank == islice_selected_rec(irec)) then
267  nrecloc = nrecloc + 1
268  recloc(nrecloc) = irec
269  endif
270  enddo
271 
272  ! user output
273  if (myrank == 0) then
274  do irec = 1, nrec
275  write(imain,*)
276  write(imain,*) 'Station # ',irec,' ',network_name(irec),station_name(irec)
277 
278  if (gather_final_distance(irec,islice_selected_rec(irec)+1) == hugeval) &
279  call exit_mpi(myrank,'Error locating receiver')
280 
281  write(imain,*) ' original x: ',sngl(st_xval(irec))
282  write(imain,*) ' original z: ',sngl(st_zval(irec))
283  write(imain,*) ' distance from source: ',sngl(distance_receiver(irec))
284  write(imain,*) 'closest estimate found: ',sngl(gather_final_distance(irec,islice_selected_rec(irec)+1)), &
285  ' m away'
286  write(imain,*) ' in element ',gather_ispec_selected_rec(irec,islice_selected_rec(irec)+1)
287  write(imain,*) ' in rank ', islice_selected_rec(irec)
288  write(imain,*) ' at xi,gamma coordinates = ',gather_xi_receiver(irec,islice_selected_rec(irec)+1), &
289  gather_gamma_receiver(irec,islice_selected_rec(irec)+1)
290  write(imain,*)
291  enddo
292 
293  write(imain,*)
294  write(imain,*) 'end of receiver detection'
295  write(imain,*)
296  call flush_imain()
297  endif
298 
299  ! deallocate arrays
300  deallocate(final_distance)
301  deallocate(gather_ispec_selected_rec)
302 
303  end subroutine locate_receivers
