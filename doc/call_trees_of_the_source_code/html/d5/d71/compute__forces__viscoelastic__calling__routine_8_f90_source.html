34  subroutine compute_forces_viscoelastic_main()
35 
36  use constants, only: source_is_moving,use_enforce_fields,alpha_lddrk,beta_lddrk
37  use specfem_par
38  use specfem_par_noise
39 
40  implicit none
41 
42  ! local parameters
43  integer :: i,iglob
44  ! non-blocking MPI
45  ! iphase: iphase = 1 is for computing outer elements (on MPI interface),
46  ! iphase = 2 is for computing inner elements
47  integer :: iphase
48 
49  ! checks if anything to do in this slice
50  if ((.not. any_elastic) .and. (.not. source_is_moving)) return
51 
52  ! enforces vanishing wavefields on axis
53  if (axisym) then
54  call enforce_zero_radial_displacements_on_the_axis()
55  endif
56 
57  ! viscous attenuation for elastic media
58  if (attenuation_viscoelastic) call compute_attenuation_viscoelastic(displ_elastic,displ_elastic_old, &
59  ispec_is_elastic,pml_boundary_conditions,e1,e11,e13)
60 
61  ! distinguishes two runs: for elements on MPI interfaces (outer), and elements within the partitions (inner)
62  do iphase = 1,2
63 
64  ! main solver for the elastic elements
65  ! visco-elastic term
66  call compute_forces_viscoelastic(accel_elastic,veloc_elastic,displ_elastic,displ_elastic_old, &
67  pml_boundary_conditions,e1,e11,e13,iphase)
68 
69  ! computes additional contributions to acceleration field
70  if (iphase == 1) then
71 
72  ! Stacey boundary conditions
73  if (stacey_absorbing_conditions) then
74  call compute_stacey_elastic(accel_elastic,veloc_elastic)
75  endif
76 
77  ! PML boundary
78  if (pml_boundary_conditions) then
79  call pml_boundary_elastic(accel_elastic,veloc_elastic,displ_elastic,displ_elastic_old)
80  endif
81 
82  ! add coupling with the acoustic side
83  if (coupled_acoustic_elastic) then
84  call compute_coupling_viscoelastic_ac()
85  endif
86 
87  ! add coupling with the poroelastic side
88  if (coupled_elastic_poro) then
89  call compute_coupling_viscoelastic_po()
90  endif
91 
92  ! add force source
93  if (.not. initialfield) then
94 
95  select case(noise_tomography)
96  case (0)
97  ! earthquake/force source
98  if (simulation_type == 1) then
99  if (source_is_moving) then
100  call compute_add_sources_viscoelastic_moving_source(accel_elastic,it,i_stage)
101  else
102  call compute_add_sources_viscoelastic(accel_elastic,it,i_stage)
103  endif
104  endif
105 
106  case (1)
107  ! noise source at master station
108  call add_point_source_noise()
109 
110  case (2)
111  ! inject generating wavefield for noise simulations
112  call add_surface_movie_noise(accel_elastic)
113  end select
114 
115  ! adjoint wavefield source
116  if (simulation_type == 3) then
117  ! adjoint sources
118  call compute_add_sources_viscoelastic_adjoint()
119  endif
120  endif
121 
122  endif
123 
124  ! enforces vanishing wavefields on axis
125  if (axisym) then
126  call enforce_zero_radial_displacements_on_the_axis()
127  endif
128 
129 #ifdef USE_MPI
130  ! LDDRK
131  ! daniel: when is this needed? veloc_elastic at it == 1 and i_stage == 1 is zero for non-initialfield simulations.
132  ! todo - please check...
133  if (time_stepping_scheme == 2) then
134  if (i_stage == 1 .and. it == 1 .and. iphase == 2 .and. (.not. initialfield)) then
135  ! debug
136  !print *,'debug veloc min/max = ',minval(veloc_elastic(:,:)),maxval(veloc_elastic(:,:))
137  veloc_elastic_lddrk_temp(:,:) = veloc_elastic(:,:)
138 
139  ! assembles velocity
140  call assemble_mpi_vector_el_blocking(veloc_elastic)
141  endif
142  endif
143 
144  ! assemble all the contributions between slices using MPI
145  if (nproc > 1 .and. ninterface_elastic > 0) then
146  ! collects all contributions on shared degrees of freedom
147  !call assemble_MPI_vector_el_blocking(accel_elastic)
148 
149  ! sends out MPI interface data
150  if (iphase == 1) then
151  ! sends accel values to corresponding MPI interface neighbors
152  call assemble_mpi_vector_el_s(accel_elastic)
153  else
154  ! waits for send/receive requests to be completed and assembles values
155  call assemble_mpi_vector_el_w(accel_elastic)
156  endif
157  endif
158 #endif
159 
160  enddo ! iphase
161 
162  ! saves boundary condition for reconstruction
163  if (pml_boundary_conditions) then
164  if (nglob_interface > 0) then
165  if (save_forward .and. simulation_type == 1) then
166  do i = 1, nglob_interface
167  write(71) accel_elastic(1,point_interface(i)),accel_elastic(2,point_interface(i)), &
168  veloc_elastic(1,point_interface(i)),veloc_elastic(2,point_interface(i)), &
169  displ_elastic(1,point_interface(i)),displ_elastic(2,point_interface(i))
170  enddo
171  endif
172  endif
173  endif
174 
175  ! multiply by the inverse of the mass matrix and update velocity
176  !! DK DK this should be vectorized
177  if (use_enforce_fields) then
178  do iglob = 1,nglob_elastic
179  if (.not. iglob_is_forced(iglob)) then
180  accel_elastic(:,iglob) = accel_elastic(:,iglob) * rmass_inverse_elastic(:,iglob)
181  endif
182  enddo
183  else
184  accel_elastic(:,:) = accel_elastic(:,:) * rmass_inverse_elastic(:,:)
185  endif
186 
187  ! time stepping
188  select case (time_stepping_scheme)
189  case (1)
190  ! Newmark
191  call update_veloc_elastic_newmark()
192  case (2)
193  ! LDDRK
194  call update_veloc_elastic_lddrk()
195  case (3)
196  ! RK
197  call update_veloc_elastic_rk()
198  end select
199 
200  end subroutine compute_forces_viscoelastic_main
201

206  subroutine compute_forces_viscoelastic_main_backward()
207 
208  use constants, only: noise_save_everywhere
209  use specfem_par
210  use specfem_par_noise
211 
212  implicit none
213 
214  ! local parameters
215  integer :: it_temp,istage_temp
216  ! non blocking MPI
217  ! iphase: iphase = 1 is for computing outer elements (on MPI interface),
218  ! iphase = 2 is for computing inner elements
219  integer :: iphase
220 
221  ! checks if anything to do
222  if (simulation_type /= 3 ) return
223 
224  ! checks if anything to do in this slice
225  if (.not. any_elastic) return
226 
227  ! timing
228  if (undo_attenuation) then
229  ! time increment
230  ! example: NSTEP = 800, NT_DUMP_ATTENUATION = 500 -> 1. subset: it_temp = (2-1)*500 + 1 = 501,502,..,800
231  ! -> 2. subset: it_temp = (2-2)*500 + 1 = 1,2,..,500
232  it_temp = (nsubset_iterations - iteration_on_subset)*nt_dump_attenuation + it_of_this_subset
233  ! time scheme
234  istage_temp = i_stage
235  else
236  ! time increment
237  ! example: NSTEP = 800 -> 800,799,..,1
238  it_temp = nstep - it + 1
239  ! time scheme
240  istage_temp = stage_time_scheme - i_stage + 1
241  endif
242 
243  ! main solver for the elastic elements
244 
245  !ZN currently we do not support plane wave source in adjoint inversion
246  if (pml_boundary_conditions) then
247  call rebuild_value_on_pml_interface_viscoelastic(it_temp)
248  endif
249 
250  if (undo_attenuation) then
251  ! viscous attenuation for elastic media
252  if (attenuation_viscoelastic) call compute_attenuation_viscoelastic(b_displ_elastic,b_displ_elastic_old, &
253  ispec_is_elastic,.false.,b_e1,b_e11,b_e13)
254  endif
255 
256  ! distinguishes two runs: for elements on MPI interfaces (outer), and elements within the partitions (inner)
257  do iphase = 1,2
258 
259  if (undo_attenuation) then
260  call compute_forces_viscoelastic(b_accel_elastic,b_veloc_elastic,b_displ_elastic,b_displ_elastic_old, &
261  .false.,b_e1,b_e11,b_e13,iphase)
262  else
263  ! todo: maybe should be b_e1,b_e11,.. here, please check...
264  call compute_forces_viscoelastic_backward(b_accel_elastic,b_displ_elastic,b_displ_elastic_old, &
265  e1,e11,e13,iphase)
266  endif
267 
268 ! ! viscous attenuation for elastic media
269 ! if (ATTENUATION_VISCOELASTIC) call compute_attenuation_viscoelastic(b_displ_elastic,b_displ_elastic_old, &
270 ! ispec_is_elastic,PML_BOUNDARY_CONDITIONS,b_e1,b_e11,b_e13)
271 !
272 ! call compute_forces_viscoelastic(b_accel_elastic,b_veloc_elastic,b_displ_elastic,b_displ_elastic_old, &
273 ! PML_BOUNDARY_CONDITIONS,b_e1,b_e11,b_e13)
274 
275  ! computes additional contributions
276  if (iphase == 1) then
277  ! Stacey boundary conditions
278  if (stacey_absorbing_conditions) then
279  if (undo_attenuation) then
280  call compute_stacey_elastic(b_accel_elastic,b_veloc_elastic)
281  else
282  call compute_stacey_elastic_backward(b_accel_elastic)
283  endif
284  endif
285 
286  ! PML boundary
287  if (pml_boundary_conditions) then
288  call pml_boundary_elastic(b_accel_elastic,b_veloc_elastic,b_displ_elastic,b_displ_elastic_old)
289  endif
290 
291  if (pml_boundary_conditions) then
292  call rebuild_value_on_pml_interface_viscoelastic(it_temp)
293  endif
294 
295  ! add coupling with the acoustic side
296  if (coupled_acoustic_elastic) then
297  call compute_coupling_viscoelastic_ac_backward()
298  endif
299 
300  ! add coupling with the poroelastic side
301  if (coupled_elastic_poro) then
302  call compute_coupling_viscoelastic_po_backward()
303  endif
304 
305  ! only on forward arrays so far implemented...
306  !if (AXISYM) then
307  ! call enforce_zero_radial_displacements_on_the_axis()
308  !endif
309 
310  ! add force source
311  if (.not. initialfield) then
312 
313  select case (noise_tomography)
314  case (0)
315  ! earthquake/force source
316  ! for backward wavefield
317  call compute_add_sources_viscoelastic(b_accel_elastic,it_temp,istage_temp)
318 
319  case (3)
320  ! noise simulation
321  ! reconstruction/backward wavefield
322  ! injects generating wavefield sources
323  if (.not. noise_save_everywhere) call add_surface_movie_noise(b_accel_elastic)
324  end select
325  endif ! if not using an initial field
326 
327  endif ! iphase
328 
329 #ifdef USE_MPI
330  ! assembling accel_elastic for elastic elements
331  if (nproc > 1 .and. ninterface_elastic > 0) then
332  if (iphase == 1) then
333  call assemble_mpi_vector_el_s(b_accel_elastic)
334  else
335  call assemble_mpi_vector_el_w(b_accel_elastic)
336  endif
337  endif
338 #endif
339 
340  enddo ! iphase
341 
342  if (pml_boundary_conditions) then
343  call rebuild_value_on_pml_interface_viscoelastic_accel(it_temp)
344  endif
345 
346  ! multiply by the inverse of the mass matrix and update velocity
347  !! DK DK this should be vectorized
348  b_accel_elastic(:,:) = b_accel_elastic(:,:) * rmass_inverse_elastic(:,:)
349 
350  ! time stepping
351  select case (time_stepping_scheme)
352  case (1)
353  ! Newmark
354  call update_veloc_elastic_newmark_backward()
355  case default
356  stop 'Time stepping scheme not implemented yet in viscoelastic backward routine'
357  end select
358 
359  end subroutine compute_forces_viscoelastic_main_backward
