33 
34  subroutine compute_forces_poro_viscous_damping()
35 
36 ! viscous damping
37 
38  use constants, only: custom_real,ngllx,ngllz,zero,use_poro_viscous_damping
39 
40  use specfem_par, only: nspec,it,simulation_type,save_forward, &
41  attenuation_poro_fluid_part, &
42  ibool,kmato,ispec_is_poroelastic, &
43  accels_poroelastic, &
44  accelw_poroelastic, &
45  velocw_poroelastic, &
46  porosity,tortuosity,permeability,poroelastcoef, &
47  jacobian,wxgll,wzgll, &
48  rx_viscous,rz_viscous,theta_e,theta_s, &
49  b_viscodampx,b_viscodampz
50 
51  implicit none
52 
53  ! local variables
54  integer :: ispec,i,j,iglob
55 
56  double precision, dimension(3):: bl_relaxed_viscoelastic,bl_unrelaxed_elastic
57  double precision :: eta_f,phi,tort
58  double precision :: permlxx,permlxz,permlzz,invpermlxx,invpermlxz,invpermlzz,detk
59  double precision :: viscodampx,viscodampz
60  double precision :: visc_x,visc_z
61 
62  ! checks if anything to do
63  if (.not. use_poro_viscous_damping) return
64 
65  ! loop over spectral elements
66  do ispec = 1,nspec
67 
68  eta_f = poroelastcoef(2,2,kmato(ispec))
69 
70  if (ispec_is_poroelastic(ispec) .and. eta_f > 0.d0) then
71  phi = porosity(kmato(ispec))
72  tort = tortuosity(kmato(ispec))
73 
74  permlxx = permeability(1,kmato(ispec))
75  permlxz = permeability(2,kmato(ispec))
76  permlzz = permeability(3,kmato(ispec))
77 
78  ! calcul of the inverse of k
79  detk = permlxx*permlzz - permlxz*permlxz
80 
81  if (detk /= zero) then
82  invpermlxx = permlzz/detk
83  invpermlxz = -permlxz/detk
84  invpermlzz = permlxx/detk
85  else
86  stop 'Permeability matrix is not invertible'
87  endif
88 
89  if (attenuation_poro_fluid_part) then
90  ! viscous attenuation is implemented following the memory variable formulation of
91  ! J. M. Carcione Wave fields in real media: wave propagation in anisotropic,
92  ! anelastic and porous media, Elsevier, p. 304-305, 2007
93  bl_relaxed_viscoelastic(1) = eta_f*invpermlxx*theta_e/theta_s
94  bl_relaxed_viscoelastic(2) = eta_f*invpermlxz*theta_e/theta_s
95  bl_relaxed_viscoelastic(3) = eta_f*invpermlzz*theta_e/theta_s
96  else
97  ! relaxed viscous coef
98  bl_unrelaxed_elastic(1) = eta_f*invpermlxx
99  bl_unrelaxed_elastic(2) = eta_f*invpermlxz
100  bl_unrelaxed_elastic(3) = eta_f*invpermlzz
101  endif
102 
103  do j = 1,ngllz
104  do i = 1,ngllx
105  iglob = ibool(i,j,ispec)
106 
107  ! computes the viscous damping term
108  if (attenuation_poro_fluid_part) then
109  ! with the unrelaxed viscous coef and add memory variable
110  viscodampx = velocw_poroelastic(1,iglob)*bl_relaxed_viscoelastic(1) &
111  + velocw_poroelastic(2,iglob)*bl_relaxed_viscoelastic(2) - rx_viscous(i,j,ispec)
112  viscodampz = velocw_poroelastic(1,iglob)*bl_relaxed_viscoelastic(2) &
113  + velocw_poroelastic(2,iglob)*bl_relaxed_viscoelastic(3) - rz_viscous(i,j,ispec)
114  else
115  ! no viscous attenuation
116  viscodampx = velocw_poroelastic(1,iglob)*bl_unrelaxed_elastic(1) &
117  + velocw_poroelastic(2,iglob)*bl_unrelaxed_elastic(2)
118  viscodampz = velocw_poroelastic(1,iglob)*bl_unrelaxed_elastic(2) &
119  + velocw_poroelastic(2,iglob)*bl_unrelaxed_elastic(3)
120  endif
121 
122  ! viscous term eta_f k^-1 dot(w)
123  visc_x = wxgll(i) * wzgll(j) * jacobian(i,j,ispec) * viscodampx
124  visc_z = wxgll(i) * wzgll(j) * jacobian(i,j,ispec) * viscodampz
125 
126  ! solid contribution
127  ! adds term + phi/tort eta_f k^-1 dot(w)
128  accels_poroelastic(1,iglob) = accels_poroelastic(1,iglob) + phi/tort * visc_x
129  accels_poroelastic(2,iglob) = accels_poroelastic(2,iglob) + phi/tort * visc_z
130 
131  ! fluid contribution
132  ! add - eta_f k^-1 dot(w)
133  accelw_poroelastic(1,iglob) = accelw_poroelastic(1,iglob) - visc_x
134  accelw_poroelastic(2,iglob) = accelw_poroelastic(2,iglob) - visc_z
135 
136  ! reconstructed/backward wavefield
137  ! viscous damping contribution
138  if (simulation_type == 1 .and. save_forward) then
139  ! stores contribution
140  b_viscodampx(i,j,ispec) = visc_x
141  b_viscodampz(i,j,ispec) = visc_z
142  endif
143 
144  enddo
145  enddo
146 
147  endif ! end of test if poroelastic element
148 
149  enddo ! end of loop over all spectral elements
150 
151  ! saves viscous contribution to disk
152  if (save_forward .and. simulation_type == 1) then
153  ! writes damping contributions to file
154  write(23,rec=it) b_viscodampx(:,:,:)
155  write(24,rec=it) b_viscodampz(:,:,:)
156  endif
157 
158  end subroutine compute_forces_poro_viscous_damping
159

165  subroutine compute_forces_poro_viscous_damping_backward()
166 
167 ! viscous damping
168 
169  use constants, only: custom_real,ngllx,ngllz,zero,use_poro_viscous_damping
170 
171  use specfem_par, only: nspec,nstep,it,simulation_type, &
172  ibool,kmato,ispec_is_poroelastic, &
173  b_accels_poroelastic, &
174  b_accelw_poroelastic, &
175  porosity,tortuosity,poroelastcoef, &
176  b_viscodampx,b_viscodampz
177 
178  implicit none
179 
180  ! local variables
181  integer :: ispec,i,j,iglob
182 
183  double precision :: eta_f,phi,tort
184  double precision :: visc_x,visc_z
185 
186  ! checks if anything to do
187  if (.not. use_poro_viscous_damping) return
188  if (simulation_type /= 3) return
189 
190  ! reads in viscous contributions for reconstructed/backward wavefield
191  read(23,rec=nstep-it+1) b_viscodampx(:,:,:)
192  read(24,rec=nstep-it+1) b_viscodampz(:,:,:)
193 
194  ! loop over spectral elements
195  do ispec = 1,nspec
196 
197  eta_f = poroelastcoef(2,2,kmato(ispec))
198 
199  if (ispec_is_poroelastic(ispec) .and. eta_f > 0.d0) then
200 
201  phi = porosity(kmato(ispec))
202  tort = tortuosity(kmato(ispec))
203 
204  do j = 1,ngllz
205  do i = 1,ngllx
206  iglob = ibool(i,j,ispec)
207 
208  ! reconstructed/backward wavefield
209  ! viscous damping contribution
210  ! kernels simulation uses previously stored contributions
211  visc_x = b_viscodampx(i,j,ispec)
212  visc_z = b_viscodampz(i,j,ispec)
213 
214  ! solid
215  b_accels_poroelastic(1,iglob) = b_accels_poroelastic(1,iglob) + phi/tort * visc_x
216  b_accels_poroelastic(2,iglob) = b_accels_poroelastic(2,iglob) + phi/tort * visc_z
217  ! fluid
218  b_accelw_poroelastic(1,iglob) = b_accelw_poroelastic(1,iglob) - visc_x
219  b_accelw_poroelastic(2,iglob) = b_accelw_poroelastic(2,iglob) - visc_z
220 
221  enddo
222  enddo
223 
224  endif ! end of test if poroelastic element
225 
226  enddo ! end of loop over all spectral elements
227 
228  end subroutine compute_forces_poro_viscous_damping_backward
