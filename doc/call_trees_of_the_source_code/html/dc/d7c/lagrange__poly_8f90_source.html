34  subroutine lagrange_any(xi,NGLL,xigll,h,hprime)
35 
36 ! subroutine to compute the Lagrange interpolants based upon the interpolation points
37 ! and their first derivatives at any point xi in [-1,1]
38 
39  implicit none
40 
41  double precision,intent(in) :: xi
42 
43  integer,intent(in) :: NGLL
44  double precision,dimension(NGLL),intent(in) :: xigll
45  double precision,dimension(NGLL),intent(out) :: h,hprime
46 
47  ! local parameters
48  integer :: dgr,i,j
49  double precision :: prod1,prod2,prod3
50  double precision :: prod2_inv
51  double precision :: sum
52  double precision :: x0,x
53 
54 ! note: this routine is hit pretty hard by the mesher, optimizing the loops here will be beneficial
55 
56  do dgr = 1,ngll
57 
58  prod1 = 1.0d0
59  prod2 = 1.0d0
60 
61  ! lagrangian interpolants
62  x0 = xigll(dgr)
63  do i = 1,ngll
64  if (i /= dgr) then
65  x = xigll(i)
66  prod1 = prod1*(xi-x)
67  prod2 = prod2*(x0-x)
68  endif
69  enddo
70 
71  ! takes inverse to avoid additional divisions
72  ! (multiplications are cheaper than divisions)
73  prod2_inv = 1.d0/prod2
74 
75  h(dgr) = prod1 * prod2_inv
76 
77  ! first derivatives
78  sum = 0.0d0
79  do i = 1,ngll
80  if (i /= dgr) then
81  prod3 = 1.0d0
82  do j = 1,ngll
83  if (j /= dgr .and. j /= i) prod3 = prod3*(xi-xigll(j))
84  enddo
85  sum = sum + prod3
86  endif
87  enddo
88 
89  hprime(dgr) = sum * prod2_inv
90 
91  enddo
92 
93  end subroutine lagrange_any

102  double precision function lagrange_deriv_gll(i,j,ZGLL,NZ)
103 
104 !------------------------------------------------------------------------
105 !
106 ! Compute the value of the derivative of the I-th
107 ! Lagrange interpolant through the
108 ! NZ Gauss-Lobatto Legendre points ZGLL at point ZGLL(j)
109 !
110 !------------------------------------------------------------------------
111 
112  implicit none
113 
114  integer :: i,j,nz
115  double precision :: zgll(0:nz-1)
116 
117  ! local parameters
118  integer :: degpoly
119 
120  double precision, external :: pnleg,pndleg
121 
122  degpoly = nz - 1
123  if (i == 0 .and. j == 0) then
124  lagrange_deriv_gll = - dble(degpoly)*(dble(degpoly)+1.d0) * 0.25d0 ! / 4.d0
125  else if (i == degpoly .and. j == degpoly) then
126  lagrange_deriv_gll = dble(degpoly)*(dble(degpoly)+1.d0) * 0.25d0 ! / 4.d0
127  else if (i == j) then
128  lagrange_deriv_gll = 0.d0
129  else
130  lagrange_deriv_gll = pnleg(zgll(j),degpoly) / &
131  (pnleg(zgll(i),degpoly)*(zgll(j)-zgll(i))) &
132  + (1.d0-zgll(j)*zgll(j))*pndleg(zgll(j),degpoly) / (dble(degpoly)* &
133  (dble(degpoly)+1.d0)*pnleg(zgll(i),degpoly)*(zgll(j)-zgll(i))*(zgll(j)-zgll(i)))
134  endif
135 
136  end function lagrange_deriv_gll

142  double precision function hgll(I,Z,ZGLL,NZ)
143 
144 !-------------------------------------------------------------
145 !
146 ! Compute the value of the Lagrangian interpolant L through
147 ! the NZ Gauss-Lobatto Legendre points ZGLL at point Z
148 ! See Nissen-Meyer et al., 2007, A two-dimensional spectral-element method for computing
149 ! spherical-earth seismograms - I. Moment-tensor source, Geophysical Journal International, p. 1087 eq. (A19)
150 ! !! Warning !! There is a minus sign missing in that paper (-1.d0)**(N+1)*(1.d0-Z)*PNDLEG(Z,N) / ALFAN
151 !
152 !-------------------------------------------------------------
153 
154  use constants, only: tinyval
155 
156  implicit none
157 
158  integer i,nz
159  double precision z
160  double precision ZGLL(0:nz-1)
161 
162  integer n
163  double precision EPS,DZ,ALFAN
164  double precision, external :: PNLEG,PNDLEG
165 
166  eps = tinyval
167  dz = z - zgll(i)
168  if (abs(dz) < eps) then
169  hgll = 1.d0
170  return
171  endif
172  n = nz - 1
173  alfan = dble(n)*(dble(n)+1.d0)
174  if (i == 0) then
175  hgll = (-1.d0)**(n+1)*(1.d0-z)*pndleg(z,n) / alfan
176  else if (i == n) then
177  hgll = (1.d0+z)*pndleg(z,n) / alfan
178  else
179  hgll = - (1.d0-z*z)*pndleg(z,n)/ (alfan*pnleg(zgll(i),n)*(z-zgll(i)))
180  endif
181 
182  end function hgll

188  double precision function hglj(I,Z,ZGLJ,NZ)
189 
190 !-------------------------------------------------------------
191 !
192 ! Compute the value of the Lagrangian interpolant L through
193 ! the NZ Gauss-Lobatto Jacobi points ZGLJ at point Z
194 ! See Nissen-Meyer et al., 2007, A two-dimensional spectral-element method for computing
195 ! spherical-earth seismograms - I. Moment-tensor source, Geophysical Journal International, p. 1088 eq. (A26)
196 !
197 !-------------------------------------------------------------
198 
199  use constants, only: tinyval
200 
201  implicit none
202 
203  integer i,nz
204  double precision z
205  double precision ZGLJ(0:nz-1)
206 
207  integer n
208  double precision EPS,DZ,ALFAN1,ALFAN2
209  double precision, external :: PNGLJ,PNDGLJ
210 
211  eps = tinyval
212  dz = z - zglj(i)
213  if (abs(dz) < eps) then
214  hglj = 1.d0
215  return
216  endif
217  n = nz - 1
218  alfan1 = dble(n)+1.d0
219  alfan2 = dble(n)*(dble(n)+2.d0)
220  if (i == 0) then
221  hglj = 2.d0*(-1.d0)**n*(z-1.0d0)*pndglj(z,n) / (alfan1*alfan2)
222  else if (i == n) then
223  hglj = (1.d0+z)*pndglj(z,n) / alfan2
224  else
225  hglj = - (1.d0-z*z)*pndglj(z,n) / (alfan2*pnglj(zglj(i),n)*(z-zglj(i)))
226  endif
227 
228  end function hglj

235 ! subroutine to compute the derivative of the interpolants of the GLJ
236 ! quadrature at the GLJ points at any given GLJ point
237 
238  double precision function poly_deriv_glj(I,j,ZGLJ,NZ)
239 
240 !------------------------------------------------------------------------
241 !
242 ! Compute the value of the derivative of the I-th
243 ! polynomial interpolant of the GLJ quadrature through the
244 ! NZ Gauss-Lobatto-Jacobi (0,1) points ZGLJ at point ZGLJ(j)
245 ! See Nissen-Meyer et al., 2007, A two-dimensional spectral-element method for computing
246 ! spherical-earth seismograms - I. Moment-tensor source, Geophysical Journal International, p. 1088 eq. (A27)
247 ! WARNING: there is an error at line 7 of their equation
248 ! \partial_{\xi}\overline{l}_{i}(\overline{\xi}_{I})=\dfrac{1}{\overline{P}_{N}(\overline{\xi}_{i})(1-\overline{\xi}_{i})}
249 !
250 !------------------------------------------------------------------------
251 
252  implicit none
253 
254  integer i,j,nz
255  double precision zglj(0:nz-1)
256 
257  integer degpoly
258 
259  double precision, external :: pnglj
260 
261  degpoly = nz - 1
262 
263  if (i == 0 .and. j == 0) then ! Case 1
264  poly_deriv_glj = -dble(degpoly)*(dble(degpoly)+2.d0)/6.d0
265  else if (i == 0 .and. 0 < j .and. j < degpoly) then ! Case 2
266  poly_deriv_glj = 2.d0*(-1)**degpoly*pnglj(zglj(j),degpoly)/((1.d0+zglj(j))*(dble(degpoly)+1.d0))
267  else if (i == 0 .and. j == degpoly) then ! Case 3
268  poly_deriv_glj = (-1)**degpoly/(dble(degpoly)+1.d0)
269  else if (0 < i .and. i < degpoly .and. j == 0) then ! Case 4
270  poly_deriv_glj = (-1)**(degpoly+1)*(dble(degpoly)+1.d0)/(2.d0*pnglj(zglj(i),degpoly)*(1.d0+zglj(i)))
271  else if (0 < i .and. i < degpoly .and. 0 < j .and. j < degpoly .and. i /= j) then ! Case 5
272  poly_deriv_glj = 1.d0/(zglj(j)-zglj(i))*pnglj(zglj(j),degpoly)/pnglj(zglj(i),degpoly)
273  else if (0 < i .and. i < degpoly .and. i == j) then ! Case 6
274  poly_deriv_glj = -1.d0/(2.d0*(1.d0+zglj(i)))
275  else if (0 < i .and. i < degpoly .and. j == degpoly) then ! Case 7
276  poly_deriv_glj = 1.d0/(pnglj(zglj(i),degpoly)*(1.d0-zglj(i)))
277  else if (i == degpoly .and. j == 0) then ! Case 8
278  poly_deriv_glj = (-1)**(degpoly+1)*(dble(degpoly)+1.d0)/4.d0
279  else if (i == degpoly .and. 0 < j .and. j < degpoly) then ! Case 9
280  poly_deriv_glj = -1.d0/(1.d0-zglj(j))*pnglj(zglj(j),degpoly)
281  else if (i == degpoly .and. j == degpoly) then ! Case 10
282  poly_deriv_glj = (dble(degpoly)*(dble(degpoly)+2.d0)-1.d0)/4.d0
283  else
284  stop 'Problem in poly_deriv_GLJ: in a perfect world this would NEVER appear'
285  endif
286 
287  end function poly_deriv_glj
