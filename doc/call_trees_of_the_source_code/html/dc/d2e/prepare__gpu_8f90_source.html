34  subroutine prepare_gpu()
35 
36  use constants, only: imain,use_mesh_coloring_gpu,approximate_hess_kl
37  use specfem_par
38  use specfem_par_gpu
39 
40  implicit none
41 
42  ! local parameters
43  real :: free_mb,used_mb,total_mb
44  real(kind=CUSTOM_REAL), dimension(:), allocatable :: cosrot_irecf, sinrot_irecf
45  real(kind=CUSTOM_REAL), dimension(:), allocatable :: rmassx,rmassz
46 
47  ! checks if anything to do
48  if (.not. gpu_mode) return
49 
50  ! GPU_MODE now defined in Par_file
51  if (myrank == 0) then
52  write(imain,*)
53  write(imain,*) "Preparing GPU Fields and Constants on Device."
54  call flush_imain()
55  endif
56 
57  ! safety checks
58  if (any_elastic .and. (.not. p_sv)) stop 'Invalid GPU simulation, SH waves not implemented yet. Please use P_SV instead.'
59  if (pml_boundary_conditions ) stop 'PML not implemented on GPU mode. Please use Stacey instead.'
60  if (axisym) stop 'Axisym not implemented on GPU yet.'
61  if (ngllx /= ngllz) stop 'GPU simulations require NGLLX == NGLLZ'
62 
63  ! initializes arrays
64  call init_host_to_dev_variable()
65 
66  ! check number of purely elastic elements
67  if (nspec_elastic /= nspec - nspec_acoustic) then
68  print *,'GPU simulation only supported for acoustic and/or elastic domain simulations'
69  stop 'Error GPU simulation'
70  endif
71 
72 !!!!!!!!!!! Parametres fournis
73 
74 ! ibool(i,j,ispec) : convertisseur numero du point GLL local (i,j) de l'element ispec => global (iglob)
75 ! ninterface : nombre d'interfaces de la partition locale avec les autres partitions
76 ! max_nibool_interfaces_ext_mesh : nombre maximum de points GLL contenus sur une interface
77 ! nibool_interfaces_ext_mesh(i) : nombre de points GLL contenus sur l'interface i
78 ! ibool_interfaces_ext_mesh(iGGL,i) : numero global iglob du ieme point GLL (iGLL) de l'interface i
79 ! numabs : tableau des elements spectraux situes en zone absorbante
80 ! abs_boundary_ij(i,j,ispecabs) : coordonnee locale i de j eme point GLL de l'element absorbant ispecabs
81 ! abs_boundary_normal(i,j,ispecabs) : i eme coordonnee du vecteur normal du j eme point GLL de l'element absorbant ispecabs
82 ! abs_boundary_jacobian1Dw(i,ispecabs) : i eme jacobienne ponderee de l'element absorbant jspecabs
83 ! nelemabs : nombre d'elements absorbant
84 ! cote_abs(ispecabs) : numero du cote (1=b, 2=r, 3=t, 4=l ) auquel appartient l'element absorbant ispecabs
85 ! ib_left : correspondance entre le numero d'element absorbant global et son numero sur le cote
86 ! ispec_is_inner : booleen vrai si l'element est a l'interieur d'une partition
87 ! sourcearray_loc(i_src,dim,i,j) : tableau de ponderation de l'intensite de la source pour chaque point GLL (i,j)
88 ! de l'element spectral qui contient la source locale i_src
89 ! ispec_selected_source(i) : numero d'element spectral contenant la source locale i
90 ! ispec_selected_rec_loc(i) : numero d'element spectral du receveur local i
91 ! nrecloc : nombre de receveurs locaux
92 ! nspec_acoustic : nombre local d'elements spectraux acoustiques
93 
94  ! prepares general fields on GPU
95  !! JC JC here we will need to add GPU support for the C-PML routines
96 
97  call prepare_constants_device(mesh_pointer, &
98  ngllx, nspec, nglob, &
99  xix, xiz, gammax, gammaz, &
100  kappastore, mustore, &
101  ibool, &
102  ninterface, max_nibool_interfaces_ext_mesh, &
103  nibool_interfaces_ext_mesh, ibool_interfaces_ext_mesh, &
104  hprime_xx,hprimewgll_xx, &
105  wxgll, &
106  stacey_absorbing_conditions, &
107  nspec_bottom, &
108  nspec_left, &
109  nspec_right, &
110  nspec_top, &
111  numabs, abs_boundary_ij, &
112  abs_boundary_normal, &
113  abs_boundary_jacobian1dw, &
114  nelemabs, &
115  cote_abs, &
116  ib_bottom, &
117  ib_left, &
118  ib_right, &
119  ib_top, &
120  ispec_is_inner, &
121  nsources_local, &
122  sourcearray_loc,source_time_function_loc, &
123  nstep,ispec_selected_source_loc, &
124  ispec_selected_rec_loc, &
125  nrecloc, &
126  cosrot_irecf,sinrot_irecf, &
127  simulation_type, &
128  use_mesh_coloring_gpu, &
129  nspec_acoustic,nspec_elastic, &
130  myrank,save_forward, &
131  xir_store_loc, &
132  gammar_store_loc)
133 
134 
135 !!! Parametres fournis
136 
137 ! rmass_inverse_acoustic : matrice acoustique inversee de taille nglob
138 ! (nglob_acoustic = nglob s'il existe des elements acoustiques)
139 ! num_phase_ispec_acoustic : max entre nb d'element spectraux acoustiques interieur et exterieur
140 ! phase_ispec_inner_acoustic(i,j) : i eme element spectral acoustique interieur si j=2 exterieur si j=1
141 ! acoustic(i) : vrai si l'element spectral i est acoustique
142 ! nelem_acoustic_surface : nombre d'elements spectraux situes sur une surface libre acoustique
143 ! free_ac_ispec : numero d'element spectral du i eme element acoustique sur surface libre
144 ! free_surface_ij(i,j,ispec) : i eme coordonnee du j eme point GLL de l'element spectral libre ispec
145 ! b_reclen_potential : place en octet prise par b_nelem_acoustic_surface * GLLX
146 ! any_elastic : vrai s'il existe des elements elastiques
147 ! num_fluid_solid_edges : nombre d'elements spectraux sur une frontiere elastique/acoustique
148 ! coupling_ac_el_ispec : tableau des elements spectraux frontiere ACOUSTIQUE
149 ! coupling_ac_el_ij : coordonnees locales des points GLL sur la frontiere elastique/acoustique
150 ! coupling_ac_el_normal(i,j,ispec) : i eme coordonne de la normale au point GLL j de l'element frontiere ispec
151 ! coupling_ac_el_jacobian1Dw(i,ispec) : jacobienne ponderee du i eme point GLL de l'element frontiere ispec
152 ! num_colors_outer_acoustic : a initialiser plus tard quand USE_COLOR_MESH sera implemente
153 ! num_colors_inner_acoustic : a initialiser plus tard quand USE_COLOR_MESH sera implemente
154 ! num_elem_colors_acoustic : a initialiser plus tard quand USE_COLOR_MESH sera implemente
155 
156 
157  ! prepares fields on GPU for acoustic simulations
158  if (any_acoustic) then
159  call prepare_fields_acoustic_device(mesh_pointer, &
160  rmass_inverse_acoustic,rhostore,kappastore, &
161  num_phase_ispec_acoustic,phase_ispec_inner_acoustic, &
162  ispec_is_acoustic, &
163  nelem_acoustic_surface, &
164  free_ac_ispec,free_surface_ij, &
165  any_elastic, num_fluid_solid_edges, &
166  coupling_ac_el_ispec,coupling_ac_el_ij, &
167  coupling_ac_el_normal,coupling_ac_el_jacobian1dw, &
168  ninterface_acoustic,inum_interfaces_acoustic, &
169  num_colors_outer_acoustic,num_colors_inner_acoustic, &
170  num_elem_colors_acoustic)
171 
172  if (simulation_type == 3) then
173  ! safety check
174  if (approximate_hess_kl) then
175  stop 'Sorry, approximate acoustic Hessian kernels not yet fully implemented for GPU simulations!'
176  endif
177  call prepare_fields_acoustic_adj_dev(mesh_pointer,approximate_hess_kl)
178  endif
179  endif
180 
181 !!! Parametres fournis
182 
183 ! rmass_inverse_elastic : matrice elastique inversee de taille nglob_acoustic
184 ! (nglob_acoustic = nglob s'il existe des elements acoustiques)
185 ! num_phase_ispec_elastic : max entre nb d'element spectraux elastiques interieur et exterieur
186 ! phase_ispec_inner_elastic(i,j) : i eme element spectral elastique interieur si j=2 exterieur si j=1
187 ! elastic(i) : vrai si l'element spectral i est elastique
188 ! num_colors_outer_elastic : a initialiser plus tard quand USE_COLOR_MESH sera implemente
189 ! num_colors_inner_elastic : a initialiser plus tard quand USE_COLOR_MESH sera implemente
190 ! num_elem_colors_elastic : a initialiser plus tard quand USE_COLOR_MESH sera implemente
191 
192  ! prepares fields on GPU for elastic simulations
193  !?!? JC JC here we will need to add GPU support for the new C-PML routines
194  if (any_elastic) then
195  ! temporary mass matrices
196  allocate(rmassx(nglob_elastic),rmassz(nglob_elastic))
197  rmassx(:) = rmass_inverse_elastic(1,:)
198  rmassz(:) = rmass_inverse_elastic(2,:)
199 
200  call prepare_fields_elastic_device(mesh_pointer, &
201  rmassx,rmassz, &
202  rho_vp,rho_vs, &
203  num_phase_ispec_elastic,phase_ispec_inner_elastic, &
204  ispec_is_elastic, &
205  nspec_left, &
206  nspec_right, &
207  nspec_top, &
208  nspec_bottom, &
209  num_colors_outer_elastic,num_colors_inner_elastic, &
210  num_elem_colors_elastic, &
211  any_anisotropy, &
212  c11store,c12store,c13store, &
213  c15store,c23store, &
214  c25store,c33store,c35store,c55store, &
215  ninterface_elastic,inum_interfaces_elastic)
216 
217 
218  if (simulation_type == 3) then
219  ! safety check
220  if (approximate_hess_kl) then
221  stop 'Sorry, approximate elastic Hessian kernels not yet fully implemented for GPU simulations!'
222  endif
223  call prepare_fields_elastic_adj_dev(mesh_pointer,ndim*nglob_ab,approximate_hess_kl)
224  endif
225 
226  ! frees memory
227  deallocate(rmassx,rmassz)
228  endif
229 
230  ! prepares fields on GPU for poroelastic simulations
231  if (any_poroelastic) then
232  stop 'todo poroelastic simulations on GPU'
233  endif
234 
235  ! prepares needed receiver array for adjoint runs
236  if (simulation_type == 2 .or. simulation_type == 3) &
237  call prepare_sim2_or_3_const_device(mesh_pointer,nrecloc,source_adjoint,nstep)
238 
239  ! synchronizes processes
240  call synchronize_all()
241 
242  ! puts acoustic initial fields onto GPU
243  if (any_acoustic) then
244  call transfer_fields_ac_to_device(nglob_ab,potential_acoustic, &
245  potential_dot_acoustic,potential_dot_dot_acoustic,mesh_pointer)
246 
247 
248 
249  if (simulation_type == 3) &
250  call transfer_b_fields_ac_to_device(nglob_ab,b_potential_acoustic, &
251  b_potential_dot_acoustic,b_potential_dot_dot_acoustic,mesh_pointer)
252  endif
253 
254  ! puts elastic initial fields onto GPU
255  if (any_elastic) then
256  ! prepares wavefields for transfering
257  allocate(tmp_displ_2d(ndim,nglob_elastic), &
258  tmp_veloc_2d(ndim,nglob_elastic), &
259  tmp_accel_2d(ndim,nglob_elastic))
260 
261  tmp_displ_2d(:,:) = displ_elastic(:,:)
262  tmp_veloc_2d(:,:) = veloc_elastic(:,:)
263  tmp_accel_2d(:,:) = accel_elastic(:,:)
264 
265  ! transfers forward fields to device with initial values
266  call transfer_fields_el_to_device(ndim*nglob_ab,tmp_displ_2d,tmp_veloc_2d,tmp_accel_2d,mesh_pointer)
267 
268  if (simulation_type == 3) then
269  tmp_displ_2d(:,:) = b_displ_elastic(:,:)
270  tmp_veloc_2d(:,:) = b_veloc_elastic(:,:)
271  tmp_accel_2d(:,:) = b_accel_elastic(:,:)
272 
273  ! transfers backward fields to device with initial values
274  call transfer_b_fields_to_device(ndim*nglob_ab,tmp_displ_2d,tmp_veloc_2d,tmp_accel_2d,mesh_pointer)
275  endif
276  endif
277 
278  ! allocates arrays for MPI transfers
279  allocate(request_send_recv_scalar_gpu(2*ninterface))
280  allocate(b_request_send_recv_scalar_gpu(2*ninterface))
281 
282  allocate(request_send_recv_vector_gpu(2*ninterface))
283  allocate(b_request_send_recv_vector_gpu(2*ninterface))
284 
285  allocate(buffer_send_scalar_gpu(max_nibool_interfaces_ext_mesh,ninterface))
286  allocate(b_buffer_send_scalar_gpu(max_nibool_interfaces_ext_mesh,ninterface))
287 
288  allocate(buffer_recv_scalar_gpu(max_nibool_interfaces_ext_mesh,ninterface))
289  allocate(b_buffer_recv_scalar_gpu(max_nibool_interfaces_ext_mesh,ninterface))
290 
291  allocate(buffer_send_vector_gpu(2,max_nibool_interfaces_ext_mesh,ninterface))
292  allocate(b_buffer_send_vector_gpu(2,max_nibool_interfaces_ext_mesh,ninterface))
293 
294  allocate(buffer_recv_vector_gpu(2,max_nibool_interfaces_ext_mesh,ninterface))
295  allocate(b_buffer_recv_vector_gpu(2,max_nibool_interfaces_ext_mesh,ninterface))
296 
297  ! synchronizes processes
298  call synchronize_all()
299 
300  ! outputs GPU usage to files for all processes
301  call output_free_device_memory(myrank)
302 
303  ! outputs usage for main process
304  if (myrank == 0) then
305  call get_free_device_memory(free_mb,used_mb,total_mb)
306 
307  write(imain,*)
308  write(imain,*) "GPU usage: free =",free_mb," MB",nint(free_mb/total_mb*100.0),"%"
309  write(imain,*) " used =",used_mb," MB",nint(used_mb/total_mb*100.0),"%"
310  write(imain,*) " total =",total_mb," MB",nint(total_mb/total_mb*100.0),"%"
311  write(imain,*)
312  call flush_imain()
313  endif
314 
315  ! frees temporary arrays
316  deallocate(cosrot_irecf,sinrot_irecf)
317 
318  ! synchronizes all processes
319  call synchronize_all()
320 
321  contains

324 
325  subroutine init_host_to_dev_variable()
326 
327 ! helper routine for array initialization and time run setup
328 
329  use constants, only: imain,iedge1,iedge2,iedge3,iedge4,ibottom,iright,itop,ileft
330 
331  implicit none
332 
333  ! local parameters
334  integer :: i_spec_free, ipoint1D, i, j, k, ispec, ispecabs, i_source
335  integer :: ispec_acoustic,ispec_elastic,iedge_acoustic,iedge_elastic
336  integer :: ier,inum
337  real(kind=CUSTOM_REAL) :: zxi,xgamma,jacobian1D
338  real(kind=CUSTOM_REAL) :: xxi,zgamma
339 
340 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
341 !! Initialisation variables pour routine prepare_constants_device
342 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
343 
344  ! user output
345  if (myrank == 0) then
346  write(imain,*) ' initializing arrays for GPU:'
347  call flush_imain()
348  endif
349 
350  ! converts to CUSTOM_REAL
351  deltatf = sngl(deltat)
352  deltatover2f = sngl(deltatover2)
353  deltatsquareover2f = sngl(deltatsquareover2)
354 
355  b_deltatf = sngl(b_deltat)
356  b_deltatover2f = sngl(b_deltatover2)
357  b_deltatsquareover2f = sngl(b_deltatsquareover2)
358 
359  nspec_ab = nspec
360  nglob_ab = nglob
361 
362  ! user output
363  if (myrank == 0) then
364  write(imain,*) ' number of MPI interfaces = ',ninterface
365  write(imain,*) ' number of acoustic elements at free surface = ',nelem_acoustic_surface
366  call flush_imain()
367  endif
368 
369  ! acoustic elements at free surface
370  allocate(free_ac_ispec(nelem_acoustic_surface))
371  if (nelem_acoustic_surface > 0) then
372  ! gets ispec indices for acoustic elements at free surface
373  free_ac_ispec(:) = acoustic_surface(1,:)
374  endif
375 
376  ! checks
377  if (nelemabs < 0) then
378  if (myrank == 0) write(imain,*) ' Warning: reading in negative nelemabs ',nelemabs,'...resetting to zero!'
379  nelemabs = 0
380  endif
381 
382  allocate(abs_boundary_ij(2,ngllx,nelemabs), &
383  abs_boundary_jacobian1dw(ngllx,nelemabs), &
384  abs_boundary_normal(ndim,ngllx,nelemabs), &
385  cote_abs(nelemabs),stat=ier)
386  if (ier /= 0 ) stop 'error allocating array abs_boundary_ispec etc.'
387 
388  if (stacey_absorbing_conditions) then
389 
390  do ispecabs = 1,nelemabs
391  ispec = numabs(ispecabs)
392 
393  !--- left absorbing boundary
394  if (codeabs(iedge4,ispecabs)) then
395  i = 1
396  do j = 1,ngllz
397  abs_boundary_ij(1,j,ispecabs) = i
398  abs_boundary_ij(2,j,ispecabs) = j
399 
400  xgamma = - xiz(i,j,ispec) * jacobian(i,j,ispec)
401  zgamma = + xix(i,j,ispec) * jacobian(i,j,ispec)
402  jacobian1d = sqrt(xgamma**2 + zgamma**2)
403 
404  abs_boundary_normal(1,j,ispecabs) = - zgamma / jacobian1d
405  abs_boundary_normal(2,j,ispecabs) = + xgamma / jacobian1d
406 
407  abs_boundary_jacobian1dw(j,ispecabs) = jacobian1d * wzgll(j)
408 
409  cote_abs(ispecabs) = 4
410  enddo
411  if (ibegin_edge4(ispecabs) == 2) abs_boundary_ij(2,1,ispecabs) = 6
412  if (iend_edge4(ispecabs) == 4) abs_boundary_ij(2,5,ispecabs) = 6
413 
414  !--- right absorbing boundary
415  else if (codeabs(iedge2,ispecabs)) then
416  i = ngllx
417  do j = 1,ngllz
418  abs_boundary_ij(1,j,ispecabs) = i
419  abs_boundary_ij(2,j,ispecabs) = j
420 
421  xgamma = - xiz(i,j,ispec) * jacobian(i,j,ispec)
422  zgamma = + xix(i,j,ispec) * jacobian(i,j,ispec)
423  jacobian1d = sqrt(xgamma**2 + zgamma**2)
424 
425  abs_boundary_normal(1,j,ispecabs) = + zgamma / jacobian1d
426  abs_boundary_normal(2,j,ispecabs) = - xgamma / jacobian1d
427 
428  abs_boundary_jacobian1dw(j,ispecabs) = jacobian1d * wzgll(j)
429 
430  cote_abs(ispecabs) = 2
431  enddo
432  if (ibegin_edge2(ispecabs) == 2) abs_boundary_ij(2,1,ispecabs) = 6
433  if (iend_edge2(ispecabs) == 4) abs_boundary_ij(2,5,ispecabs) = 6
434 
435  !--- bottom absorbing boundary
436  else if (codeabs(iedge1,ispecabs)) then
437  j = 1
438  do i = 1,ngllx
439  abs_boundary_ij(1,i,ispecabs) = i
440  abs_boundary_ij(2,i,ispecabs) = j
441 
442  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
443  zxi = - gammax(i,j,ispec) * jacobian(i,j,ispec)
444  jacobian1d = sqrt(xxi**2 + zxi**2)
445 
446  abs_boundary_normal(1,i,ispecabs) = + zxi / jacobian1d
447  abs_boundary_normal(2,i,ispecabs) = - xxi / jacobian1d
448 
449  abs_boundary_jacobian1dw(i,ispecabs) = jacobian1d * wxgll(i)
450 
451  cote_abs(ispecabs) = 1
452 
453  enddo
454  if (ibegin_edge1(ispecabs) == 2 .or. codeabs_corner(1,ispecabs)) abs_boundary_ij(1,1,ispecabs) = 6
455  if (iend_edge1(ispecabs) == 4 .or. codeabs_corner(2,ispecabs)) abs_boundary_ij(1,5,ispecabs) = 6
456 
457  !--- top absorbing boundary
458  else if (codeabs(iedge3,ispecabs)) then
459  j = ngllz
460  do i = 1,ngllx
461  abs_boundary_ij(1,i,ispecabs) = i
462  abs_boundary_ij(2,i,ispecabs) = j
463 
464  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
465  zxi = - gammax(i,j,ispec) * jacobian(i,j,ispec)
466  jacobian1d = sqrt(xxi**2 + zxi**2)
467 
468  abs_boundary_normal(1,i,ispecabs) = - zxi / jacobian1d
469  abs_boundary_normal(2,i,ispecabs) = + xxi / jacobian1d
470 
471  abs_boundary_jacobian1dw(i,ispecabs) = jacobian1d * wxgll(i)
472 
473  cote_abs(ispecabs) = 3
474  enddo
475  if (ibegin_edge3(ispecabs) == 2 .or. codeabs_corner(3,ispecabs)) abs_boundary_ij(1,1,ispecabs) = 6
476  if (iend_edge3(ispecabs) == 4 .or. codeabs_corner(4,ispecabs)) abs_boundary_ij(1,5,ispecabs) = 6
477 
478  endif
479  enddo
480  endif ! STACEY_ABSORBING_CONDITIONS
481 
482  ! sources
483  ! counts sources in this process slice
484  nsources_local = 0
485  do i = 1, nsources
486  if (myrank == islice_selected_source(i)) then
487  nsources_local = nsources_local + 1
488  endif
489  enddo
490 
491  ! user output
492  if (myrank == 0) then
493  write(imain,*) ' number of sources = ',nsources
494  write(imain,*) ' number of sources in this process slice = ',nsources_local
495  call flush_imain()
496  endif
497 
498  allocate(source_time_function_loc(nsources_local,nstep))
499  allocate(ispec_selected_source_loc(nsources_local))
500  j = 0
501  do i = 1, nsources
502  if (myrank == islice_selected_source(i)) then
503  if (j > nsources_local) stop 'Error with the number of local sources'
504  j = j + 1
505  source_time_function_loc(j,:) = source_time_function(i,:,1)
506  ispec_selected_source_loc(j) = ispec_selected_source(i)
507  endif
508  enddo
509 
510  if (nsources_local > 0) then
511  allocate(sourcearray_loc(nsources_local,ndim,ngllx,ngllx))
512  else
513  allocate(sourcearray_loc(1,1,1,1))
514  endif
515 
516  k = 0
517  do i_source = 1,nsources
518  if (myrank == islice_selected_source(i_source)) then
519  ! source belongs to this process
520  k = k + 1
521  sourcearray_loc(k,:,:,:) = sourcearrays(i_source,:,:,:)
522  sourcearray_loc(k,:,:,:) = sourcearrays(i_source,:,:,:)
523  endif
524  enddo
525 
526  ! receivers
