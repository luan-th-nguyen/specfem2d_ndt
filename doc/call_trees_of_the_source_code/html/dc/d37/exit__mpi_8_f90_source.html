36 !-----------------------------------------------
37  subroutine exit_mpi(myrank,error_msg)
38 
39 #ifdef USE_MPI
40  use mpi
41 #endif
42  use constants, only: max_string_len,imain,istandard_output,output_files
43 
44  implicit none
45 
46  ! identifier for error message file
47  integer, parameter :: ierror = 30
48 
49  integer :: myrank
50  character(len=*) :: error_msg
51 
52  character(len=MAX_STRING_LEN) :: outputname
53 
54  ! write error message to screen
55  write(*,*) error_msg(1:len(error_msg))
56  write(*,*) 'Error detected, aborting MPI... proc ',myrank
57 
58 ! write error message to file
59  write(outputname,"('/error_message',i6.6,'.txt')") myrank
60  open(unit=ierror,file=trim(output_files)//outputname,status='unknown')
61  write(ierror,*) error_msg(1:len(error_msg))
62  write(ierror,*) 'Error detected, aborting MPI... proc ',myrank
63  close(ierror)
64 
65 ! close output file
66  if (myrank == 0 .and. imain /= istandard_output) close(imain)
67 
68  call abort_mpi()
69 
70  end subroutine exit_mpi
