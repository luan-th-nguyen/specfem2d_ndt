34  subroutine write_postscript_snapshot()
35 
36  use constants, only: imain
37 
38  use specfem_par, only: myrank,p_sv,it, &
39  potential_acoustic,potential_gravitoacoustic, &
40  potential_gravito,displ_elastic,displs_poroelastic, &
41  potential_dot_acoustic,potential_dot_gravitoacoustic, &
42  potential_dot_gravito,veloc_elastic,velocs_poroelastic, &
43  potential_dot_dot_acoustic,potential_dot_dot_gravitoacoustic, &
44  potential_dot_dot_gravito,accel_elastic,accels_poroelastic
45 
46  use specfem_par_movie, only: imagetype_postscript
47 
48  implicit none
49 
50  ! user output
51  if (myrank == 0) then
52  write(imain,*)
53  write(imain,*) 'Writing PostScript vector plot for time step ',it
54  call flush_imain()
55  endif
56 
57  ! determines postscript output type
58  if (p_sv) then
59  select case (imagetype_postscript)
60  case (1)
61  ! displacement
62  if (myrank == 0) write(imain,*) 'drawing displacement vector as small arrows...'
63  call compute_vector_whole_medium(potential_acoustic,potential_gravitoacoustic, &
64  potential_gravito,displ_elastic,displs_poroelastic)
65  case (2)
66  ! velocity
67  if (myrank == 0) write(imain,*) 'drawing velocity vector as small arrows...'
68  call compute_vector_whole_medium(potential_dot_acoustic,potential_dot_gravitoacoustic, &
69  potential_dot_gravito,veloc_elastic,velocs_poroelastic)
70  case (3)
71  ! acceleration
72  if (myrank == 0) write(imain,*) 'drawing acceleration vector as small arrows...'
73  call compute_vector_whole_medium(potential_dot_dot_acoustic,potential_dot_dot_gravitoacoustic, &
74  potential_dot_dot_gravito,accel_elastic,accels_poroelastic)
75  case default
76  call exit_mpi(myrank,'wrong type for PostScript snapshots')
77  end select
78 
79  ! postscript plotting
80  call plot_post()
81 
82  else
83  call exit_mpi(myrank,'cannot draw a SH scalar field as a vector plot, turn PostScript plots off')
84  endif
85 
86  ! user output
87  if (myrank == 0) then
88  write(imain,*) 'PostScript file written'
89  call flush_imain()
90  endif
91 
92  end subroutine write_postscript_snapshot
