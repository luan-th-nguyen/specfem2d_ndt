34  subroutine save_databases()
35 
36 ! generates the databases for the solver
37 
38  use constants, only: imain,iout,max_string_len,output_files
39  use part_unstruct_par, only: nspec,iproc
40  use shared_parameters, only: nproc
41 
42  implicit none
43 
44  ! local parameters
45  integer :: ier
46  character(len=MAX_STRING_LEN) :: prname
47 
48  do iproc = 0, nproc-1
49 
50  ! opens Database file
51  write(prname, "(a,i5.5,a)") './'//trim(output_files)//'Database',iproc,'.bin'
52  open(unit=iout,file=trim(prname),status='unknown',action='write',form='unformatted',iostat=ier)
53  if (ier /= 0 ) stop 'Error saving databases; check that directory OUTPUT_FILES exists'
54 
55  ! saves header infos and simulation setup
56  call save_databases_init()
57 
58  ! sources setup
59  call save_databases_sources()
60 
61  ! mesh node coordinates and mesh properties
62  call save_databases_coorg_elem()
63 
64  ! attenuation
65  call save_databases_attenuation()
66 
67  ! material properties
68  call save_databases_materials()
69 
70  ! saves mpi/partition interface
71  call save_databases_interfaces()
72 
73  ! absorbing boundary
74  call save_databases_absorbing()
75 
76  ! acoustic forcing
77  call save_databases_acoustic_forcing()
78 
79  ! acoustic free surface
80  call save_databases_free_surf()
81 
82  ! coupled domain edges
83  call save_databases_coupled()
84 
85  ! tangential detection curve
86  call save_databases_tangential()
87 
88  ! axial elements
89  call save_databases_axial_elements()
90 
91  ! closes Database file
92  close(iout)
93 
94  ! user output
95  if (iproc == 0) write(imain,*)
96  write(imain,*) 'slice ',iproc,' has number of spectral elements =',nspec
97  call flush_imain()
98 
99  enddo
100 
101  end subroutine save_databases

372  nedges_coupled_loc,nedges_acporo_coupled_loc, &

396  subroutine save_databases_materials()
397 
398  use constants, only: iout,isotropic_material,anisotropic_material,poroelastic_material
399  use part_unstruct_par
400  use shared_parameters
401 
402  implicit none
403 
404  ! local parameters
405  double precision :: val0,val1,val2,val3,val4,val5,val6,val7,val8,val9,val10,val11,val12
406  integer :: i,indic
407 
408  ! material set header
409  ! 'Material sets (num 1 rho vp vs 0 0 QKappa Qmu 0 0 0 0 0 0) or '
410  ! '(num 2 rho c11 c13 c15 c33 c35 c55 c12 c23 c25 0 0 0) or '
411  ! '(num 3 rhos rhof phi c k_xx k_xz k_zz Ks Kf Kfr etaf mufr Qmu)'
412 
413  do i = 1,nbmodels
414  ! material type
415  indic = icodemat(i)
416 
417  if (indic == isotropic_material) then
418  ! isotropic
419  val0 = rho_s_read(i)
420  val1 = cp(i)
421  val2 = cs(i)
422  val3 = comp_g(i)
423  val4 = 0.d0
424  val5 = qkappa(i)
425  val6 = qmu(i)
426  val7 = 0.d0
427  val8 = 0.d0
428  val9 = 0.d0
429  val10 = 0.d0
430  val11 = 0.d0
431  val12 = 0.d0
432  ! old
433  !write(IOUT) i,icodemat(i),rho_s_read(i),cp(i),cs(i),0,0,QKappa(i),Qmu(i),0,0,0,0,0,0
434 
435  else if (indic == anisotropic_material) then
436  ! anisotropic
437  val0 = rho_s_read(i)
438  val1 = aniso3(i)
439  val2 = aniso4(i)
440  val3 = aniso5(i)
441  val4 = aniso6(i)
442  val5 = aniso7(i)
443  val6 = aniso8(i)
444  val7 = aniso9(i)
445  val8 = aniso10(i)
446  val9 = aniso11(i)
447  val10 = aniso12(i)
448  val11 = 0.d0
449  val12 = 0.d0
450  ! old
451  !write(IOUT) i,icodemat(i),rho_s_read(i), &
452  ! aniso3(i),aniso4(i),aniso5(i),aniso6(i), &
453  ! aniso7(i),aniso8(i),aniso9(i),aniso10(i),aniso11(i),aniso12(i),0,0
454 
455  else if (indic == poroelastic_material) then
456  ! poro-elastic
457  val0 = rho_s_read(i)
458  val1 = rho_f_read(i)
459  val2 = phi_read(i)
460  val3 = tortuosity_read(i)
461  val4 = permxx_read(i)
462  val5 = permxz_read(i)
463  val6 = permzz_read(i)
464  val7 = kappa_s_read(i)
465  val8 = kappa_f_read(i)
466  val9 = kappa_fr_read(i)
467  val10 = eta_f_read(i)
468  val11 = mu_fr_read(i)
469  val12 = qmu(i)
470  ! old
471  !write(IOUT) i,icodemat(i),rho_s_read(i),rho_f_read(i),phi_read(i),tortuosity_read(i), &
472  ! permxx_read(i),permxz_read(i),permzz_read(i),kappa_s_read(i), &
473  ! kappa_f_read(i),kappa_fr_read(i),eta_f_read(i),mu_fr_read(i),Qmu(i)
474 
475  else if (indic <= 0) then
476  ! external material
477  ! The values will be read from an external tomo file
478  val0 = rho_s_read(i)
479  val1 = cp(i)
480  val2 = cs(i)
481  val3 = 0.d0
482  val4 = 0.d0
483  val5 = qkappa(i)
484  val6 = qmu(i)
485  val7 = 0.d0
486  val8 = 0.d0
487  val9 = 0.d0
488  val10 = 0.d0
489  val11 = 0.d0
490  val12 = 0.d0
491  ! old
492  !write(IOUT) i,icodemat(i),rho_s_read(i),cp(i),cs(i),0,0,QKappa(i),Qmu(i),0,0,0,0,0,0
493  else
494  ! case should not occur
495  stop 'Unknown material code'
496  endif
497 
498  ! check format with file src/specfem2D/read_materials.f90
499  write(iout) i,indic,val0,val1,val2,val3,val4,val5,val6,val7,val8,val9,val10,val11,val12
500  enddo
501 
502  ! writes out material properties
503  ! 'Arrays kmato and knods for each bloc:'
504 
505 ! DK DK add support for using PML in MPI mode with external mesh
506 ! call write_partition_database(IOUT, iproc, nspec, num_material, ngnod, 2)
507  call write_partition_database(iout, iproc, nspec, num_material, region_pml_external_mesh, ngnod, 2)
508 
509  end subroutine save_databases_materials
