34  subroutine add_manual_crack(npgeo_ori)
35 
36  use constants, only: nb_points_to_add_to_npgeo,fast_numbering
37  use specfem_par
38 
39  implicit none
40 
41  integer,intent(in) :: npgeo_ori
42 
43  ! local parameters
44  integer :: ispec,ispec2,ignod
45  integer :: check_nb_points_to_add_to_npgeo,current_last_point,original_value
46  logical :: already_found_a_crack_element
47 
48  ! safety check
49  if (nproc > 1) stop 'currently only serial runs are handled when adding a crack manually'
50 
51 !! DK DK material number 2 indicates the spectral elements that form the left vertical side of the crack
52  check_nb_points_to_add_to_npgeo = count(kmato == 2)
53  print *
54  print *,'adding a crack manually'
55  print *,'need to add ',nb_points_to_add_to_npgeo,' npgeo mesh points to do that'
56 
57  if (check_nb_points_to_add_to_npgeo /= nb_points_to_add_to_npgeo) &
58  stop 'must have check_nb_points_to_add_to_npgeo == NB_POINTS_TO_ADD_TO_NPGEO when adding a crack manually'
59 
60  if (ngnod /= 4) stop 'must currently have ngnod == 4 when adding a crack manually'
61 
62  if (fast_numbering) stop 'must not have FAST_NUMBERING when adding a crack manually'
63 
64  !! DK DK modify arrays "knods" and "coorg" to introduce the crack manually by duplicating and splitting the nodes
65  already_found_a_crack_element = .false.
66  current_last_point = npgeo_ori
67 
68  do ispec = 1,nspec-1
69 !! DK DK my convention is to introduce a vertical crack between two elements with material numbers 2 and 3
70  if (kmato(ispec) == 2 .and. kmato(ispec+1) == 3) then
71 
72  print *,'adding a crack between elements ',ispec,' and ',ispec+1
73 
74 !! DK DK duplicate and split the lower-right corner of this element,
75 !! DK DK except if it is the first crack element found, because then it is the crack
76 !! DK DK tip and thus it should be assembled rather than split.
77 !! DK DK Lower-right corner of an element is local npgeo point #2
78  if (already_found_a_crack_element .and. knods(2,ispec) <= npgeo_ori) then
79  current_last_point = current_last_point + 1
80  original_value = knods(2,ispec)
81 !! DK DK split this point number in all the elements in which it appears
82  do ispec2 = 1,nspec
83 ! do this only for elements that define the left vertical edge of the crack
84  if (kmato(ispec2) /= 2) cycle
85  do ignod = 1,ngnod
86  if (knods(ignod,ispec2) == original_value) then
87  knods(ignod,ispec2) = current_last_point
88  coorg(:,current_last_point) = coorg(:,original_value)
89  endif
90  enddo
91  enddo
92  endif
93 
94 !! DK DK duplicate and split the upper-right corner of this element
95  already_found_a_crack_element = .true.
96 
97 !! DK DK Upper-right corner of an element is local npgeo point #3
98  if (knods(3,ispec) <= npgeo_ori) then
99 
100  current_last_point = current_last_point + 1
101  original_value = knods(3,ispec)
102 !! DK DK split this point number in all the elements in which it appears
103  do ispec2 = 1,nspec
104 ! do this only for elements that define the left vertical edge of the crack
105  if (kmato(ispec2) /= 2) cycle
106  do ignod = 1,ngnod
107  if (knods(ignod,ispec2) == original_value) then
108  knods(ignod,ispec2) = current_last_point
109  coorg(:,current_last_point) = coorg(:,original_value)
110  endif
111  enddo
112  enddo
113  endif
114 
115  endif ! of if (kmato(ispec) == 2 .and. kmato(ispec+1) == 3)
116 
117  enddo ! ispec
118 
119  if (current_last_point /= npgeo) then
120  print *,'current_last_point = ',current_last_point
121  print *,'npgeo_new = ',npgeo
122  stop 'did not find the right total number of points, should have current_last_point == npgeo_new'
123  endif
124 
125  end subroutine add_manual_crack
