34  subroutine is_in_convex_quadrilateral(elmnt_coords, x_coord, z_coord, is_in)
35 
36  implicit none
37 
38  double precision, dimension(2,4) :: elmnt_coords
39  double precision, intent(in) :: x_coord, z_coord
40  logical, intent(out) :: is_in
41 
42  real :: x1, x2, x3, x4, z1, z2, z3, z4
43  real :: normal1, normal2, normal3, normal4
44 
45  x1 = elmnt_coords(1,1)
46  x2 = elmnt_coords(1,2)
47  x3 = elmnt_coords(1,3)
48  x4 = elmnt_coords(1,4)
49  z1 = elmnt_coords(2,1)
50  z2 = elmnt_coords(2,2)
51  z3 = elmnt_coords(2,3)
52  z4 = elmnt_coords(2,4)
53 
54  normal1 = (z_coord-z1) * (x2-x1) - (x_coord-x1) * (z2-z1)
55  normal2 = (z_coord-z2) * (x3-x2) - (x_coord-x2) * (z3-z2)
56  normal3 = (z_coord-z3) * (x4-x3) - (x_coord-x3) * (z4-z3)
57  normal4 = (z_coord-z4) * (x1-x4) - (x_coord-x4) * (z1-z4)
58 
59  if ((normal1 < 0) .or. (normal2 < 0) .or. (normal3 < 0) .or. (normal4 < 0)) then
60  is_in = .false.
61  else
62  is_in = .true.
63  endif
64 
65  end subroutine is_in_convex_quadrilateral
