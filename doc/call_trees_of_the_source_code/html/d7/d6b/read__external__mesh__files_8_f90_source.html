37  !-----------------------------------------------
38  ! Read the mesh and storing it in array 'elmnts' (which is allocated here).
39  ! 'remove_min_to_start_at_zero' is used to have the numbering of the nodes starting at '0'.
40  ! 'nelmnts' is the number of elements, 'nnodes' is the number of nodes in the mesh.
41  !-----------------------------------------------
42 
43  subroutine read_external_mesh_file(filename, remove_min_to_start_at_zero, ngnod)
44 
45  use constants, only: max_string_len,imain
46  use part_unstruct_par, only: elmnts,nelmnts,nnodes
47 
48  implicit none
49 
50  character(len=MAX_STRING_LEN), intent(in) :: filename
51  integer, intent(out) :: remove_min_to_start_at_zero
52  integer, intent(in) :: ngnod
53 
54  integer :: i,ier
55 
56  ! user output
57  write(imain,*)
58  write(imain,*) 'Reading data from external mesh file: ',trim(filename)
59  call flush_imain()
60 
61 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
62  open(unit=990, file=trim(filename), form='unformatted' , status='old', action='read',iostat=ier)
63 #else
64  open(unit=990, file=trim(filename), form='formatted' , status='old', action='read',iostat=ier)
65 #endif
66  if (ier /= 0) then
67  print *,'Error opening file: ',trim(filename)
68  stop 'Error read external mesh file'
69  endif
70 
71 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
72  read(990) nelmnts
73 #else
74  read(990,*) nelmnts
75 #endif
76 
77  allocate(elmnts(0:ngnod*nelmnts-1))
78 
79  do i = 0, nelmnts-1
80  if (ngnod == 4) then
81 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
82  read(990) elmnts(i*ngnod), elmnts(i*ngnod+1), elmnts(i*ngnod+2), elmnts(i*ngnod+3)
83 #else
84  read(990,*) elmnts(i*ngnod), elmnts(i*ngnod+1), elmnts(i*ngnod+2), elmnts(i*ngnod+3)
85 #endif
86  else if (ngnod == 9) then
87 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
88  read(990) elmnts(i*ngnod), elmnts(i*ngnod+1), elmnts(i*ngnod+2), elmnts(i*ngnod+3), &
89  elmnts(i*ngnod+4), elmnts(i*ngnod+5), elmnts(i*ngnod+6), elmnts(i*ngnod+7), elmnts(i*ngnod+8)
90 #else
91  read(990,*) elmnts(i*ngnod), elmnts(i*ngnod+1), elmnts(i*ngnod+2), elmnts(i*ngnod+3), &
92  elmnts(i*ngnod+4), elmnts(i*ngnod+5), elmnts(i*ngnod+6), elmnts(i*ngnod+7), elmnts(i*ngnod+8)
93 #endif
94  else
95  stop 'error, ngnod should be either 4 or 9 for external meshes'
96  endif
97  enddo
98 
99  close(990)
100 
101  remove_min_to_start_at_zero = minval(elmnts)
102  elmnts(:) = elmnts(:) - remove_min_to_start_at_zero
103 
104  nnodes = maxval(elmnts) + 1
105 
106  ! user output
107  write(imain,*) 'Total number of spectral elements :',nelmnts
108  write(imain,*)
109  call flush_imain()
110 
111 
112  end subroutine read_external_mesh_file

281  subroutine read_external_acoustic_surface(filename, num_material, &
282  nbmodels, icodemat, phi_material, remove_min_to_start_at_zero)
283 
284  use constants, only: max_string_len,anisotropic_material
285  use part_unstruct_par, only: nelmnts,nelem_acoustic_surface,acoustic_surface
286 
287  implicit none
288 
289  character(len=MAX_STRING_LEN), intent(in) :: filename
290  integer, dimension(0:nelmnts-1) :: num_material
291  integer, intent(in) :: nbmodels
292  integer, dimension(1:nbmodels), intent(in) :: icodemat
293  double precision, dimension(1:nbmodels), intent(in) :: phi_material
294  integer, intent(in) :: remove_min_to_start_at_zero
295 
296  ! local parameters
297  integer, dimension(:,:), allocatable :: acoustic_surface_tmp
298  integer :: nelmnts_surface
299  integer :: i,ier
300  integer :: imaterial_number
301 
302 
303 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
304  open(unit=993, file=trim(filename), form='unformatted' , status='old', action='read', iostat=ier)
305 #else
306  open(unit=993, file=trim(filename), form='formatted' , status='old', action='read', iostat=ier)
307 #endif
308  if (ier /= 0) then
309  print *,'Error opening file: ',trim(filename)
310  stop 'Error read acoustic surface file'
311  endif
312 
313 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
314  read(993) nelmnts_surface
315 #else
316  read(993,*) nelmnts_surface
317 #endif
318 
319  allocate(acoustic_surface_tmp(4,nelmnts_surface))
320 
321  do i = 1, nelmnts_surface
322 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
323  read(993) acoustic_surface_tmp(1,i), acoustic_surface_tmp(2,i), acoustic_surface_tmp(3,i), acoustic_surface_tmp(4,i)
324 #else
325  read(993,*) acoustic_surface_tmp(1,i), acoustic_surface_tmp(2,i), acoustic_surface_tmp(3,i), acoustic_surface_tmp(4,i)
326 #endif
327  enddo
328 
329  close(993)
330 
331  acoustic_surface_tmp(1,:) = acoustic_surface_tmp(1,:) - remove_min_to_start_at_zero
332  acoustic_surface_tmp(3,:) = acoustic_surface_tmp(3,:) - remove_min_to_start_at_zero
333  acoustic_surface_tmp(4,:) = acoustic_surface_tmp(4,:) - remove_min_to_start_at_zero
334 
335  nelem_acoustic_surface = 0
336  do i = 1, nelmnts_surface
337  imaterial_number = num_material(acoustic_surface_tmp(1,i))
338  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_material(imaterial_number) >= 1.d0) then
339  nelem_acoustic_surface = nelem_acoustic_surface + 1
340  endif
341  enddo
342 
343  allocate(acoustic_surface(4,nelem_acoustic_surface),stat=ier)
344  if (ier /= 0) stop 'Error allocating acoustic_surface array'
345 
346  nelem_acoustic_surface = 0
347  do i = 1, nelmnts_surface
348  imaterial_number = num_material(acoustic_surface_tmp(1,i))
349  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_material(imaterial_number) >= 1.d0) then
350  nelem_acoustic_surface = nelem_acoustic_surface + 1
351  acoustic_surface(:,nelem_acoustic_surface) = acoustic_surface_tmp(:,i)
352  endif
353  enddo
354 
355  end subroutine read_external_acoustic_surface

368  subroutine read_external_abs_surface(filename, remove_min_to_start_at_zero)
369 
370  use constants, only: max_string_len
371  use part_unstruct_par, only: abs_surface,nelemabs
372 
373  implicit none
374 
375  character(len=MAX_STRING_LEN), intent(in) :: filename
376  integer, intent(in) :: remove_min_to_start_at_zero
377 
378  integer :: i,ier
379 
380 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
381  open(unit=994, file=trim(filename), form='unformatted' , status='old', action='read', iostat=ier)
382 #else
383  open(unit=994, file=trim(filename), form='formatted' , status='old', action='read', iostat=ier)
384 #endif
385  if (ier /= 0) then
386  print *,'Error opening file: ',trim(filename)
387  stop 'Error read absorbing surface file'
388  endif
389 
390 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
391  read(994) nelemabs
392 #else
393  read(994,*) nelemabs
394 #endif
395 
396  allocate(abs_surface(5,nelemabs))
397 
398  do i = 1, nelemabs
399 #ifdef USE_BINARY_FOR_EXTERNAL_MESH_DATABASE
400  read(994) abs_surface(1,i), abs_surface(2,i), abs_surface(3,i), abs_surface(4,i), abs_surface(5,i)
401 #else
402  read(994,*) abs_surface(1,i), abs_surface(2,i), abs_surface(3,i), abs_surface(4,i), abs_surface(5,i)
403 #endif
404 
405  if (abs_surface(2,i) /= 2) then
406  print *,'Only two nodes per absorbing element can be listed.'
407  print *,'If one of your elements has more than one edge along a given absorbing contour'
408  print *,'(e.g., if that contour has a corner) then list it twice,'
409  print *,'putting the first edge on the first line and the second edge on the second line.'
410  print *,'If one of your elements has a single point along the absording contour rather than a full edge, do NOT list it'
411  print *,'(it would have no weight in the contour integral anyway because it would consist of a single point).'
412  print *,'If you use 9-node elements, list only the first and last points of the edge and not the intermediate point'
413  print *,'located around the middle of the edge; the right 9-node curvature will be restored automatically by the code.'
414 
415  stop 'only two nodes per element should be listed for absorbing edges'
416  endif
417 
418  if (abs_surface(5,i) < 1 .or. abs_surface(5,i) > 4) then
419  stop 'absorbing element type must be between 1 (IBOTTOM) and 4 (ILEFT)'
420  endif
421 
422  enddo
423 
424  close(994)
425 
426  abs_surface(1,:) = abs_surface(1,:) - remove_min_to_start_at_zero
427  abs_surface(3,:) = abs_surface(3,:) - remove_min_to_start_at_zero
428  abs_surface(4,:) = abs_surface(4,:) - remove_min_to_start_at_zero
429 
430  end subroutine read_external_abs_surface
