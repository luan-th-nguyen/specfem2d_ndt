45 __global__ void compute_kernels_ani_cudakernel(int* ispec_is_elastic,
46  int* d_ibool,
47  realw* accel,
48  realw* b_displ,
49  realw* epsilondev_xx,realw* epsilondev_yy,realw* epsilondev_xy,
50  realw* epsilondev_xz,realw* epsilondev_yz,
51  realw* b_epsilondev_xx,realw* b_epsilondev_yy,realw* b_epsilondev_xy,
52  realw* b_epsilondev_xz,realw* b_epsilondev_yz,
53  realw* rho_kl,
54  realw deltat,
55  realw* cijkl_kl,
56  realw* epsilon_trace_over_3,
57  realw* b_epsilon_trace_over_3,
58  int NSPEC_AB) {
59 
60  int ispec = blockIdx.x + blockIdx.y*gridDim.x;
61  int ijk = threadIdx.x;
62  int ijk_ispec = ijk + NGLL3*ispec;
63  int ijk21_ispec = ijk + 21*NGLL3*ispec;
64 
65  realw prod[21];
66  realw eps[6];
67  realw b_eps[6];
68  realw epsdev[6];
69  realw b_epsdev[6];
70  realw eps_trace_over_3,b_eps_trace_over_3;
71  int i,j;
72 
73  // handles case when there is 1 extra block (due to rectangular grid)
74  if (ispec < NSPEC_AB) {
75 
76  // elastic elements only
77  if (ispec_is_elastic[ispec]) {
78  int iglob = d_ibool[ijk + NGLL3_PADDED*ispec] - 1;
79 
80  // anisotropic kernels:
81  // density kernel
82  rho_kl[ijk_ispec] += deltat * (accel[3*iglob]*b_displ[3*iglob]+
83  accel[3*iglob+1]*b_displ[3*iglob+1]+
84  accel[3*iglob+2]*b_displ[3*iglob+2]);
85 
86 
87  // anisotropic kernel
88  epsdev[0] = epsilondev_xx[ijk_ispec];
89  epsdev[1] = epsilondev_yy[ijk_ispec];
90  epsdev[2] = epsilondev_xy[ijk_ispec];
91  epsdev[3] = epsilondev_xz[ijk_ispec];
92  epsdev[4] = epsilondev_yz[ijk_ispec];
93 
94  b_epsdev[0] = b_epsilondev_xx[ijk_ispec];
95  b_epsdev[1] = b_epsilondev_yy[ijk_ispec];
96  b_epsdev[2] = b_epsilondev_xy[ijk_ispec];
97  b_epsdev[3] = b_epsilondev_xz[ijk_ispec];
98  b_epsdev[4] = b_epsilondev_yz[ijk_ispec];
99 
100  eps_trace_over_3 = epsilon_trace_over_3[ijk_ispec];
101  b_eps_trace_over_3 = b_epsilon_trace_over_3[ijk_ispec];
102 
103  //! Building of the local matrix of the strain tensor
104  //! for the adjoint field and the regular backward field
105  //!eps11 et eps22
106  eps[0] = epsdev[0] + eps_trace_over_3;
107  eps[1] = epsdev[1] + eps_trace_over_3;
108  //!eps33
109  eps[2] = -(eps[0]+eps[1])+3*eps_trace_over_3;
110  //!eps23
111  eps[3] = epsdev[4];
112  //!eps13
113  eps[4] = epsdev[3];
114  //!eps12
115  eps[5] = epsdev[2];
116 
117  // backward arrays
118  b_eps[0] = b_epsdev[0] + b_eps_trace_over_3;
119  b_eps[1] = b_epsdev[1] + b_eps_trace_over_3;
120  b_eps[2] = -(b_eps[0]+b_eps[1])+3*b_eps_trace_over_3;
121  b_eps[3] = b_epsdev[4];
122  b_eps[4] = b_epsdev[3];
123  b_eps[5] = b_epsdev[2];
124 
125  //! Computing the 21 strain products without assuming eps(i)*b_eps(j) = eps(j)*b_eps(i)
126  int p = 0;
127  for( i=0; i<6; i++){
128  for( j=i; j<6; j++){
129  prod[p] = eps[i] * b_eps[j];
130  if (j > i) {
131  prod[p] = prod[p] + eps[j]*b_eps[i];
132  if (j > 2 && i < 3) { prod[p] = prod[p]*2; }
133  }
134  if (i > 2) { prod[p] = prod[p]*4; }
135  p++;
136  }
137  }
138 
139  // all 21 anisotropic coefficients
140  for( i=0; i<21; i++){
141  cijkl_kl[i+ijk21_ispec] += deltat * prod[i];
142  }
143 
144  }
145  }
146 }

150 __global__ void compute_kernels_cudakernel(int* ispec_is_elastic,
151  int* d_ibool,
152  realw* accel,
153  realw* b_displ,
154  realw* rho_kl,
155  realw* mu_kl,
156  realw* kappa_kl,
157  int NSPEC_AB,
158  realw* dsxx,
159  realw* dsxz,
160  realw* dszz,
161  realw* b_dsxx,
162  realw* b_dsxz,
163  realw* b_dszz,
164  realw deltat) {
165 
166  int ispec = blockIdx.x + blockIdx.y*gridDim.x;
167  int ij = threadIdx.x;
168  int ij_ispec = ij + NGLL2*ispec;
169 
170 
171  // handles case when there is 1 extra block (due to rectangular grid)
172  if (ispec < NSPEC_AB) {
173 
174  // elastic elements only
175  if (ispec_is_elastic[ispec]) {
176  int iglob = d_ibool[ij + NGLL2_PADDED*ispec] - 1 ;
177 
178  // isotropic kernels:
179  // density kernel
180  rho_kl[ij_ispec] += (accel[2*iglob]*b_displ[2*iglob]+
181  accel[2*iglob+1]*b_displ[2*iglob+1]);
182 
183 
184  realw prod = (dsxx[iglob] + dszz[iglob]) * (b_dsxx[iglob] + b_dszz[iglob]);
185 
186  // shear modulus kernel
187  mu_kl[ij_ispec] += (dsxx[iglob] * b_dsxx[iglob] + dszz[iglob] * b_dszz[iglob] +
188  0.5* dsxz[iglob] * b_dsxz[iglob] -
189  prod/3) * deltat;
190 
191  // bulk modulus kernel
192  kappa_kl[ij_ispec] += prod * deltat;
193 
194  }
195  }
196 }

246 __device__ void compute_gradient_kernel(int ij,
247  int ispec,
248  realw* scalar_field,
249  realw* vector_field_element,
250  realw* d_hprime_xx,
251  realw* d_xix,realw* d_xiz,
252  realw* d_gammax,realw* d_gammaz,
253  realw rhol) {
254 
255  realw temp1l,temp3l;
256  realw hp1,hp3;
257  realw xixl,xizl,gammaxl,gammazl;
258  realw rho_invl;
259  int l,offset,offset1,offset3;
260 
261  const int NGLL2_ALIGN = NGLL2_PADDED;
262 
263  int J = (ij/NGLLX);
264  int I = (ij-J*NGLLX);
265 
266  // derivative along x
267  temp1l = 0.f;
268  for( l=0; l<NGLLX;l++){
269  hp1 = d_hprime_xx[l*NGLLX+I];
270  offset1 = J*NGLLX+l;
271  temp1l += scalar_field[offset1]*hp1;
272  }
273 
274  // derivative along z
275  temp3l = 0.f;
276  for( l=0; l<NGLLX;l++){
277  // assumes hprime_xx == hprime_zz
278  hp3 = d_hprime_xx[l*NGLLX+J];
279  offset3 = l*NGLLX+I;
280  temp3l += scalar_field[offset3]*hp3;
281  }
282 
283  offset = ispec*NGLL2_ALIGN + ij;
284 
285  xixl = d_xix[offset];
286  xizl = d_xiz[offset];
287  gammaxl = d_gammax[offset];
288  gammazl = d_gammaz[offset];
289 
290  rho_invl = 1.0f / rhol;
291 
292  // derivatives of acoustic scalar potential field on GLL points
293  vector_field_element[0] = (temp1l*xixl + temp3l*gammaxl) * rho_invl;
294  vector_field_element[1] = (temp1l*xizl + temp3l*gammazl) * rho_invl;
295 }

300 __global__ void compute_kernels_acoustic_kernel(int* ispec_is_acoustic,
301  int* d_ibool,
302  realw* rhostore,
303  realw* kappastore,
304  realw* d_hprime_xx,
305  realw* d_xix,realw* d_xiz,
306  realw* d_gammax,realw* d_gammaz,
307  realw* potential_dot_dot_acoustic,
308  realw* b_potential_acoustic,
309  realw* b_potential_dot_dot_acoustic,
310  realw* rho_ac_kl,
311  realw* kappa_ac_kl,
312  int NSPEC_AB,
313  realw deltat) {
314 
315  int ispec = blockIdx.x + blockIdx.y*gridDim.x;
316  int ij = threadIdx.x;
317 
318  // local and global indices
319  int ij_ispec = ij + NGLL2*ispec;
320  int ij_ispec_padded = ij + NGLL2_PADDED*ispec;
321  int iglob;
322 
323  // shared memory between all threads within this block
324  __shared__ realw scalar_field_displ[NGLL2];
325  __shared__ realw scalar_field_accel[NGLL2];
326 
327  int active = 0;
328 
329  // handles case when there is 1 extra block (due to rectangular grid)
330  if (ispec < NSPEC_AB) {
331  // acoustic elements only
332  if (ispec_is_acoustic[ispec]) {
333  active = 1;
334 
335  // copy field values
336  iglob = d_ibool[ij_ispec_padded] - 1;
337  scalar_field_displ[ij] = b_potential_acoustic[iglob];
338  scalar_field_accel[ij] = potential_dot_dot_acoustic[iglob];
339  }
340  }
341 
342  // synchronizes threads
343  __syncthreads();
344 
345  if (active) {
346  realw accel_elm[2];
347  realw b_displ_elm[2];
348  realw rhol,kappal;
349 
350  // gets material parameter
351  rhol = rhostore[ij_ispec_padded];
352 
353  // displacement vector from backward field
354  compute_gradient_kernel(ij,ispec,scalar_field_displ,b_displ_elm,
355  d_hprime_xx,
356  d_xix,d_xiz,d_gammax,d_gammaz,
357  rhol);
358 
359  // acceleration vector
360  compute_gradient_kernel(ij,ispec,scalar_field_accel,accel_elm,
361  d_hprime_xx,
362  d_xix,d_xiz,d_gammax,d_gammaz,
363  rhol);
364 
365  // density kernel
366  rho_ac_kl[ij_ispec] -= rhol * (accel_elm[0]*b_displ_elm[0] +
367  accel_elm[1]*b_displ_elm[1] ) * deltat;
368 
369  // bulk modulus kernel
370  kappal = kappastore[ij_ispec];
371  kappa_ac_kl[ij_ispec] -= potential_dot_dot_acoustic[iglob] * b_potential_acoustic[iglob] * deltat / kappal ;
372  } // active
373 }

420 __global__ void compute_kernels_hess_el_cudakernel(int* ispec_is_elastic,
421  int* d_ibool,
422  realw* accel,
423  realw* b_accel,
424  realw* hess_kl,
425  int NSPEC_AB) {
426 
427  int ispec = blockIdx.x + blockIdx.y*gridDim.x;
428  int ij = threadIdx.x;
429 
430  // handles case when there is 1 extra block (due to rectangular grid)
431  if (ispec < NSPEC_AB) {
432 
433  // elastic elements only
434  if (ispec_is_elastic[ispec]) {
435  int iglob = d_ibool[ij + NGLL2_PADDED*ispec] - 1;
436 
437  // approximate hessian
438  hess_kl[ij + NGLL2*ispec] += (accel[2*iglob]*b_accel[2*iglob]+
439  accel[2*iglob+1]*b_accel[2*iglob+1]);
440  }
441  }
442 }
