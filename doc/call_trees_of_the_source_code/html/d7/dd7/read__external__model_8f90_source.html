34 
35  subroutine read_external_model()
36 
37 ! reads in external model files
38 
39  use constants, only: custom_real,ngllx,ngllz,tinyval,imain,in_data_files
40 
41  use specfem_par, only: nspec,nglob,ibool,ispec_is_elastic,ispec_is_anisotropic, &
42  coord,kmato,model,myrank,setup_with_binary_database
43 
44  ! external model parameters
45  use specfem_par, only: rhoext,vpext,vsext,gravityext,nsqext, &
46  qkappa_attenuationext,qmu_attenuationext,c11ext,c13ext,c15ext,c33ext,c35ext,c55ext,c12ext,c23ext,c25ext
47 
48  implicit none
49 
50  ! Local variables
51  integer :: i,j,ispec,itmp
52  integer :: ier
53  real(kind=CUSTOM_REAL) :: tmp1,tmp2
54  double precision :: vs_val,vp_val,rho_val
55  character(len=150) :: inputname
56 
57  ! note: we read in external models once the basic mesh with its geometry and GLL points has been setup.
58  ! External models define new velocity/material parameters which need to be defined on all GLL points.
59 
60  select case (trim(model))
61  case ('legacy')
62  ! old model format
63  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_model_velocity.dat_input'
64  if (myrank == 0) write(imain,*) ' reading external files: ','DATA/proc*****_model_velocity.dat_input'
65 
66  open(unit=1001,file=inputname,status='old',action='read',iostat=ier)
67  if (ier /= 0) stop 'Error opening DATA/proc*****_model_velocity.dat_input file.'
68  do ispec = 1,nspec
69  do j = 1,ngllz
70  do i = 1,ngllx
71  ! format: #unused #unused #unused #rho #vp #vs
72  read(1001,*) itmp,tmp1,tmp2,rho_val,vp_val,vs_val
73  rhoext(i,j,ispec) = rho_val
74  vpext(i,j,ispec) = vp_val
75  vsext(i,j,ispec) = vs_val
76  enddo
77  enddo
78  enddo
79  close(1001)
80  ! default no attenuation
81  qkappa_attenuationext(:,:,:) = 9999.d0
82  qmu_attenuationext(:,:,:) = 9999.d0
83 
84  case ('ascii')
85  ! ascii model format
86  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_rho_vp_vs.dat'
87  if (myrank == 0) write(imain,*) ' reading external files: ','DATA/proc*****_rho_vp_vs.dat'
88 
89  open(unit=1001,file=inputname,status='old',action='read',iostat=ier)
90  if (ier /= 0) stop 'Error opening DATA/proc*****_rho_vp_vs.dat file.'
91  do ispec = 1,nspec
92  do j = 1,ngllz
93  do i = 1,ngllx
94  ! format: #unused #unused #rho #vp #vs
95  read(1001,*) tmp1,tmp2,rho_val,vp_val,vs_val
96  rhoext(i,j,ispec) = rho_val
97  vpext(i,j,ispec) = vp_val
98  vsext(i,j,ispec) = vs_val
99  enddo
100  enddo
101  enddo
102  close(1001)
103  ! default no attenuation
104  qkappa_attenuationext(:,:,:) = 9999.d0
105  qmu_attenuationext(:,:,:) = 9999.d0
106 
107  case ('binary','gll')
108  ! binary formats
109  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_rho.bin'
110  if (myrank == 0) write(imain,*) ' reading external files: ','DATA/proc*****_rho.bin, .._vp.bin, .._vs.bin'
111 
112  open(unit = 1001, file = inputname, status='old',action='read',form='unformatted',iostat=ier)
113  if (ier /= 0) stop 'Error opening DATA/proc*****_rho.bin file.'
114 
115  read(1001) rhoext
116  close(1001)
117  print *, 'rho', minval(rhoext), maxval(rhoext)
118 
119  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_vp.bin'
120  open(unit = 1001, file = inputname, status='old',action='read',form='unformatted',iostat=ier)
121  if (ier /= 0) stop 'Error opening DATA/proc*****_vp.bin file.'
122 
123  read(1001) vpext
124  close(1001)
125 
126  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_vs.bin'
127  open(unit = 1001, file = inputname, status='old',action='read',form='unformatted',iostat=ier)
128  if (ier /= 0) stop 'Error opening DATA/proc*****_vs.bin file.'
129 
130  read(1001) vsext
131  close(1001)
132 
133  ! default no attenuation
134  qkappa_attenuationext(:,:,:) = 9999.d0
135  qmu_attenuationext(:,:,:) = 9999.d0
136 
137  case ('binary_voigt')
138  ! Voigt model
139  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_rho.bin'
140  if (myrank == 0) write(imain,*) ' reading external files: ','DATA/proc*****_rho.bin, .._c11.bin, .._c55.bin'
141 
142  open(unit = 1001, file = inputname,status='old',action='read',form='unformatted',iostat=ier)
143  if (ier /= 0) stop 'Error opening DATA/proc*****_rho.bin file.'
144 
145  read(1001) rhoext
146  close(1001)
147  print *, 'rho', minval(rhoext), maxval(rhoext)
148 
149  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_c11.bin'
150  open(unit = 1001, file = inputname,status='old',action='read',form='unformatted', iostat=ier)
151  if (ier /= 0) stop 'Error opening DATA/proc*****_c11.bin file.'
152 
153  read(1001) c11ext
154  close(1001)
155  print *, 'c11ext', minval(c11ext), maxval(c11ext)
156 
157  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_c13.bin'
158  open(unit = 1001, file = inputname,status='old',action='read',form='unformatted', iostat=ier)
159  if (ier /= 0) stop 'Error opening DATA/proc*****_c13.bin file.'
160 
161  read(1001) c13ext
162  close(1001)
163  print *, 'c13ext', minval(c13ext), maxval(c13ext)
164 
165  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_c15.bin'
166  open(unit = 1001, file = inputname,status='old',action='read',form='unformatted', iostat=ier)
167  if (ier /= 0) stop 'Error opening DATA/proc*****_c15.bin file.'
168 
169  read(1001) c15ext
170  close(1001)
171  print *, 'c15ext', minval(c15ext), maxval(c15ext)
172 
173  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_c33.bin'
174  open(unit = 1001, file = inputname,status='old',action='read',form='unformatted', iostat=ier)
175  if (ier /= 0) stop 'Error opening DATA/proc*****_c33.bin file.'
176 
177  read(1001) c33ext
178  close(1001)
179  print *, 'c33ext', minval(c33ext), maxval(c33ext)
180 
181  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_c35.bin'
182  open(unit = 1001, file = inputname,status='old',action='read',form='unformatted', iostat=ier)
183  if (ier /= 0) stop 'Error opening DATA/proc*****_c35.bin file.'
184 
185  read(1001) c35ext
186  close(1001)
187  print *, 'c35ext', minval(c35ext), maxval(c35ext)
188 
189  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_c55.bin'
190  open(unit = 1001, file = inputname,status='old',action='read',form='unformatted', iostat=ier)
191  if (ier /= 0) stop 'Error opening DATA/proc*****_c55.bin file.'
192 
193  read(1001) c55ext
194  close(1001)
195  print *, 'c55ext', minval(c55ext), maxval(c55ext)
196 
197  do ispec = 1,nspec
198  do j = 1,ngllz
199  do i = 1,ngllx
200  if (c55ext(i,j,ispec) < tinyval) then
201  c33ext(i,j,ispec) = 0.0_custom_real
202  c12ext(i,j,ispec) = 0.0_custom_real
203  c23ext(i,j,ispec) = 0.0_custom_real
204  vpext(i,j,ispec) = 1500.0
205  vsext(i,j,ispec) = 0.0_custom_real
206  else
207  c12ext(i,j,ispec) = 1.d-6
208  c23ext(i,j,ispec) = 1.d-6
209  vpext(i,j,ispec) = sqrt(c33ext(i,j,ispec)/rhoext(i,j,ispec))
210  vsext(i,j,ispec) = sqrt(c55ext(i,j,ispec)/rhoext(i,j,ispec))
211  endif
212  c25ext(i,j,ispec) = 0.0_custom_real
213  enddo
214  enddo
215  enddo
216 
217  ! default no attenuation
218  qkappa_attenuationext(:,:,:) = 9999.d0
219  qmu_attenuationext(:,:,:) = 9999.d0
220 
221  case ('external')
222  ! generic model defined in external files
223  call define_external_model(coord,kmato,ibool,rhoext,vpext,vsext, &
224  qkappa_attenuationext,qmu_attenuationext,gravityext,nsqext, &
225  c11ext,c13ext,c15ext,c33ext,c35ext,c55ext,c12ext,c23ext,c25ext,nspec,nglob)
226  case ('marmousi')
227  ! marmousi type model
228  call define_external_model_from_marmousi(coord,ibool,rhoext,vpext,vsext, &
229  qkappa_attenuationext,qmu_attenuationext,gravityext,nsqext, &
230  c11ext,c13ext,c15ext,c33ext,c35ext,c55ext,c12ext,c23ext,c25ext,nspec,nglob)
231 
232  case ('tomo')
233  ! tomographic file
234  call define_external_model_from_tomo_file()
235 
236  case default
237  print *,"Error: unrecognized model = ",trim(model)
238  print *,"Invalid MODEL chosen, please check your Par_file settings..."
239  stop 'Invalid MODEL parameter'
240 
241  end select
242 
243  ! check that the external model that has just been defined makes sense
244  if (trim(model) == 'external' .or. trim(model) == 'tomo' .or. trim(model) == 'marmousi') then
245  ! checks velocities for each element
246  do ispec = 1,nspec
247  do j = 1,ngllz
248  do i = 1,ngllx
249  ! anisotropic elastic
250  if (c11ext(i,j,ispec) > tinyval .or. c13ext(i,j,ispec) > tinyval .or. c15ext(i,j,ispec) > tinyval .or. &
251  c33ext(i,j,ispec) > tinyval .or. c35ext(i,j,ispec) > tinyval .or. c55ext(i,j,ispec) > tinyval) then
252  ! vp, vs : assign dummy values, trick to avoid floating point errors in the case of an anisotropic medium
253  vpext(i,j,ispec) = 20.d0
254  vsext(i,j,ispec) = 10.d0
255  endif
256 
257  ! check that the element type is not redefined compared to what is defined initially in DATA/Par_file
258  if ((c11ext(i,j,ispec) > tinyval .or. c13ext(i,j,ispec) > tinyval .or. c15ext(i,j,ispec) > tinyval .or. &
259  c33ext(i,j,ispec) > tinyval .or. c35ext(i,j,ispec) > tinyval .or. c55ext(i,j,ispec) > tinyval) &
260  .and. .not. ispec_is_anisotropic(ispec)) &
261  stop 'Error: non anisotropic material in DATA/Par_file or &
262  &external mesh redefined as anisotropic in define_external_model()'
263 
264  ! acoustic element
265  if (vsext(i,j,ispec) < tinyval .and. (ispec_is_elastic(ispec) .or. ispec_is_anisotropic(ispec))) &
266  stop 'Error: non acoustic material in DATA/Par_file or &
267  &external mesh redefined as acoustic in define_external_model()'
268 
269  ! elastic element
270  if (vsext(i,j,ispec) > tinyval .and. .not. ispec_is_elastic(ispec)) &
271  stop 'Error: acoustic material in DATA/Par_file or &
272  &external mesh redefined as non acoustic in define_external_model()'
273  enddo
274  enddo
275  enddo
276  endif
277 
278  ! resets domain flags
279  if (setup_with_binary_database /= 2) then
280  call get_simulation_domains_from_external_models()
281  endif
282 
283  end subroutine read_external_model
