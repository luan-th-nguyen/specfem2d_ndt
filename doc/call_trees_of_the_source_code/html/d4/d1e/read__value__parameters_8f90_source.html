126  subroutine read_next_line(iin,ignore_junk,string_read)
127 
128  use constants, only: max_string_len
129 
130  implicit none
131 
132  logical ignore_junk
133  character(len=MAX_STRING_LEN) string_read
134 
135  integer ios,iin,index_equal_sign
136 
137  do
138  ! daniel: actually MAX_STRING_LEN set to 512...
139  read(unit=iin,fmt="(a256)",iostat=ios) string_read
140  if (ios /= 0) stop 'error while reading input file'
141 
142 ! suppress leading white spaces, if any
143  string_read = adjustl(string_read)
144 
145 ! suppress trailing carriage return (ASCII code 13) if any (e.g. if input text file coming from Windows/DOS)
146  if (index(string_read,achar(13)) > 0) string_read = string_read(1:index(string_read,achar(13))-1)
147 
148 ! exit loop when we find the first line that is not a comment or a white line
149  if (len_trim(string_read) == 0) cycle
150  if (string_read(1:1) /= '#') exit
151 
152  enddo
153 
154 ! suppress trailing white spaces, if any
155  string_read = string_read(1:len_trim(string_read))
156 
157 ! suppress trailing comments, if any
158  if (index(string_read,'#') > 0) string_read = string_read(1:index(string_read,'#')-1)
159 
160 ! suppress leading junk (up to the first equal sign, included) if needed
161  if (ignore_junk) then
162  index_equal_sign = index(string_read,'=')
163  if (index_equal_sign <= 1 .or. index_equal_sign == len_trim(string_read)) stop 'incorrect syntax detected in DATA/Par_file'
164  string_read = string_read(index_equal_sign + 1:len_trim(string_read))
165  endif
166 
167 ! suppress leading and trailing white spaces again, if any, after having suppressed the leading junk
168  string_read = adjustl(string_read)
169  string_read = string_read(1:len_trim(string_read))
170 
171  end subroutine read_next_line
