35 
36 ! computes adjoint sensitivity kernel contributions
37 !
38 ! see e.g. Tromp et al. (2005) for elastic calculation
39 ! and Morency et al. (2009) for poroelastic calculation

49  call compute_kernels_ac()
50  endif
51 
52  ! elastic simulations
53  if (any_elastic) then
54  call compute_kernels_el()
55  endif
56 
57  ! poro-elastic simulations
58  if (any_poroelastic) then
59  call compute_kernels_po()
60  endif
61 
62  ! computes an approximative Hessian for preconditioning kernels
63  if (approximate_hess_kl) then
64  call compute_kernels_hessian()
65  endif
66 
67  end subroutine compute_kernels

72 
73  subroutine compute_kernels_el()
74 
75 ! elastic kernel calculations
76 ! see e.g. Tromp et al. (2005)
77 
78  use constants, only: custom_real,ngllx,ngllz,half,two,four_thirds
79 
80  use specfem_par, only: ispec_is_elastic,rho_k, & ! AXISYM,
81  rho_kl,mu_kl,kappa_kl,rhop_kl,beta_kl,alpha_kl,bulk_c_kl,bulk_beta_kl, &
82  nglob,nspec,ibool,accel_elastic,b_displ_elastic, &
83  density,poroelastcoef,kmato,assign_external_model,rhoext,vsext,vpext, &
84  deltat,p_sv,displ_elastic, &
85  mu_k,kappa_k,ibool,hprime_xx,hprime_zz,xix,xiz,gammax,gammaz, &
86  gpu_mode
87 
88  use specfem_par_gpu, only: mesh_pointer,deltatf
89 
90  use specfem_par, only: c11_k,c13_k,c15_k,c33_k,c35_k,c55_k,ispec_is_anisotropic, &
91  rho_kl,c11_kl,c13_kl,c15_kl,c33_kl,c35_kl,c55_kl
92 
93  implicit none
94 
95  !local variables
96  integer :: i,j,k,ispec,iglob
97  real(kind=CUSTOM_REAL) :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
98  real(kind=CUSTOM_REAL) :: dux_dxl,duz_dxl,dux_dzl,duz_dzl
99  real(kind=CUSTOM_REAL) :: b_dux_dxi,b_dux_dgamma,b_duz_dxi,b_duz_dgamma
100  real(kind=CUSTOM_REAL) :: b_dux_dxl,b_duz_dxl,b_dux_dzl,b_duz_dzl
101  real(kind=CUSTOM_REAL) :: dsxx,dsxz,dszz
102  real(kind=CUSTOM_REAL) :: b_dsxx,b_dsxz,b_dszz
103  real(kind=CUSTOM_REAL) :: rhol,mul,kappal
104 
105  ! Jacobian matrix and determinant
106  double precision :: xixl,xizl,gammaxl,gammazl
107 
108  ! elastic kernels
109  if (.not. gpu_mode) then
110  ! updates kernels on CPU
111  do ispec = 1,nspec
112  if (ispec_is_elastic(ispec)) then
113  do j = 1,ngllz; do i = 1,ngllx
114  ! derivative along x and along z
115  dux_dxi = 0._custom_real
116  duz_dxi = 0._custom_real
117  dux_dgamma = 0._custom_real
118  duz_dgamma = 0._custom_real
119  b_dux_dxi = 0._custom_real
120  b_duz_dxi = 0._custom_real
121  b_dux_dgamma = 0._custom_real
122  b_duz_dgamma = 0._custom_real
123 
124  ! first double loop over GLL points to compute and store gradients
125  ! we can merge the two loops because NGLLX == NGLLZ
126  do k = 1,ngllx
127  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
128  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
129  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
130  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
131 
132 
133  b_dux_dxi = b_dux_dxi + b_displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
134  b_duz_dxi = b_duz_dxi + b_displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
135  b_dux_dgamma = b_dux_dgamma + b_displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
136  b_duz_dgamma = b_duz_dgamma + b_displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
137 
138  enddo
139 
140  xixl = xix(i,j,ispec)
141  xizl = xiz(i,j,ispec)
142  gammaxl = gammax(i,j,ispec)
143  gammazl = gammaz(i,j,ispec)
144 
145  ! derivatives of displacement
146  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
147  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
148  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
149  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
150 
151  b_dux_dxl = b_dux_dxi*xixl + b_dux_dgamma*gammaxl
152  b_dux_dzl = b_dux_dxi*xizl + b_dux_dgamma*gammazl
153 
154  b_duz_dxl = b_duz_dxi*xixl + b_duz_dgamma*gammaxl
155  b_duz_dzl = b_duz_dxi*xizl + b_duz_dgamma*gammazl
156 
157  iglob = ibool(i,j,ispec)
158 
159  ! isotropic kernel contributions
160  if (p_sv) then
161  ! P-SV waves
162  dsxx = dux_dxl
163  dsxz = half * (duz_dxl + dux_dzl)
164  dszz = duz_dzl
165 
166  b_dsxx = b_dux_dxl
167  b_dsxz = half * (b_duz_dxl + b_dux_dzl)
168  b_dszz = b_duz_dzl
169 
170  kappa_k(iglob) = (dsxx + dszz) * (b_dsxx + b_dszz)
171  mu_k(iglob) = dsxx * b_dsxx + dszz * b_dszz + &
172  2._custom_real * dsxz * b_dsxz - 1._custom_real/3._custom_real * kappa_k(iglob)
173  else
174  ! SH (membrane) waves
175  mu_k(iglob) = dux_dxl * b_dux_dxl + dux_dzl * b_dux_dzl
176  endif
177 
178  ! Voigt kernels, e.g., see Sieminski, 2007a,b
179  if (ispec_is_anisotropic(ispec)) then
180  c11_k(iglob) = dux_dxl*b_dux_dxl
181  c13_k(iglob) = dux_dxl*b_duz_dzl + duz_dzl*b_dux_dxl
182  c15_k(iglob) = 2*(dux_dxl*half*(b_dux_dzl+b_duz_dxl)+&
183  half*(dux_dzl+duz_dxl)*b_dux_dxl)
184  c33_k(iglob) = duz_dzl*b_duz_dzl
185  c35_k(iglob) = 2*(duz_dzl*half*(b_dux_dzl+b_duz_dxl)+&
186  half*(dux_dzl+duz_dxl)*b_duz_dzl)
187  c55_k(iglob) = 4*half*(dux_dzl+duz_dxl)*half*(b_dux_dzl+b_duz_dxl)
188  endif
189  enddo; enddo
190  endif
191  enddo
192 
193  do iglob = 1,nglob
194  rho_k(iglob) = accel_elastic(1,iglob)*b_displ_elastic(1,iglob) + accel_elastic(2,iglob)*b_displ_elastic(2,iglob)
195  enddo
196 
197  else
198  ! updates kernels on GPU
199  call compute_kernels_elastic_cuda(mesh_pointer,deltatf)
200  endif
201 
202  do ispec = 1, nspec
203  if (ispec_is_elastic(ispec)) then
204  ! isotropic kernels
205  do j = 1, ngllz
206  do i = 1, ngllx
207  iglob = ibool(i,j,ispec)
208  if (.not. assign_external_model) then
209  rhol = density(1,kmato(ispec))
210  mul = poroelastcoef(2,1,kmato(ispec))
211  !if (AXISYM) then ! ABAB !! Warning !! This is false for plane strain (look for: bulk modulus plane strain) CHECK Kappa
212  kappal = poroelastcoef(3,1,kmato(ispec)) - four_thirds * mul
213  !else
214  ! kappal = poroelastcoef(3,1,kmato(ispec)) - mul
215  !endif
216  else
217  rhol = rhoext(i,j,ispec)
218  mul = rhoext(i,j,ispec)*vsext(i,j,ispec)*vsext(i,j,ispec)
219  !if (AXISYM) then ! ABAB !! Warning !! This is false for plane strain (look for: bulk modulus plane strain) CHECK Kappa
220  kappal = rhoext(i,j,ispec)*vpext(i,j,ispec)*vpext(i,j,ispec) - four_thirds * mul
221  !else
222  ! kappal = rhoext(i,j,ispec)*vpext(i,j,ispec)*vpext(i,j,ispec) - mul
223  !endif
224  endif
225 
226  ! for parameterization (rho,mu,kappa): "primary" kernels
227  ! density kernel
228  rho_kl(i,j,ispec) = rho_kl(i,j,ispec) - rhol * rho_k(iglob) * deltat
229  ! shear modulus kernel
230  mu_kl(i,j,ispec) = mu_kl(i,j,ispec) - two * mul * mu_k(iglob) * deltat
231  ! bulk modulus kernel
232  kappa_kl(i,j,ispec) = kappa_kl(i,j,ispec) - kappal * kappa_k(iglob) * deltat
233 
234  ! for parameterization (rho,beta,alpha):
235  ! rho prime kernel
236  rhop_kl(i,j,ispec) = rho_kl(i,j,ispec) + kappa_kl(i,j,ispec) + mu_kl(i,j,ispec)
237  ! Vs kernel
238  ! ABAB !! Warning !! This is possibly false for plane strain (look for: bulk modulus plane strain) CHECK Kappa
239  beta_kl(i,j,ispec) = two * (mu_kl(i,j,ispec) - four_thirds * mul/kappal * kappa_kl(i,j,ispec))
240  ! Vp kernel
241  ! ABAB !! Warning !! This is possibly false for plane strain (look for: bulk modulus plane strain) Check Kappa
242  alpha_kl(i,j,ispec) = two * (1._custom_real + four_thirds * mul/kappal) * kappa_kl(i,j,ispec)
243  ! for bulk velocity c parameterization (rho,bulk_c,beta):
244  bulk_c_kl(i,j,ispec) = two * kappa_kl(i,j,ispec)
245  bulk_beta_kl(i,j,ispec) = two * mu_kl(i,j,ispec)
246  enddo
247  enddo
248 
249  ! Voigt kernels, e.g., see Sieminski, 2007a,b
250  if (ispec_is_anisotropic(ispec)) then
251  do j = 1, ngllz
252  do i = 1, ngllx
253  iglob = ibool(i,j,ispec)
254  rho_kl(i,j,ispec) = rho_kl(i,j,ispec) - rho_k(iglob) * deltat
255  c11_kl(i,j,ispec) = c11_kl(i,j,ispec) - c11_k(iglob) * deltat
256  c13_kl(i,j,ispec) = c13_kl(i,j,ispec) - c13_k(iglob) * deltat
257  c15_kl(i,j,ispec) = c15_kl(i,j,ispec) - c15_k(iglob) * deltat
258  c33_kl(i,j,ispec) = c33_kl(i,j,ispec) - c33_k(iglob) * deltat
259  c35_kl(i,j,ispec) = c35_kl(i,j,ispec) - c35_k(iglob) * deltat
260  c55_kl(i,j,ispec) = c55_kl(i,j,ispec) - c55_k(iglob) * deltat
261  rhop_kl(i,j,ispec) = rho_kl(i,j,ispec) + c11_kl(i,j,ispec) + &
262  c13_kl(i,j,ispec) + c15_kl(i,j,ispec) + c33_kl(i,j,ispec) + &
263  c35_kl(i,j,ispec) + c55_kl(i,j,ispec)
264  enddo
265  enddo
266  endif
267 
268  endif ! elastic
269  enddo
270 
271  end subroutine compute_kernels_el
272

277  subroutine compute_kernels_ac()
278 
279 ! acoustic kernel calculations
280 ! see e.g. Tromp et al. (2005)
281 
282  use constants, only: custom_real,ngllx,ngllz,zero,half,two
283 
284  use specfem_par, only: nspec,ispec_is_acoustic,ibool,kappal_ac_global,rhol_ac_global, &
285  poroelastcoef,density,kmato,assign_external_model,rhoext,vpext,deltat, &
286  hprime_xx,hprime_zz,xix,xiz,gammax,gammaz, &
287  potential_acoustic,b_potential_acoustic,b_potential_dot_dot_acoustic, &
288  accel_ac,b_accel_ac,b_displ_ac, &
289  rho_ac_kl,kappa_ac_kl,rhop_ac_kl,alpha_ac_kl,gpu_mode
290 
291  use specfem_par_gpu, only: mesh_pointer,deltatf
292 
293  implicit none
294 
295  !local variables
296  integer :: i,j,k,ispec,iglob
297  real(kind=CUSTOM_REAL) :: tempx1l,tempx2l,b_tempx1l,b_tempx2l,bb_tempx1l,bb_tempx2l
298  double precision :: xixl,xizl,gammaxl,gammazl
299 
300  if (.not. gpu_mode) then
301  ! kernels on CPU
302  do ispec = 1, nspec
303  if (ispec_is_acoustic(ispec)) then
304  do j = 1, ngllz
305  do i = 1, ngllx
306  iglob = ibool(i,j,ispec)
307  if (.not. assign_external_model) then
308  kappal_ac_global(iglob) = poroelastcoef(3,1,kmato(ispec))
309  rhol_ac_global(iglob) = density(1,kmato(ispec))
310  else
311  kappal_ac_global(iglob) = rhoext(i,j,ispec)*vpext(i,j,ispec)*vpext(i,j,ispec)
312  rhol_ac_global(iglob) = rhoext(i,j,ispec)
313  endif
314 
315  ! calcul the displacement by computing the gradient of potential / rho
316  ! and calcul the acceleration by computing the gradient of potential_dot_dot / rho
317  tempx1l = zero
318  tempx2l = zero
319  b_tempx1l = zero
320  b_tempx2l = zero
321  bb_tempx1l = zero
322  bb_tempx2l = zero
323  do k = 1,ngllx
324  ! derivative along x
325  !tempx1l = tempx1l + potential_dot_dot_acoustic(ibool(k,j,ispec))*hprime_xx(i,k)
326  tempx1l = tempx1l + potential_acoustic(ibool(k,j,ispec))*hprime_xx(i,k) !!! YANGL
327  b_tempx1l = b_tempx1l + b_potential_acoustic(ibool(k,j,ispec))*hprime_xx(i,k)
328  bb_tempx1l = bb_tempx1l + b_potential_dot_dot_acoustic(ibool(k,j,ispec))*hprime_xx(i,k)
329  ! derivative along z
330  !tempx2l = tempx2l + potential_dot_dot_acoustic(ibool(i,k,ispec))*hprime_zz(j,k)
331  tempx2l = tempx2l + potential_acoustic(ibool(i,k,ispec))*hprime_zz(j,k) !!! YANGL
332  b_tempx2l = b_tempx2l + b_potential_acoustic(ibool(i,k,ispec))*hprime_zz(j,k)
333  bb_tempx2l = bb_tempx2l + b_potential_dot_dot_acoustic(ibool(i,k,ispec))*hprime_zz(j,k)
334  enddo
335 
336  xixl = xix(i,j,ispec)
337  xizl = xiz(i,j,ispec)
338  gammaxl = gammax(i,j,ispec)
339  gammazl = gammaz(i,j,ispec)
340 
341  ! derivatives of potential
342  accel_ac(1,iglob) = (tempx1l*xixl + tempx2l*gammaxl) / rhol_ac_global(iglob)
343  accel_ac(2,iglob) = (tempx1l*xizl + tempx2l*gammazl) / rhol_ac_global(iglob)
344  b_displ_ac(1,iglob) = (b_tempx1l*xixl + b_tempx2l*gammaxl) / rhol_ac_global(iglob)
345  b_displ_ac(2,iglob) = (b_tempx1l*xizl + b_tempx2l*gammazl) / rhol_ac_global(iglob)
346  b_accel_ac(1,iglob) = (bb_tempx1l*xixl + bb_tempx2l*gammaxl) / rhol_ac_global(iglob)
347  b_accel_ac(2,iglob) = (bb_tempx1l*xizl + bb_tempx2l*gammazl) / rhol_ac_global(iglob)
348  enddo !i = 1, NGLLX
349  enddo !j = 1, NGLLZ
350  endif
351  enddo
352 
353  do ispec = 1,nspec
354  if (ispec_is_acoustic(ispec)) then
355  do j = 1, ngllz
356  do i = 1, ngllx
357  iglob = ibool(i,j,ispec)
358  ! YANGL
359  !!!! old expression (from elastic kernels)
360  !!!rho_ac_kl(i,j,ispec) = rho_ac_kl(i,j,ispec) - rhol_ac_global(iglob) * &
361  !!! dot_product(accel_ac(:,iglob),b_displ_ac(:,iglob)) * deltat
362  !!!kappa_ac_kl(i,j,ispec) = kappa_ac_kl(i,j,ispec) - kappal_ac_global(iglob) * &
363  !!! potential_dot_dot_acoustic(iglob)/kappal_ac_global(iglob) * &
364  !!! b_potential_dot_dot_acoustic(iglob)/kappal_ac_global(iglob)&
365  !!! * deltat
366  !!!! new expression (from PDE-constrained optimization, coupling terms changed as well)
367  rho_ac_kl(i,j,ispec) = rho_ac_kl(i,j,ispec) + rhol_ac_global(iglob) * &
368  dot_product(accel_ac(:,iglob),b_displ_ac(:,iglob)) * deltat
369  kappa_ac_kl(i,j,ispec) = kappa_ac_kl(i,j,ispec) + kappal_ac_global(iglob) * &
370  potential_acoustic(iglob)/kappal_ac_global(iglob) * &
371  b_potential_dot_dot_acoustic(iglob)/kappal_ac_global(iglob) * deltat
372  ! YANGL
373  rhop_ac_kl(i,j,ispec) = rho_ac_kl(i,j,ispec) + kappa_ac_kl(i,j,ispec)
374  alpha_ac_kl(i,j,ispec) = two * kappa_ac_kl(i,j,ispec)
375  enddo
376  enddo
377  endif
378  enddo
379  else
380  ! on GPU
381  call compute_kernels_acoustic_cuda(mesh_pointer,deltatf)
382  endif
383 
384  end subroutine compute_kernels_ac

390  subroutine compute_kernels_po()
391 
392 ! kernel calculations
393 ! see e.g. Morency et al. (2009)
394 
395  use constants, only: custom_real,four_thirds,ngllx,ngllz,two,half
396 
397  use specfem_par, only: nglob,nspec,ispec_is_poroelastic,ibool,deltat, &
398  kmato,permeability, &
399  accels_poroelastic,accelw_poroelastic,velocw_poroelastic, &
400  b_displs_poroelastic,b_displw_poroelastic, &
401  epsilondev_s,b_epsilondev_s, &
402  epsilondev_w,b_epsilondev_w, &
403  rhot_k,rhof_k,sm_k,eta_k,b_k,c_k, &
404  rhot_kl,rhof_kl,sm_kl,eta_kl,b_kl,c_kl,m_kl,m_k, &
405  mufr_kl,mufr_k,rhob_kl,rhofb_kl, &
406  mufrb_kl,phi_kl,rhobb_kl,rhofbb_kl,phib_kl,cpi_kl,cpii_kl,cs_kl,ratio_kl, &
407  gpu_mode
408  implicit none
409 
410  !local variables
411  integer :: i,j,ispec,iglob
412  real(kind=CUSTOM_REAL) :: rholb,dd1
413  real(kind=CUSTOM_REAL) :: ratio
414  real(kind=CUSTOM_REAL) :: dsxx,dsxz,dszz,dszx_xz
415  real(kind=CUSTOM_REAL) :: b_dsxx,b_dsxz,b_dszz,b_dszx_xz
416  real(kind=CUSTOM_REAL) :: dwxx,dwzz,b_dwxx,b_dwzz
417 
418  ! to evaluate cpI, cpII, and cs, and rI (poroelastic medium)
419  double precision :: phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar
420  double precision :: D_biot,H_biot,C_biot,M_biot
421  double precision :: B_biot
422  double precision :: perm_xx
423  double precision :: afactor,bfactor,cfactor
424  double precision :: gamma1,gamma2,gamma3,gamma4
425  double precision :: cpIsquare,cpIIsquare,cssquare
426 
427  integer :: material
428 
429  ! safety check
430  if (gpu_mode) stop 'Error poroelastic kernels not implemented on GPUs yet'
431 
432  ! kernel contributions on global nodes
433  do iglob = 1,nglob
434  rhot_k(iglob) = accels_poroelastic(1,iglob) * b_displs_poroelastic(1,iglob) + &
435  accels_poroelastic(2,iglob) * b_displs_poroelastic(2,iglob)
436 
437  rhof_k(iglob) = accelw_poroelastic(1,iglob) * b_displs_poroelastic(1,iglob) + &
438  accelw_poroelastic(2,iglob) * b_displs_poroelastic(2,iglob) + &
439  accels_poroelastic(1,iglob) * b_displw_poroelastic(1,iglob) + &
440  accels_poroelastic(2,iglob) * b_displw_poroelastic(2,iglob)
441 
442  sm_k(iglob) = accelw_poroelastic(1,iglob) * b_displw_poroelastic(1,iglob) + &
