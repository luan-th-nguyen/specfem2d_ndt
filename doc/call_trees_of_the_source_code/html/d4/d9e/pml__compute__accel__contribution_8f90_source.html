34 ! calculates contribution from each C-PML element to update acceleration to the global mesh
35 
36 ! second-order accurate convolution term calculation from equation (21) of
37 ! Shumin Wang, Robert Lee, and Fernando L. Teixeira,
38 ! Anisotropic-medium PML for vector FETD with modified basis functions,
39 ! IEEE Transactions on Antennas and Propagation, vol. 54, no. 1, (2006)
40 
41 
42  subroutine pml_compute_accel_contribution_acoustic(ispec,nglob, &
43  potential_acoustic,potential_acoustic_old,potential_dot_acoustic, &
44  potential_dot_dot_acoustic_PML,r_xiplus1)
45 
46 ! for acoustic elements
47 
48  use constants, only: custom_real,ngllx,ngllz,cpml_x_only,cpml_z_only,alpha_lddrk,beta_lddrk,c_lddrk, &
49  nglj,two_thirds
50 
51  use specfem_par, only: time_stepping_scheme,i_stage,it,deltat, &
52  assign_external_model,rhoext,vpext,density,poroelastcoef,kmato, &
53  ibool,jacobian, &
54  wxgll,wzgll, &
55  rmemory_potential_acoustic,rmemory_potential_acoustic_lddrk, &
56  axisym,is_on_the_axis,coord,wxglj
57 
58  ! PML arrays
59  use specfem_par, only: nspec_pml,ispec_is_pml,spec_to_pml,region_cpml, &
60  k_x_store,k_z_store,d_x_store,d_z_store,alpha_x_store,alpha_z_store
61 
62  implicit none
63 
64  integer,intent(in) :: ispec
65 
66  integer,intent(in) :: nglob
67  real(kind=CUSTOM_REAL), dimension(nglob),intent(in) :: potential_acoustic,potential_acoustic_old,potential_dot_acoustic
68 
69  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(out) :: potential_dot_dot_acoustic_PML
70  real(kind=CUSTOM_REAL), dimension(NGLJ,NGLLZ),intent(in) :: r_xiplus1
71 
72  ! local parameters
73  integer :: i,j,iglob
74 
75  ! PML
76  integer :: ispec_PML
77  integer :: CPML_region_local
78  integer :: singularity_type
79  double precision :: kappa_x,kappa_z,d_x,d_z,alpha_x,alpha_z,beta_x,beta_z
80  double precision :: time_n,time_nsub1
81  double precision :: A0,A1,A2,A3,A4,bb_1,coef0_1,coef1_1,coef2_1,bb_2,coef0_2,coef1_2,coef2_2
82 
83  ! material properties of the acoustic medium
84  real(kind=CUSTOM_REAL) :: mul_relaxed,lambdal_relaxed,kappal,cpl,rhol
85  real(kind=CUSTOM_REAL) :: fac
86 
87  ! checks if anything to do in this slice
88  if (nspec_pml == 0) return
89  if (.not. ispec_is_pml(ispec)) return
90 
91  ! timing
92  select case (time_stepping_scheme)
93  case (1)
94  ! Newmark
95  time_n = (it-1) * deltat
96  time_nsub1 = (it-2) * deltat
97  case (2)
98  ! LDDRK
99  time_n = (it-1) * deltat + c_lddrk(i_stage) * deltat
100  case default
101  stop 'Sorry, time stepping scheme for PML accel contribution not implemented yet'
102  end select
103 
104  ! local PML element
105  ispec_pml = spec_to_pml(ispec)
106  cpml_region_local = region_cpml(ispec)
107 
108  do j = 1,ngllz
109  do i = 1,ngllx
110 
111  kappa_x = k_x_store(i,j,ispec_pml)
112  kappa_z = k_z_store(i,j,ispec_pml)
113 
114  d_x = d_x_store(i,j,ispec_pml)
115  d_z = d_z_store(i,j,ispec_pml)
116 
117  alpha_x = alpha_x_store(i,j,ispec_pml)
118  alpha_z = alpha_z_store(i,j,ispec_pml)
119 
120  beta_x = alpha_x + d_x / kappa_x
121  beta_z = alpha_z + d_z / kappa_z
122 
123  ! the subroutine of l_parameter_computation is located at the end of compute_forces_viscoelastic.F90
124  call l_parameter_computation(time_n,deltat,kappa_x,beta_x,alpha_x,kappa_z,beta_z,alpha_z, &
125  cpml_region_local,a0,a1,a2,a3,a4,singularity_type, &
126  bb_1,coef0_1,coef1_1,coef2_1,bb_2,coef0_2,coef1_2,coef2_2)
127 
128  iglob = ibool(i,j,ispec)
129 
130  select case (time_stepping_scheme)
131  case (1)
132  ! Newmark
133  rmemory_potential_acoustic(1,i,j,ispec_pml) = coef0_1 * rmemory_potential_acoustic(1,i,j,ispec_pml) + &
134  coef1_1 * potential_acoustic(iglob) + coef2_1 * potential_acoustic_old(iglob)
135 
136  if (singularity_type == 0) then
137  rmemory_potential_acoustic(2,i,j,ispec_pml) = coef0_2 * rmemory_potential_acoustic(2,i,j,ispec_pml) + &
138  coef1_2 * potential_acoustic(iglob) + coef2_2 * potential_acoustic_old(iglob)
139  else
140  rmemory_potential_acoustic(2,i,j,ispec_pml) = coef0_2 * rmemory_potential_acoustic(2,i,j,ispec_pml) + &
141  coef1_2 * time_n * potential_acoustic(iglob) + coef2_2 * time_nsub1 * potential_acoustic_old(iglob)
142  endif
143 
144  case (2)
145  ! LDDRK
146  rmemory_potential_acoustic_lddrk(1,i,j,ispec_pml) = &
147  alpha_lddrk(i_stage) * rmemory_potential_acoustic_lddrk(1,i,j,ispec_pml) + &
148  deltat * (-bb_1 * rmemory_potential_acoustic(1,i,j,ispec_pml) + potential_acoustic(iglob))
149  rmemory_potential_acoustic(1,i,j,ispec_pml) = rmemory_potential_acoustic(1,i,j,ispec_pml) + &
150  beta_lddrk(i_stage) * rmemory_potential_acoustic_lddrk(1,i,j,ispec_pml)
151 
152  if (singularity_type == 0) then
153  rmemory_potential_acoustic_lddrk(2,i,j,ispec_pml) = &
154  alpha_lddrk(i_stage) * rmemory_potential_acoustic_lddrk(2,i,j,ispec_pml) + &
155  deltat * (-bb_2 * rmemory_potential_acoustic(2,i,j,ispec_pml) + potential_acoustic(iglob))
156  rmemory_potential_acoustic(2,i,j,ispec_pml) = rmemory_potential_acoustic(2,i,j,ispec_pml) + &
157  beta_lddrk(i_stage) * rmemory_potential_acoustic_lddrk(2,i,j,ispec_pml)
158  else
159  rmemory_potential_acoustic_lddrk(2,i,j,ispec_pml) = &
160  alpha_lddrk(i_stage) * rmemory_potential_acoustic_lddrk(2,i,j,ispec_pml) + &
161  deltat * (-bb_2 * rmemory_potential_acoustic(2,i,j,ispec_pml) + potential_acoustic(iglob) * time_n)
162  rmemory_potential_acoustic(2,i,j,ispec_pml) = rmemory_potential_acoustic(2,i,j,ispec_pml) + &
163  beta_lddrk(i_stage) * rmemory_potential_acoustic_lddrk(2,i,j,ispec_pml)
164  endif
165  end select
166 
167  ! material properties
168  if (assign_external_model) then
169  rhol = rhoext(i,j,ispec)
170  cpl = vpext(i,j,ispec)
171  !assuming that in fluid(acoustic) part input cpl is defined by sqrt(kappal/rhol), &
172  !which is not the same as in cpl input in elastic part
173  kappal = rhol * cpl * cpl ! CHECK Kappa : it is ok here because we are in acoustic elements
174  else
175  rhol = density(1,kmato(ispec))
176  lambdal_relaxed = poroelastcoef(1,1,kmato(ispec))
177  mul_relaxed = poroelastcoef(2,1,kmato(ispec))
178  if (axisym) then ! CHECK kappa
179  kappal = lambdal_relaxed + two_thirds * mul_relaxed
180  else
181  kappal = lambdal_relaxed + mul_relaxed
182  endif
183  endif
184 
185  fac = jacobian(i,j,ispec) / kappal
186 
187  if (axisym) then
188  if (is_on_the_axis(ispec)) then
189  potential_dot_dot_acoustic_pml(i,j)= wxglj(i) * wzgll(j) * fac * r_xiplus1(i,j) * &
190  (a1 * potential_dot_acoustic(iglob) + a2 * potential_acoustic(iglob) + &
191  a3 * rmemory_potential_acoustic(1,i,j,ispec_pml) + &
192  a4 * rmemory_potential_acoustic(2,i,j,ispec_pml))
193  else
194  potential_dot_dot_acoustic_pml(i,j)= wxgll(i) * wzgll(j) * fac * coord(1,ibool(i,j,ispec)) * &
195  (a1 * potential_dot_acoustic(iglob) + a2 * potential_acoustic(iglob) + &
196  a3 * rmemory_potential_acoustic(1,i,j,ispec_pml) + &
197  a4 * rmemory_potential_acoustic(2,i,j,ispec_pml))
198  endif
199  else
200  potential_dot_dot_acoustic_pml(i,j)= wxgll(i) * wzgll(j) * fac * &
201  (a1 * potential_dot_acoustic(iglob) + a2 * potential_acoustic(iglob) + &
202  a3 * rmemory_potential_acoustic(1,i,j,ispec_pml) + a4 * rmemory_potential_acoustic(2,i,j,ispec_pml))
203  endif
204  enddo
205  enddo
206 
207  end subroutine pml_compute_accel_contribution_acoustic

213 
214  subroutine pml_compute_accel_contribution_elastic(ispec,nglob, &
215  dummy_loc,displ_elastic_old,veloc_elastic, &
216  accel_elastic_PML,r_xiplus1)
217 ! for elastic elements
218 
219  use constants, only: custom_real,ndim,ngllx,ngllz,cpml_x_only,cpml_z_only,alpha_lddrk,beta_lddrk,c_lddrk, &
220  nglj,two_thirds
221 
222  use specfem_par, only: time_stepping_scheme,i_stage,it,deltat, &
223  assign_external_model,rhoext,density,kmato, &
224  ibool,jacobian, &
225  wxgll,wzgll, &
226  rmemory_displ_elastic,rmemory_displ_elastic_lddrk, &
227  axisym,is_on_the_axis,coord,wxglj
228 
229  ! PML arrays
230  use specfem_par, only: nspec_pml,ispec_is_pml,spec_to_pml,region_cpml, &
231  k_x_store,k_z_store,d_x_store,d_z_store,alpha_x_store,alpha_z_store
232 
233  implicit none
234 
235  integer,intent(in) :: ispec,nglob
236 
237  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ),intent(in) :: dummy_loc
238  real(kind=CUSTOM_REAL), dimension(NDIM,nglob),intent(in) :: veloc_elastic,displ_elastic_old
239 
240  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ),intent(out) :: accel_elastic_PML
241 
242  real(kind=CUSTOM_REAL), dimension(NGLJ,NGLLZ),intent(in) :: r_xiplus1
243 
244  ! local parameters
245  integer :: i,j,iglob
246 
247  ! PML
248  integer :: ispec_PML
249  integer :: CPML_region_local
250  integer :: singularity_type
251  double precision :: kappa_x,kappa_z,d_x,d_z,alpha_x,alpha_z,beta_x,beta_z
252  double precision :: time_n,time_nsub1
253  double precision :: A0,A1,A2,A3,A4,bb_1,coef0_1,coef1_1,coef2_1,bb_2,coef0_2,coef1_2,coef2_2
254 
255  ! material properties of the acoustic medium
256  real(kind=CUSTOM_REAL) :: rhol
257 
258  ! checks if anything to do in this slice
259  if (nspec_pml == 0) return
260  if (.not. ispec_is_pml(ispec)) return
261 
262  ! timing
263  select case (time_stepping_scheme)
264  case (1)
265  ! Newmark
266  time_n = (it-1) * deltat
267  time_nsub1 = (it-2) * deltat
268  case (2)
269  ! LDDRK
270  time_n = (it-1) * deltat + c_lddrk(i_stage) * deltat
271  case default
272  stop 'Sorry, time stepping scheme for PML accel contribution not implemented yet'
273  end select
274 
275  ! initializes
276  accel_elastic_pml(:,:,:) = 0._custom_real
277 
278  ! local PML element
279  ispec_pml = spec_to_pml(ispec)
280  cpml_region_local = region_cpml(ispec)
281 
282  do j = 1,ngllz
283  do i = 1,ngllx
284 
285  iglob = ibool(i,j,ispec)
286 
287  kappa_x = k_x_store(i,j,ispec_pml)
288  kappa_z = k_z_store(i,j,ispec_pml)
289 
290  d_x = d_x_store(i,j,ispec_pml)
291  d_z = d_z_store(i,j,ispec_pml)
292 
293  alpha_x = alpha_x_store(i,j,ispec_pml)
294  alpha_z = alpha_z_store(i,j,ispec_pml)
295 
296  beta_x = alpha_x + d_x / kappa_x
297  beta_z = alpha_z + d_z / kappa_z
298 
299  call l_parameter_computation(time_n,deltat,kappa_x,beta_x,alpha_x,kappa_z,beta_z,alpha_z, &
300  cpml_region_local,a0,a1,a2,a3,a4,singularity_type, &
301  bb_1,coef0_1,coef1_1,coef2_1,bb_2,coef0_2,coef1_2,coef2_2)
302 
303  select case (time_stepping_scheme)
304  case (1)
305  ! Newmark
306  rmemory_displ_elastic(1,1,i,j,ispec_pml) = coef0_1 * rmemory_displ_elastic(1,1,i,j,ispec_pml) + &
307  coef1_1 * dummy_loc(1,i,j) + coef2_1 * displ_elastic_old(1,iglob)
308  rmemory_displ_elastic(1,2,i,j,ispec_pml) = coef0_1 * rmemory_displ_elastic(1,2,i,j,ispec_pml) + &
309  coef1_1 * dummy_loc(2,i,j) + coef2_1 * displ_elastic_old(2,iglob)
310 
311  if (singularity_type == 0) then
312  rmemory_displ_elastic(2,1,i,j,ispec_pml) = coef0_2 * rmemory_displ_elastic(2,1,i,j,ispec_pml) + &
313  coef1_2 * dummy_loc(1,i,j) + coef2_2 * displ_elastic_old(1,iglob)
314  rmemory_displ_elastic(2,2,i,j,ispec_pml) = coef0_2 * rmemory_displ_elastic(2,2,i,j,ispec_pml) + &
315  coef1_2 * dummy_loc(2,i,j) + coef2_2 * displ_elastic_old(2,iglob)
316  else
317  rmemory_displ_elastic(2,1,i,j,ispec_pml) = coef0_2 * rmemory_displ_elastic(2,1,i,j,ispec_pml) + &
318  coef1_2 * time_n * dummy_loc(1,i,j) + &
319  coef2_2 * time_nsub1 * displ_elastic_old(1,iglob)
320  rmemory_displ_elastic(2,2,i,j,ispec_pml) = coef0_2 * rmemory_displ_elastic(2,2,i,j,ispec_pml) + &
321  coef1_2 * time_n * dummy_loc(2,i,j) + &
322  coef2_2 * time_nsub1 * displ_elastic_old(2,iglob)
323  endif
324 
325  case (2)
326  ! LDDRK
327  rmemory_displ_elastic_lddrk(1,1,i,j,ispec_pml) = &
328  alpha_lddrk(i_stage) * rmemory_displ_elastic_lddrk(1,1,i,j,ispec_pml) + &
329  deltat * (-bb_1 * rmemory_displ_elastic(1,1,i,j,ispec_pml) + dummy_loc(1,i,j))
330  rmemory_displ_elastic(1,1,i,j,ispec_pml) = rmemory_displ_elastic(1,1,i,j,ispec_pml) + &
331  beta_lddrk(i_stage) * rmemory_displ_elastic_lddrk(1,1,i,j,ispec_pml)
332 
333  rmemory_displ_elastic_lddrk(1,2,i,j,ispec_pml) = &
334  alpha_lddrk(i_stage) * rmemory_displ_elastic_lddrk(1,2,i,j,ispec_pml) + &
335  deltat * (-bb_1 * rmemory_displ_elastic(1,2,i,j,ispec_pml) + dummy_loc(2,i,j))
336  rmemory_displ_elastic(1,2,i,j,ispec_pml) = rmemory_displ_elastic(1,2,i,j,ispec_pml) + &
337  beta_lddrk(i_stage) * rmemory_displ_elastic_lddrk(1,2,i,j,ispec_pml)
338 
339  if (singularity_type == 0) then
340  rmemory_displ_elastic_lddrk(2,1,i,j,ispec_pml) = &
341  alpha_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,1,i,j,ispec_pml) + &
342  deltat * (-bb_2 * rmemory_displ_elastic(2,1,i,j,ispec_pml) + dummy_loc(1,i,j))
343  rmemory_displ_elastic(2,1,i,j,ispec_pml) = rmemory_displ_elastic(2,1,i,j,ispec_pml) + &
344  beta_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,1,i,j,ispec_pml)
345 
346  rmemory_displ_elastic_lddrk(2,2,i,j,ispec_pml) = &
347  alpha_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,2,i,j,ispec_pml) + &
348  deltat * (-bb_2 * rmemory_displ_elastic(2,2,i,j,ispec_pml) + dummy_loc(2,i,j))
349  rmemory_displ_elastic(2,2,i,j,ispec_pml) = rmemory_displ_elastic(2,2,i,j,ispec_pml) + &
350  beta_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,2,i,j,ispec_pml)
351  else
352  rmemory_displ_elastic_lddrk(2,1,i,j,ispec_pml) = &
353  alpha_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,1,i,j,ispec_pml) + &
354  deltat * (-bb_2 * rmemory_displ_elastic(2,1,i,j,ispec_pml) + dummy_loc(1,i,j) * time_n)
355  rmemory_displ_elastic(2,1,i,j,ispec_pml) = rmemory_displ_elastic(2,1,i,j,ispec_pml) + &
356  beta_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,1,i,j,ispec_pml)
357 
358  rmemory_displ_elastic_lddrk(2,2,i,j,ispec_pml) = &
359  alpha_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,2,i,j,ispec_pml) + &
360  deltat * (-bb_2 * rmemory_displ_elastic(2,2,i,j,ispec_pml) + dummy_loc(2,i,j) * time_n)
361  rmemory_displ_elastic(2,2,i,j,ispec_pml) = rmemory_displ_elastic(2,2,i,j,ispec_pml) + &
362  beta_lddrk(i_stage) * rmemory_displ_elastic_lddrk(2,2,i,j,ispec_pml)
363  endif
364 
365  case default
366  stop 'Time stepping scheme not implemented yet for PML accel contribution'
367  end select
368 
369  if (assign_external_model) then
370  rhol = rhoext(i,j,ispec)
371  else
372  rhol = density(1,kmato(ispec))
373  endif
374 
375  if (axisym) then
376  if (is_on_the_axis(ispec)) then
377  accel_elastic_pml(1,i,j) = wxglj(i)*wzgll(j)*rhol*jacobian(i,j,ispec)*r_xiplus1(i,j) * &
378  ( a1 * veloc_elastic(1,iglob) + a2 * dummy_loc(1,i,j) + &
379  a3 * rmemory_displ_elastic(1,1,i,j,ispec_pml) + a4 * rmemory_displ_elastic(2,1,i,j,ispec_pml))
380  accel_elastic_pml(2,i,j) = wxglj(i)*wzgll(j)*rhol*jacobian(i,j,ispec)*r_xiplus1(i,j) * &
381  ( a1 * veloc_elastic(2,iglob) + a2 * dummy_loc(2,i,j) + &
382  a3 * rmemory_displ_elastic(1,2,i,j,ispec_pml) + a4 * rmemory_displ_elastic(2,2,i,j,ispec_pml))
383  else
384  accel_elastic_pml(1,i,j) = wxgll(i)*wzgll(j)*rhol*jacobian(i,j,ispec)*coord(1,ibool(i,j,ispec))* &
385  ( a1 * veloc_elastic(1,iglob) + a2 * dummy_loc(1,i,j) + &
386  a3 * rmemory_displ_elastic(1,1,i,j,ispec_pml) + a4 * rmemory_displ_elastic(2,1,i,j,ispec_pml))
387  accel_elastic_pml(2,i,j) = wxgll(i)*wzgll(j)*rhol*jacobian(i,j,ispec)*coord(1,ibool(i,j,ispec))* &
388  ( a1 * veloc_elastic(2,iglob) + a2 * dummy_loc(2,i,j) + &
389  a3 * rmemory_displ_elastic(1,2,i,j,ispec_pml) + a4 * rmemory_displ_elastic(2,2,i,j,ispec_pml))
390  endif
391  else
392  ! default
393  accel_elastic_pml(1,i,j) = wxgll(i)*wzgll(j)*rhol*jacobian(i,j,ispec) * &
394  ( a1 * veloc_elastic(1,iglob) + a2 * dummy_loc(1,i,j) + &
395  a3 * rmemory_displ_elastic(1,1,i,j,ispec_pml) + a4 * rmemory_displ_elastic(2,1,i,j,ispec_pml))
