34  subroutine compute_forces_acoustic_main()
35 
36  use constants, only: source_is_moving,use_enforce_fields,alpha_lddrk,beta_lddrk
37  use specfem_par
38 
39  implicit none
40 
41  ! local parameters
42  integer :: i
43  ! non-blocking MPI
44  ! iphase: iphase = 1 is for computing outer elements (on MPI interface),
45  ! iphase = 2 is for computing inner elements
46  integer :: iphase
47 
48  ! checks if anything to do in this slice
49  if ((.not. any_acoustic) .and. (.not. source_is_moving)) return
50 
51  ! free surface for an acoustic medium
52  call enforce_acoustic_free_surface(potential_dot_dot_acoustic,potential_dot_acoustic,potential_acoustic)
53 
54 !! DK DK QUENTIN visco begin
55  ! viscoacoustic attenuation for fluid media
56  if (attenuation_viscoacoustic) call compute_attenuation_acoustic(potential_acoustic,potential_acoustic_old, &
57  ispec_is_acoustic,pml_boundary_conditions,e1_fluid)
58 !! DK DK QUENTIN visco end
59 
60  ! distinguishes two runs: for elements on MPI interfaces, and elements within the partitions
61  do iphase = 1,2
62 
63  ! main solver for the acoustic elements
64  call compute_forces_acoustic(potential_dot_dot_acoustic,potential_dot_acoustic,potential_acoustic, &
65  pml_boundary_conditions,potential_acoustic_old,iphase,e1_fluid)
66 
67  ! PML boundary conditions enforces zero potentials on boundary
68  if (pml_boundary_conditions) then
69  call pml_boundary_acoustic(potential_dot_dot_acoustic,potential_dot_acoustic, &
70  potential_acoustic,potential_acoustic_old)
71  endif
72 
73  ! computes additional contributions
74  if (iphase == 1) then
75  ! Stacey boundary conditions
76  if (stacey_absorbing_conditions) then
77  call compute_stacey_acoustic(potential_dot_dot_acoustic,potential_dot_acoustic)
78  endif
79 
80  ! add acoustic forcing at a rigid boundary
81  if (acoustic_forcing .and. (.not. use_enforce_fields)) then
82  call add_acoustic_forcing_at_rigid_boundary(potential_dot_dot_acoustic)
83  endif
84 
85  ! applies to coupling in case of MPI partitioning:
86  ! coupling interfaces might not be properly detected if one material domain is only in an another slice.
87  ! in such a case, the common nodes would not be detected as belonging to a coupling interface.
88  ! something to do in future...
89 
90  ! add coupling with the elastic side
91  if (coupled_acoustic_elastic) then
92  if (simulation_type == 1) then
93  call compute_coupling_acoustic_el(displ_elastic,displ_elastic_old,potential_dot_dot_acoustic)
94  endif
95 
96  ! coupling for adjoint wavefields
97  if (simulation_type == 3) then
98  ! note: handles adjoint runs coupling between adjoint potential and adjoint elastic wavefield
99  ! adjoint definition: \partial_t^2 \bfs^\dagger = - \frac{1}{\rho} \bfnabla \phi^\dagger
100  !
101  ! coupling with adjoint wavefields
102  call compute_coupling_acoustic_el(accel_elastic_adj_coupling,displ_elastic_old,potential_dot_dot_acoustic)
103  endif
104  endif
105 
106  ! add coupling with the poroelastic side
107  if (coupled_acoustic_poro) then
108  call compute_coupling_acoustic_po()
109  endif
110 
111  ! add force source
112  if (.not. initialfield) then
113  if (simulation_type == 1) then
114  if (source_is_moving) then
115  call compute_add_sources_acoustic_moving_source(potential_dot_dot_acoustic,it,i_stage)
116  else
117  call compute_add_sources_acoustic(potential_dot_dot_acoustic,it,i_stage)
118  endif
119  else if (simulation_type == 3) then
120  ! adjoint sources
121  call compute_add_sources_acoustic_adjoint()
122  endif
123  endif
124  endif ! iphase == 1
125 
126 #ifdef USE_MPI
127  ! assembling potential_dot_dot or b_potential_dot_dot for acoustic elements
128  if (nproc > 1 .and. ninterface_acoustic > 0) then
129  if (iphase == 1) then
130  call assemble_mpi_scalar_ac_s(potential_dot_dot_acoustic)
131  else
132  call assemble_mpi_scalar_ac_w(potential_dot_dot_acoustic)
133  endif
134 
135  if (time_stepping_scheme == 2) then
136  ! LDDRK
137  if (i_stage == 1 .and. it == 1 .and. iphase == 2 .and. (.not. initialfield)) then
138  potential_dot_acoustic_temp(:) = potential_dot_acoustic(:)
139  call assemble_mpi_scalar_ac_blocking(potential_dot_acoustic)
140  endif
141  endif
142  endif
143 #endif
144 
145  enddo ! iphase
146 
147  ! PML saves interface values
148  if (pml_boundary_conditions) then
149  if (nglob_interface > 0) then
150  if (save_forward .and. simulation_type == 1) then
151  do i = 1, nglob_interface
152  write(72) potential_dot_dot_acoustic(point_interface(i)), potential_dot_acoustic(point_interface(i)), &
153  potential_acoustic(point_interface(i))
154  enddo
155  endif
156  endif
157  endif
158 
159  ! multiply by the inverse of the mass matrix
160  !! DK DK this should be vectorized
161  potential_dot_dot_acoustic(:) = potential_dot_dot_acoustic(:) * rmass_inverse_acoustic(:)
162 
163  ! update velocity
164  select case (time_stepping_scheme)
165  case (1)
166  ! Newmark scheme
167  call update_veloc_acoustic_newmark()
168  case (2)
169  ! LDDRK scheme
170  call update_veloc_acoustic_lddrk()
171  case (3)
172  ! RK scheme
173  call update_veloc_acoustic_rk()
174  case default
175  stop 'Invalid time stepping scheme for compute forces routine!'
176  end select
177 
178  ! free surface for an acoustic medium
179  call enforce_acoustic_free_surface(potential_dot_dot_acoustic,potential_dot_acoustic,potential_acoustic)
180 
181  end subroutine compute_forces_acoustic_main
182 
183 !
184 !-------------------------------------------------------------------------------------
185 !
186 
187  subroutine compute_forces_acoustic_main_backward()
188 
189  use specfem_par
190 
191  implicit none
192 
193  ! local parameters
194  integer :: it_temp,istage_temp
195  ! non-blocking MPI
196  ! iphase: iphase = 1 is for computing outer elements (on MPI interface),
197  ! iphase = 2 is for computing inner elements
198  integer :: iphase
199 
200  ! checks
201  if (simulation_type /= 3 ) return
202 
203  ! checks if anything to do in this slice
204  if (.not. any_acoustic) return
205 
206  ! timing
207  if (undo_attenuation) then
208  ! time increment
209  ! example: NSTEP = 800, NT_DUMP_ATTENUATION = 500 -> 1. subset: it_temp = (2-1)*500 + 1 = 501,502,..,800
210  ! -> 2. subset: it_temp = (2-2)*500 + 1 = 1,2,..,500
211  it_temp = (nsubset_iterations - iteration_on_subset)*nt_dump_attenuation + it_of_this_subset
212  ! time scheme
213  istage_temp = i_stage
214  else
215  ! time increment
216  it_temp = nstep - it + 1
217  ! time scheme
218  istage_temp = stage_time_scheme - i_stage + 1
219  endif
220 
221  ! PML restores interface values
222  if (pml_boundary_conditions) then
223  call rebuild_value_on_pml_interface_acoustic(it_temp,b_potential_acoustic,b_potential_dot_acoustic)
224  endif
225 
226  ! free surface for an acoustic medium
227  call enforce_acoustic_free_surface(b_potential_dot_dot_acoustic,b_potential_dot_acoustic,b_potential_acoustic)
228 
229  ! distinguishes two runs: for elements on MPI interfaces, and elements within the partitions
230  do iphase = 1,2
231 
232  ! main solver for the acoustic elements
233  if (undo_attenuation) then
234  call compute_forces_acoustic(b_potential_dot_dot_acoustic,b_potential_dot_acoustic,b_potential_acoustic, &
235  .false.,b_potential_acoustic_old,iphase,e1_fluid)
236  else
237  call compute_forces_acoustic_backward(b_potential_dot_dot_acoustic,b_potential_acoustic,iphase)
238  endif
239 
240  ! PML boundary conditions
241  if (pml_boundary_conditions) then
242  ! enforces zero potentials on boundary
243  call pml_boundary_acoustic(b_potential_dot_dot_acoustic,b_potential_dot_acoustic, &
244  b_potential_acoustic,b_potential_acoustic_old)
245 
246  ! restores potentials on interface
247  call rebuild_value_on_pml_interface_acoustic(it_temp,b_potential_acoustic,b_potential_dot_acoustic)
248  endif
249 
250  ! computes additional contributions
251  if (iphase == 1) then
252  ! Stacey boundary conditions
253  if (stacey_absorbing_conditions) then
254  if (undo_attenuation) then
255  call compute_stacey_acoustic(b_potential_dot_dot_acoustic,b_potential_dot_acoustic)
256  else
257  call compute_stacey_acoustic_backward(b_potential_dot_dot_acoustic)
258  endif
259  endif
260 
261  ! add acoustic forcing at a rigid boundary
262  if (acoustic_forcing) then
263  call add_acoustic_forcing_at_rigid_boundary(b_potential_dot_dot_acoustic)
264  endif
265 
266  ! add coupling with the elastic side
267  if (coupled_acoustic_elastic) then
268  call compute_coupling_acoustic_el_backward(b_displ_elastic,b_potential_dot_dot_acoustic)
269  endif
270 
271  ! add coupling with the poroelastic side
272  if (coupled_acoustic_poro) then
273  call compute_coupling_acoustic_po_backward()
274  endif
275 
276  ! PML restores interface values
277  if (pml_boundary_conditions) then
278  call rebuild_value_on_pml_interface_acoustic(it_temp,b_potential_acoustic,b_potential_dot_acoustic)
279  endif
280 
281  ! add force source
282  if (.not. initialfield) then
283  ! backward wavefield
284  call compute_add_sources_acoustic(b_potential_dot_dot_acoustic,it_temp,istage_temp)
285  endif
286 
287  endif ! iphase == 1
288 
289 #ifdef USE_MPI
290  ! assembling potential_dot_dot or b_potential_dot_dot for acoustic elements
291  if (nproc > 1 .and. ninterface_acoustic > 0) then
292  if (iphase == 1) then
293  call assemble_mpi_scalar_ac_s(b_potential_dot_dot_acoustic)
294  else
295  call assemble_mpi_scalar_ac_w(b_potential_dot_dot_acoustic)
296  endif
297  endif
298 #endif
299 
300  enddo ! iphase
301 
302  ! PML restores interface values
303  if (pml_boundary_conditions) then
304  call rebuild_value_on_pml_interface_acoustic_accel(it_temp,b_potential_dot_dot_acoustic)
305  endif
306 
307  ! multiply by the inverse of the mass matrix
308  b_potential_dot_dot_acoustic(:) = b_potential_dot_dot_acoustic(:) * rmass_inverse_acoustic(:)
309 
310  ! update velocity
311  select case (time_stepping_scheme)
312  case (1)
313  ! Newmark
314  call update_veloc_acoustic_newmark_backward()
315  case default
316  stop 'Sorry, time stepping scheme not implemented yet for backward computations'
317  end select
318 
319  ! free surface for an acoustic medium
320  call enforce_acoustic_free_surface(b_potential_dot_dot_acoustic,b_potential_dot_acoustic,b_potential_acoustic)
321 
322  end subroutine compute_forces_acoustic_main_backward
