55  subroutine update_veloc_elastic_lddrk()
56 
57 ! updates acceleration,velocity and displacement in elastic regions (crust/mantle,inner core)
58 
59  use constants, only: alpha_lddrk,beta_lddrk
60  use specfem_par
61 
62  implicit none
63 
64  ! local parameters
65 
66  !! DK DK this should be vectorized
67  veloc_elastic_lddrk(:,:) = alpha_lddrk(i_stage) * veloc_elastic_lddrk(:,:) + deltat * accel_elastic(:,:)
68  displ_elastic_lddrk(:,:) = alpha_lddrk(i_stage) * displ_elastic_lddrk(:,:) + deltat * veloc_elastic(:,:)
69 
70  if (i_stage == 1 .and. it == 1 .and. (.not. initialfield)) then
71  veloc_elastic_lddrk_temp(:,:) = veloc_elastic_lddrk_temp(:,:) + beta_lddrk(i_stage) * veloc_elastic_lddrk(:,:)
72  veloc_elastic(:,:) = veloc_elastic_lddrk_temp(:,:)
73  else
74  veloc_elastic(:,:) = veloc_elastic(:,:) + beta_lddrk(i_stage) * veloc_elastic_lddrk(:,:)
75  endif
76 
77  displ_elastic(:,:) = displ_elastic(:,:) + beta_lddrk(i_stage) * displ_elastic_lddrk(:,:)
78 
79  end subroutine update_veloc_elastic_lddrk

88  subroutine update_veloc_acoustic_lddrk()
89 
90 ! updates velocity potential (corrector)
91 
92  use constants, only: alpha_lddrk,beta_lddrk
93  use specfem_par
94 
95  implicit none
96 
97  !! DK DK this should be vectorized
98  potential_dot_acoustic_lddrk(:) = alpha_lddrk(i_stage) * potential_dot_acoustic_lddrk(:) + &
99  deltat * potential_dot_dot_acoustic(:)
100  potential_acoustic_lddrk(:) = alpha_lddrk(i_stage) * potential_acoustic_lddrk(:) + &
101  deltat * potential_dot_acoustic(:)
102 
103 !! DK DK QUENTIN visco begin
104  e1_fluid_lddrk(:,:,:,:) = alpha_lddrk(i_stage) * e1_fluid_lddrk(:,:,:,:) + deltat * dot_e1_fluid(:,:,:,:)
105 !! DK DK QUENTIN visco end
106 
107  if (i_stage == 1 .and. it == 1 .and. (.not. initialfield)) then
108  !! DK DK this should be vectorized
109  potential_dot_acoustic_temp(:) = potential_dot_acoustic_temp(:) + &
110  beta_lddrk(i_stage) * potential_dot_acoustic_lddrk(:)
111  potential_dot_acoustic(:) = potential_dot_acoustic_temp(:)
112 
113 !! DK DK QUENTIN visco begin
114  e1_fluid(:,:,:,:) = e1_fluid(:,:,:,:) + beta_lddrk(i_stage) * e1_fluid_lddrk(:,:,:,:)
115 !! DK DK QUENTIN visco end
116  else
117  potential_dot_acoustic(:) = potential_dot_acoustic(:) + beta_lddrk(i_stage) * potential_dot_acoustic_lddrk(:)
118 
119 !! DK DK QUENTIN visco begin
120  e1_fluid(:,:,:,:) = e1_fluid(:,:,:,:) + beta_lddrk(i_stage) * e1_fluid_lddrk(:,:,:,:)
121 !! DK DK QUENTIN visco end
122  endif
123 
124  !! DK DK this should be vectorized
125  potential_acoustic(:) = potential_acoustic(:) + beta_lddrk(i_stage) * potential_acoustic_lddrk(:)
126 
127  end subroutine update_veloc_acoustic_lddrk
