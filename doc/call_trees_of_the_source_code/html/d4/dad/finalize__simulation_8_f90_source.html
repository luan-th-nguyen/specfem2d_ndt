34  subroutine finalize_simulation()

36 #ifdef USE_MPI

46  integer :: i,ispec,j,iglob
47  integer :: ier
48  real(kind=4),dimension(:,:,:),allocatable :: rho_save, vp_save, vs_save, kappa_save, x_save, z_save
49  double precision :: mul_unrelaxed_elastic,lambdal_unrelaxed_elastic
50  character(len=MAX_STRING_LEN) :: inputname,outputname,outputname2
51 
52  ! writes out kernel files
53  if (simulation_type == 3) then
54  call save_adjoint_kernels()
55  endif
56 
57  ! saves model files
58  if (trim(save_model) /= 'default' .and. trim(save_model) /= '.false.') then
59  ! allocates temporary arrays for file storage
60  allocate(rho_save(ngllx,ngllz,nspec))
61  allocate(vp_save(ngllx,ngllz,nspec))
62  allocate(vs_save(ngllx,ngllz,nspec))
63  allocate(kappa_save(ngllx,ngllz,nspec))
64  allocate(x_save(ngllx,ngllz,nspec))
65  allocate(z_save(ngllx,ngllz,nspec))
66  do ispec= 1,nspec
67  do j = 1,ngllz
68  do i = 1,ngllx
69  rho_save(i,j,ispec) = density(1,kmato(ispec))
70  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec))
71  mul_unrelaxed_elastic = poroelastcoef(2,1,kmato(ispec))
72  if (axisym) then ! CHECK kappa
73  kappa_save(i,j,ispec) = lambdal_unrelaxed_elastic + two_thirds * mul_unrelaxed_elastic
74  vp_save(i,j,ispec) = sqrt((kappa_save(i,j,ispec) + four_thirds *mul_unrelaxed_elastic)/density(1,kmato(ispec)))
75  else
76  kappa_save(i,j,ispec) = lambdal_unrelaxed_elastic + mul_unrelaxed_elastic
77  vp_save(i,j,ispec) = sqrt((kappa_save(i,j,ispec) + mul_unrelaxed_elastic)/density(1,kmato(ispec)))
78  endif
79 
80  vs_save(i,j,ispec) = sqrt(mul_unrelaxed_elastic/density(1,kmato(ispec)))
81 
82  iglob = ibool(i,j,ispec)
83  x_save(i,j,ispec) = coord(1,iglob)
84  z_save(i,j,ispec) = coord(2,iglob)
85  enddo
86  enddo
87  enddo
88 
89  ! SMNSR For compatibility with NUMBER_OF_SIMULTANEOUS_RUNS we have to change the lines trim(IN_DATA_FILES)//'proc'
90

92  if (trim(save_model) == 'legacy') then

96  if (ier /= 0) call exit_mpi(myrank,'Error opening model file proc**_model_velocity.dat_input')

98  do j = 1,ngllz
99  do i = 1,ngllx
100  iglob = ibool(i,j,ispec)
101  write(1001,'(I10,5e15.5e4)') iglob, x_save(i,j,ispec),z_save(i,j,ispec),rho_save(i,j,ispec), &
102  vp_save(i,j,ispec),vs_save(i,j,ispec)
103  enddo
104  enddo
105  enddo
106  close(1001)
107 
108  else if (trim(save_model) == 'ascii') then
109  ! ascii format
110  write(inputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_rho_vp_vs.dat'

112  if (ier /= 0) call exit_mpi(myrank,'Error opening model file proc**_rho_vp_vs.dat')

114  do j = 1,ngllz
115  do i = 1,ngllx
116  iglob = ibool(i,j,ispec)
117  write(1001,'(5e15.5e4)') x_save(i,j,ispec),z_save(i,j,ispec),rho_save(i,j,ispec),vp_save(i,j,ispec),vs_save(i,j,ispec)
118  enddo
119  enddo
120  enddo
121  close(1001)
122

134  close(172)

136  open(unit=172,file=outputname,status='unknown',form='unformatted',iostat=ier)

139  close(172)

144  close(172)
145  write(outputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_z.bin'
146  open(unit=172,file=outputname,status='unknown',form='unformatted',iostat=ier)
147  if (ier /= 0) call exit_mpi(myrank,'Error opening model file proc**_z.bin')
148  write(172) z_save
149  close(172)
150

152  open(unit=172,file=outputname,status='unknown',form='unformatted',iostat=ier)
153  if (ier /= 0) call exit_mpi(myrank,'Error opening model file proc**_jacobian.bin')
154  write(172) jacobian
155  close(172)
156 
157  else
158  stop 'Save Model not implemented for external and tomo'
159  endif !Type of model
160  endif !save model

263  endif
264 
265  endif !any elastic
