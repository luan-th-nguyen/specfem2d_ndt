37  subroutine build_is_on_the_axis()
38 
39 ! This subroutine set the vector of logicals is_on_the_axis. is_on_the_axis(ispec)
40 ! should be .true. if the element number ispec is located near the axis.
41 ! The vector build_is_on_the_axis has been initialized to .false. before.
42 
43  use specfem_par, only: nspec, ispec_of_axial_elements, is_on_the_axis
44 
45  implicit none
46 
47  ! local parameters
48  integer :: ispec
49 
50  is_on_the_axis(:) = .false.
51 
52  do ispec = 1,nspec
53  ! checks if ispec is contained in list ispec_of_axial_element
54  if (any(ispec_of_axial_elements(:) == ispec)) is_on_the_axis(ispec) = .true.
55  enddo
56 
57  end subroutine build_is_on_the_axis

64  subroutine check_compatibility_axisym()
65 
66 ! This subroutine check the parameters of the run and stop the program (or display a warning) if the configuration is not
67 ! compatible with axisymetric simulations.
68 ! _Not compatible fundamentaly :
69 ! If it is on the axis the source need to be in the x direction (test if sourceangle =0 /180+-epsilon)
70 ! Warning if the source is not on the axis (circular source)
71 ! _Not implemented yet (or not tested):
72 ! poroelasticity, anisotropy, Stacey absorbing boundaries, time stepping scheme /= 1, PML rotated, adjoint
73 ! simulations, periodic conditions, noise tomographies
74 
75  use constants, only: pi,two,tinyval
76 
77  use specfem_par, only: any_poroelastic, rotate_pml_activate, &
78  stacey_absorbing_conditions, simulation_type, save_forward, &
79  nsources, source_type, ispec_selected_source, add_periodic_conditions, &
80  anglesource, is_on_the_axis, ispec_is_elastic, islice_selected_source,myrank
81 
82  use specfem_par_noise, only: noise_tomography
83 
84  implicit none
85 
86  ! Local parameters
87  integer :: isource
88 
89  if (any_poroelastic) &
90  call exit_mpi(myrank,'Poroelasticity is not implemented for axisymmetric simulations')
91  if (rotate_pml_activate) &
92  call exit_mpi(myrank,'ROTATE_PML_ACTIVATE is not implemented for axisymmetric simulations')
93  if (stacey_absorbing_conditions) &
94  call exit_mpi(myrank,'Stacey boundary conditions are not implemented for axisymmetric simulations,use PML instead')
95  if (simulation_type /= 1) &
96  call exit_mpi(myrank,'Just axisymmetric FORWARD simulations are possible so far')
97  if (save_forward) &
98  call exit_mpi(myrank,'SAVE_FORWARD has presently not been tested with axisymmetric simulations')
99  if (add_periodic_conditions) &
100  call exit_mpi(myrank,'Periodic conditions (ADD_PERIODIC_CONDITIONS) are not implemented for axisymmetric simulations')
101  if (noise_tomography /= 0) &
102  call exit_mpi(myrank,'Axisymmetric noise tomographies are not possible yet')
103 
104  ! Check sources
105  ! Loop on the sources :
106  do isource = 1,nsources
107  if (myrank == islice_selected_source(isource)) then
108  ! If the source is not an elastic force or an acoustic pressure
109  if (source_type(isource) /= 1) then
110  call exit_mpi(myrank,'Axisymmetry : just elastic force or acoustic pressure sources has been tested so far)')
111  endif
112  ! If the source is on an axial element
113  if (is_on_the_axis(ispec_selected_source(isource))) then
114  ! ... or if the source is (at r=0) on an elastic axial element.
115  if (ispec_is_elastic(ispec_selected_source(isource))) then
116  if (((anglesource(isource) > tinyval) .and. (anglesource(isource) < pi) ) & ! ... and has a radial component.
117  .or. ( (anglesource(isource) > pi) .and. (anglesource(isource) < two*pi))) then
118  print *, '***** WARNING *****'
119  print *, 'Axisymmetry: U_r(r=0)=0, Radial component of axial source will be ignored (anglesource /= 0 modulo 180)'
120  print *
121  endif
122  endif
123  else ! If the source is not on an axial element
124  print *, '***** WARNING *****'
125  print *, 'Axisymmetry: physically a non axial source is a circular source!'
126  print *
127  endif
128  endif
129  enddo
130 
131  end subroutine check_compatibility_axisym

137  subroutine enforce_zero_radial_displacements_on_the_axis()
138 
139 ! This subroutine enforces zero displacement, velocity and acceleration on the axis for elastic elements;
140 ! for acoustic elements we do not need to do anything, some gradient components
141 ! will be set to zero on the axis later in the code, when they are computed
142 
143  use constants, only: nglj,ngllz,zero
144 
145  use specfem_par, only: ispec_is_elastic, ibool, nelem_on_the_axis, ispec_of_axial_elements, is_on_the_axis, &
146  displ_elastic, veloc_elastic, accel_elastic
147 
148  implicit none
149 
150  ! local variables
151  integer :: i_on_the_axis,ispec_axis,i,j
152 
153  do i_on_the_axis = 1,nelem_on_the_axis ! Loop on the elements on the axis
154  ispec_axis = ispec_of_axial_elements(i_on_the_axis)
155 
156  ! if the element is acoustic we do not need to do anything, some gradient components
157  ! will be set to zero on the axis later in the code, when they are computed
158 
159  ! if the element is elastic
160  if (ispec_is_elastic(ispec_axis)) then
161  do j = 1,ngllz ! Loop on the GLL/GLJ points
162  do i = 1,nglj
163  if (is_on_the_axis(ispec_axis) .and. i == 1) then ! If the point scanned is on the axis
164  displ_elastic(1,ibool(i,j,ispec_axis)) = zero ! enforce the radial displacement to zero
165  veloc_elastic(1,ibool(i,j,ispec_axis)) = zero ! enforce the radial velocity to zero
166  accel_elastic(1,ibool(i,j,ispec_axis)) = zero ! enforce the radial acceleration to zero
167  endif
168  enddo
169  enddo
170  endif
171  enddo
172 
173  end subroutine enforce_zero_radial_displacements_on_the_axis
