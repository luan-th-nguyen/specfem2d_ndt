33 
34  subroutine read_regions()
35 
36 ! reads in material definitions in DATA/Par_file and outputs to num_material
37 
38  use constants, only: imain,anisotropic_material,poroelastic_material,tinyval
39 
40  use shared_parameters, only: nbregions,nbmodels,num_material,icodemat,cp,cs, &
41  rho_s_read,qkappa,qmu,aniso3,aniso4,aniso5,aniso6,aniso7,aniso8,aniso9,aniso10,aniso11, &
42  nelmnts,nxread,nzread
43  implicit none
44 
45  ! local parameters
46  integer :: iregion,ix_start,ix_end,iz_start,iz_end,imaterial_number
47  integer :: i,j,ielem,ier
48  integer :: reread_nbregions
49  double precision :: vpregion,vsregion,poisson_ratio
50  logical :: is_overwriting
51  integer :: id_already_set
52 
53  integer,external :: err_occurred
54 
55  ! user output
56  write(imain,*) 'Regions:'
57  write(imain,*) ' Nb of regions in the mesh = ',nbregions
58  write(imain,*)
59 
60  ! assigns materials to mesh elements
61  allocate(num_material(nelmnts),stat=ier)
62  if (ier /= 0) stop 'Error allocating num_material array'
63  num_material(:) = 0
64 
65  ! this call positions again the read header to the line with nbregions. we can then call next line to get the table
66  call read_value_integer_p(reread_nbregions, 'mesher.nbregions')
67  if (err_occurred() /= 0) stop 'Error reading parameter nbregions in Par_file'
68 
69  ! checks
70  if (reread_nbregions /= nbregions) stop 'Error re-reading parameter nbregions in Par_file'
71  if (nbmodels < 1) stop 'Invalid number of model definitions'
72 
73  ! read the material numbers for each region
74  do iregion = 1,nbregions
75 
76  ! reads in region range
77  ! format: #ix start #ix end #iz start #iz end
78  call read_region_coordinates_p(ix_start,ix_end,iz_start,iz_end,imaterial_number)
79 
80  ! check
81  if (imaterial_number < 1) stop 'Negative material number not allowed!'
82  if (ix_start < 1) stop 'Left coordinate of region negative!'
83  if (ix_end > nxread) stop 'Right coordinate of region too high!'
84  if (iz_start < 1) stop 'Bottom coordinate of region negative!'
85  if (iz_end > nzread) stop 'Top coordinate of region too high!'
86 
87  if (iregion == 1) write(imain,*) '------'
88  write(imain,*) 'Region ',iregion
89  write(imain,*) 'IX from ',ix_start,' to ',ix_end
90  write(imain,*) 'IZ from ',iz_start,' to ',iz_end
91 
92  ! determines region domain
93  if (icodemat(imaterial_number) /= anisotropic_material .and. icodemat(imaterial_number) /= poroelastic_material) then
94 
95  ! isotropic material
96  vpregion = cp(imaterial_number)
97  vsregion = cs(imaterial_number)
98 
99  poisson_ratio = 0.5d0*(vpregion*vpregion-2.d0*vsregion*vsregion) / (vpregion*vpregion-vsregion*vsregion)
100 
101  write(imain,*) 'Material # ',imaterial_number,' isotropic'
102  if (vsregion < tinyval) then
103  write(imain,*) 'Material is fluid'
104  else
105  write(imain,*) 'Material is solid'
106  endif
107  write(imain,*) 'vp = ',sngl(vpregion)
108  write(imain,*) 'vs = ',sngl(vsregion)
109  write(imain,*) 'rho = ',sngl(rho_s_read(imaterial_number))
110  write(imain,*) 'Poisson''s ratio = ',sngl(poisson_ratio)
111  write(imain,*) 'QKappa = ',sngl(qkappa(imaterial_number))
112  write(imain,*) 'Qmu = ',sngl(qmu(imaterial_number))
113 
114  if (poisson_ratio <= -1.00001d0 .or. poisson_ratio >= 0.50001d0) stop 'incorrect value of Poisson''s ratio'
115 
116  else if (icodemat(imaterial_number) == poroelastic_material) then
117 
118  ! poroelastic material
119  write(imain,*) 'Material # ',imaterial_number,' isotropic'
120  write(imain,*) 'Material is poroelastic'
121 
122  else
123 
124  ! anisotropic material
125  write(imain,*) 'Material # ',imaterial_number,' anisotropic'
126  write(imain,*) 'cp = ',sngl(cp(imaterial_number))
127  write(imain,*) 'cs = ',sngl(cs(imaterial_number))
128  write(imain,*) 'c11 = ',sngl(aniso3(imaterial_number))
129  write(imain,*) 'c13 = ',sngl(aniso4(imaterial_number))
130  write(imain,*) 'c15 = ',sngl(aniso5(imaterial_number))
131  write(imain,*) 'c33 = ',sngl(aniso6(imaterial_number))
132  write(imain,*) 'c35 = ',sngl(aniso7(imaterial_number))
133  write(imain,*) 'c55 = ',sngl(aniso8(imaterial_number))
134  write(imain,*) 'c12 = ',sngl(aniso9(imaterial_number))
135  write(imain,*) 'c23 = ',sngl(aniso10(imaterial_number))
136  write(imain,*) 'c25 = ',sngl(aniso11(imaterial_number))
137  write(imain,*) 'rho = ',sngl(rho_s_read(imaterial_number))
138  write(imain,*) 'QKappa = ',sngl(qkappa(imaterial_number))
139  write(imain,*) 'Qmu = ',sngl(qmu(imaterial_number))
140  endif
141 
142  ! store density and velocity model
143  is_overwriting = .false.
144  do j = iz_start,iz_end
145  do i = ix_start,ix_end
146  ! element index
147  ielem = (j-1)*nxread+i
148  ! checks if element has been already assigned
149  if (num_material(ielem) /= 0) then
150  is_overwriting = .true.
151  id_already_set = num_material(ielem)
152  endif
153  ! sets new material id for element
154  num_material(ielem) = imaterial_number
155  enddo
156  enddo
157 
158  ! user output
159  if (is_overwriting) then
160  write(imain,*) '*************************************'
161  write(imain,*) 'Warning: Element range from this region is overwriting material numbers previously set on elements.'
162  write(imain,*) ' This indicates that your region range is overlapping the region for material ',id_already_set
163  write(imain,*) ' If your regions should be exclusive, please fix the region definitions in the Par_file!'
164  write(imain,*) '*************************************'
165  endif
166  write(imain,*) '------'
167  call flush_imain()
168 
169  enddo
170 
171  if (minval(num_material) <= 0) stop 'Velocity model not entirely set...'
172 
173  end subroutine read_regions
