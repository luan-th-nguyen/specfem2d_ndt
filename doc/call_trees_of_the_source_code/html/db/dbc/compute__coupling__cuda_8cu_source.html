46 __global__ void compute_coupling_acoustic_el_kernel(realw* displ,
47  realw* potential_dot_dot_acoustic,
48  int num_coupling_ac_el_faces,
49  int* coupling_ac_el_ispec,
50  int* coupling_ac_el_ij,
51  realw* coupling_ac_el_normal,
52  realw* coupling_ac_el_jacobian1Dw,
53  int* d_ibool) {
54 
55  int igll = threadIdx.x;
56  int iface = blockIdx.x + gridDim.x*blockIdx.y;
57 
58  int i,j,iglob,ispec;
59  realw displ_x,displ_z,displ_n;
60  realw nx,nz;
61  realw jacobianw;
62 
63  if (iface < num_coupling_ac_el_faces){
64 
65  // don't compute points outside NGLLSQUARE==NGLL2==25
66  // way 2: no further check needed since blocksize = 25
67  // if (igll<NGLL2) {
68 
69  // "-1" from index values to convert from Fortran-> C indexing
70  ispec = coupling_ac_el_ispec[iface] - 1;
71 
72  i = coupling_ac_el_ij[INDEX3(NDIM,NGLLX,0,igll,iface)] - 1;
73  j = coupling_ac_el_ij[INDEX3(NDIM,NGLLX,1,igll,iface)] - 1;
74 
75  iglob = d_ibool[INDEX3_PADDED(NGLLX,NGLLX,i,j,ispec)] - 1;
76 
77  // elastic displacement on global point
78  displ_x = displ[iglob*2] ; // (1,iglob)
79  displ_z = displ[iglob*2+1] ; // (2,iglob)
80 
81  // gets associated normal on GLL point
82  nx = coupling_ac_el_normal[INDEX3(NDIM,NGLLX,0,igll,iface)]; // (1,igll,iface)
83  nz = coupling_ac_el_normal[INDEX3(NDIM,NGLLX,1,igll,iface)]; // (2,igll,iface)
84 
85  // calculates displacement component along normal
86  // (normal points outwards of acoustic element)
87  displ_n = displ_x*nx + displ_z*nz;
88 
89  // gets associated, weighted jacobian
90  jacobianw = coupling_ac_el_jacobian1Dw[INDEX2(NGLLX,igll,iface)];
91 
92  // continuity of pressure and normal displacement on global point
93 
94  // note: Newmark time scheme together with definition of scalar potential:
95  // pressure = - chi_dot_dot
96  // requires that this coupling term uses the updated displacement at time step [t+delta_t],
97  // which is done at the very beginning of the time loop
98  // (see e.g. Chaljub & Vilotte, Nissen-Meyer thesis...)
99  // it also means you have to calculate and update this here first before
100  // calculating the coupling on the elastic side for the acceleration...
101  atomicAdd(&potential_dot_dot_acoustic[iglob],+ jacobianw*displ_n);
102 
103  // }
104  }
105 }

109 extern "C"
110 void FC_FUNC_(compute_coupling_ac_el_cuda,
111  COMPUTE_COUPLING_AC_EL_CUDA)(long* Mesh_pointer,
112  int* iphasef,
113  int* num_coupling_ac_el_facesf) {
114  TRACE("compute_coupling_ac_el_cuda");
115  //double start_time = get_time();
116 
117  Mesh* mp = (Mesh*)(*Mesh_pointer); //get mesh pointer out of fortran integer container
118  int iphase = *iphasef;
119 
120  // only adds this contribution for first pass
121  if (iphase != 1) return;
122 
123  int num_coupling_ac_el_faces = *num_coupling_ac_el_facesf;
124 
125  // way 1: exact blocksize to match NGLLSQUARE
126  int blocksize = NGLLX;
127 
128  int num_blocks_x, num_blocks_y;
129  get_blocks_xy(num_coupling_ac_el_faces,&num_blocks_x,&num_blocks_y);
130 
131  dim3 grid(num_blocks_x,num_blocks_y);
132  dim3 threads(blocksize,1,1);
133 
134  // launches GPU kernel
135  compute_coupling_acoustic_el_kernel<<<grid,threads>>>(mp->d_displ,
136  mp->d_potential_dot_dot_acoustic,
137  num_coupling_ac_el_faces,
138  mp->d_coupling_ac_el_ispec,
139  mp->d_coupling_ac_el_ijk,
140  mp->d_coupling_ac_el_normal,
141  mp->d_coupling_ac_el_jacobian2Dw,
142  mp->d_ibool);
143 
144  // adjoint simulations
145  if (mp->simulation_type == 3) {
146  compute_coupling_acoustic_el_kernel<<<grid,threads>>>(mp->d_b_displ,
147  mp->d_b_potential_dot_dot_acoustic,
148  num_coupling_ac_el_faces,
149  mp->d_coupling_ac_el_ispec,
150  mp->d_coupling_ac_el_ijk,
151  mp->d_coupling_ac_el_normal,
152  mp->d_coupling_ac_el_jacobian2Dw,
153  mp->d_ibool);
154  }
155 
156 
157 #ifdef ENABLE_VERY_SLOW_ERROR_CHECKING
158  //double end_time = get_time();
159  //printf("Elapsed time: %e\n",end_time-start_time);
160  exit_on_cuda_error("compute_coupling_acoustic_el_kernel");
161 #endif
162 }

171 __global__ void compute_coupling_elastic_ac_kernel(realw* potential_dot_dot_acoustic,
172  realw* accel,
173  int num_coupling_ac_el_faces,
174  int* coupling_ac_el_ispec,
175  int* coupling_ac_el_ij,
176  realw* coupling_ac_el_normal,
177  realw* coupling_ac_el_jacobian1Dw,
178  int* d_ibool) {
179 
180  int igll = threadIdx.x;
181  int iface = blockIdx.x + gridDim.x*blockIdx.y;
182 
183  int i,j,iglob,ispec;
184  realw pressure;
185  realw nx,nz;
186  realw jacobianw;
187 
188 
189  if (iface < num_coupling_ac_el_faces){
190 
191  // don't compute points outside NGLLSQUARE==NGLL2==25
192  // way 2: no further check needed since blocksize = 25
193  // if (igll<NGLL2) {
194 
195  // "-1" from index values to convert from Fortran-> C indexing
196  ispec = coupling_ac_el_ispec[iface] - 1;
197 
198  i = coupling_ac_el_ij[INDEX3(NDIM,NGLLX,0,igll,iface)] - 1;
199  j = coupling_ac_el_ij[INDEX3(NDIM,NGLLX,1,igll,iface)] - 1;
200 
201  iglob = d_ibool[INDEX3_PADDED(NGLLX,NGLLX,i,j,ispec)] - 1;
202 
203  // gets associated normal on GLL point
204  // note: normal points away from acoustic element
205  nx = coupling_ac_el_normal[INDEX3(NDIM,NGLLX,0,igll,iface)]; // (1,igll,iface)
206  nz = coupling_ac_el_normal[INDEX3(NDIM,NGLLX,1,igll,iface)]; // (2,igll,iface)
207 
208  // gets associated, weighted jacobian
209  jacobianw = coupling_ac_el_jacobian1Dw[INDEX2(NGLLX,igll,iface)];
210 
211 
212  pressure = - potential_dot_dot_acoustic[iglob];
213 
214 
215  // continuity of displacement and pressure on global point
216  //
217  // note: Newmark time scheme together with definition of scalar potential:
218  // pressure = - chi_dot_dot
219  // requires that this coupling term uses the *UPDATED* pressure (chi_dot_dot), i.e.
220  // pressure at time step [t + delta_t]
221  // (see e.g. Chaljub & Vilotte, Nissen-Meyer thesis...)
222  // it means you have to calculate and update the acoustic pressure first before
223  // calculating this term...
224  atomicAdd(&accel[iglob*2],+ jacobianw*nx*pressure);
225  atomicAdd(&accel[iglob*2+1],+ jacobianw*nz*pressure);
226 
227  // }
228  }
229 }
