33 
34  subroutine compute_forces_acoustic_backward(b_potential_dot_dot_acoustic,b_potential_acoustic,iphase)
35 
36 
37 ! compute forces in the acoustic elements in forward simulation and in adjoint simulation in adjoint inversion
38 
39  use constants, only: custom_real,ngllx,ngllz,nglj,cpml_x_only,cpml_z_only,iright,ileft,ibottom,itop, &
40  zero,one,two,two_thirds,iedge1,iedge2,iedge3,iedge4
41 
42  use specfem_par, only: nglob, &
43  assign_external_model,ibool,kmato,ispec_is_acoustic, &
44  density,poroelastcoef,xix,xiz,gammax,gammaz,jacobian, &
45  vpext,rhoext, &
46  hprime_xx,hprimewgll_xx, &
47  hprime_zz,hprimewgll_zz,wxgll,wzgll, &
48  axisym,coord, is_on_the_axis,hprimebar_xx,hprimebarwglj_xx,xiglj,wxglj
49 
50  ! overlapping communication
51  use specfem_par, only: nspec_inner_acoustic,nspec_outer_acoustic,phase_ispec_inner_acoustic
52 
53  implicit none
54 
55  real(kind=CUSTOM_REAL), dimension(nglob) :: b_potential_dot_dot_acoustic, b_potential_acoustic
56 
57  integer,intent(in) :: iphase
58 
59  ! local parameters
60  integer :: ispec,i,j,k,iglob
61 
62  ! spatial derivatives
63  real(kind=CUSTOM_REAL) :: dux_dxi,dux_dgamma,dux_dxl,dux_dzl
64  real(kind=CUSTOM_REAL) :: xxi
65 
66  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: tempx1,tempx2
67  real(kind=CUSTOM_REAL), dimension(NGLJ,NGLLZ) :: r_xiplus1
68 
69  ! Jacobian matrix and determinant
70  real(kind=CUSTOM_REAL) :: xixl,xizl,gammaxl,gammazl,jacobianl
71 
72  ! material properties of the acoustic medium
73  real(kind=CUSTOM_REAL) :: mul_relaxed,lambdal_relaxed,kappal,cpl,rhol
74 
75  integer :: num_elements,ispec_p
76 
77  ! choses inner/outer elements
78  if (iphase == 1) then
79  num_elements = nspec_outer_acoustic
80  else
81  num_elements = nspec_inner_acoustic
82  endif
83 
84  ! loop over spectral elements
85  do ispec_p = 1,num_elements
86 
87  ! returns element id from stored element list
88  ispec = phase_ispec_inner_acoustic(ispec_p,iphase)
89 
90  ! acoustic spectral element
91  if (.not. ispec_is_acoustic(ispec)) cycle
92 
93  rhol = density(1,kmato(ispec))
94 
95  ! first double loop over GLL points to compute and store gradients
96  do j = 1,ngllz
97  do i = 1,ngllx
98  ! derivative along x and along z
99  dux_dxi = 0._custom_real; dux_dgamma = 0._custom_real
100 
101  ! first double loop over GLL points to compute and store gradients
102  ! we can merge the two loops because NGLLX == NGLLZ
103  do k = 1,ngllx
104  if (axisym) then
105  if (is_on_the_axis(ispec)) then
106  dux_dxi = dux_dxi + b_potential_acoustic(ibool(k,j,ispec)) * hprimebar_xx(i,k)
107  else
108  dux_dxi = dux_dxi + b_potential_acoustic(ibool(k,j,ispec)) * hprime_xx(i,k)
109  endif
110  else
111  dux_dxi = dux_dxi + b_potential_acoustic(ibool(k,j,ispec)) * hprime_xx(i,k)
112  endif
113  dux_dgamma = dux_dgamma + b_potential_acoustic(ibool(i,k,ispec)) * hprime_zz(j,k)
114  enddo
115 
116  xixl = xix(i,j,ispec)
117  xizl = xiz(i,j,ispec)
118  gammaxl = gammax(i,j,ispec)
119  gammazl = gammaz(i,j,ispec)
120 
121  ! derivatives of potential
122  dux_dxl = dux_dxi * xixl + dux_dgamma * gammaxl
123  dux_dzl = dux_dxi * xizl + dux_dgamma * gammazl
124 
125  if (axisym .and. is_on_the_axis(ispec) .and. i == 1) then ! dchi/dr=rho * u_r=0 on the axis
126  dux_dxl = zero
127  endif
128 
129  jacobianl = jacobian(i,j,ispec)
130 
131  ! if external density model
132  if (assign_external_model) then
133  rhol = rhoext(i,j,ispec)
134  endif
135 
136  if (axisym) then
137  if (is_on_the_axis(ispec) .and. i == 1) then
138  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
139  r_xiplus1(i,j) = xxi
140  else if (is_on_the_axis(ispec)) then
141  r_xiplus1(i,j) = coord(1,ibool(i,j,ispec))/(xiglj(i)+one)
142  endif
143  endif
144 
145  ! for acoustic medium also add integration weights
146  if (axisym) then
147  if (is_on_the_axis(ispec)) then
148  tempx1(i,j) = wzgll(j) * r_xiplus1(i,j) * jacobianl * (xixl * dux_dxl + xizl * dux_dzl) / rhol
149  tempx2(i,j) = wxglj(i) * r_xiplus1(i,j) * jacobianl * (gammaxl * dux_dxl + gammazl * dux_dzl) / rhol
150  else
151  tempx1(i,j) = wzgll(j) * coord(1,ibool(i,j,ispec)) * jacobianl * (xixl * dux_dxl + xizl * dux_dzl) / rhol
152  tempx2(i,j) = wxgll(i) * coord(1,ibool(i,j,ispec)) * jacobianl * (gammaxl * dux_dxl + gammazl * dux_dzl) / rhol
153  endif
154  else
155  tempx1(i,j) = wzgll(j) * jacobianl * (xixl * dux_dxl + xizl * dux_dzl) / rhol
156  tempx2(i,j) = wxgll(i) * jacobianl * (gammaxl * dux_dxl + gammazl * dux_dzl) / rhol
157  endif
158  enddo
159  enddo
160 
161  ! first double loop over GLL points to compute and store gradients
162  do j = 1,ngllz
163  do i = 1,ngllx
164  iglob = ibool(i,j,ispec)
165  if (assign_external_model) then
166  rhol = rhoext(i,j,ispec)
167  cpl = vpext(i,j,ispec)
168  !assuming that in fluid(acoustic) part input cpl is defined by sqrt(kappal/rhol), &
169  !which is not the same as in cpl input in elastic part
170  kappal = rhol * cpl * cpl ! CHECK kappa
171  else
172  lambdal_relaxed = poroelastcoef(1,1,kmato(ispec))
173  mul_relaxed = poroelastcoef(2,1,kmato(ispec))
174  if (axisym) then ! CHECK kappa
175  kappal = lambdal_relaxed + two_thirds * mul_relaxed
176  else
177  kappal = lambdal_relaxed + mul_relaxed
178  endif
179  rhol = density(1,kmato(ispec))
180  endif
181 
182  enddo
183  enddo
184 !
185 ! second double-loop over GLL to compute all the terms
186 !
187  do j = 1,ngllz
188  do i = 1,ngllx
189  iglob = ibool(i,j,ispec)
190  ! along x direction and z direction
191  ! and assemble the contributions
192  if (axisym) then
193  if (is_on_the_axis(ispec)) then
194  do k = 1,ngllx
195  b_potential_dot_dot_acoustic(iglob) = b_potential_dot_dot_acoustic(iglob) - &
196  (tempx1(k,j) * hprimebarwglj_xx(k,i) + tempx2(i,k) * hprimewgll_zz(k,j))
197  enddo
198  else
199  do k = 1,ngllx
200  b_potential_dot_dot_acoustic(iglob) = b_potential_dot_dot_acoustic(iglob) - &
201  (tempx1(k,j) * hprimewgll_xx(k,i) + tempx2(i,k) * hprimewgll_zz(k,j))
202  enddo
203  endif
204  else
205  do k = 1,ngllx
206  b_potential_dot_dot_acoustic(iglob) = b_potential_dot_dot_acoustic(iglob) - &
207  (tempx1(k,j) * hprimewgll_xx(k,i) + tempx2(i,k) * hprimewgll_zz(k,j))
208  enddo
209  endif
210  enddo ! second loop over the GLL points
211  enddo
212 
213  enddo ! end of loop over all spectral elements
214 
215  end subroutine compute_forces_acoustic_backward
