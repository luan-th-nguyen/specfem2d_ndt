34  subroutine write_color_image_snaphot()
35 
36 #ifdef USE_MPI
37  use mpi
38 #endif
39 
40  use constants, only: imain,ngllx,ngllz,remove_pmls_from_jpeg_images
41 
42  use specfem_par, only: myrank,nspec,it,nproc, &
43  assign_external_model,ibool,kmato,density,rhoext,p_sv, &
44  potential_acoustic,potential_dot_acoustic,potential_dot_dot_acoustic, &
45  potential_gravito,potential_dot_gravito,potential_dot_dot_gravito, &
46  potential_gravitoacoustic,potential_dot_gravitoacoustic,potential_dot_dot_gravitoacoustic, &
47  displ_elastic,veloc_elastic,accel_elastic, &
48  displs_poroelastic,velocs_poroelastic,accels_poroelastic
49 
50  ! PML arrays
51  use specfem_par, only: pml_boundary_conditions,ispec_is_pml
52 
53  use specfem_par_movie, only: vector_field_display,image_color_data,iglob_image_color, &
54  imagetype_jpeg,nb_pixel_loc, &
55  nx_image_color,nz_image_color, &
56  num_pixel_loc
57 
58 #ifdef USE_MPI
59  use specfem_par_movie, only: data_pixel_recv,data_pixel_send,nb_pixel_per_proc,num_pixel_recv
60 #endif
61 
62  implicit none
63 
64  !local variables
65  integer :: i,j,k,ispec,iglob,iproc
66  double precision :: rhol
67 
68  if (myrank == 0) then
69  write(imain,*)
70  write(imain,*) 'Creating color image of size ',nx_image_color,' x ',nz_image_color,' for time step ',it
71  call flush_imain()
72  endif
73 
74  if (imagetype_jpeg >= 1 .and. imagetype_jpeg <= 3) then
75  if (myrank == 0) write(imain,*) 'drawing scalar image of part of the displacement vector...'
76  call compute_vector_whole_medium(potential_acoustic,potential_gravitoacoustic, &
77  potential_gravito,displ_elastic,displs_poroelastic)
78 
79  else if (imagetype_jpeg >= 4 .and. imagetype_jpeg <= 6) then
80  if (myrank == 0) write(imain,*) 'drawing scalar image of part of the velocity vector...'
81  call compute_vector_whole_medium(potential_dot_acoustic,potential_dot_gravitoacoustic, &
82  potential_dot_gravito,veloc_elastic,velocs_poroelastic)
83 
84  else if (imagetype_jpeg >= 7 .and. imagetype_jpeg <= 9) then
85  if (myrank == 0) write(imain,*) 'drawing scalar image of part of the acceleration vector...'
86  call compute_vector_whole_medium(potential_dot_dot_acoustic,potential_dot_dot_gravitoacoustic, &
87  potential_dot_dot_gravito,accel_elastic,accels_poroelastic)
88 
89  else if (imagetype_jpeg >= 11 .and. imagetype_jpeg <= 13) then
90  ! allocation for normalized representation in JPEG image
91  ! for an atmosphere model
92  if (myrank == 0) write(imain,*) 'drawing scalar image of part of normalized displacement vector...'
93  call compute_vector_whole_medium(potential_acoustic,potential_gravitoacoustic, &
94  potential_gravito,displ_elastic,displs_poroelastic)
95 
96  do ispec = 1,nspec
97  do j = 1,ngllz
98  do i = 1,ngllx
99  if (assign_external_model) then
100  rhol = rhoext(i,j,ispec)
101  else
102  rhol = density(1,kmato(ispec))
103  endif
104  iglob = ibool(i,j,ispec)
105  vector_field_display(1,iglob) = sqrt(rhol) * vector_field_display(1,iglob)
106  vector_field_display(2,iglob) = sqrt(rhol) * vector_field_display(2,iglob)
107  enddo
108  enddo
109  enddo
110 
111  else if (imagetype_jpeg >= 14 .and. imagetype_jpeg <= 16) then
112  ! allocation for normalized representation in JPEG image
113  ! for an atmosphere model
114  call compute_vector_whole_medium(potential_dot_acoustic,potential_dot_gravitoacoustic, &
115  potential_dot_gravito,veloc_elastic,velocs_poroelastic)
116 
117  do ispec = 1,nspec
118  do j = 1,ngllz
119  do i = 1,ngllx
120  if (assign_external_model) then
121  rhol = rhoext(i,j,ispec)
122  else
123  rhol = density(1,kmato(ispec))
124  endif
125  iglob = ibool(i,j,ispec)
126  vector_field_display(1,iglob) = sqrt(rhol) * vector_field_display(1,iglob)
127  vector_field_display(2,iglob) = sqrt(rhol) * vector_field_display(2,iglob)
128  enddo
129  enddo
130  enddo
131 
132  else if (imagetype_jpeg == 10 .and. p_sv) then
133  if (myrank == 0) write(imain,*) 'drawing image of pressure field...'
134  call compute_pressure_whole_medium()
135 
136  else if (imagetype_jpeg == 10 .and. .not. p_sv) then
137  call exit_mpi(myrank,'cannot draw pressure field for SH (membrane) waves')
138 
139  else
140  call exit_mpi(myrank,'wrong type for JPEG snapshots')
141  endif
142 
143 !! DK DK quick hack to remove the PMLs from JPEG images if needed: set the vector field to zero there
144  if (pml_boundary_conditions .and. remove_pmls_from_jpeg_images) then
145  do ispec = 1,nspec
146  if (ispec_is_pml(ispec)) then
147  do j = 1,ngllz
148  do i = 1,ngllx
149  iglob = ibool(i,j,ispec)
150  vector_field_display(1,iglob) = 0.d0
151  vector_field_display(2,iglob) = 0.d0
152  enddo
153  enddo
154  endif
155  enddo
156  endif
157 !! DK DK quick hack to remove the PMLs from JPEG images if needed
158 
159  image_color_data(:,:) = 0.d0
160 
161  do k = 1, nb_pixel_loc
162  j = ceiling(real(num_pixel_loc(k)) / real(nx_image_color))
163  i = num_pixel_loc(k) - (j-1)*nx_image_color
164 
165  ! avoid edge effects
166  if (i < 1 ) i = 1
167  if (j < 1 ) j = 1
168 
169  if (i > nx_image_color ) i = nx_image_color
170  if (j > nz_image_color ) j = nz_image_color
171 
172  if (p_sv) then
173  ! P-SH waves, plot a component of vector, its norm, or else pressure
174  if (iglob_image_color(i,j) /= -1) then
175  if (imagetype_jpeg == 1 .or. imagetype_jpeg == 4 .or. imagetype_jpeg == 7 .or. &
176  imagetype_jpeg == 11 .or. imagetype_jpeg == 14) then
177  ! draw the X component of the vector
178  image_color_data(i,j) = vector_field_display(1,iglob_image_color(i,j))
179 
180  else if (imagetype_jpeg == 2 .or. imagetype_jpeg == 5 .or. imagetype_jpeg == 8 .or. &
181  imagetype_jpeg == 12 .or. imagetype_jpeg == 15) then
182  ! draw the Z component of the vector
183  image_color_data(i,j) = vector_field_display(2,iglob_image_color(i,j))
184 
185  else if (imagetype_jpeg == 3 .or. imagetype_jpeg == 6 .or. imagetype_jpeg == 9 .or. &
186  imagetype_jpeg == 13 .or. imagetype_jpeg == 16) then
187  ! draw the norm of the vector
188  image_color_data(i,j) = sqrt(vector_field_display(1,iglob_image_color(i,j))**2 &
189  + vector_field_display(2,iglob_image_color(i,j))**2)
190 
191  else if (imagetype_jpeg == 10) then
192  ! by convention we have stored pressure in the 2. component of the array
193  image_color_data(i,j) = vector_field_display(2,iglob_image_color(i,j))
194 
195  else
196  call exit_mpi(myrank,'wrong type for JPEG snapshots')
197  endif
198  endif
199 
200  else
201  ! SH (membrane) waves, plot y-component
202  if (iglob_image_color(i,j) /= -1) image_color_data(i,j) = vector_field_display(1,iglob_image_color(i,j))
203  endif
204  enddo
205 
206  ! assembling array image_color_data on process zero for color output
207 #ifdef USE_MPI
208  if (nproc > 1) then
209  if (myrank == 0) then
210  do iproc = 1, nproc-1
211  call recv_dp(data_pixel_recv(1), nb_pixel_per_proc(iproc), iproc, 43)
212 
213  do k = 1, nb_pixel_per_proc(iproc)
214  j = ceiling(real(num_pixel_recv(k,iproc+1)) / real(nx_image_color))
215  i = num_pixel_recv(k,iproc+1) - (j-1)*nx_image_color
216 
217  ! avoid edge effects
218  if (i < 1) i = 1
219  if (j < 1) j = 1
220 
221  if (i > nx_image_color) i = nx_image_color
222  if (j > nz_image_color) j = nz_image_color
223 
224  image_color_data(i,j) = data_pixel_recv(k)
225  enddo
226  enddo
227  else
228  do k = 1, nb_pixel_loc
229  j = ceiling(real(num_pixel_loc(k)) / real(nx_image_color))
230  i = num_pixel_loc(k) - (j-1)*nx_image_color
231 
232  ! avoid edge effects
233  if (i < 1) i = 1
234  if (j < 1) j = 1
235 
236  if (i > nx_image_color) i = nx_image_color
237  if (j > nz_image_color) j = nz_image_color
238 
239  if (p_sv) then ! P-SH waves, plot a component of vector, its norm, or else pressure
240 
241  if (imagetype_jpeg == 1 .or. imagetype_jpeg == 4 .or. imagetype_jpeg == 7 .or. &
242  imagetype_jpeg == 11 .or. imagetype_jpeg == 14) then
243  data_pixel_send(k) = vector_field_display(1,iglob_image_color(i,j)) ! draw the X component of the vector
244 
245  else if (imagetype_jpeg == 2 .or. imagetype_jpeg == 5 .or. imagetype_jpeg == 8 .or. &
246  imagetype_jpeg == 12 .or. imagetype_jpeg == 15) then
247  data_pixel_send(k) = vector_field_display(2,iglob_image_color(i,j)) ! draw the Z component of the vector
248 
249  else if (imagetype_jpeg == 3 .or. imagetype_jpeg == 6 .or. imagetype_jpeg == 9 .or. &
250  imagetype_jpeg == 13 .or. imagetype_jpeg == 16) then
251  data_pixel_send(k) = sqrt(vector_field_display(1,iglob_image_color(i,j))**2 + &
252  vector_field_display(2,iglob_image_color(i,j))**2) ! draw the norm of the vector
253 
254  else if (imagetype_jpeg == 10) then
255  ! by convention we have stored pressure in the 2. component of the array
256  data_pixel_send(k) = vector_field_display(2,iglob_image_color(i,j))
257 
258  else
259  call exit_mpi(myrank,'wrong type for JPEG snapshots')
260  endif
261 
262  else ! SH (membrane) waves, plot y-component
263  if (iglob_image_color(i,j) /= -1) data_pixel_send(k) = vector_field_display(1,iglob_image_color(i,j))
264  endif
265  enddo
266 
267  call send_dp(data_pixel_send(1), nb_pixel_loc, 0, 43)
268 
269  endif
270  endif
271  call synchronize_all()
272 #else
273  ! dummy to avoid compiler warning
274  iproc = nproc
275 #endif
276 
277  call synchronize_all()
278 
279  ! creates image
280  if (myrank == 0) then
281  call create_color_image()
282 
283  ! user output
284  write(imain,*) 'Color image created'
285  call flush_imain()
286  endif
287  call synchronize_all()
288 
289  end subroutine write_color_image_snaphot
