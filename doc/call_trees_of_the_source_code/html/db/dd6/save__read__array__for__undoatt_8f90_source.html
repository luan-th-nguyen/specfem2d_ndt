35 
36  use constants, only: iout_undo_att,max_string_len,output_files
37 
38  use specfem_par, only: myrank,iteration_on_subset, &
39  any_acoustic,any_elastic,attenuation_viscoelastic, &
40  potential_acoustic,potential_dot_acoustic,potential_dot_dot_acoustic, &
41  displ_elastic,veloc_elastic,accel_elastic, &
42  e1,e11,e13
43 
44  implicit none
45 
46  ! local parameters
47  integer :: iteration_on_subset_tmp
48  integer :: ier
49  character(len=MAX_STRING_LEN) :: outputname
50 
51  ! current subset iteration
52  iteration_on_subset_tmp = iteration_on_subset
53 
54  ! saves frame of the forward simulation
55 
56  write(outputname,'(a,i6.6,a,i6.6,a)') 'proc',myrank,'_save_frame_at',iteration_on_subset_tmp,'.bin'
57  open(unit=iout_undo_att ,file=trim(output_files)//outputname, &
58  status='unknown',form='unformatted',action='write',iostat=ier)
59  if (ier /= 0 ) call exit_mpi(myrank,'Error opening file proc***_save_frame_at** for writing')
60 
61  if (any_acoustic) then
62  write(iout_undo_att) potential_dot_dot_acoustic
63  write(iout_undo_att) potential_dot_acoustic
64  write(iout_undo_att) potential_acoustic
65  endif
66 
67  if (any_elastic) then
68  write(iout_undo_att) accel_elastic
69  write(iout_undo_att) veloc_elastic
70  write(iout_undo_att) displ_elastic
71 
72  if (attenuation_viscoelastic) then
73  write(iout_undo_att) e1
74  write(iout_undo_att) e11
75  write(iout_undo_att) e13
76  endif
77  endif
78 
79  close(iout_undo_att)
80 
81  end subroutine save_forward_arrays_undoatt
