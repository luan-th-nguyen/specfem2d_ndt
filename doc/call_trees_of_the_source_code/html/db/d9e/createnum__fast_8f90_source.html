34  subroutine createnum_fast()
35 
36 ! same as subroutine "createnum_slow" but with a faster algorithm
37 
38  use constants, only: imain,ngllx,ngllz,ndim,hugeval,smallvaltol,zero
39 
40  use specfem_par, only: knods,ibool,shape2d,coorg,nglob,nspec,ngnod,myrank
41 
42  implicit none
43 
44 ! additional arrays needed for this fast version
45  integer, dimension(:), allocatable :: locval,ind,ninseg,iglob,iwork
46  logical, dimension(:), allocatable :: ifseg
47  double precision, dimension(:), allocatable :: xp,yp,work
48 
49  integer :: ispec,nseg,ioff,iseg,ig,i,j
50  integer :: nxyz,ntot,ieoff,ilocnum,iy,ix,in,nnum
51 
52  double precision :: xmaxval,xminval,ymaxval,yminval,xtol,xtypdist
53  double precision :: xcor,ycor
54 
55 
56 !---- create global mesh numbering
57  if (myrank == 0) then
58  write(imain,*)
59  write(imain,*) 'Generating global mesh numbering (fast version)...'
60  write(imain,*)
61  call flush_imain()
62  endif
63 
64  nxyz = ngllx*ngllz
65  ntot = nxyz*nspec
66 
67  allocate(locval(ntot))
68  allocate(ind(ntot))
69  allocate(ninseg(ntot))
70  allocate(iglob(ntot))
71  allocate(ifseg(ntot))
72  allocate(xp(ntot))
73  allocate(yp(ntot))
74  allocate(work(ntot))
75  allocate(iwork(ntot))
76 
77 ! compute coordinates of the grid points
78  do ispec = 1,nspec
79  ieoff = nxyz*(ispec - 1)
80  ilocnum = 0
81 
82  do iy = 1,ngllx
83  do ix = 1,ngllx
84 
85  ilocnum = ilocnum + 1
86 
87  xcor = zero
88  ycor = zero
89  do in = 1,ngnod
90  nnum = knods(in,ispec)
91  xcor = xcor + shape2d(in,ix,iy)*coorg(1,nnum)
92  ycor = ycor + shape2d(in,ix,iy)*coorg(2,nnum)
93  enddo
94 
95  xp(ilocnum + ieoff) = xcor
96  yp(ilocnum + ieoff) = ycor
97 
98  enddo
99  enddo
100 
101  enddo
102 
103 ! %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
104 
105 ! Establish initial pointers
106  do ispec = 1,nspec
107  ieoff = nxyz*(ispec -1)
108  do ix = 1,nxyz
109  locval(ix+ieoff) = ix+ieoff
110  enddo
111  enddo
112 
113 ! set up a local geometric tolerance
114 
115  xtypdist = +hugeval
116 
117  do ispec = 1,nspec
118 
119  xminval = +hugeval
120  yminval = +hugeval
121  xmaxval = -hugeval
122  ymaxval = -hugeval
123  ieoff = nxyz*(ispec-1)
124  do ilocnum = 1,nxyz
125  xmaxval = max(xp(ieoff+ilocnum),xmaxval)
126  xminval = min(xp(ieoff+ilocnum),xminval)
127  ymaxval = max(yp(ieoff+ilocnum),ymaxval)
128  yminval = min(yp(ieoff+ilocnum),yminval)
129  enddo
130 
131 ! compute the minimum typical "size" of an element in the mesh
132  xtypdist = min(xtypdist,xmaxval-xminval)
133  xtypdist = min(xtypdist,ymaxval-yminval)
134 
135  enddo
136 
137 ! define a tolerance, small with respect to the minimum size
138  xtol = smallvaltol * xtypdist
139 
140  ifseg(:) = .false.
141  nseg = 1
142  ifseg(1) = .true.
143  ninseg(1) = ntot
144 
145  do j = 1,ndim
146 ! Sort within each segment
147  ioff=1
148  do iseg= 1,nseg
149  if (j == 1) then
150  call rank (xp(ioff),ind,ninseg(iseg))
151  else
152  call rank (yp(ioff),ind,ninseg(iseg))
153  endif
154  call swap(xp(ioff),work,ind,ninseg(iseg))
155  call swap(yp(ioff),work,ind,ninseg(iseg))
156  call iswap(locval(ioff),iwork,ind,ninseg(iseg))
157  ioff=ioff+ninseg(iseg)
158  enddo
159 ! Check for jumps in current coordinate
160  if (j == 1) then
161  do i = 2,ntot
162  if (abs(xp(i)-xp(i-1)) > xtol) ifseg(i)=.true.
163  enddo
164  else
165  do i = 2,ntot
166  if (abs(yp(i)-yp(i-1)) > xtol) ifseg(i)=.true.
167  enddo
168  endif
169 ! Count up number of different segments
170  nseg = 0
171  do i = 1,ntot
172  if (ifseg(i)) then
173  nseg = nseg+1
174  ninseg(nseg) = 1
175  else
176  ninseg(nseg) = ninseg(nseg) + 1
177  endif
178  enddo
179  enddo
180 !
181 ! Assign global node numbers (now sorted lexicographically!)
182 !
183  ig = 0
184  do i = 1,ntot
185  if (ifseg(i)) ig=ig+1
186  iglob(locval(i)) = ig
187  enddo
188 
189  nglob = ig
190 
191 ! %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
192 
193 ! get result in my format
194  do ispec= 1,nspec
195  ieoff = nxyz*(ispec - 1)
196  ilocnum = 0
197  do iy = 1,ngllx
198  do ix = 1,ngllx
199  ilocnum = ilocnum + 1
200  ibool(ix,iy,ispec) = iglob(ilocnum + ieoff)
201  enddo
202  enddo
203  enddo
204 
205  deallocate(locval)
206  deallocate(ind)
207  deallocate(ninseg)
208  deallocate(iglob)
209  deallocate(ifseg)
210  deallocate(xp)
211  deallocate(yp)
212  deallocate(work)
213  deallocate(iwork)
214 
215 ! check the numbering obtained
216  if (minval(ibool) /= 1 .or. maxval(ibool) /= nglob) call exit_mpi(myrank,'Error while generating global numbering')
217 
218  end subroutine createnum_fast

223  subroutine rank(A,IND,N)
224 !
225 ! Use Heap Sort (p 233 Numerical Recipes)
226 !
227  implicit none
228 
229  integer N
230  double precision A(n)
231  integer IND(n)
232 
233  integer i,j,l,ir,indx
234  double precision q
235 
236  do j = 1,n
237  ind(j)=j
238  enddo
239 
240  if (n == 1) return
241  l=n/2+1
242  ir=n
243  100 continue
244  if (l > 1) then
245  l=l-1
246  indx=ind(l)
247  q=a(indx)
248  ELSE
249  indx=ind(ir)
250  q=a(indx)
251  ind(ir)=ind(1)
252  ir=ir-1
253  if (ir == 1) then
254  ind(1)=indx
255  return
256  endif
257  endif
258  i=l
259  j=l+l
260  200 continue
261  if (j <= ir) then
262  if (j < ir) then
263  if (a(ind(j)) < a(ind(j+1))) j=j+1
264  endif
265  if (q < a(ind(j))) then
266  ind(i)=ind(j)
267  i=j
268  j=j+j
269  ELSE
270  j=ir+1
271  endif
272  goto 200
273  endif
274  ind(i)=indx
275  goto 100
276 
277  end subroutine rank
