34 ! for poroelastic solver: update memory variables with fourth-order Runge-Kutta time scheme for attenuation
35 
36  subroutine compute_attenuation_poro_fluid_part()
37 
38  use constants, only: zero,ngllx,ngllz,alpha_lddrk,beta_lddrk
39 
40  use specfem_par, only: nspec,ispec_is_poroelastic,poroelastcoef,kmato,permeability,ibool, &
41  velocw_poroelastic,time_stepping_scheme,deltat,i_stage,time_stepping_scheme, &
42  rx_viscous,rz_viscous,viscox,viscoz, &
43  rx_viscous_force_rk,rx_viscous_initial_rk,rz_viscous_force_rk,rz_viscous_initial_rk, &
44  rx_viscous_lddrk,rz_viscous_lddrk, &
45  alphaval,betaval,gammaval,theta_e,theta_s,thetainv,attenuation_poro_fluid_part
46 
47  implicit none
48 
49  ! local variables
50  integer :: i,j,ispec,iglob
51  double precision :: eta_f,permlxx,permlxz,permlzz,detk,invpermlxx,invpermlxz,invpermlzz, &
52  Sn,Snp1,weight_rk
53  double precision, dimension(3) :: bl_unrelaxed_elastic
54  double precision, dimension(NGLLX,NGLLZ) :: viscox_loc,viscoz_loc
55 
56  ! checks if anything to do
57  if (.not. attenuation_poro_fluid_part) return
58 
59  ! loop over spectral elements
60  do ispec = 1,nspec
61 
62  ! only for poroelastic elements
63  if (.not. ispec_is_poroelastic(ispec)) cycle
64 
65  ! fluid viscosity
66  eta_f = poroelastcoef(2,2,kmato(ispec))
67 
68  ! only if viscous
69  if (eta_f > 0.d0) then
70  permlxx = permeability(1,kmato(ispec))
71  permlxz = permeability(2,kmato(ispec))
72  permlzz = permeability(3,kmato(ispec))
73 
74  ! calcul of the inverse of k
75  detk = permlxx * permlzz - permlxz * permlxz
76  if (detk /= zero) then
77  invpermlxx = permlzz/detk
78  invpermlxz = -permlxz/detk
79  invpermlzz = permlxx/detk
80  else
81  stop 'Permeability matrix is not invertible'
82  endif
83 
84  ! relaxed viscous coef
85  bl_unrelaxed_elastic(1) = eta_f*invpermlxx
86  bl_unrelaxed_elastic(2) = eta_f*invpermlxz
87  bl_unrelaxed_elastic(3) = eta_f*invpermlzz
88 
89  do j = 1,ngllz
90  do i = 1,ngllx
91  iglob = ibool(i,j,ispec)
92  viscox_loc(i,j) = velocw_poroelastic(1,iglob) * bl_unrelaxed_elastic(1) + &
93  velocw_poroelastic(2,iglob) * bl_unrelaxed_elastic(2)
94  viscoz_loc(i,j) = velocw_poroelastic(1,iglob) * bl_unrelaxed_elastic(2) + &
95  velocw_poroelastic(2,iglob) * bl_unrelaxed_elastic(3)
96 
97  ! time stepping
98  select case (time_stepping_scheme)
99  case (1)
100  ! Newmark
101  ! evolution rx_viscous
102  sn = - (1.d0 - theta_e/theta_s)/theta_s*viscox(i,j,ispec)
103  snp1 = - (1.d0 - theta_e/theta_s)/theta_s*viscox_loc(i,j)
104  rx_viscous(i,j,ispec) = alphaval * rx_viscous(i,j,ispec) + betaval * sn + gammaval * snp1
105 
106  ! evolution rz_viscous
107  sn = - (1.d0 - theta_e/theta_s)/theta_s*viscoz(i,j,ispec)
108  snp1 = - (1.d0 - theta_e/theta_s)/theta_s*viscoz_loc(i,j)
109  rz_viscous(i,j,ispec) = alphaval * rz_viscous(i,j,ispec) + betaval * sn + gammaval * snp1
110 
111  case (2)
112  ! LDDRK
113  sn = - (1.d0 - theta_e/theta_s)/theta_s*viscox(i,j,ispec)
114  rx_viscous_lddrk(i,j,ispec) = alpha_lddrk(i_stage) * rx_viscous_lddrk(i,j,ispec) + &
115  deltat * (sn + thetainv * rx_viscous(i,j,ispec))
116  rx_viscous(i,j,ispec)= rx_viscous(i,j,ispec)+beta_lddrk(i_stage) * rx_viscous_lddrk(i,j,ispec)
117 
118  sn = - (1.d0 - theta_e/theta_s)/theta_s*viscoz(i,j,ispec)
119  rz_viscous_lddrk(i,j,ispec)= alpha_lddrk(i_stage) * rz_viscous_lddrk(i,j,ispec)+&
120  deltat * (sn + thetainv * rz_viscous(i,j,ispec))
121  rz_viscous(i,j,ispec)= rz_viscous(i,j,ispec)+beta_lddrk(i_stage) * rz_viscous_lddrk(i,j,ispec)
122 
123  case (3)
124  ! Runge-Kutta
125  ! x-component
126  sn = - (1.d0 - theta_e/theta_s)/theta_s*viscox(i,j,ispec)
127  rx_viscous_force_rk(i,j,ispec,i_stage) = deltat * (sn + thetainv * rx_viscous(i,j,ispec))
128 
129  if (i_stage == 1 .or. i_stage == 2 .or. i_stage == 3) then
130  if (i_stage == 1) weight_rk = 0.5d0
131  if (i_stage == 2) weight_rk = 0.5d0
132  if (i_stage == 3) weight_rk = 1.0d0
133 
134  if (i_stage == 1) then
135  rx_viscous_initial_rk(i,j,ispec) = rx_viscous(i,j,ispec)
136  endif
137  rx_viscous(i,j,ispec) = rx_viscous_initial_rk(i,j,ispec) + weight_rk * rx_viscous_force_rk(i,j,ispec,i_stage)
138 
139  else if (i_stage == 4) then
140  rx_viscous(i,j,ispec) = rx_viscous_initial_rk(i,j,ispec) + &
141  1.0d0 / 6.0d0 * (rx_viscous_force_rk(i,j,ispec,i_stage) + &
142  2.0d0 * rx_viscous_force_rk(i,j,ispec,i_stage) + &
143  2.0d0 * rx_viscous_force_rk(i,j,ispec,i_stage) + &
144  rx_viscous_force_rk(i,j,ispec,i_stage))
145  endif
146 
147  ! z-component
148  sn = - (1.d0 - theta_e/theta_s)/theta_s*viscoz(i,j,ispec)
149  rz_viscous_force_rk(i,j,ispec,i_stage) = deltat * (sn + thetainv * rz_viscous(i,j,ispec))
150 
151  if (i_stage == 1 .or. i_stage == 2 .or. i_stage == 3) then
152  if (i_stage == 1) weight_rk = 0.5d0
153  if (i_stage == 2) weight_rk = 0.5d0
154  if (i_stage == 3) weight_rk = 1.0d0
155 
156  if (i_stage == 1) then
157  rz_viscous_initial_rk(i,j,ispec) = rz_viscous(i,j,ispec)
158  endif
159  rz_viscous(i,j,ispec) = rz_viscous_initial_rk(i,j,ispec) + weight_rk * rz_viscous_force_rk(i,j,ispec,i_stage)
160 
161  else if (i_stage == 4) then
162  rz_viscous(i,j,ispec) = rz_viscous_initial_rk(i,j,ispec) + &
163  1.0d0 / 6.0d0 * (rz_viscous_force_rk(i,j,ispec,i_stage) + &
164  2.0d0 * rz_viscous_force_rk(i,j,ispec,i_stage) + &
165  2.0d0 * rz_viscous_force_rk(i,j,ispec,i_stage) + &
166  rz_viscous_force_rk(i,j,ispec,i_stage))
167  endif
168 
169  case default
170  stop 'Time stepping scheme not implemented yet for poro_fluid attenuation'
171  end select
172  enddo
173  enddo
174 
175  if (time_stepping_scheme == 1) then
176  ! Newmark
177  ! save visco for Runge-Kutta scheme when used together with Newmark
178  viscox(:,:,ispec) = viscox_loc(:,:)
179  viscoz(:,:,ispec) = viscoz_loc(:,:)
180  endif
181 
182  endif ! viscous element
183 
184  enddo ! end of spectral element loop
185 
186  end subroutine compute_attenuation_poro_fluid_part
