33 
34  subroutine write_movie_output()
35 
36 ! outputs snapshots for movies
37 
38  use constants, only: max_string_len,custom_real,ndim,noise_movie_output,output_files
39 
40  use specfem_par, only: myrank,it,nstep,nspec,nglob,ibool, &
41  potential_acoustic,potential_dot_acoustic,potential_dot_dot_acoustic, &
42  displ_elastic,veloc_elastic,accel_elastic, &
43  b_displ_elastic,rho_k,rho_kl, &
44  any_acoustic,any_elastic,gpu_mode,p_sv
45 
46  use specfem_par_gpu, only: mesh_pointer,tmp_displ_2d,tmp_veloc_2d,tmp_accel_2d,nglob_ab
47 
48  use specfem_par_noise, only: noise_tomography,mask_noise, &
49  surface_movie_y_or_z_noise,noise_output_rhokl,noise_output_array,noise_output_ncol
50 
51  use specfem_par_movie, only: output_postscript_snapshot,output_color_image,output_wavefield_dumps, &
52  nstep_between_output_images
53 
54  implicit none
55 
56  ! local parameters
57  integer :: ier
58  logical :: ex, is_opened
59  character(len=MAX_STRING_LEN) :: noise_output_file
60 
61  ! checks if anything to do
62  if (.not. (mod(it,nstep_between_output_images) == 0 .or. it == 5 .or. it == nstep)) return
63 
64  ! transfers arrays from GPU to CPU
65  if (gpu_mode) then
66  ! Fields transfer for imaging
67  ! acoustic domains
68  if (any_acoustic ) then
69  call transfer_fields_ac_from_device(nglob_ab,potential_acoustic,potential_dot_acoustic, &
70  potential_dot_dot_acoustic,mesh_pointer)
71  endif
72  ! elastic domains
73  if (any_elastic) then
74  call transfer_fields_el_from_device(ndim*nglob_ab,tmp_displ_2d,tmp_veloc_2d,tmp_accel_2d,mesh_pointer)
75  if (p_sv) then
76  ! P-SV waves
77  displ_elastic(1,:) = tmp_displ_2d(1,:)
78  displ_elastic(2,:) = tmp_displ_2d(2,:)
79  veloc_elastic(1,:) = tmp_veloc_2d(1,:)
80  veloc_elastic(2,:) = tmp_veloc_2d(2,:)
81  accel_elastic(1,:) = tmp_accel_2d(1,:)
82  accel_elastic(2,:) = tmp_accel_2d(2,:)
83  else
84  ! SH waves
85  displ_elastic(1,:) = tmp_displ_2d(1,:)
86  veloc_elastic(1,:) = tmp_veloc_2d(1,:)
87  accel_elastic(1,:) = tmp_accel_2d(1,:)
88  endif
89  endif
90  endif
91 
92  ! noise simulations
93  if (.not. gpu_mode) then
94  if (noise_tomography == 3 .and. noise_movie_output) then
95  ! load ensemble forward source
96  inquire(unit=501,exist=ex,opened=is_opened)
97  if (.not. is_opened) then
98  open(unit=501,file=trim(output_files)//'noise_eta.bin',access='direct', &
99  recl=nglob*custom_real,action='read',iostat=ier)
100  if (ier /= 0) call exit_mpi(myrank,'Error opening noise eta file')
101  endif
102  ! for both P_SV/SH cases
103  read(unit=501,rec=it) surface_movie_y_or_z_noise ! either y (SH) or z (P_SV) component
104 
105  ! load product of fwd, adj wavefields
106  call spec2glob(nspec,nglob,ibool,rho_kl,noise_output_rhokl)
107 
108  ! prepares array
109  ! noise distribution
110  noise_output_array(1,:) = surface_movie_y_or_z_noise(:) * mask_noise(:)
111 
112  ! P_SV/SH-case
113  noise_output_array(2,:) = b_displ_elastic(1,:)
114  noise_output_array(3,:) = accel_elastic(1,:)
115 
116  ! rho kernel on global nodes
117  noise_output_array(4,:) = rho_k(:)
118 
119  ! rho kernel on global nodes from local kernel (for comparison)
120  noise_output_array(5,:) = noise_output_rhokl(:)
121 
122  ! writes out to text file
123  write(noise_output_file,"(a,i6.6,a)") trim(output_files)//'noise_snapshot_all_',it,'.txt'
124  call snapshots_noise(noise_output_ncol,nglob,noise_output_file,noise_output_array)
125 
126  ! re-initializes noise array
127  surface_movie_y_or_z_noise(:) = 0._custom_real
128  endif
129  endif
130 
131  ! output_postscript_snapshot
132  if (output_postscript_snapshot) then
133  call write_postscript_snapshot()
134  endif
135 
136  ! display color image
137  if (output_color_image) then
138  call write_color_image_snaphot()
139  endif
140 
141  ! dump the full (local) wavefield to a file
142  ! note: in the case of MPI, in the future it would be more convenient to output a single file
143  ! rather than one for each myrank
144  if (output_wavefield_dumps) then
145  call write_wavefield_dumps()
146  endif
147 
148  end subroutine write_movie_output
