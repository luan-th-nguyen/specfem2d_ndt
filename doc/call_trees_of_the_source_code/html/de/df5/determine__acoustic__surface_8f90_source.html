33 
34  subroutine determine_acoustic_surface()
35 
36  use constants, only: anisotropic_material,tinyval
37 
38  use part_unstruct_par, only: nelem_acoustic_surface,acoustic_surface, &
39  nxread,nzread,elmnts
40 
41  use shared_parameters, only: axisym,ngnod,num_material,icodemat,phi_read,xmin_param, &
42  absorbbottom,absorbleft,absorbright,absorbtop
43 
44  implicit none
45 
46  ! local parameters
47  integer :: i,j,ier
48  integer :: imaterial_number
49 
50  ! count the number of acoustic free-surface elements
51  nelem_acoustic_surface = 0
52 
53  ! if the surface is absorbing, it cannot be free at the same time
54  if (.not. absorbtop) then
55  j = nzread
56  do i = 1,nxread
57  imaterial_number = num_material((j-1)*nxread+i)
58  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
59  nelem_acoustic_surface = nelem_acoustic_surface + 1
60  endif
61  enddo
62  endif
63  if (.not. absorbbottom) then
64  j = 1
65  do i = 1,nxread
66  imaterial_number = num_material((j-1)*nxread+i)
67  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
68  nelem_acoustic_surface = nelem_acoustic_surface + 1
69  endif
70  enddo
71  endif
72  ! in the axisymmetric case if xmin == 0 the axis is a symmetry axis and thus cannot be a free surface as well
73  if (.not. absorbleft .and. .not. (axisym .and. abs(xmin_param) < tinyval)) then
74  i = 1
75  do j = 1,nzread
76  imaterial_number = num_material((j-1)*nxread+i)
77  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
78  nelem_acoustic_surface = nelem_acoustic_surface + 1
79  endif
80  enddo
81  endif
82  if (.not. absorbright) then
83  i = nxread
84  do j = 1,nzread
85  imaterial_number = num_material((j-1)*nxread+i)
86  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
87  nelem_acoustic_surface = nelem_acoustic_surface + 1
88  endif
89  enddo
90  endif
91 
92  ! allocates surface elements
93  allocate(acoustic_surface(4,nelem_acoustic_surface),stat=ier)
94  if (ier /= 0) stop 'Error allocating acoustic_surface array'
95 
96  nelem_acoustic_surface = 0
97 
98  if (.not. absorbtop) then
99  j = nzread
100  do i = 1,nxread
101  imaterial_number = num_material((j-1)*nxread+i)
102  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
103  nelem_acoustic_surface = nelem_acoustic_surface + 1
104  acoustic_surface(1,nelem_acoustic_surface) = (j-1)*nxread + (i-1)
105  acoustic_surface(2,nelem_acoustic_surface) = 2
106  acoustic_surface(3,nelem_acoustic_surface) = elmnts(3+ngnod*((j-1)*nxread+i-1))
107  acoustic_surface(4,nelem_acoustic_surface) = elmnts(2+ngnod*((j-1)*nxread+i-1))
108  endif
109  enddo
110  endif
111  if (.not. absorbbottom) then
112  j = 1
113  do i = 1,nxread
114  imaterial_number = num_material((j-1)*nxread+i)
115  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
116  nelem_acoustic_surface = nelem_acoustic_surface + 1
117  acoustic_surface(1,nelem_acoustic_surface) = (j-1)*nxread + (i-1)
118  acoustic_surface(2,nelem_acoustic_surface) = 2
119  acoustic_surface(3,nelem_acoustic_surface) = elmnts(0+ngnod*((j-1)*nxread+i-1))
120  acoustic_surface(4,nelem_acoustic_surface) = elmnts(1+ngnod*((j-1)*nxread+i-1))
121  endif
122  enddo
123  endif
124  ! in the axisymmetric case if xmin == 0 the axis is a symmetry axis and thus cannot be a free surface as well
125  if (.not. absorbleft .and. .not. (axisym .and. abs(xmin_param) < tinyval)) then
126  i = 1
127  do j = 1,nzread
128  imaterial_number = num_material((j-1)*nxread+i)
129  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
130  nelem_acoustic_surface = nelem_acoustic_surface + 1
131  acoustic_surface(1,nelem_acoustic_surface) = (j-1)*nxread + (i-1)
132  acoustic_surface(2,nelem_acoustic_surface) = 2
133  acoustic_surface(3,nelem_acoustic_surface) = elmnts(0+ngnod*((j-1)*nxread+i-1))
134  acoustic_surface(4,nelem_acoustic_surface) = elmnts(3+ngnod*((j-1)*nxread+i-1))
135  endif
136  enddo
137  endif
138  if (.not. absorbright) then
139  i = nxread
140  do j = 1,nzread
141  imaterial_number = num_material((j-1)*nxread+i)
142  if (icodemat(imaterial_number) /= anisotropic_material .and. phi_read(imaterial_number) >= 1.d0) then
143  nelem_acoustic_surface = nelem_acoustic_surface + 1
144  acoustic_surface(1,nelem_acoustic_surface) = (j-1)*nxread + (i-1)
145  acoustic_surface(2,nelem_acoustic_surface) = 2
146  acoustic_surface(3,nelem_acoustic_surface) = elmnts(1+ngnod*((j-1)*nxread+i-1))
147  acoustic_surface(4,nelem_acoustic_surface) = elmnts(2+ngnod*((j-1)*nxread+i-1))
148  endif
149  enddo
150  endif
151 
152  end subroutine determine_acoustic_surface
