35  subroutine set_source_parameters()
36 
37 ! gets source parameters and determines simulation start/onset time
38 
39  use constants, only: ngllx,ngllz,ndim,imain,tinyval,pi,f_ormsby
40 
41  use specfem_par, only: myrank,nsources,source_type,time_function_type, &
42  x_source,z_source,mxx,mzz,mxz,f0_source,tshift_src,factor,anglesource, &
43  t0,initialfield,user_t0
44 
45  implicit none
46 
47  ! local parameters
48  integer :: i_source
49  double precision, dimension(NSOURCES) :: t0_source
50  double precision :: min_tshift_src_original,hdur
51 
52  ! note: see also routine in read_source_file.f90 which reads in source parameters
53 
54  ! checks the input
55  do i_source = 1,nsources
56 
57  ! half-duration of the marmousi_ormsby_wavelet whose freq signature is 5-10-60-80
58  ! so here we choose 35Hz as dominant frequency
59  if (time_function_type(i_source) == 11) then
60  ! warning
61  if (f0_source(i_source) /= f_ormsby) then
62  print *,'warning: Marmousi Ormsby wavelet dominant frequency will be set to 35 Hz'
63  f0_source(i_source) = f_ormsby
64  endif
65  endif
66 
67  ! checks source type
68  if (.not. initialfield) then
69  if (source_type(i_source) == 1) then
70  ! force
71  if (myrank == 0) then
72  ! user output
73  write(imain,212) x_source(i_source),z_source(i_source),f0_source(i_source),tshift_src(i_source), &
74  factor(i_source),anglesource(i_source)
75  endif
76  else if (source_type(i_source) == 2) then
77  ! moment tensor
78  if (myrank == 0) then
79  ! user output
80  write(imain,222) x_source(i_source),z_source(i_source),f0_source(i_source),tshift_src(i_source), &
81  factor(i_source),mxx(i_source),mzz(i_source),mxz(i_source)
82  endif
83  else
84  call exit_mpi(myrank,'Unknown source type number !')
85  endif
86  endif
87 
88  ! half-duration of source
89  hdur = 1.d0 / f0_source(i_source)
90 
91  ! sets source start times, shifted by the given (non-zero) time-shift
92  if (time_function_type(i_source) == 5 .or. time_function_type(i_source) == 11) then
93  ! Heaviside or Ormsby
94  t0_source(i_source) = 2.0d0 * hdur + tshift_src(i_source)
95  else
96  t0_source(i_source) = 1.20d0 * hdur + tshift_src(i_source)
97  endif
98 
99  enddo ! do i_source= 1,NSOURCES
100 
101  ! initializes simulation start time
102  if (nsources == 1) then
103  ! simulation start time
104  t0 = t0_source(1)
105  ! sets source time shift relative to simulation start time
106  min_tshift_src_original = tshift_src(1)
107  tshift_src(1) = 0.d0
108  else
109  ! starts with earliest start time
110  t0 = minval( t0_source(:) )
111  ! sets source time shifts relative to simulation start time
112  min_tshift_src_original = minval( tshift_src(:) )
113  tshift_src(:) = t0_source(:) - t0
114  endif
115 
116  ! checks if user set USER_T0 to fix simulation start time
117  ! note: USER_T0 has to be positive
118  if (user_t0 > 0.d0) then
119  ! user cares about origin time and time shifts of the CMTSOLUTION
120  ! and wants to fix simulation start time to a constant start time
121  ! time 0 on time axis will correspond to given origin time
122 
123  ! notifies user
124  if (myrank == 0) then
125  write(imain,*)
126  write(imain,*) ' using USER_T0 . . . . . . . . . = ',user_t0
127  write(imain,*) ' original t0 . . . . . . . . . = ',t0
128  write(imain,*) ' min_tshift_src_original . . . = ',min_tshift_src_original
129  write(imain,*)
130  call flush_imain()
131  endif
132 
133  ! checks if automatically set t0 is too small
134  ! note: times in seismograms are shifted by t0(1)
135  if (t0 <= user_t0 + min_tshift_src_original) then
136 
137  ! sets new simulation start time such that
138  ! simulation starts at t = - t0 = - USER_T0
139  t0 = user_t0
140 
141  ! notifies user
142  if (myrank == 0) then
143  write(imain,*) ' fix new simulation start time . = ', - t0
144  endif
145 
146  ! loops over all sources
147  do i_source = 1,nsources
148  ! half-duration of source
149  hdur = 1.d0 / f0_source(i_source)
150 
151  ! sets the given, initial time shifts
152  if (time_function_type(i_source) == 5) then
153  tshift_src(i_source) = t0_source(i_source) - 2.0d0 * hdur
154  else
155  tshift_src(i_source) = t0_source(i_source) - 1.20d0 * hdur
156  endif
157  ! user output
158  if (myrank == 0) then
159  write(imain,*) ' source ',i_source,'uses tshift = ',tshift_src(i_source)
160  endif
161  enddo
162  ! user output
163  if (myrank == 0) then
164  write(imain,*)
165  endif
166 
167  else
168  ! start time needs to be at least t0 for numerical stability
169  ! notifies user
170  if (myrank == 0) then
171  write(imain,*) 'Error: USER_T0 is too small'
172  write(imain,*) ' must make one of three adjustements:'
173  write(imain,*) ' - increase USER_T0 to be at least: ',t0
174  write(imain,*) ' - decrease time shift tshift_src in SOURCE file'
175  write(imain,*) ' - increase frequency f0 in SOURCE file'
176  endif
177  call exit_mpi(myrank,'Error USER_T0 is set but too small')
178  endif
179  else if (user_t0 < 0.d0) then
180  if (myrank == 0) then
181  write(imain,*) 'error: USER_T0 is negative, must be set zero or positive!'
182  endif
183  call exit_mpi(myrank,'Error negative USER_T0 parameter in constants.h')
184  endif
185 
186  ! checks onset times
187  if (.not. initialfield) then
188 
189  ! loops over sources
190  do i_source = 1,nsources
191 
192  ! excludes Dirac and Heaviside sources
193  if (time_function_type(i_source) /= 4 .and. time_function_type(i_source) /= 5) then
194 
195  ! user output
196  if (myrank == 0) then
197  write(imain,*) ' Onset time. . . . . . = ',- (t0+tshift_src(i_source))
198  write(imain,*) ' Fundamental period. . = ',1.d0/f0_source(i_source)
199  write(imain,*) ' Fundamental frequency = ',f0_source(i_source)
200  endif
201 
202  ! checks source onset time
203  if (t0+tshift_src(i_source) < 1.d0/f0_source(i_source)) then
204  call exit_mpi(myrank,'Onset time too small')
205  else
206  if (myrank == 0) then
207  write(imain,*) ' The onset time is ok'
208  endif
209  endif
210  endif
211  enddo
212 
213  endif
214 
215 
216  ! output formats
217 212 format(//,5x,'Source Type. . . . . . . . . . . . . . = Collocated Force',/5x, &
218  'X-position (meters). . . . . . . . . . =',1pe20.10,/5x, &
219  'Y-position (meters). . . . . . . . . . =',1pe20.10,/5x, &
220  'Fundamental frequency (Hz) . . . . . . =',1pe20.10,/5x, &
221  'Time delay (s) . . . . . . . . . . . . =',1pe20.10,/5x, &
222  'Multiplying factor . . . . . . . . . . =',1pe20.10,/5x, &
223  'Angle from vertical direction (deg). . =',1pe20.10,/5x)
224 
225 222 format(//,5x,'Source Type. . . . . . . . . . . . . . = Moment-tensor',/5x, &
226  'X-position (meters). . . . . . . . . . =',1pe20.10,/5x, &
227  'Y-position (meters). . . . . . . . . . =',1pe20.10,/5x, &
228  'Fundamental frequency (Hz) . . . . . . =',1pe20.10,/5x, &
229  'Time delay (s) . . . . . . . . . . . . =',1pe20.10,/5x, &
230  'Multiplying factor . . . . . . . . . . =',1pe20.10,/5x, &
231  'Mxx. . . . . . . . . . . . . . . . . . =',1pe20.10,/5x, &
232  'Mzz. . . . . . . . . . . . . . . . . . =',1pe20.10,/5x, &
233  'Mxz. . . . . . . . . . . . . . . . . . =',1pe20.10)
234 
235  end subroutine set_source_parameters
