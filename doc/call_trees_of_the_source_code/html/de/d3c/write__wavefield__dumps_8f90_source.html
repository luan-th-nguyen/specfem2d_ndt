33 
34  subroutine write_wavefield_dumps()
35 
36  use constants, only: imain,size_real,ngllx,ngllz,output_files
37 
38  use specfem_par, only: myrank,nglob,nspec, &
39  ibool,coord,p_sv,it,simulation_type, &
40  potential_acoustic,potential_gravitoacoustic, &
41  potential_gravito,displ_elastic,displs_poroelastic, &
42  potential_dot_acoustic,veloc_elastic,velocs_poroelastic, &
43  potential_dot_dot_acoustic,accel_elastic,accels_poroelastic
44 
45  use specfem_par_movie, only: this_is_the_first_time_we_dump,mask_ibool,imagetype_wavefield_dumps, &
46  use_binary_for_wavefield_dumps,vector_field_display
47 
48  implicit none
49 
50  !local variables
51  integer :: i,j,ispec,iglob,icounter,nb_of_values_to_save
52  integer :: ier
53  ! name of wavefield snapshot file
54  character(len=150) :: wavefield_file
55 
56  if (myrank == 0) then
57  write(imain,*)
58  write(imain,*) 'Dumping the wave field to a file for time step ',it
59  call flush_imain()
60  endif
61 
62  if (this_is_the_first_time_we_dump) then
63 
64  if (.not. allocated(mask_ibool)) allocate(mask_ibool(nglob))
65 
66 ! save the grid separately once and for all
67  if (use_binary_for_wavefield_dumps) then
68  write(wavefield_file,"(a,i3.3,a)") trim(output_files)//'wavefield_grid_for_dumps_',myrank,'.bin'
69  open(unit=27,file=wavefield_file,form='unformatted',access='direct',status='unknown', &
70  action='write',recl=2*size_real,iostat=ier)
71  if (ier /= 0) call exit_mpi(myrank,'Error opening file wavefield_grid_for_dumps_**.bin')
72  else
73  write(wavefield_file,"(a,i3.3,a)") trim(output_files)//'wavefield_grid_for_dumps_',myrank,'.txt'
74  open(unit=27,file=wavefield_file,status='unknown',action='write',iostat=ier)
75  if (ier /= 0) call exit_mpi(myrank,'Error opening file wavefield_grid_for_dumps_**.txt')
76  endif
77 
78  icounter = 0
79  mask_ibool(:) = .false.
80  do ispec = 1,nspec
81  do j = 1,ngllz
82  do i = 1,ngllx
83  iglob = ibool(i,j,ispec)
84  if (.not. mask_ibool(iglob)) then
85  icounter = icounter + 1
86  mask_ibool(iglob) = .true.
87  if (use_binary_for_wavefield_dumps) then
88  write(27,rec=icounter) sngl(coord(1,iglob)),sngl(coord(2,iglob))
89  else
90  write(27,'(2e16.6)') coord(1,iglob),coord(2,iglob)
91  endif
92  endif
93  enddo
94  enddo
95  enddo
96 
97  close(27)
98 
99  ! save nglob to a file once and for all
100  write(wavefield_file,"(a,i3.3,a)") trim(output_files)//'wavefield_grid_value_of_nglob_',myrank,'.txt'
101  open(unit=27,file=wavefield_file,status='unknown',action='write',iostat=ier)
102  if (ier /= 0) call exit_mpi(myrank,'Error opening file wavefield_grid_value_of_nglob_**.txt')
103  write(27,*) icounter
104  close(27)
105  if (icounter /= nglob) stop 'error: should have icounter == nglob in wavefield dumps'
106 
107  this_is_the_first_time_we_dump = .false.
108 
109  endif
110 
111  if (imagetype_wavefield_dumps == 1) then
112  if (myrank == 0) write(imain,*) 'dumping the displacement vector...'
113  call compute_vector_whole_medium(potential_acoustic,potential_gravitoacoustic, &
114  potential_gravito,displ_elastic,displs_poroelastic)
115  else if (imagetype_wavefield_dumps == 2) then
116  if (myrank == 0) write(imain,*) 'dumping the velocity vector...'
117  call compute_vector_whole_medium(potential_dot_acoustic,potential_gravitoacoustic, &
118  potential_gravito,veloc_elastic,velocs_poroelastic)
119 
120  else if (imagetype_wavefield_dumps == 3) then
121  if (myrank == 0) write(imain,*) 'dumping the acceleration vector...'
122  call compute_vector_whole_medium(potential_dot_dot_acoustic,potential_gravitoacoustic, &
123  potential_gravito,accel_elastic,accels_poroelastic)
124 
125  else if (imagetype_wavefield_dumps == 4 .and. p_sv) then
126  if (myrank == 0) write(imain,*) 'dumping the pressure field...'
127  call compute_pressure_whole_medium()
128 
129  else if (imagetype_wavefield_dumps == 4 .and. .not. p_sv) then
130  call exit_mpi(myrank,'cannot dump the pressure field for SH (membrane) waves')
131 
132  else
133  call exit_mpi(myrank,'wrong type of flag for wavefield dumping')
134  endif
135 
136  if (use_binary_for_wavefield_dumps) then
137  if (p_sv .and. .not. imagetype_wavefield_dumps == 4) then
138  nb_of_values_to_save = 2
139  else
140  nb_of_values_to_save = 1
141  endif
142  write(wavefield_file,"(a,i7.7,'_',i2.2,'_',i3.3,a)") trim(output_files)//'wavefield',it,simulation_type,myrank,'.bin'
143  open(unit=27,file=wavefield_file,form='unformatted',access='direct',status='unknown', &
144  action='write',recl=nb_of_values_to_save*size_real,iostat=ier)
145  if (ier /= 0) call exit_mpi(myrank,'Error opening file wavefield**.bin')
146  else
147  write(wavefield_file,"(a,i7.7,'_',i2.2,'_',i3.3,a)") trim(output_files)//'wavefield',it,simulation_type,myrank,'.txt'
148  open(unit=27,file=wavefield_file,status='unknown',action='write',iostat=ier)
149  if (ier /= 0) call exit_mpi(myrank,'Error opening file wavefield**.txt')
150  endif
151 
152  icounter = 0
153  mask_ibool(:) = .false.
154  do ispec = 1,nspec
155  do j = 1,ngllz
156  do i = 1,ngllx
157  iglob = ibool(i,j,ispec)
158  if (.not. mask_ibool(iglob)) then
159  icounter = icounter + 1
160  mask_ibool(iglob) = .true.
161  if (use_binary_for_wavefield_dumps) then
162  if (p_sv) then
163  ! P-SV waves
164  if (imagetype_wavefield_dumps == 4) then
165  ! by convention we use the 2. component of the array to store the pressure above
166  write(27,rec=icounter) sngl(vector_field_display(2,iglob))
167  else
168  write(27,rec=icounter) sngl(vector_field_display(1,iglob)),sngl(vector_field_display(2,iglob))
169  endif
170  else
171  ! SH case
172  write(27,rec=icounter) sngl(vector_field_display(1,iglob))
173  endif
174  else
175  if (p_sv) then
176  ! P-SV waves
177  if (imagetype_wavefield_dumps == 4) then
178  ! by convention we use the 2. component of the array to store the pressure above
179  write(27,*) sngl(vector_field_display(2,iglob))
180  else
181  write(27,*) sngl(vector_field_display(1,iglob)),sngl(vector_field_display(2,iglob))
182  endif
183  else
184  ! SH case
185  write(27,*) sngl(vector_field_display(1,iglob))
186  endif
187  endif
188  endif
189  enddo
190  enddo
191  enddo
192 
193  close(27)
194 
195  if (myrank == 0) then
196  write(imain,*) 'Wave field dumped'
197  call flush_imain()
198  endif
199 
200  end subroutine write_wavefield_dumps
