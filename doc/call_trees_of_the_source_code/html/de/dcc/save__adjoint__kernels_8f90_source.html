33 
34  subroutine save_adjoint_kernels()
35 
36 ! saves adjoint sensitivity kernels to file
37 
38  use constants, only: ngllx,ngllz,imain,approximate_hess_kl
39 
40  use specfem_par, only: myrank, nspec, ibool, coord, save_ascii_kernels, &
41  any_acoustic, any_elastic, any_poroelastic, &
42  rho_ac_kl, kappa_ac_kl, alpha_ac_kl, rhop_ac_kl, &
43  rho_kl, kappa_kl, mu_kl, rhop_kl, alpha_kl, beta_kl, &
44  bulk_c_kl, bulk_beta_kl, &
45  rhorho_ac_hessian_final1, rhorho_ac_hessian_final2, &
46  rhorho_el_hessian_final1, rhorho_el_hessian_final2, &
47  rhot_kl, rhof_kl, sm_kl, eta_kl, mufr_kl, b_kl, &
48  c_kl, m_kl, rhob_kl, rhofb_kl, phi_kl, mufrb_kl, &
49  rhobb_kl, rhofbb_kl, phib_kl, cpi_kl, cpii_kl, cs_kl, ratio_kl, gpu_mode
50 
51  use specfem_par, only: ispec_is_anisotropic, c11_kl, c13_kl, c15_kl, c33_kl, c35_kl, c55_kl
52 
53  implicit none
54 
55  integer :: i, j, ispec, iglob
56  double precision :: xx, zz
57 
58  ! user output
59  if (myrank == 0) then
60  write(imain,*)
61  write(imain,*) 'Writing Kernels file'
62  write(imain,*)
63  call flush_imain()
64  endif
65 
66  if (any_acoustic) then
67  ! acoustic domain
68  if (save_ascii_kernels) then
69  ! ascii format
70  do ispec = 1, nspec
71  do j = 1, ngllz
72  do i = 1, ngllx
73  iglob = ibool(i,j,ispec)
74  xx = coord(1,iglob)
75  zz = coord(2,iglob)
76  write(95,'(4e15.5e4)') xx,zz,rho_ac_kl(i,j,ispec),kappa_ac_kl(i,j,ispec)
77  write(96,'(4e15.5e4)') xx,zz,rhop_ac_kl(i,j,ispec),alpha_ac_kl(i,j,ispec)
78  enddo
79  enddo
80  enddo
81  close(95)
82  close(96)
83  else
84  ! binary format
85  write(200) rho_ac_kl
86  write(201) kappa_ac_kl
87  write(202) rhop_ac_kl
88  write(203) alpha_ac_kl
89  close(200)
90  close(201)
91  close(202)
92  close(203)
93  ! Hessian kernels
94  if (approximate_hess_kl) then
95  write(212) rhorho_ac_hessian_final1
96  write(213) rhorho_ac_hessian_final2
97  close(212)
98  close(213)
99  endif
100  endif
101  endif
102 
103  if (any_elastic) then
104  ! elastic domains
105  if (save_ascii_kernels) then
106  ! ascii format
107  do ispec = 1, nspec
108  do j = 1, ngllz
109  do i = 1, ngllx
110  iglob = ibool(i,j,ispec)
111  xx = coord(1,iglob)
112  zz = coord(2,iglob)
113  if (count(ispec_is_anisotropic(:) .eqv. .true.) >= 1) then
114  ! anisotropic
115  write(97,'(9e15.5e4)') xx, zz, rho_kl(i,j,ispec), c11_kl(i,j,ispec), &
116  c13_kl(i,j,ispec), c15_kl(i,j,ispec), c33_kl(i,j,ispec), c35_kl(i,j,ispec), &
117  c55_kl(i,j,ispec)
118  else
119  ! isotropic
120  ! parameterization (rho,kappa,mu) "primary" kernels
121  write(97,'(5e15.5e4)') xx,zz,rho_kl(i,j,ispec),kappa_kl(i,j,ispec),mu_kl(i,j,ispec)
122  ! parameterization (rho,Vp,Vs)
123  write(98,'(5e15.5e4)') xx,zz,rhop_kl(i,j,ispec),alpha_kl(i,j,ispec),beta_kl(i,j,ispec)
124  endif
125  enddo
126  enddo
127  enddo
128  close(97)
129  close(98)
130  else
131  ! binary format
132  write(204) rho_kl
133  if (count(ispec_is_anisotropic(:) .eqv. .true.) >= 1) then ! anisotropic
134  write(205) c11_kl
135  write(206) c13_kl
136  write(207) c15_kl
137  write(208) c33_kl
138  write(209) c35_kl
139  write(210) c55_kl
140  else
141  write(205) kappa_kl
142  write(206) mu_kl
143  write(207) rhop_kl
144  write(208) alpha_kl
145  write(209) beta_kl
146  write(210) bulk_c_kl
147  write(211) bulk_beta_kl
148  close(211)
149  endif
150  close(204)
151  close(205)
152  close(202)
153  close(207)
154  close(208)
155  close(209)
156  close(210)
157  ! Hessian kernels
158  if (approximate_hess_kl) then
159  write(214) rhorho_el_hessian_final1
160  write(215) rhorho_el_hessian_final2
161  close(214)
162  close(215)
163  endif
164  endif
165  endif
166 
167  if (any_poroelastic) then
168  ! poro-elastic domains
169  ! checks
170  if (gpu_mode) stop 'poroelastic kernel output not implemented on GPUs yet'
171  if (.not. save_ascii_kernels) stop 'poroelastic simulations must use SAVE_ASCII_KERNELS'
172  ! ascii format
173  do ispec = 1, nspec
174  do j = 1, ngllz
175  do i = 1, ngllx
176  iglob = ibool(i,j,ispec)
177  xx = coord(1,iglob)
178  zz = coord(2,iglob)
179  write(144,'(5e11.3)') xx,zz,mufr_kl(i,j,ispec),b_kl(i,j,ispec),c_kl(i,j,ispec)
180  write(155,'(5e11.3)') xx,zz,m_kl(i,j,ispec),rhot_kl(i,j,ispec),rhof_kl(i,j,ispec)
181 
182  write(16,'(5e11.3)') xx,zz,sm_kl(i,j,ispec),eta_kl(i,j,ispec)
183 
184  write(17,'(5e11.3)') xx,zz,mufrb_kl(i,j,ispec),b_kl(i,j,ispec),c_kl(i,j,ispec)
185  write(18,'(5e11.3)') xx,zz,m_kl(i,j,ispec),rhob_kl(i,j,ispec),rhofb_kl(i,j,ispec)
186 
187  write(19,'(5e11.3)') xx,zz,phi_kl(i,j,ispec),eta_kl(i,j,ispec)
188  write(20,'(5e11.3)') xx,zz,cpi_kl(i,j,ispec),cpii_kl(i,j,ispec),cs_kl(i,j,ispec)
189  write(21,'(5e11.3)') xx,zz,rhobb_kl(i,j,ispec),rhofbb_kl(i,j,ispec),ratio_kl(i,j,ispec)
190  write(22,'(5e11.3)') xx,zz,phib_kl(i,j,ispec),eta_kl(i,j,ispec)
191  enddo
192  enddo
193  enddo
194  close(144)
195  close(155)
196  close(16)
197  close(17)
198  close(18)
199  close(19)
200  close(20)
201  close(21)
202  close(22)
203  endif
204 
205  end subroutine save_adjoint_kernels
