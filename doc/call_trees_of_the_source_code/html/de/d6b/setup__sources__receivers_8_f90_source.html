70  subroutine setup_sources()
71 
72  use constants, only: ngllx,ngllz,ndim,imain,iin, &
73 #ifndef use_mpi
74  output_files, &
75 #endif
76  max_string_len
77 
78  use specfem_par, only: nsources,initialfield,source_type, &
79  coord,ibool,nglob,nspec,nelem_acoustic_surface,acoustic_surface, &
80  ispec_is_elastic,ispec_is_poroelastic, &
81  x_source,z_source,ispec_selected_source, &
82  islice_selected_source, &
83  xigll,zigll,npgeo, &
84  nproc,myrank,xi_source,gamma_source,coorg,knods,ngnod, &
85  iglob_source
86  implicit none
87 
88  ! Local variables
89  integer :: ispec_acoustic_surface
90  integer :: ixmin, ixmax, izmin, izmax,i_source,ispec
91 
92  ! user output
93  if (myrank == 0) then
94  write(imain,*)
95  write(imain,*) 'sources:'
96  call flush_imain()
97  endif
98 
99  ! locates sources
100  do i_source = 1,nsources
101 
102  if (source_type(i_source) == 1) then
103  ! collocated force source
104  call locate_source(ibool,coord,nspec,nglob,xigll,zigll, &
105  x_source(i_source),z_source(i_source), &
106  ispec_selected_source(i_source),islice_selected_source(i_source), &
107  nproc,myrank,xi_source(i_source),gamma_source(i_source),coorg,knods,ngnod,npgeo, &
108  iglob_source(i_source),.true.) ! flag .true. indicates force source
109 
110  ! check
111  if (myrank == islice_selected_source(i_source)) then
112  ! checks that acoustic source is not exactly on the free surface because pressure is zero there
113  do ispec_acoustic_surface = 1,nelem_acoustic_surface
114  ispec = acoustic_surface(1,ispec_acoustic_surface)
115  ixmin = acoustic_surface(2,ispec_acoustic_surface)
116  ixmax = acoustic_surface(3,ispec_acoustic_surface)
117  izmin = acoustic_surface(4,ispec_acoustic_surface)
118  izmax = acoustic_surface(5,ispec_acoustic_surface)
119  if (.not. ispec_is_elastic(ispec) .and. .not. ispec_is_poroelastic(ispec) .and. &
120  ispec == ispec_selected_source(i_source)) then
121  if ((izmin == 1 .and. izmax == 1 .and. ixmin == 1 .and. ixmax == ngllx .and. &
122  gamma_source(i_source) < -0.99d0) .or. &
123  (izmin == ngllz .and. izmax == ngllz .and. ixmin == 1 .and. ixmax == ngllx .and. &
124  gamma_source(i_source) > 0.99d0) .or. &
125  (izmin == 1 .and. izmax == ngllz .and. ixmin == 1 .and. ixmax == 1 .and. &
126  xi_source(i_source) < -0.99d0) .or. &
127  (izmin == 1 .and. izmax == ngllz .and. ixmin == ngllx .and. ixmax == ngllx .and. &
128  xi_source(i_source) > 0.99d0) .or. &
129  (izmin == 1 .and. izmax == 1 .and. ixmin == 1 .and. ixmax == 1 .and. &
130  gamma_source(i_source) < -0.99d0 .and. xi_source(i_source) < -0.99d0) .or. &
131  (izmin == 1 .and. izmax == 1 .and. ixmin == ngllx .and. ixmax == ngllx .and. &
132  gamma_source(i_source) < -0.99d0 .and. xi_source(i_source) > 0.99d0) .or. &
133  (izmin == ngllz .and. izmax == ngllz .and. ixmin == 1 .and. ixmax == 1 .and. &
134  gamma_source(i_source) > 0.99d0 .and. xi_source(i_source) < -0.99d0) .or. &
135  (izmin == ngllz .and. izmax == ngllz .and. ixmin == ngllx .and. ixmax == ngllx .and. &
136  gamma_source(i_source) > 0.99d0 .and. xi_source(i_source) > 0.99d0)) then
137  call exit_mpi(myrank,'an acoustic source cannot be located exactly '// &
138  'on the free surface because pressure is zero there')
139  endif
140  endif
141  enddo
142 
143  endif
144 
145  else if (source_type(i_source) == 2) then
146  ! moment-tensor source
147  ! note: iglob_source is not really needed for moment-tensor sources, but left as argument since it's already allocated
148  ! and might help for future routines...
149  call locate_source(ibool,coord,nspec,nglob,xigll,zigll, &
150  x_source(i_source),z_source(i_source), &
151  ispec_selected_source(i_source),islice_selected_source(i_source), &
152  nproc,myrank,xi_source(i_source),gamma_source(i_source),coorg,knods,ngnod,npgeo, &
153  iglob_source(i_source),.false.) ! flag .false. indicates moment-tensor source
154 
155  else if (.not. initialfield) then
156 
157  call exit_mpi(myrank,'incorrect source type')
158 
159  endif
160 
161  enddo ! do i_source= 1,NSOURCES
162 
163 !! DK DK this below not supported in the case of MPI yet, we should do a MPI_GATHER() of the values
164 !! DK DK and use "if (myrank == islice_selected_rec(irec)) then" to display the right sources
165 !! DK DK and receivers carried by each mesh slice, and not fictitious values coming from other slices
166 #ifndef USE_MPI
167  if (myrank == 0) then
168  ! write actual source locations to file
169  ! note that these may differ from input values, especially if source_surf = .true. in SOURCE
170  ! note that the exact source locations are determined from (ispec,xi,gamma) values
171  open(unit=14,file=trim(output_files)//'for_information_SOURCE_actually_used',status='unknown')
172  do i_source= 1,nsources
173  write(14,*) x_source(i_source), z_source(i_source)
174  enddo
175  close(14)
176  endif
177 #endif
178 
179  ! synchronizes all processes
180  call synchronize_all()
181 
182  end subroutine setup_sources

188  subroutine setup_receivers()
189 
190 #ifdef USE_MPI
191  use constants, only: imain,iin,mygroup,in_data_files
192 #else
193  use constants, only: imain,iin,mygroup,in_data_files,output_files,iout
194 #endif
195  use specfem_par
196 
197  implicit none
198 
199  ! Local variables
200  integer :: nrec_tot_found
201  integer :: ier
202  integer :: irec,irec_local
203 
204  character(len=MAX_STRING_LEN) :: stations_filename,path_to_add,dummystring
205 
206  ! user output
207  call synchronize_all()
208  if (myrank == 0) then
209  write(imain,*)
210  write(imain,*) 'receivers:'
211  call flush_imain()
212  endif
213 
214  stations_filename = trim(in_data_files)//'STATIONS'
215 
216  ! see if we are running several independent runs in parallel
217  ! if so, add the right directory for that run
218  ! (group numbers start at zero, but directory names start at run0001, thus we add one)
219  ! a negative value for "mygroup" is a convention that indicates that groups (i.e. sub-communicators, one per run) are off
220  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
221  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
222  stations_filename = path_to_add(1:len_trim(path_to_add))//stations_filename(1:len_trim(stations_filename))
223  endif
224 
225  ! get number of stations from receiver file
226  open(unit=iin,file=trim(stations_filename),status='old',action='read',iostat=ier)
227  if (ier /= 0) call exit_mpi(myrank,'No file '//trim(stations_filename)//', exit')
228  nrec = 0
229  do while(ier == 0)
230  read(iin,"(a)",iostat=ier) dummystring
231  if (ier == 0) nrec = nrec + 1
232  enddo
233  close(iin)
234 
235  if (myrank == 0) then
236  write(imain,*)
237  write(imain,*) 'Total number of receivers = ',nrec
238  write(imain,*)
239  call flush_imain()
240  endif
241 
242  if (nrec < 1) call exit_mpi(myrank,'need at least one receiver')
243 
244  ! receiver information
245  allocate(ispec_selected_rec(nrec), &
246  st_xval(nrec), &
247  st_zval(nrec), &
248  xi_receiver(nrec), &
249  gamma_receiver(nrec), &
250  station_name(nrec), &
251  network_name(nrec), &
252  recloc(nrec), &
253  islice_selected_rec(nrec), &
254  x_final_receiver(nrec), &
255  z_final_receiver(nrec),stat=ier)
256  if (ier /= 0) stop 'Error allocating receiver arrays'
257 
258  ! locate receivers in the mesh
259  call locate_receivers(ibool,coord,nspec,nglob,xigll,zigll, &
260  nrec,nrecloc,recloc,islice_selected_rec,nproc,myrank, &
261  st_xval,st_zval,ispec_selected_rec, &
262  xi_receiver,gamma_receiver,station_name,network_name, &
263  x_source(1),z_source(1), &
264  coorg,knods,ngnod,npgeo, &
265  x_final_receiver,z_final_receiver)
266 
267 !! DK DK this below not supported in the case of MPI yet, we should do a MPI_GATHER() of the values
268 !! DK DK and use "if (myrank == islice_selected_rec(irec)) then" to display the right sources
269 !! DK DK and receivers carried by each mesh slice, and not fictitious values coming from other slices
270 #ifndef USE_MPI
271  if (myrank == 0) then
272  ! write out actual station locations (compare with STATIONS from meshfem2D)
273  ! NOTE: this will be written out even if use_existing_STATIONS = .true.
274  open(unit=iout,file=trim(output_files)//'for_information_STATIONS_actually_used',status='unknown')
275  do irec = 1,nrec
276  write(iout,"('S',i4.4,' AA ',f20.7,1x,f20.7,' 0.0 0.0')") &
277  irec,x_final_receiver(irec),z_final_receiver(irec)
278  enddo
279  close(iout)
280  endif
281 #endif
282 
283  ! synchronizes all processes
284  call synchronize_all()
285 
286  ! checks that the sum of the number of receivers in each slice is nrec
287  call sum_all_i(nrecloc,nrec_tot_found)
288  if (myrank == 0) then
289  ! checks total
290  if (nrec_tot_found /= nrec) then
291  print *,'nrec_tot_found,nrec (should be equal) = ',nrec_tot_found,nrec
292  call exit_mpi(myrank,'problem when dispatching the receivers')
293  endif
294 
295  ! user output
296  if (myrank == 0) then
297  write(imain,*) 'found a total of ',nrec_tot_found,' receivers, this is okay'
298  write(imain,*)
299  call flush_imain()
300  endif
301  endif
302 
303  ! synchronizes all processes
304  call synchronize_all()
305 
306  ! checks if acoustic receiver is exactly on the free surface because pressure is zero there
307  call setup_receivers_check_acoustic()
308 
309  ! create a local array for receivers
310  allocate(ispec_selected_rec_loc(nrecloc))
311  irec_local = 0
312  do irec = 1, nrec
313  if (myrank == islice_selected_rec(irec)) then
314  if (irec_local > nrecloc) stop 'Error with the number of local sources'
315  irec_local = irec_local + 1
316  ispec_selected_rec_loc(irec_local) = ispec_selected_rec(irec)
317  endif
318  enddo
319 
320  end subroutine setup_receivers

327  subroutine setup_receivers_check_acoustic()
328 
329 ! checks if acoustic receiver is exactly on the free surface because pressure is zero there
330 
331  use constants, only: ngllx,ngllz,imain
332 
333  use specfem_par, only: myrank,ispec_selected_rec,nrecloc,recloc, &
334  xi_receiver,gamma_receiver,seismotype, &
335  nelem_acoustic_surface,acoustic_surface,ispec_is_acoustic
336  implicit none
337 
338  ! Local variables
339  integer :: irec,irecloc
340  integer :: ispec,ispec_acoustic_surface
341  integer :: ixmin, ixmax, izmin, izmax
342 
343  ! check if acoustic receiver is exactly on the free surface because pressure is zero there
344  do ispec_acoustic_surface = 1,nelem_acoustic_surface
345  ispec = acoustic_surface(1,ispec_acoustic_surface)
346  ixmin = acoustic_surface(2,ispec_acoustic_surface)
347  ixmax = acoustic_surface(3,ispec_acoustic_surface)
348  izmin = acoustic_surface(4,ispec_acoustic_surface)
349  izmax = acoustic_surface(5,ispec_acoustic_surface)
350  do irecloc = 1,nrecloc
351  irec = recloc(irecloc)
352  if (ispec_is_acoustic(ispec) .and. ispec == ispec_selected_rec(irec)) then
353  if ((izmin == 1 .and. izmax == 1 .and. ixmin == 1 .and. ixmax == ngllx .and. &
354  gamma_receiver(irec) < -0.99d0) .or. &
355  (izmin == ngllz .and. izmax == ngllz .and. ixmin == 1 .and. ixmax == ngllx .and. &
356  gamma_receiver(irec) > 0.99d0) .or. &
357  (izmin == 1 .and. izmax == ngllz .and. ixmin == 1 .and. ixmax == 1 .and. &
358  xi_receiver(irec) < -0.99d0) .or. &
359  (izmin == 1 .and. izmax == ngllz .and. ixmin == ngllx .and. ixmax == ngllx .and. &
360  xi_receiver(irec) > 0.99d0) .or. &
361  (izmin == 1 .and. izmax == 1 .and. ixmin == 1 .and. ixmax == 1 .and. &
362  gamma_receiver(irec) < -0.99d0 .and. xi_receiver(irec) < -0.99d0) .or. &
363  (izmin == 1 .and. izmax == 1 .and. ixmin == ngllx .and. ixmax == ngllx .and. &
364  gamma_receiver(irec) < -0.99d0 .and. xi_receiver(irec) > 0.99d0) .or. &
365  (izmin == ngllz .and. izmax == ngllz .and. ixmin == 1 .and. ixmax == 1 .and. &
366  gamma_receiver(irec) > 0.99d0 .and. xi_receiver(irec) < -0.99d0) .or. &
367  (izmin == ngllz .and. izmax == ngllz .and. ixmin == ngllx .and. ixmax == ngllx .and. &
368  gamma_receiver(irec) > 0.99d0 .and. xi_receiver(irec) > 0.99d0)) then
369  ! checks
370  if (seismotype == 4) then
371  call exit_mpi(myrank,'an acoustic pressure receiver cannot be located exactly '// &
372  'on the free surface because pressure is zero there')
373  else
374  write(imain,*) '**********************************************************************'
375  write(imain,*) '*** Warning: acoustic receiver located exactly on the free surface ***'
376  write(imain,*) '*** Warning: tangential component will be zero there ***'
377  write(imain,*) '**********************************************************************'
378  write(imain,*)
379  endif
380  endif
381  endif
382  enddo
383  enddo
384 
385  end subroutine setup_receivers_check_acoustic

392  subroutine setup_adjoint_sources
393 
394 ! compute source array for adjoint source
395 
396  use constants, only: custom_real,ngllx,ngllz,ndim,max_string_len,imain
397 
398  use specfem_par, only: nadj_rec_local,nrec,nrecloc,nstep,nproc,simulation_type,su_format, &
399  myrank,islice_selected_rec,seismotype, &
400  network_name,station_name,source_adjoint
401 
402  implicit none
403 
404  ! local parameters
405  integer :: irec,irec_local
406  character(len=MAX_STRING_LEN) :: adj_source_file
407 
408  ! number of adjoint receivers in this slice
409  nadj_rec_local = 0
410 
411  ! note: SIMULATION_TYPE == 2 for "pure" adjoint simulations is not fully supported yet,
412  ! the run would stop after initialization.
413  ! it is left here for compatibility with 3D versions however and in case we plan to support it in future...
414 
415  ! adjoint calculation
416  if (simulation_type == 2 .or. simulation_type == 3) then
417  ! user output
418  if (myrank == 0) then
419  write(imain,*)
420  write(imain,*) 'adjoint sources:'
421  call flush_imain()
422  endif
423 
424  allocate(source_adjoint(nrecloc,nstep,2))
425 
426  ! counts number of adjoint sources in this slice
427  do irec = 1,nrec
428  ! counts local adjoint receiver stations
429  if (myrank == islice_selected_rec(irec)) then
430  ! check that the source proc number is okay
431  if (islice_selected_rec(irec) < 0 .or. islice_selected_rec(irec) > nproc-1) then
432  call exit_mpi(myrank,'something is wrong with the source proc number in adjoint simulation')
433  endif
434  ! counter
435  nadj_rec_local = nadj_rec_local + 1
436  endif
437  enddo
438 
439  ! reads in adjoint source files
440  if (.not. su_format) then
441  ! user output
442  if (myrank == 0) then
443  write(imain,*) ' reading ASCII adjoint source files'
444  call flush_imain()
445  endif
446 
447  ! reads in ascii adjoint source files **.adj
448  irec_local = 0
449  do irec = 1, nrec
450  ! compute only adjoint source arrays in the local proc
451  if (myrank == islice_selected_rec(irec)) then
452  irec_local = irec_local + 1
453  adj_source_file = trim(network_name(irec))//'.'//trim(station_name(irec))
454  call read_adj_source(irec_local,adj_source_file)
455  endif
456  enddo
457  ! checks
458  if (irec_local /= nadj_rec_local) stop 'Error invalid number of local adjoint sources found'
459  else
460  ! user output
461  if (myrank == 0) then
462  write(imain,*) ' reading SU-format adjoint source files'
463  call flush_imain()
464  endif
465 
466  ! (SU_FORMAT)
467  call read_adj_source_su(seismotype)
468  endif
469 
470  ! user output
471  if (myrank == 0) then
472  write(imain,*) ' number of adjoint sources = ',nrec
473  call flush_imain()
474  endif
475  endif ! SIMULATION_TYPE == 3
476 
477  ! synchronizes all processes
478  call synchronize_all()
479 
480  end subroutine setup_adjoint_sources

485 
486  subroutine setup_source_receiver_tangentials()
487 
488 ! tangential computation
489 
490 #ifdef USE_MPI
491  use mpi
492 #endif
493 
494  use constants, only: pi,hugeval,output_files
495  use specfem_par
496 
497  implicit none
498 
499  ! local parameters
500  integer :: i,irec,i_source
501  integer :: ier,nrec_alloc
502  integer :: irecloc
503  double precision :: x_final_receiver_dummy, z_final_receiver_dummy
504 
505  ! receiver arrays
506  if (nrecloc > 0) then
507  nrec_alloc = nrecloc
508  else
509  ! dummy allocation
510  nrec_alloc = 1
511  endif
512 
513  allocate(anglerec_irec(nrecloc), &
514  cosrot_irec(nrecloc), &
515  sinrot_irec(nrecloc), &
516  rec_tangential_detection_curve(nrecloc),stat=ier)
517  if (ier /= 0) stop 'Error allocating tangential arrays'
518 
519  ! checks angle
520  if (rec_normal_to_surface .and. abs(anglerec) > 1.d-6) &
521  stop 'anglerec should be zero when receivers are normal to the topography'
522 
523  ! convert receiver angle to radians
524  anglerec = anglerec * pi / 180.d0
525 
526  anglerec_irec(:) = anglerec
527  cosrot_irec(:) = cos(anglerec_irec(:))
528  sinrot_irec(:) = sin(anglerec_irec(:))
529 
530  ! tangential computation
531  ! for receivers
532  if (rec_normal_to_surface) then
533  irecloc = 0
534  do irec = 1, nrec
535  if (myrank == islice_selected_rec(irec)) then
536  irecloc = irecloc + 1
537  distmin = hugeval
538  do i = 1, nnodes_tangential_curve
539  dist_current = sqrt((x_final_receiver(irec)-nodes_tangential_curve(1,i))**2 + &
540  (z_final_receiver(irec)-nodes_tangential_curve(2,i))**2)
541  if (dist_current < distmin) then
542  n1_tangential_detection_curve = i
543  distmin = dist_current
544  endif
545  enddo
546 
547  rec_tangential_detection_curve(irecloc) = n1_tangential_detection_curve
548  call tri_quad(n_tangential_detection_curve, n1_tangential_detection_curve, &
549  nnodes_tangential_curve)
550 
551  call compute_normal_vector( anglerec_irec(irecloc), &
552  nodes_tangential_curve(1,n_tangential_detection_curve(1)), &
553  nodes_tangential_curve(1,n_tangential_detection_curve(2)), &
554  nodes_tangential_curve(1,n_tangential_detection_curve(3)), &
555  nodes_tangential_curve(1,n_tangential_detection_curve(4)), &
556  nodes_tangential_curve(2,n_tangential_detection_curve(1)), &
557  nodes_tangential_curve(2,n_tangential_detection_curve(2)), &
558  nodes_tangential_curve(2,n_tangential_detection_curve(3)), &
559  nodes_tangential_curve(2,n_tangential_detection_curve(4)) )
560  endif
561 
562  enddo
563  cosrot_irec(:) = cos(anglerec_irec(:))
564  sinrot_irec(:) = sin(anglerec_irec(:))
565  endif
566 
567  ! for the source
568  if (force_normal_to_surface) then
569  do i_source = 1,nsources
570  if (myrank == islice_selected_source(i_source)) then
571  distmin = hugeval
