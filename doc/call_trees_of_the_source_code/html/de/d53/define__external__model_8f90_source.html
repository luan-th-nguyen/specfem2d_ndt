44 ! -------------------------------------------------------------------------------------
45 ! Dummy example of this routine, to be used as a template that users can modify.
46 ! To use it you will need to rename it as define_external_model() (i.e., get rid of "_dummy")
47 ! and suppress the existing define_external_model() routine provided below for the AK135F global Earth model.
48 ! -------------------------------------------------------------------------------------
49 
50 ! users can modify this routine to assign any different external model (rho, vp, vs)
51 ! based on the x and y coordinates of that grid point and the material number of the region it belongs to

62  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec), intent(out) :: QKappa_attenuation,Qmu_attenuation
63  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec), intent(out) :: gravity,Nsq
64  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec), intent(out) :: c11,c15,c13,c33,c35,c55,c12,c23,c25
65 
66  ! local parameters
67  integer :: i,j,ispec,iglob
68 
69  double precision :: x,z
70 
71 ! dummy routine here, just to demonstrate how the model can be assigned
72 ! and how such a routine can be written
73 
74  ! user output
75  if (myrank == 0) then
76  write(imain,*) ' external model: ','dummy example'
77  call flush_imain()
78  endif
79 
80 ! remove gravity
81 ! leave these arrays here even if you do not assign them to use them because they need to be cleared
82  gravity(:,:,:) = 0.d0
83  nsq(:,:,:) = 0.d0
84 
85  ! default no attenuation
86  qkappa_attenuation(:,:,:) = 9999.d0
87  qmu_attenuation(:,:,:) = 9999.d0
88 
89 ! loop on all the elements of the mesh, and inside each element loop on all the GLL points
90  do ispec = 1,nspec
91  do j = 1,ngllz
92  do i = 1,ngllx
93 
94  iglob = ibool(i,j,ispec)
95 
96  x = coord(1,iglob)
97  z = coord(2,iglob)
98 
99  if (material_element(ispec) == 1 .or. x < 1700.d0 .or. z >= 2300.d0) then
100  rho(i,j,ispec) = 2000.d0 + 10.d0*x - 32.17d0*z**3
101  vp(i,j,ispec) = 3000.d0 - 7.12d0*cos(x)
102  vs(i,j,ispec) = vp(i,j,ispec) / sqrt(3.d0)
103  qkappa_attenuation(i,j,ispec) = 9999. ! this means no attenuation
104  qmu_attenuation(i,j,ispec) = 9999. ! this means no attenuation
105  c11(i,j,ispec) = 169.d9
106  c13(i,j,ispec) = 122.d9
107  c15(i,j,ispec) = 0.d0
108  c33(i,j,ispec) = c11(i,j,ispec)
109  c35(i,j,ispec) = 0.d0
110  c55(i,j,ispec) = 75.3d9
111  c12(i,j,ispec) = 0.d0
112  c23(i,j,ispec) = 0.d0
113  c25(i,j,ispec) = 0.d0
114 
115  else if (material_element(ispec) == 2) then
116  rho(i,j,ispec) = 2500.d0
117  vp(i,j,ispec) = 3600.d0
118  vs(i,j,ispec) = vp(i,j,ispec) / 2.d0
119  qkappa_attenuation(i,j,ispec) = 120.
120  qmu_attenuation(i,j,ispec) = 120.
121  c11(i,j,ispec) = 0.d0 ! this means no anisotropy
122  c13(i,j,ispec) = 0.d0
123  c15(i,j,ispec) = 0.d0
124  c33(i,j,ispec) = 0.d0
125  c35(i,j,ispec) = 0.d0
126  c55(i,j,ispec) = 0.d0
127  c12(i,j,ispec) = 0.d0
128  c23(i,j,ispec) = 0.d0
129  c25(i,j,ispec) = 0.d0
130 
131  else
132  ! this case should not occur
133  write(imain,*) 'Error: invalid flag number in external model is equal to ',material_element(ispec)
134  stop 'wrong flag number in external model; exiting...'
135  endif
136 
137  !! ABAB Do not forget these 3 lines otherwise PML may not work !! TODO check that
138  density(1,kmato(ispec)) = rho(i,j,ispec)
139  poroelastcoef(3,1,kmato(ispec)) = rho(i,j,ispec) * vp(i,j,ispec) * vp(i,j,ispec)
140  poroelastcoef(2,1,kmato(ispec)) = rho(i,j,ispec) * vs(i,j,ispec) * vs(i,j,ispec)
141  !! ABAB Do not forget these 3 lines otherwise PML may not work !! TODO check that
142 
143  enddo
144  enddo
145  enddo
146 
147  end subroutine define_external_model_dummy
