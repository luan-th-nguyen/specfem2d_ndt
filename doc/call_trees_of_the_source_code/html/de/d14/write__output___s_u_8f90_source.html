34  subroutine write_output_su(x_source,z_source,irec,buffer_binary,number_of_components)
35 
36  use specfem_par, only: nstep,nrec,deltat,seismotype,st_xval, &
37  nstep_between_output_seismos,seismo_offset,seismo_current,p_sv, &
38  st_zval,subsamp_seismos
39 
40  implicit none
41 
42  double precision,intent(in) :: x_source,z_source
43  integer,intent(in) :: irec,number_of_components
44 
45  ! to write seismograms in single precision SEP and double precision binary
46  double precision, dimension(NSTEP_BETWEEN_OUTPUT_SEISMOS/subsamp_seismos,nrec,number_of_components),intent(in) :: buffer_binary
47 
48  ! local parameters
49  integer :: deltat_int2
50  integer :: isample
51 
52  ! scaling factor for Seismic Unix xsu dislay
53  double precision, parameter :: FACTORXSU = 1.d0
54 
55  integer(kind=2) :: header2(2)
56 
57  if (seismo_offset == 0) then
58 
59  if (deltat*1.0d6 > 2**15) then
60  deltat_int2 = 0
61  else
62  deltat_int2 = nint(deltat*1.0d6, kind=2) ! deltat (unit: 10^{-6} second)
63  endif
64 
65  ! write SU headers (refer to Seismic Unix for details)
66  write(12,rec=(irec-1)*60+(irec-1)*nstep+1) irec ! receiver ID
67  write(12,rec=(irec-1)*60+(irec-1)*nstep+10) nint(st_xval(irec)-x_source) ! offset
68  write(12,rec=(irec-1)*60+(irec-1)*nstep+19) nint(x_source) ! source location xs
69  write(12,rec=(irec-1)*60+(irec-1)*nstep+20) nint(z_source) ! source location zs
70  write(12,rec=(irec-1)*60+(irec-1)*nstep+21) nint(st_xval(irec)) ! receiver location xr
71  write(12,rec=(irec-1)*60+(irec-1)*nstep+22) nint(st_zval(irec)) ! receiver location zr
72  if (nrec > 1) write(12,rec=(irec-1)*60+(irec-1)*nstep+48) sngl(st_xval(2)-st_xval(1)) ! receiver interval
73  header2(1)=0 ! dummy
74  header2(2)=int(nstep, kind=2)
75  write(12,rec=(irec-1)*60+(irec-1)*nstep+29) header2
76  header2(1)=deltat_int2
77  header2(2)=0 ! dummy
78  write(12,rec=(irec-1)*60+(irec-1)*nstep+30) header2
79  if (seismotype /= 4 .and. seismotype /= 6 .and. p_sv) then
80  ! headers
81  if (seismo_offset == 0) then
82  write(14,rec=(irec-1)*60+(irec-1)*nstep+1) irec
83  write(14,rec=(irec-1)*60+(irec-1)*nstep+10) nint(st_xval(irec)-x_source)
84  write(14,rec=(irec-1)*60+(irec-1)*nstep+19) nint(x_source)
85  write(14,rec=(irec-1)*60+(irec-1)*nstep+20) nint(z_source)
86  write(14,rec=(irec-1)*60+(irec-1)*nstep+21) nint(st_xval(irec))
87  write(14,rec=(irec-1)*60+(irec-1)*nstep+22) nint(st_zval(irec))
88  if (nrec > 1) write(14,rec=(irec-1)*60+(irec-1)*nstep+48) sngl(st_xval(2)-st_xval(1))
89  header2(1)=0 ! dummy
90  header2(2)=int(nstep, kind=2)
91  write(14,rec=(irec-1)*60+(irec-1)*nstep+29) header2
92  header2(1)=deltat_int2
93  header2(2)=0 ! dummy
94  write(14,rec=(irec-1)*60+(irec-1)*nstep+30) header2
95  endif
96  endif
97 
98  endif
99 
100  ! the "60" in the following corresponds to 240 bytes header (note the reclength is 4 bytes)
101  do isample = 1, seismo_current
102  write(12,rec=irec*60+(irec-1)*nstep+seismo_offset+isample) sngl(buffer_binary(isample,irec,1))
103  if (seismotype /= 4 .and. seismotype /= 6 .and. p_sv) then
104  write(14,rec=irec*60+(irec-1)*nstep+seismo_offset+isample) sngl(buffer_binary(isample,irec,2))
105  endif
106  enddo
107 
108  end subroutine write_output_su
