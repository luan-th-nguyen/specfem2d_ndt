34 ! compute spline coefficients
35 
36  subroutine spline_construction(xpoint,ypoint,npoint,tangent_first_point,tangent_last_point,spline_coefficients)
37 
38  implicit none
39 
40 ! tangent to the spline imposed at the first and last points
41  double precision, intent(in) :: tangent_first_point,tangent_last_point
42 
43 ! number of input points and coordinates of the input points
44  integer, intent(in) :: npoint
45  double precision, dimension(npoint), intent(in) :: xpoint,ypoint
46 
47 ! spline coefficients output by the routine
48  double precision, dimension(npoint), intent(out) :: spline_coefficients
49 
50  integer :: i
51 
52  double precision, dimension(:), allocatable :: temporary_array
53 
54  allocate(temporary_array(npoint))
55 
56  spline_coefficients(1) = - 1.d0 / 2.d0
57 
58  temporary_array(1) = (3.d0/(xpoint(2)-xpoint(1)))*((ypoint(2)-ypoint(1))/(xpoint(2)-xpoint(1))-tangent_first_point)
59 
60  do i = 2,npoint-1
61 
62  spline_coefficients(i) = ((xpoint(i)-xpoint(i-1))/(xpoint(i+1)-xpoint(i-1))-1.d0) &
63  / ((xpoint(i)-xpoint(i-1))/(xpoint(i+1)-xpoint(i-1))*spline_coefficients(i-1)+2.d0)
64 
65  temporary_array(i) = (6.d0*((ypoint(i+1)-ypoint(i))/(xpoint(i+1)-xpoint(i)) &
66  - (ypoint(i)-ypoint(i-1))/(xpoint(i)-xpoint(i-1)))/(xpoint(i+1)-xpoint(i-1)) &
67  - (xpoint(i)-xpoint(i-1))/(xpoint(i+1)-xpoint(i-1))*temporary_array(i-1)) &
68  / ((xpoint(i)-xpoint(i-1))/(xpoint(i+1)-xpoint(i-1))*spline_coefficients(i-1)+2.d0)
69 
70  enddo
71 
72  spline_coefficients(npoint) = ((3.d0/(xpoint(npoint)-xpoint(npoint-1))) &
73  * (tangent_last_point-(ypoint(npoint)-ypoint(npoint-1))/(xpoint(npoint)-xpoint(npoint-1))) &
74  - 1.d0/2.d0*temporary_array(npoint-1))/(1.d0/2.d0*spline_coefficients(npoint-1)+1.d0)
75 
76  do i = npoint-1,1,-1
77  spline_coefficients(i) = spline_coefficients(i)*spline_coefficients(i+1) + temporary_array(i)
78  enddo
79 
80  deallocate(temporary_array)
81 
82  end subroutine spline_construction

88 ! evaluate a spline
89 
90  subroutine spline_evaluation(xpoint,ypoint,spline_coefficients,npoint,x_evaluate_spline,y_spline_obtained)
91 
92  implicit none
93 
94 ! number of input points and coordinates of the input points
95  integer, intent(in) :: npoint
96  double precision, dimension(npoint), intent(in) :: xpoint,ypoint
97 
98 ! spline coefficients to use
99  double precision, dimension(npoint), intent(in) :: spline_coefficients
100 
101 ! abscissa at which we need to evaluate the value of the spline
102  double precision, intent(in):: x_evaluate_spline
103 
104 ! ordinate evaluated by the routine for the spline at this abscissa
105  double precision, intent(out):: y_spline_obtained
106 
107  integer :: index_loop,index_lower,index_higher
108 
109  double precision :: coef1,coef2
110 
111 ! initialize to the whole interval
112  index_lower = 1
113  index_higher = npoint
114 
115 ! determine the right interval to use, by dichotomy
116  do while (index_higher - index_lower > 1)
117 ! compute the middle of the interval
118  index_loop = (index_higher + index_lower) / 2
119  if (xpoint(index_loop) > x_evaluate_spline) then
120  index_higher = index_loop
121  else
122  index_lower = index_loop
123  endif
124  enddo
125 
126 ! test that the interval obtained does not have a size of zero
127 ! (this could happen for instance in the case of duplicates in the input list of points)
128  if (xpoint(index_higher) == xpoint(index_lower)) stop 'incorrect interval found in spline evaluation'
129 
130  coef1 = (xpoint(index_higher) - x_evaluate_spline) / (xpoint(index_higher) - xpoint(index_lower))
131  coef2 = (x_evaluate_spline - xpoint(index_lower)) / (xpoint(index_higher) - xpoint(index_lower))
132 
133  y_spline_obtained = coef1*ypoint(index_lower) + coef2*ypoint(index_higher) + &
134  ((coef1**3 - coef1)*spline_coefficients(index_lower) + &
135  (coef2**3 - coef2)*spline_coefficients(index_higher))*((xpoint(index_higher) - xpoint(index_lower))**2)/6.d0
136 
137  end subroutine spline_evaluation
