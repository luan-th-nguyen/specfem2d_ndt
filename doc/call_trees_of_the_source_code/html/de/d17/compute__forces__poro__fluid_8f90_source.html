34  subroutine compute_forces_poro_fluid(displs_poroelastic,displw_poroelastic,accelw_poroelastic,epsilondev_w,iphase)
35 
36 ! compute forces for the fluid poroelastic part
37 
38  use constants, only: custom_real,ndim,ngllx,ngllz,two,zero, &
39  iedge1,iedge2,iedge3,iedge4,alpha_lddrk,beta_lddrk
40 
41  use specfem_par, only: nglob,nspec,nglob_poroelastic,nspec_poroelastic_b, &
42  attenuation_viscoelastic,deltat, &
43  ibool,ispec_is_poroelastic, &
44  xix,xiz,gammax,gammaz,jacobian, &
45  e11,e13,hprime_xx,hprimewgll_xx,hprime_zz,hprimewgll_zz,wxgll,wzgll, &
46  inv_tau_sigma_nu2,phi_nu2,mu_nu2,n_sls, &
47  simulation_type, &
48  e11_lddrk,e13_lddrk, &
49  e11_initial_rk,e13_initial_rk,e11_force_rk, e13_force_rk, &
50  time_stepping_scheme,i_stage
51 
52  use specfem_par, only: displs_poroelastic_old
53 
54  ! overlapping communication
55  use specfem_par, only: nspec_inner_poroelastic,nspec_outer_poroelastic,phase_ispec_inner_poroelastic
56 
57  implicit none
58 
59  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_poroelastic), intent(in) :: displs_poroelastic,displw_poroelastic
60  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_poroelastic), intent(inout) :: accelw_poroelastic
61 
62  real(kind=CUSTOM_REAL), dimension(4,NGLLX,NGLLZ,nspec_poroelastic_b),intent(out) :: epsilondev_w
63 
64  integer,intent(in) :: iphase
65 
66  ! local variables
67  integer :: ispec,i,j,k,iglob
68  real(kind=CUSTOM_REAL) :: weight_rk
69  real(kind=CUSTOM_REAL) :: e11_sum,e13_sum
70  integer :: i_sls
71 
72  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec) ::dux_dxl_n,duz_dzl_n,duz_dxl_n,dux_dzl_n, &
73  !nsub1 denote discrete time step n-1
74  dux_dxl_nsub1,duz_dzl_nsub1,duz_dxl_nsub1,dux_dzl_nsub1
75 
76  ! spatial derivatives
77  real(kind=CUSTOM_REAL) :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
78  real(kind=CUSTOM_REAL) :: dwx_dxi,dwx_dgamma,dwz_dxi,dwz_dgamma
79  real(kind=CUSTOM_REAL) :: dux_dxl,duz_dxl,dux_dzl,duz_dzl
80  real(kind=CUSTOM_REAL) :: dwx_dxl,dwz_dxl,dwx_dzl,dwz_dzl
81  real(kind=CUSTOM_REAL) :: sigma_xx,sigma_xz,sigma_zz
82  real(kind=CUSTOM_REAL) :: sigmap
83 
84  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: tempx1,tempx2,tempz1,tempz2
85  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: tempx1p,tempx2p,tempz1p,tempz2p
86 
87  real(kind=CUSTOM_REAL), dimension(4,NGLLX,NGLLZ) :: epsilondev_loc
88 
89 ! Jacobian matrix and determinant
90  real(kind=CUSTOM_REAL) :: xixl,xizl,gammaxl,gammazl,jacobianl
91 
92 ! material properties of the poroelastic medium
93  real(kind=CUSTOM_REAL) :: mu_relaxed_viscoelastic,lambda_relaxed_viscoelastic,lambdalplus2mu_relaxed_viscoel
94  real(kind=CUSTOM_REAL) :: mu_G,lambdal_G,lambdalplus2mul_G
95 
96  double precision :: phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar
97  double precision :: D_biot,H_biot,C_biot,M_biot
98 
99 ! for attenuation
100  real(kind=CUSTOM_REAL) :: phinu2,tauinvnu2,theta_n_u,theta_nsub1_u
101  real(kind=CUSTOM_REAL) :: bb,coef0,coef1,coef2
102 
103  integer :: num_elements,ispec_p
104 
105 ! implement attenuation
106  if (attenuation_viscoelastic) then
107 
108  ! safety check
109  if (simulation_type == 3) &
110  stop 'ATTENUATION_VISCOELASTIC not fully implemented yet for poroelastic kernel simulations'
111 
112 ! compute Grad(displs_poroelastic) at time step n for attenuation
113  call compute_gradient_attenuation(displs_poroelastic,dux_dxl_n,duz_dxl_n, &
114  dux_dzl_n,duz_dzl_n,xix,xiz,gammax,gammaz,ibool,ispec_is_poroelastic,hprime_xx,hprime_zz,nspec,nglob)
115 
116 ! compute Grad(displs_poroelastic) at time step n-1 for attenuation
117  call compute_gradient_attenuation(displs_poroelastic_old,dux_dxl_nsub1,duz_dxl_nsub1, &
118  dux_dzl_nsub1,duz_dzl_nsub1,xix,xiz,gammax,gammaz,ibool,ispec_is_poroelastic,hprime_xx,hprime_zz,nspec,nglob)
119 
120 ! update memory variables with fourth-order Runge-Kutta time scheme for attenuation
121 ! loop over spectral elements
122  do ispec = 1,nspec
123 
124  if (ispec_is_poroelastic(ispec)) then
125  do j = 1,ngllz; do i = 1,ngllx
126  theta_n_u = dux_dxl_n(i,j,ispec) + duz_dzl_n(i,j,ispec)
127  theta_nsub1_u = dux_dxl_nsub1(i,j,ispec) + duz_dzl_nsub1(i,j,ispec)
128 
129  ! loop on all the standard linear solids
130  do i_sls = 1,n_sls
131  phinu2 = phi_nu2(i,j,ispec,i_sls)
132  tauinvnu2 = inv_tau_sigma_nu2(i,j,ispec,i_sls)
133 
134  ! update e1, e11, e13 in convolution formation with modified recursive convolution scheme on basis of
135  ! second-order accurate convolution term calculation from equation (21) of
136  ! Shumin Wang, Robert Lee, and Fernando L. Teixeira,
137  ! Anisotropic-medium PML for vector FETD with modified basis functions,
138  ! IEEE Transactions on Antennas and Propagation, vol. 54, no. 1, (2006)
139  ! evolution e1 ! no need since we are just considering shear attenuation
140  if (time_stepping_scheme == 1) then
141  ! Newmark
142  bb = tauinvnu2; coef0 = exp(-bb * deltat)
143  if (abs(bb) > 1e-5_custom_real) then
144  coef1 = (1._custom_real - exp(-bb * deltat / 2._custom_real)) / bb
145  coef2 = (1._custom_real - exp(-bb* deltat / 2._custom_real)) * exp(-bb * deltat / 2._custom_real)/ bb
146  else
147  coef1 = deltat / 2._custom_real
148  coef2 = deltat / 2._custom_real
149  endif
150 
151  e11(i,j,ispec,i_sls) = coef0 * e11(i,j,ispec,i_sls) + &
152  phinu2 * (coef1 * (dux_dxl_n(i,j,ispec)-theta_n_u/two) + &
153  coef2 * (dux_dxl_nsub1(i,j,ispec)-theta_nsub1_u/two))
154 
155  e13(i,j,ispec,i_sls) = coef0 * e13(i,j,ispec,i_sls) + &
156  phinu2 * (coef1 * (dux_dzl_n(i,j,ispec) + duz_dxl_n(i,j,ispec)) + &
157  coef2 * (dux_dzl_nsub1(i,j,ispec) + duz_dxl_nsub1(i,j,ispec)))
158 
159  ! update e1, e11, e13 in ADE formation with LDDRK scheme
160  ! evolution e1 ! no need since we are just considering shear attenuation
161  else if (time_stepping_scheme == 2) then
162  ! LDDRK
163  e11_lddrk(i,j,ispec,i_sls) = alpha_lddrk(i_stage) * e11_lddrk(i,j,ispec,i_sls) + &
164  deltat * ((dux_dxl_n(i,j,ispec)-theta_n_u/two) * phinu2) - &
165  deltat * (e11(i,j,ispec,i_sls) * tauinvnu2)
166  e11(i,j,ispec,i_sls) = e11(i,j,ispec,i_sls)+beta_lddrk(i_stage)*e11_lddrk(i,j,ispec,i_sls)
167 
168  e13_lddrk(i,j,ispec,i_sls) = alpha_lddrk(i_stage) * e13_lddrk(i,j,ispec,i_sls) + &
169  deltat * ((dux_dzl_n(i,j,ispec) + duz_dxl_n(i,j,ispec))*phinu2) - &
170  deltat * (e13(i,j,ispec,i_sls) * tauinvnu2)
171  e13(i,j,ispec,i_sls) = e13(i,j,ispec,i_sls)+beta_lddrk(i_stage) * e13_lddrk(i,j,ispec,i_sls)
172 
173  ! update e1, e11, e13 in ADE formation with classical Runge-Kutta scheme
174  ! evolution e1 ! no need since we are just considering shear attenuation
175  else if (time_stepping_scheme == 3) then
176  ! RK
177  e11_force_rk(i,j,ispec,i_sls,i_stage) = deltat * ((dux_dxl_n(i,j,ispec)-theta_n_u/two) * phinu2 - &
178  e11(i,j,ispec,i_sls) * tauinvnu2)
179  if (i_stage == 1 .or. i_stage == 2 .or. i_stage == 3) then
180  if (i_stage == 1)weight_rk = 0.5_custom_real
181  if (i_stage == 2)weight_rk = 0.5_custom_real
182  if (i_stage == 3)weight_rk = 1._custom_real
183 
184  if (i_stage == 1) e11_initial_rk(i,j,ispec,i_sls) = e11(i,j,ispec,i_sls)
185  e11(i,j,ispec,i_sls) = e11_initial_rk(i,j,ispec,i_sls) + weight_rk * e11_force_rk(i,j,ispec,i_sls,i_stage)
186  else if (i_stage == 4) then
187  e11(i,j,ispec,i_sls) = e11_initial_rk(i,j,ispec,i_sls) + 1._custom_real / 6._custom_real * &
188  (e11_force_rk(i,j,ispec,i_sls,1) + 2._custom_real * e11_force_rk(i,j,ispec,i_sls,2) + &
189  2._custom_real * e11_force_rk(i,j,ispec,i_sls,3) + e11_force_rk(i,j,ispec,i_sls,4))
190  endif
191 
192  e13_force_rk(i,j,ispec,i_sls,i_stage) = deltat * ((dux_dzl_n(i,j,ispec) + duz_dxl_n(i,j,ispec))*phinu2 - &
193  e13(i,j,ispec,i_sls) * tauinvnu2)
194  if (i_stage == 1 .or. i_stage == 2 .or. i_stage == 3) then
195  if (i_stage == 1)weight_rk = 0.5_custom_real
196  if (i_stage == 2)weight_rk = 0.5_custom_real
197  if (i_stage == 3)weight_rk = 1._custom_real
198  if (i_stage == 1) e13_initial_rk(i,j,ispec,i_sls) = e13(i,j,ispec,i_sls)
199  e13(i,j,ispec,i_sls) = e13_initial_rk(i,j,ispec,i_sls) + weight_rk * e13_force_rk(i,j,ispec,i_sls,i_stage)
200  else if (i_stage == 4) then
201  e13(i,j,ispec,i_sls) = e13_initial_rk(i,j,ispec,i_sls) + 1._custom_real / 6._custom_real * &
202  (e13_force_rk(i,j,ispec,i_sls,1) + 2._custom_real * e13_force_rk(i,j,ispec,i_sls,2) + &
203  2._custom_real * e13_force_rk(i,j,ispec,i_sls,3) + e13_force_rk(i,j,ispec,i_sls,4))
204  endif
205  endif
206  enddo
207  enddo
208  enddo
209  endif
210  enddo
211 
212  endif ! end of test on attenuation
213 
214  ! choses inner/outer elements
215  if (iphase == 1) then
216  num_elements = nspec_outer_poroelastic
217  else
218  num_elements = nspec_inner_poroelastic
219  endif
220 
221  ! loop over spectral elements
222  do ispec_p = 1,num_elements
223 
224  ! returns element id from stored element list
225  ispec = phase_ispec_inner_poroelastic(ispec_p,iphase)
226 
227 !---
228 !--- poroelastic spectral element
229 !---
230 
231  if (.not. ispec_is_poroelastic(ispec)) cycle
232 
233  ! get poroelastic properties of current spectral element
234  call get_poroelastic_material(ispec,phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar)
235 
236  ! Biot coefficients for the input phi
237  call get_poroelastic_biot_coeff(phi,kappa_s,kappa_f,kappa_fr,mu_fr,d_biot,h_biot,c_biot,m_biot)
238 
239  !The RHS has the form : div T_f -rho_f/rho_bar div T - eta_fk^-1.partial t w
240  !where T = G:grad u_s + C_biot div w I
241  !and T_f = C_biot div u_s I + M_biot div w I
242  mu_g = mu_fr
243  lambdal_g = h_biot - 2._custom_real*mu_fr
244  lambdalplus2mul_g = lambdal_g + two*mu_g
245 
246  ! first double loop over GLL points to compute and store gradients
247  do j = 1,ngllz
248  do i = 1,ngllx
249  ! derivative along x and along z for u_s and w
250  dux_dxi = zero
251  duz_dxi = zero
252 
253  dux_dgamma = zero
254  duz_dgamma = zero
255 
256  dwx_dxi = zero
257  dwz_dxi = zero
258 
259  dwx_dgamma = zero
260  dwz_dgamma = zero
261 
262  ! first double loop over GLL points to compute and store gradients
263  ! we can merge the two loops because NGLLX == NGLLZ
264  do k = 1,ngllx
265  dux_dxi = dux_dxi + displs_poroelastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
266  duz_dxi = duz_dxi + displs_poroelastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
267  dux_dgamma = dux_dgamma + displs_poroelastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
268  duz_dgamma = duz_dgamma + displs_poroelastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
269 
270  dwx_dxi = dwx_dxi + displw_poroelastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
271  dwz_dxi = dwz_dxi + displw_poroelastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
272  dwx_dgamma = dwx_dgamma + displw_poroelastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
273  dwz_dgamma = dwz_dgamma + displw_poroelastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
274  enddo
275 
276  xixl = xix(i,j,ispec)
277  xizl = xiz(i,j,ispec)
278  gammaxl = gammax(i,j,ispec)
279  gammazl = gammaz(i,j,ispec)
280 
281  ! derivatives of displacement
282  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
283  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
284 
285  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
286  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
287 
288  dwx_dxl = dwx_dxi*xixl + dwx_dgamma*gammaxl
289  dwx_dzl = dwx_dxi*xizl + dwx_dgamma*gammazl
290 
291  dwz_dxl = dwz_dxi*xixl + dwz_dgamma*gammaxl
292  dwz_dzl = dwz_dxi*xizl + dwz_dgamma*gammazl
293 
294  ! compute stress tensor (include attenuation if needed)
295  if (attenuation_viscoelastic) then
296 ! Dissipation only controlled by frame share attenuation in poroelastic (see Morency & Tromp, GJI 2008).
297 ! attenuation is implemented following the memory variable formulation of
298 ! J. M. Carcione, Seismic modeling in viscoelastic media, Geophysics,
299 ! vol. 58(1), p. 110-120 (1993). More details can be found in
300 ! J. M. Carcione, D. Kosloff and R. Kosloff, Wave propagation simulation in a linear
301 ! viscoelastic medium, Geophysical Journal International, vol. 95, p. 597-611 (1988).
302 
303 ! When implementing viscoelasticity according to the Carcione 1993 paper, attenuation is
304 ! non-causal rather than causal i.e. wave speed up instead of slowing down
305 ! when attenuation is turned on. We fixed that issue (which is not incorrect but non traditional)
306 ! by taking the unrelaxed state (infinite frequency) as a reference instead of the relaxed state (zero frequency)
307 ! and also using equations in Carcione's 2007 book.
308 ! See file doc/old_problem_attenuation_reference_Specfem2D_fixed_by_Xie_Zhinan.pdf
309 ! and doc/how_we_modified_Carcione_1993_to_make_it_causal_and_include_the_missing_1_over_L_factor.pdf
310 
311 ! See also J. M. Carcione, Seismic modeling in viscoelastic media, Geophysics,
312 ! vol. 58(1), p. 110-120 (1993) for two memory-variable mechanisms (page 112).
313 
314 ! and J. M. Carcione, D. Kosloff and R. Kosloff, Wave propagation simulation
315 ! in a linear viscoelastic medium, Geophysical Journal International,
316 ! vol. 95, p. 597-611 (1988) for two memory-variable mechanisms (page 604).
317 
318 ! compute unrelaxed elastic coefficients from formulas in Carcione 2007 page 125
319  lambda_relaxed_viscoelastic = (lambdal_g + mu_g) - mu_g / mu_nu2(i,j,ispec)
320  mu_relaxed_viscoelastic = mu_g / mu_nu2(i,j,ispec)
321  lambdalplus2mu_relaxed_viscoel = lambda_relaxed_viscoelastic + two*mu_relaxed_viscoelastic
322 
323 ! compute the stress using the unrelaxed Lame parameters (Carcione 2007 page 125)
324  sigma_xx = (lambdal_g + 2.0 * mu_g)*dux_dxl + mu_g*duz_dzl + c_biot*(dwx_dxl + dwz_dzl)
325  sigma_xz = mu_g*(duz_dxl + dux_dzl)
326  sigma_zz = (lambdal_g + 2.0 * mu_g)*duz_dzl + mu_g*dux_dxl + c_biot*(dwx_dxl + dwz_dzl)
327 
328  sigmap = c_biot*(dux_dxl + duz_dzl) + m_biot*(dwx_dxl + dwz_dzl)
329 
330 ! add the memory variables using the relaxed parameters (Carcione 2007 page 125)
331 ! beware: there is a bug in Carcione's equation (2c) of his 1993 paper for sigma_zz, we fixed it in the code below
332  e11_sum = 0._custom_real
333  e13_sum = 0._custom_real
334 
335  do i_sls = 1,n_sls
336  e11_sum = e11_sum + e11(i,j,ispec,i_sls)
337  e13_sum = e13_sum + e13(i,j,ispec,i_sls)
338  enddo
339 
340 ! mu_G is the relaxed modulus. Note that it is defined as the
341 ! frame modulus (in compute_forces_poro_solid.f90), which Christina Morency noted
342 ! mu_fr, which is in her case equivalent to the solid phase shear
343 ! modulus, and whose value is entered in Par_file for example
344  sigma_xx = sigma_xx + two * mu_relaxed_viscoelastic * e11_sum
345  sigma_xz = sigma_xz + mu_relaxed_viscoelastic * e13_sum
346  sigma_zz = sigma_zz - two * mu_relaxed_viscoelastic * e11_sum
347 
348  else
349 
350 ! no attenuation
351  sigma_xx = lambdalplus2mul_g*dux_dxl + lambdal_g*duz_dzl + c_biot*(dwx_dxl + dwz_dzl)
352  sigma_xz = mu_g*(duz_dxl + dux_dzl)
353  sigma_zz = lambdalplus2mul_g*duz_dzl + lambdal_g*dux_dxl + c_biot*(dwx_dxl + dwz_dzl)
354 
355  sigmap = c_biot*(dux_dxl + duz_dzl) + m_biot*(dwx_dxl + dwz_dzl)
356 
357  endif
358 
359  ! kernels calculation
360  if (simulation_type == 3) then
361  epsilondev_loc(1,i,j) = dux_dxl
362  epsilondev_loc(2,i,j) = duz_dzl
363 
364  epsilondev_loc(3,i,j) = dwx_dxl
365  epsilondev_loc(4,i,j) = dwz_dzl
366 
367  ! see now in compute_kernels_po() routine...
368  !iglob = ibool(i,j,ispec)
369  !C_k(iglob) = ((dux_dxl + duz_dzl) * (b_dwx_dxl + b_dwz_dzl) + &
370  ! (dwx_dxl + dwz_dzl) * (b_dux_dxl + b_duz_dzl)) * C_biot
371  !M_k(iglob) = (dwx_dxl + dwz_dzl) * (b_dwx_dxl + b_dwz_dzl) * M_biot
372  endif
373 
374  jacobianl = jacobian(i,j,ispec)
375 
376 ! weak formulation term based on stress tensor (symmetric form)
377 ! also add GLL integration weights
378  tempx1(i,j) = wzgll(j)*jacobianl*(sigma_xx*xixl+sigma_xz*xizl)
379  tempz1(i,j) = wzgll(j)*jacobianl*(sigma_xz*xixl+sigma_zz*xizl)
380 
381  tempx2(i,j) = wxgll(i)*jacobianl*(sigma_xx*gammaxl+sigma_xz*gammazl)
382  tempz2(i,j) = wxgll(i)*jacobianl*(sigma_xz*gammaxl+sigma_zz*gammazl)
383 
384  tempx1p(i,j) = wzgll(j)*jacobianl*sigmap*xixl
385  tempz1p(i,j) = wzgll(j)*jacobianl*sigmap*xizl
386 
387  tempx2p(i,j) = wxgll(i)*jacobianl*sigmap*gammaxl
388  tempz2p(i,j) = wxgll(i)*jacobianl*sigmap*gammazl
389 
390  enddo
391  enddo
392 
393 !
394 ! second double-loop over GLL to compute all the terms
395 !
396  do j = 1,ngllz
397  do i = 1,ngllx
398 
399  iglob = ibool(i,j,ispec)
400 
401  ! along x direction and z direction
