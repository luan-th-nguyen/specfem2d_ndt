33 
34  subroutine compute_stacey_elastic(accel_elastic,veloc_elastic)
35 
36 ! absorbing boundaries
37 !
38 ! Clayton-Engquist condition if elastic
39 
40  use constants, only: custom_real,ngllx,ngllz,ndim, &
41  zero,one,two,two_thirds,four_thirds,iedge1,iedge2,iedge3,iedge4
42 
43  use specfem_par, only: axisym,nglob,nelemabs,it,any_elastic, &
44  assign_external_model,ibool,kmato,numabs,ispec_is_elastic, &
45  codeabs,codeabs_corner,density,poroelastcoef,xix,xiz,gammax,gammaz,jacobian, &
46  vpext,vsext,rhoext,wxgll,wzgll,p_sv, &
47  simulation_type,save_forward, &
48  b_absorb_elastic_left,b_absorb_elastic_right, &
49  b_absorb_elastic_bottom,b_absorb_elastic_top, &
50  ib_left,ib_right,ib_bottom,ib_top, &
51  stacey_absorbing_conditions,deltat
52 
53  ! initialfield
54  use specfem_par, only: v0x_left,v0z_left,v0x_right,v0z_right,v0x_bot,v0z_bot, &
55  t0x_left,t0z_left,t0x_right,t0z_right,t0x_bot,t0z_bot, &
56  add_bielak_conditions_bottom,add_bielak_conditions_right, &
57  add_bielak_conditions_top,add_bielak_conditions_left, &
58  initialfield,over_critical_angle, &
59  anglesource,anglesource_refl,a_plane,b_plane,c_plane,c_inc,c_refl,time_offset
60 
61  ! for Bielak
62  use specfem_par, only: x_source,z_source,f0_source,coord
63 
64  implicit none
65 
66  real(kind=CUSTOM_REAL), dimension(NDIM,nglob),intent(inout) :: accel_elastic
67  real(kind=CUSTOM_REAL), dimension(NDIM,nglob),intent(in) :: veloc_elastic
68 
69  ! local parameters
70  integer :: ispecabs,ispec,i,j,iglob
71  real(kind=CUSTOM_REAL) :: weight,xxi,zxi,xgamma,zgamma,jacobian1D
72  real(kind=CUSTOM_REAL) :: nx,nz,vx,vy,vz,vn,rho_vp,rho_vs,tx,ty,tz
73 
74  ! material properties of the elastic medium
75  real(kind=CUSTOM_REAL) :: mul_unrelaxed_elastic,lambdal_unrelaxed_elastic, &
76  lambdaplus2mu_unrelaxed_elastic,kappal,cpl,csl,rhol
77 
78  ! for analytical initial plane wave for Bielak's conditions
79  double precision :: veloc_horiz,veloc_vert,dxUx,dzUx,dxUz,dzUz,traction_x_t0,traction_z_t0
80  integer :: count_left,count_right,count_bottom
81 
82  ! checks if anything to do
83  if (.not. stacey_absorbing_conditions) return
84  if (.not. any_elastic) return
85 
86  ! Clayton-Engquist condition if elastic
87  count_left=1
88  count_right=1
89  count_bottom=1
90  do ispecabs = 1,nelemabs
91 
92  ispec = numabs(ispecabs)
93 
94  ! only for elastic elements, skip others
95  if (.not. ispec_is_elastic(ispec) ) cycle
96 
97  ! get elastic parameters of current spectral element
98  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec))
99  mul_unrelaxed_elastic = poroelastcoef(2,1,kmato(ispec))
100 
101  lambdaplus2mu_unrelaxed_elastic = lambdal_unrelaxed_elastic + two * mul_unrelaxed_elastic
102 
103  rhol = density(1,kmato(ispec))
104 
105  if (axisym) then ! CHECK kappa
106  kappal = lambdal_unrelaxed_elastic + two_thirds*mul_unrelaxed_elastic
107  cpl = sqrt((kappal + four_thirds * mul_unrelaxed_elastic)/rhol) ! CHECK kappa
108  else
109  kappal = lambdal_unrelaxed_elastic + mul_unrelaxed_elastic
110  cpl = sqrt((kappal + mul_unrelaxed_elastic)/rhol) ! CHECK kappa
111  endif
112 
113  csl = sqrt(mul_unrelaxed_elastic/rhol)
114 
115  !--- left absorbing boundary
116  if (codeabs(iedge4,ispecabs)) then
117  i = 1
118  do j = 1,ngllz
119  ! Clayton-Engquist condition if elastic
120  iglob = ibool(i,j,ispec)
121 
122  veloc_horiz = 0._custom_real
123  veloc_vert = 0._custom_real
124  traction_x_t0 = 0._custom_real
125  traction_z_t0 = 0._custom_real
126 
127  ! for analytical initial plane wave for Bielak's conditions
128  ! left or right edge, horizontal normal vector
129  if (add_bielak_conditions_left .and. initialfield) then
130  if (.not. over_critical_angle) then
131  call compute_bielak_conditions(coord,iglob,nglob,it,deltat,dxux,dxuz,dzux,dzuz,veloc_horiz,veloc_vert, &
132  x_source(1), z_source(1), a_plane, b_plane, c_plane, anglesource(1), anglesource_refl, &
133  c_inc, c_refl, time_offset,f0_source(1))
134 
135  traction_x_t0 = lambdaplus2mu_unrelaxed_elastic * dxux + lambdal_unrelaxed_elastic * dzuz
136  traction_z_t0 = mul_unrelaxed_elastic * (dxuz + dzux)
137  else
138  veloc_horiz = v0x_left(count_left,it)
139  veloc_vert = v0z_left(count_left,it)
140  traction_x_t0 = t0x_left(count_left,it)
141  traction_z_t0 = t0z_left(count_left,it)
142  count_left = count_left+1
143  endif
144  endif
145 
146  ! external velocity model
147  if (assign_external_model) then
148  cpl = vpext(i,j,ispec)
149  csl = vsext(i,j,ispec)
150  rhol = rhoext(i,j,ispec)
151  endif
152 
153  rho_vp = rhol*cpl
154  rho_vs = rhol*csl
155 
156  ! normal pointing left
157  xgamma = - xiz(i,j,ispec) * jacobian(i,j,ispec)
158  zgamma = + xix(i,j,ispec) * jacobian(i,j,ispec)
159  jacobian1d = sqrt(xgamma**2 + zgamma**2)
160  nx = - zgamma / jacobian1d
161  nz = + xgamma / jacobian1d
162 
163  if (p_sv) then
164  ! P_SV case
165  vx = veloc_elastic(1,iglob) - veloc_horiz
166  vz = veloc_elastic(2,iglob) - veloc_vert
167  vn = nx*vx + nz*vz
168 
169  ! first-order Clayton-Engquist (1977) uses tractions
170  ! T_normal = - rho * vp * veloc_normal
171  ! with velocity's normal component: veloc_normal = vn * normal
172  ! T_tangential = - rho * vs * veloc_tangential
173  ! with velocity's tangential component: veloc_tangential = v - vn * normal
174  ! total traction
175  ! T = T_normal + T_tangential
176  tx = rho_vp*vn*nx + rho_vs*(vx-vn*nx)
177  tz = rho_vp*vn*nz + rho_vs*(vz-vn*nz)
178 
179  ! second-order Stacey (1988), named P3 uses derivatives du / dx_tangential
180  ! T_tangential2 = + rho * vs * (2 * vs - vp) du_normal / dx_tangential
181  ! and
182  ! T_normal2 = - rho * vs * (2 * vs - vp) du_tangential / dx_tangential
183  ! and total traction is
184  ! T_normal + Tnormal2 + T_tangential + T_tangential2
185  ! derivatives: grad_tangential(f) is the dot-product grad(f) * vector_tangential
186 
187 ! daniel debug: todo - still one could try to evaluate this second-order boundary...
188 !
189 ! ! u_normal
190 ! displn = nx*displ_elastic(1,iglob) + nz*displ_elastic(2,iglob)
191 ! displnx = displn * nx
192 ! displnz = displn * nz
193 ! ! u_tangential
194 ! displtx = displ_elastic(1,iglob) - displn * nx
195 ! displtz = displ_elastic(2,iglob) - displn * nz
196 ! displt_norm = sqrt(displtx**2 + displtz**2)
197 ! if (displt_norm > 0._CUSTOM_REAL) then
198 ! ! unit vector in tangential direction
199 ! unit_tx = displtx / displt_norm
200 ! unit_tz = displtz / displt_norm
201 !
202 ! ! derivative along x
203 ! tempx1l = 0._CUSTOM_REAL
204 ! tempx2l = 0._CUSTOM_REAL
205 ! do k = 1,NGLLX
206 ! iglob = ibool(k,j,ispec)
207 ! tempx1l = tempx1l + displnx * hprime_xx(i,k)
208 ! tempx2l = tempx2l + displnz * hprime_xx(i,k)
209 ! enddo
210 ! ! derivative along z
211 ! tempz1l = 0._CUSTOM_REAL
212 ! tempz2l = 0._CUSTOM_REAL
213 ! do k = 1,NGLLX
214 ! iglob = ibool(i,k,ispec)
215 ! tempz1l = tempz1l + displnx * hprime_zz(j,k)
216 ! tempz2l = tempz2l + displnz * hprime_zz(j,k)
217 ! enddo
218 ! ! derivatives of displacement du_normal / du
219 ! dux_dxl = (tempx1l*xixl + tempz1l*gammaxl)
220 ! dux_dzl = (tempx1l*xizl + tempz1l*gammazl)
221 !
222 ! duz_dxl = (tempx2l*xixl + tempz2l*gammaxl)
223 ! duz_dzl = (tempx2l*xizl + tempz2l*gammazl)
224 !
225 ! ! derivative du_normal / dx_tangential (along tangential vector)
226 ! derivx = (dux_dxl * unit_tx + dux_dzl * unit_tz) * unit_tx
227 ! derivz = (duz_dxl * unit_tx + duz_dzl * unit_tz) * unit_tz
228 !
229 ! tx = tx + rho_vs *( 2._CUSTOM_REAL * csl - cpl) * derivx
230 ! tz = tz + rho_vs *( 2._CUSTOM_REAL * csl - cpl) * derivz
231 ! endif
232 
233  else
234  ! SH case
235  vy = veloc_elastic(1,iglob)
236  ty = rho_vs*vy
237  endif
238 
239  weight = jacobian1d * wzgll(j)
240 
241  if (p_sv) then
242  ! P_SV case
243  accel_elastic(1,iglob) = accel_elastic(1,iglob) - (tx + traction_x_t0)*weight
244  accel_elastic(2,iglob) = accel_elastic(2,iglob) - (tz + traction_z_t0)*weight
245  else
246  ! SH case
247  accel_elastic(1,iglob) = accel_elastic(1,iglob) - ty*weight
248  endif
249 
250  if (save_forward .and. simulation_type == 1) then
251  if (p_sv) then
252  ! P-SV waves
253  b_absorb_elastic_left(1,j,ib_left(ispecabs),it) = (tx + traction_x_t0)*weight
254  b_absorb_elastic_left(2,j,ib_left(ispecabs),it) = (tz + traction_z_t0)*weight
255  else
256  ! SH (membrane) waves
257  b_absorb_elastic_left(1,j,ib_left(ispecabs),it) = ty*weight
258  endif
259  endif
260  enddo
261  endif ! end of left absorbing boundary
262 
263  !--- right absorbing boundary
264  if (codeabs(iedge2,ispecabs)) then
265  i = ngllx
266  do j = 1,ngllz
267  ! Clayton-Engquist condition if elastic
268  iglob = ibool(i,j,ispec)
269 
270  veloc_horiz = 0._custom_real
271  veloc_vert = 0._custom_real
272  traction_x_t0 = 0._custom_real
273  traction_z_t0 = 0._custom_real
274 
275  ! for analytical initial plane wave for Bielak's conditions
276  ! left or right edge, horizontal normal vector
277  if (add_bielak_conditions_right .and. initialfield) then
278  if (.not. over_critical_angle) then
279  call compute_bielak_conditions(coord,iglob,nglob,it,deltat,dxux,dxuz,dzux,dzuz,veloc_horiz,veloc_vert, &
280  x_source(1), z_source(1), a_plane, b_plane, c_plane, anglesource(1), anglesource_refl, &
281  c_inc, c_refl, time_offset,f0_source(1))
282 
283  traction_x_t0 = lambdaplus2mu_unrelaxed_elastic * dxux + lambdal_unrelaxed_elastic * dzuz
284  traction_z_t0 = mul_unrelaxed_elastic * (dxuz + dzux)
285  else
286  veloc_horiz = v0x_right(count_right,it)
287  veloc_vert = v0z_right(count_right,it)
288  traction_x_t0 = t0x_right(count_right,it)
289  traction_z_t0 = t0z_right(count_right,it)
290  count_right = count_right+1
291  endif
292  endif
293 
294  ! external velocity model
295  if (assign_external_model) then
296  cpl = vpext(i,j,ispec)
297  csl = vsext(i,j,ispec)
298  rhol = rhoext(i,j,ispec)
299  endif
300 
301  rho_vp = rhol*cpl
302  rho_vs = rhol*csl
303 
304  ! normal pointing right
305  xgamma = - xiz(i,j,ispec) * jacobian(i,j,ispec)
306  zgamma = + xix(i,j,ispec) * jacobian(i,j,ispec)
307  jacobian1d = sqrt(xgamma**2 + zgamma**2)
308  nx = + zgamma / jacobian1d
309  nz = - xgamma / jacobian1d
310 
311  if (p_sv) then
312  ! P_SV case
313  vx = veloc_elastic(1,iglob) - veloc_horiz
314  vz = veloc_elastic(2,iglob) - veloc_vert
315  vn = nx*vx + nz*vz
316  tx = rho_vp*vn*nx + rho_vs*(vx-vn*nx)
317  tz = rho_vp*vn*nz + rho_vs*(vz-vn*nz)
318  else
319  ! SH case
320  vy = veloc_elastic(1,iglob)
321  ty = rho_vs*vy
322  endif
323 
324  weight = jacobian1d * wzgll(j)
325 
326  if (p_sv) then
327  ! P_SV case
328  accel_elastic(1,iglob) = accel_elastic(1,iglob) - (tx - traction_x_t0)*weight
329  accel_elastic(2,iglob) = accel_elastic(2,iglob) - (tz - traction_z_t0)*weight
330  else
331  ! SH case
332  accel_elastic(1,iglob) = accel_elastic(1,iglob) - ty*weight
333  endif
334 
335  if (save_forward .and. simulation_type == 1) then
336  if (p_sv) then
337  ! P-SV waves
338  b_absorb_elastic_right(1,j,ib_right(ispecabs),it) = (tx - traction_x_t0)*weight
339  b_absorb_elastic_right(2,j,ib_right(ispecabs),it) = (tz - traction_z_t0)*weight
340  else
341  ! SH (membrane) waves
342  b_absorb_elastic_right(1,j,ib_right(ispecabs),it) = ty*weight
343  endif
344  endif
345  enddo
346  endif ! end of right absorbing boundary
347 
348  !--- bottom absorbing boundary
349  if (codeabs(iedge1,ispecabs)) then
350  j = 1
351  do i = 1,ngllx
352  ! Clayton-Engquist condition if elastic
353  iglob = ibool(i,j,ispec)
354 
355  veloc_horiz = 0._custom_real
356  veloc_vert = 0._custom_real
357  traction_x_t0 = 0._custom_real
358  traction_z_t0 = 0._custom_real
359 
360  ! for analytical initial plane wave for Bielak's conditions
361  ! top or bottom edge, vertical normal vector
362  if (add_bielak_conditions_bottom .and. initialfield) then
363  if (.not. over_critical_angle) then
364  call compute_bielak_conditions(coord,iglob,nglob,it,deltat,dxux,dxuz,dzux,dzuz,veloc_horiz,veloc_vert, &
365  x_source(1), z_source(1), a_plane, b_plane, c_plane, anglesource(1), anglesource_refl, &
366  c_inc, c_refl, time_offset,f0_source(1))
367 
368  traction_x_t0 = mul_unrelaxed_elastic * (dxuz + dzux)
369  traction_z_t0 = lambdal_unrelaxed_elastic * dxux + lambdaplus2mu_unrelaxed_elastic * dzuz
370  else
371  veloc_horiz = v0x_bot(count_bottom,it)
372  veloc_vert = v0z_bot(count_bottom,it)
373  traction_x_t0 = t0x_bot(count_bottom,it)
374  traction_z_t0 = t0z_bot(count_bottom,it)
375  count_bottom = count_bottom+1
376  endif
377  endif
378 
379  ! external velocity model
380  if (assign_external_model) then
381  cpl = vpext(i,j,ispec)
382  csl = vsext(i,j,ispec)
383  rhol = rhoext(i,j,ispec)
384  endif
385 
386  rho_vp = rhol*cpl
387  rho_vs = rhol*csl
388 
389  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
390  zxi = - gammax(i,j,ispec) * jacobian(i,j,ispec)
391  jacobian1d = sqrt(xxi**2 + zxi**2)
392  nx = + zxi / jacobian1d
393  nz = - xxi / jacobian1d
394 
395  if (p_sv) then
396  ! P_SV case
397  vx = veloc_elastic(1,iglob) - veloc_horiz
398  vz = veloc_elastic(2,iglob) - veloc_vert
399  vn = nx*vx+nz*vz
400  tx = rho_vp*vn*nx+rho_vs*(vx-vn*nx)
401  tz = rho_vp*vn*nz+rho_vs*(vz-vn*nz)
402  else
403  ! SH case
404  vy = veloc_elastic(1,iglob)
405  ty = rho_vs*vy
406  endif
407 
408 ! exclude corners to make sure there is no contradiction on the normal
409 ! for Stacey absorbing conditions but not for incident plane waves;
410 ! thus subtract nothing i.e. zero in that case
411  if ((codeabs_corner(1,ispecabs) .and. i == 1) .or. (codeabs_corner(2,ispecabs) .and. i == ngllx)) then
412  tx = 0._custom_real
413  ty = 0._custom_real
414  tz = 0._custom_real
415  endif
416 
417  weight = jacobian1d * wxgll(i)
418 
419  if (p_sv) then
420  ! P_SV case
421  accel_elastic(1,iglob) = accel_elastic(1,iglob) - (tx + traction_x_t0)*weight
422  accel_elastic(2,iglob) = accel_elastic(2,iglob) - (tz + traction_z_t0)*weight
423  else
424  ! SH case
425  accel_elastic(1,iglob) = accel_elastic(1,iglob) - ty*weight
426  endif
427 
428  if (save_forward .and. simulation_type == 1) then
429  if (p_sv) then
430  ! P-SV waves
431  b_absorb_elastic_bottom(1,i,ib_bottom(ispecabs),it) = (tx + traction_x_t0)*weight
432  b_absorb_elastic_bottom(2,i,ib_bottom(ispecabs),it) = (tz + traction_z_t0)*weight
433  else
434  ! SH (membrane) waves
435  b_absorb_elastic_bottom(1,i,ib_bottom(ispecabs),it) = ty*weight
436  endif
437  endif
438  enddo
439  endif ! end of bottom absorbing boundary
440 
441  !--- top absorbing boundary
442  if (codeabs(iedge3,ispecabs)) then
443  j = ngllz
444  do i = 1,ngllx
445  ! Clayton-Engquist condition if elastic
446  iglob = ibool(i,j,ispec)
447 
448  veloc_horiz = 0._custom_real
449  veloc_vert = 0._custom_real
450  traction_x_t0 = 0._custom_real
451  traction_z_t0 = 0._custom_real
452 
453  ! for analytical initial plane wave for Bielak's conditions
454  ! top or bottom edge, vertical normal vector
455  if (add_bielak_conditions_top .and. initialfield) then
456  ! at the top there is no test for whether we are above the critical angle
457  ! because a critical angle can only exist when the top edge is a free surface, not in an infinite medium
458  call compute_bielak_conditions(coord,iglob,nglob,it,deltat,dxux,dxuz,dzux,dzuz,veloc_horiz,veloc_vert, &
459  x_source(1), z_source(1), a_plane, b_plane, c_plane, anglesource(1), anglesource_refl, &
460  c_inc, c_refl, time_offset,f0_source(1))
461 
462  traction_x_t0 = mul_unrelaxed_elastic * (dxuz + dzux)
463  traction_z_t0 = lambdal_unrelaxed_elastic * dxux + lambdaplus2mu_unrelaxed_elastic * dzuz
464  endif
465 
466  ! external velocity model
467  if (assign_external_model) then
468  cpl = vpext(i,j,ispec)
469  csl = vsext(i,j,ispec)
470  rhol = rhoext(i,j,ispec)
471  endif
472 
473  rho_vp = rhol*cpl
474  rho_vs = rhol*csl
475 
476  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
477  zxi = - gammax(i,j,ispec) * jacobian(i,j,ispec)
478  jacobian1d = sqrt(xxi**2 + zxi**2)
479  nx = - zxi / jacobian1d
480  nz = + xxi / jacobian1d
481 
482  if (p_sv) then
483  ! P_SV case
484  vx = veloc_elastic(1,iglob) - veloc_horiz
485  vz = veloc_elastic(2,iglob) - veloc_vert
486  vn = nx*vx+nz*vz
487  tx = rho_vp*vn*nx+rho_vs*(vx-vn*nx)
488  tz = rho_vp*vn*nz+rho_vs*(vz-vn*nz)
489  else
490  ! SH case
491  vy = veloc_elastic(1,iglob)
492  ty = rho_vs*vy
493  endif
494 
495 ! exclude corners to make sure there is no contradiction on the normal
496 ! for Stacey absorbing conditions but not for incident plane waves;
497 ! thus subtract nothing i.e. zero in that case
498  if ((codeabs_corner(3,ispecabs) .and. i == 1) .or. (codeabs_corner(4,ispecabs) .and. i == ngllx)) then
499  tx = 0._custom_real
500  ty = 0._custom_real
501  tz = 0._custom_real
502  endif
503 
504  weight = jacobian1d * wxgll(i)
505 
506  if (p_sv) then
507  ! P_SV case
508  accel_elastic(1,iglob) = accel_elastic(1,iglob) - (tx - traction_x_t0)*weight
509  accel_elastic(2,iglob) = accel_elastic(2,iglob) - (tz - traction_z_t0)*weight
510  else
511  ! SH case
512  accel_elastic(1,iglob) = accel_elastic(1,iglob) - ty*weight
513  endif
514 
515  if (save_forward .and. simulation_type == 1) then
516  if (p_sv) then
517  ! P-SV waves
518  b_absorb_elastic_top(1,i,ib_top(ispecabs),it) = (tx - traction_x_t0)*weight
519  b_absorb_elastic_top(2,i,ib_top(ispecabs),it) = (tz - traction_z_t0)*weight
520  else
