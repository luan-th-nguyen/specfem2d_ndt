33 
34  subroutine create_color_image()
35 
36 ! display a given field as a red and blue color JPEG image
37 
38 ! to display the snapshots : display image*.jpg
39 
40  use constants, only: tinyval,hugeval,stability_threshold,output_files,imain
41 
42  use specfem_par, only: myrank,it,nsources,p_sv,nrec
43 
44  use specfem_par_movie, only: image_color_data,iglob_image_color,nx_image_color,nz_image_color, &
45  isnapshot_number,cutsnaps,image_color_vp_display, &
46  use_snapshot_number_in_filename,power_display_color, &
47  draw_sources_and_receivers, &
48  ix_image_color_source,iy_image_color_source,ix_image_color_receiver,iy_image_color_receiver, &
49  use_constant_max_amplitude,constant_max_amplitude_to_use
50 
51  implicit none
52 
53  ! local parameters
54  integer :: i
55 
56  ! for the JPEG library
57  character(len=1), dimension(3,NX_IMAGE_color,NZ_IMAGE_color) :: JPEG_raw_image
58  integer :: ix,iy,R,G,B
59  double precision :: amplitude_max,normalized_value,vpmin,vpmax,x1
60  character(len=100) :: filename
61  logical :: do_warning
62 
63  ! size of cross and square in pixels drawn to represent the source and the receivers in JPEG pictures
64  integer :: half_width_cross, thickness_cross, half_size_square
65 
66 ! make the size of the source and receiver symbols depend on the size of the picture
67 ! using a rule of thumb
68  thickness_cross = 1
69  if (nx_image_color > 2000 .or. nz_image_color > 2000) then
70  half_width_cross = 6
71  half_size_square = 4
72  else if (nx_image_color <= 100 .or. nz_image_color <= 100) then
73  half_width_cross = 2
74  half_size_square = 1
75  else if (nx_image_color <= 250 .or. nz_image_color <= 250) then
76  half_width_cross = 3
77  half_size_square = 2
78  else
79  half_width_cross = 5
80  half_size_square = 3
81  endif
82 
83 ! open the image file
84 ! slightly change the beginning of the file name depending if we use the time step of the image number, to avoid confusion
85  if (use_snapshot_number_in_filename) then
86  isnapshot_number = isnapshot_number + 1
87  write(filename,"(a,i7.7,a)") trim(output_files)//'img',isnapshot_number,'.jpg'
88  else
89  write(filename,"(a,i7.7,a)") trim(output_files)//'image',it,'.jpg'
90  endif
91 
92 ! compute maximum amplitude
93  if (.not. use_constant_max_amplitude) then
94  amplitude_max = maxval(abs(image_color_data))
95  else
96  amplitude_max = constant_max_amplitude_to_use
97 ! in case of a pre-defined and constant maximum, truncate all values that are outside that constant range
98  where(image_color_data > +constant_max_amplitude_to_use) image_color_data = +constant_max_amplitude_to_use
99  where(image_color_data < -constant_max_amplitude_to_use) image_color_data = -constant_max_amplitude_to_use
100  endif
101  ! user output
102  write(imain,*) 'Color image maximum amplitude = ',amplitude_max
103 
104 
105 ! this trick checks for NaN (Not a Number), which is not even equal to itself
106  if (amplitude_max > stability_threshold .or. amplitude_max < 0 .or. amplitude_max /= amplitude_max) then
107  print *,'Warning: failed creating color image, maximum value of amplitude in image color is invalid'
108  print *,'amplitude max = ',amplitude_max,' with threshold at ', stability_threshold
109  print *,'Please check your simulation setup...'
110  call exit_mpi(myrank,'code became unstable and blew up (image_color_data)')
111  endif
112  do_warning = .false.
113 
114  vpmin = hugeval
115  vpmax = tinyval
116  do iy= 1,nz_image_color
117  do ix= 1,nx_image_color
118 ! negative values in image_color_vp_display are a flag indicating a water layer to color in light blue later
119  if (iglob_image_color(ix,iy) > -1 .and. image_color_vp_display(ix,iy) >= 0) then
120  vpmin = min(vpmin,image_color_vp_display(ix,iy))
121  vpmax = max(vpmax,image_color_vp_display(ix,iy))
122  endif
123  enddo
124  enddo
125 
126 ! in the image format, the image starts in the upper-left corner
127  do iy=nz_image_color,1,-1
128  do ix= 1,nx_image_color
129 
130 ! check if pixel is defined or not (can be above topography for instance)
131  if (iglob_image_color(ix,iy) == -1) then
132 
133 ! use white to display undefined region above topography to avoid visual confusion with a water layer
134  r = 255
135  g = 255
136  b = 255
137 
138 ! suppress small amplitudes considered as noise and display the background velocity model instead
139  else if (abs(image_color_data(ix,iy)) < amplitude_max * cutsnaps) then
140 
141 ! use P velocity model as background where amplitude is negligible
142  if ((p_sv) .and. ((vpmax-vpmin)/max(vpmin, tinyval) > 0.02d0)) then
143  x1 = (image_color_vp_display(ix,iy)-vpmin)/(vpmax-vpmin)
144  else
145  x1 = 0.5d0
146  endif
147 
148 ! rescale to avoid very dark gray levels
149  x1 = x1*0.7 + 0.2
150  if (x1 > 1.d0) x1=1.d0
151 
152 ! invert scale: white = vpmin, dark gray = vpmax
153  x1 = 1.d0 - x1
154 
155 ! map to [0,255]
156  x1 = x1 * 255.d0
157 
158  r = nint(x1)
159  if (r < 0) r = 0
160  if (r > 255) r = 255
161  g = r
162  b = r
163 
164 ! negative values in image_color_vp_display are a flag indicating a water layer to color in light blue
165  if (image_color_vp_display(ix,iy) < 0) then
166 ! use light blue to display water
167 !!!!!! R = 204
168 !!!!!! G = 255
169 !!!!!! B = 255
170  r = 135 !!! LightSkyBlue
171  g = 206
172  b = 250
173  endif
174 
175  else
176 
177 ! define normalized image data in [-1:1] and convert to nearest integer
178 ! keeping in mind that data values can be negative
179  if (amplitude_max >= tinyval) then
180  normalized_value = image_color_data(ix,iy) / amplitude_max
181  else
182  normalized_value = image_color_data(ix,iy) / tinyval
183  endif
184  ! check value (isNaN)
185  if (normalized_value /= normalized_value) then
186  ! will be set to zero
187  normalized_value = 0.d0
188  do_warning = .true.
189  endif
190 
191 ! suppress values outside of [-1:+1]
192  if (normalized_value < -1.d0) normalized_value = -1.d0
193  if (normalized_value > 1.d0) normalized_value = 1.d0
194 
195 ! use red if positive value, blue if negative, no green
196  if (normalized_value >= 0.d0) then
197  r = nint(255.d0*normalized_value**power_display_color)
198  g = 0
199  b = 0
200  else
201  r = 0
202  g = 0
203  b = nint(255.d0*abs(normalized_value)**power_display_color)
204  endif
205 
206  endif
207 
208 ! for JPEG
209  jpeg_raw_image(1,ix,nz_image_color-iy+1) = char(r)
210  jpeg_raw_image(2,ix,nz_image_color-iy+1) = char(g)
211  jpeg_raw_image(3,ix,nz_image_color-iy+1) = char(b)
212 
213  enddo
214  enddo
215 
216  if (do_warning) then
217  print *,'Warning: normalized value is invalid, likely due to unstable simulation! process rank is ',myrank
218  print *,'Warning: normalized_value = ',normalized_value,' will be set to zero'
219  endif
220 
221 !
222 !---- draw position of the sources and receivers
223 !
224  if (draw_sources_and_receivers) then
225 
226 ! draw position of the sources with orange crosses
227  do i = 1,nsources
228 
229 ! avoid edge effects for source or receiver symbols that can be partly outside of the image
230  do iy = max(iy_image_color_source(i) - half_width_cross,1), min(iy_image_color_source(i) + half_width_cross,nz_image_color)
231  do ix = max(ix_image_color_source(i) - thickness_cross,1), min(ix_image_color_source(i) + thickness_cross,nx_image_color)
232 ! use orange color
233  r = 255
234  g = 157
235  b = 0
236 ! for JPEG
237  jpeg_raw_image(1,ix,nz_image_color-iy+1) = char(r)
238  jpeg_raw_image(2,ix,nz_image_color-iy+1) = char(g)
239  jpeg_raw_image(3,ix,nz_image_color-iy+1) = char(b)
240  enddo
241  enddo
242 
243 ! avoid edge effects for source or receiver symbols that can be partly outside of the image
244  do iy = max(iy_image_color_source(i) - thickness_cross,1), min(iy_image_color_source(i) + thickness_cross,nz_image_color)
245  do ix = max(ix_image_color_source(i) - half_width_cross,1), min(ix_image_color_source(i) + half_width_cross,nx_image_color)
246 ! use orange color
247  r = 255
248  g = 157
249  b = 0
250 ! for JPEG
251  jpeg_raw_image(1,ix,nz_image_color-iy+1) = char(r)
252  jpeg_raw_image(2,ix,nz_image_color-iy+1) = char(g)
253  jpeg_raw_image(3,ix,nz_image_color-iy+1) = char(b)
254  enddo
255  enddo
256 
257  enddo
258 
259 ! draw position of the receivers with green squares
260  do i = 1,nrec
261 ! avoid edge effects for source or receiver symbols that can be partly outside of the image
262  do iy = max(iy_image_color_receiver(i) - half_size_square,1), &
263  min(iy_image_color_receiver(i) + half_size_square,nz_image_color)
264  do ix = max(ix_image_color_receiver(i) - half_size_square,1), &
265  min(ix_image_color_receiver(i) + half_size_square,nx_image_color)
266 ! use dark green color
267  r = 30
268  g = 180
269  b = 60
270 ! for JPEG
271  jpeg_raw_image(1,ix,nz_image_color-iy+1) = char(r)
272  jpeg_raw_image(2,ix,nz_image_color-iy+1) = char(g)
273  jpeg_raw_image(3,ix,nz_image_color-iy+1) = char(b)
274  enddo
275  enddo
276  enddo
277 
278  endif
279 
280 ! for JPEG
281  call write_jpeg_image(jpeg_raw_image,nx_image_color,nz_image_color,filename)
282 
283  end subroutine create_color_image
