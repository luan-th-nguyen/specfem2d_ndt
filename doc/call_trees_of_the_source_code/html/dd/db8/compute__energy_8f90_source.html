34  subroutine compute_and_output_energy()
35 
36  use constants, only: iout_energy,custom_real
37 
38  use specfem_par, only: gpu_mode,myrank,it,deltat,kinetic_energy,potential_energy,t0
39 
40  implicit none
41 
42  ! local parameters
43  real(kind=CUSTOM_REAL) :: kinetic_energy_total,potential_energy_total
44 
45  ! safety check
46  if (gpu_mode) stop 'Error computing energy for output is not implemented on GPUs yet'
47

50

59  endif
60 
61  end subroutine compute_and_output_energy

65 !
66 
67  subroutine compute_energy()
68 
69 ! compute kinetic and potential energy in the solid (acoustic elements are excluded)
70 
71  use constants, only: custom_real,ngllx,ngllz,nglj,ndim,zero,two
72 
73  use specfem_par, only: axisym,is_on_the_axis,myrank,nspec,kinetic_energy,potential_energy, &
74  ibool,hprime_xx,hprime_zz,hprimebar_xx,xix,xiz,gammax,gammaz,jacobian,wxgll,wzgll, &
75  displ_elastic,veloc_elastic, &
76  displs_poroelastic,displw_poroelastic,velocs_poroelastic,velocw_poroelastic, &
77  potential_dot_acoustic,potential_dot_gravitoacoustic,potential_dot_gravito, &
78  vsext,vpext,rhoext,poroelastcoef,density,kmato,assign_external_model, &
79  ispec_is_poroelastic,ispec_is_elastic,p_sv
80 
81  implicit none
82 
83 ! local variables
84  integer :: i,j,k,ispec
85  real(kind=CUSTOM_REAL) :: cpl,csl,kappal
86  real(kind=CUSTOM_REAL) :: mu_G,lambdal_G,lambdalplus2mul_G
87 
88  ! Jacobian matrix and determinant
89  double precision :: xixl,xizl,gammaxl,gammazl,jacobianl
90  double precision :: rhol
91  ! to evaluate cpI, cpII, and cs, and rI (poroelastic medium)
92  double precision :: phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar
93  double precision :: D_biot,H_biot,C_biot,M_biot
94  double precision :: mul_unrelaxed_elastic,lambdal_unrelaxed_elastic,lambdaplus2mu_unrelaxed_elastic
95 
96  double precision :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
97  double precision :: dwx_dxi,dwx_dgamma,dwz_dxi,dwz_dgamma
98  double precision :: dux_dxl,duz_dxl,dux_dzl,duz_dzl
99  double precision :: dwx_dxl,dwz_dxl,dwx_dzl,dwz_dzl
100  ! vector field in an element
101  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ) :: vector_field_element
102  ! pressure in an element
103  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: pressure_element
104 
105  ! loop over spectral elements
106  do ispec = 1,nspec ! TODO uncomment
107  !---
108  !--- elastic spectral element
109  !---
110  if (ispec_is_elastic(ispec)) then
111 
112  ! checks wave type
113  if (.not. p_sv) then
114  call exit_mpi(myrank,'output energy for SH waves not implemented yet')
115  endif
116 
117  ! get relaxed elastic parameters of current spectral element
118  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec))
119  mul_unrelaxed_elastic = poroelastcoef(2,1,kmato(ispec))
120  lambdaplus2mu_unrelaxed_elastic = poroelastcoef(3,1,kmato(ispec))
121 
122  rhol = density(1,kmato(ispec))
123 
124  ! double loop over GLL points
125  do j = 1,ngllz
126  do i = 1,ngllx
127 
128  !--- if external medium, get elastic parameters of current grid point
129  if (assign_external_model) then
130  cpl = vpext(i,j,ispec)
131  csl = vsext(i,j,ispec)
132  rhol = rhoext(i,j,ispec)
133  mul_unrelaxed_elastic = rhol*csl*csl
134  lambdal_unrelaxed_elastic = rhol*cpl*cpl - two*mul_unrelaxed_elastic
135  lambdaplus2mu_unrelaxed_elastic = lambdal_unrelaxed_elastic + two*mul_unrelaxed_elastic
136  endif
137 
138  ! derivative along x and along z
139  dux_dxi = 0._custom_real
140  duz_dxi = 0._custom_real
141 
142  dux_dgamma = 0._custom_real
143  duz_dgamma = 0._custom_real
144 
145  ! first double loop over GLL points to compute and store gradients
146  ! we can merge the two loops because NGLLX == NGLLZ
147 
148  if (axisym) then
149  if (is_on_the_axis(ispec)) then
150  do k = 1,nglj
151  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
152  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprimebar_xx(i,k)
153  enddo
154  else
155  do k = 1,nglj
156  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
157  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
158  enddo
159  endif
160  else
161  do k = 1,ngllx
162  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
163  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
164  enddo
165  endif
166 
167  do k = 1,ngllx
168  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
169  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
170  enddo
171 
172  xixl = xix(i,j,ispec)
173  xizl = xiz(i,j,ispec)
174  gammaxl = gammax(i,j,ispec)
175  gammazl = gammaz(i,j,ispec)
176  jacobianl = jacobian(i,j,ispec)
177 
178  ! derivatives of displacement
179  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
180  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
181 
182  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
183  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
184 
185  ! compute kinetic energy ! TODO ABAB This integral over space should be adapted for axisym geometries if needed
186  kinetic_energy = kinetic_energy &
187  + rhol * (veloc_elastic(1,ibool(i,j,ispec))**2 + veloc_elastic(2,ibool(i,j,ispec))**2) &
188  *wxgll(i)*wzgll(j)*jacobianl / two
189 
190  ! compute potential energy ! TODO ABAB This integral over space should be adapted for axisym geometries if needed
191  potential_energy = potential_energy &
192  + (lambdaplus2mu_unrelaxed_elastic*dux_dxl**2 &
193  + lambdaplus2mu_unrelaxed_elastic*duz_dzl**2 &
194  + two*lambdal_unrelaxed_elastic*dux_dxl*duz_dzl &
195  + mul_unrelaxed_elastic*(dux_dzl + duz_dxl)**2)*wxgll(i)*wzgll(j)*jacobianl / two
196 
197  enddo
198  enddo
199 
200  !---
201  !--- poroelastic spectral element
202  !---
203  else if (ispec_is_poroelastic(ispec)) then
204 
205  ! get unrelaxed elastic parameters of current spectral element
206  !for now replaced by solid, fluid, and frame parameters of current spectral element
207 
208  ! gets poroelastic material
209  call get_poroelastic_material(ispec,phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar)
210 
211  ! Biot coefficients for the input phi
212  call get_poroelastic_biot_coeff(phi,kappa_s,kappa_f,kappa_fr,mu_fr,d_biot,h_biot,c_biot,m_biot)
213 
214  !The RHS has the form : div T -phi/c div T_f + phi/ceta_fk^-1.partial t w
215  !where T = G:grad u_s + C div w I
216  !and T_f = C div u_s I + M div w I
217  !we are expressing lambdaplus2mu, lambda, and mu for G, C, and M
218  mu_g = mu_fr
219  lambdal_g = h_biot - two*mu_fr
220  lambdalplus2mul_g = lambdal_g + two*mu_g
221 
222  ! first double loop over GLL points to compute and store gradients
223  do j = 1,ngllz
224  do i = 1,ngllx
225 
226  ! derivative along x and along z
227  dux_dxi = zero
228  duz_dxi = zero
229 
230  dux_dgamma = zero
231  duz_dgamma = zero
232 
233  dwx_dxi = zero
234  dwz_dxi = zero
235 
236  dwx_dgamma = zero
237  dwz_dgamma = zero
238 
239  ! first double loop over GLL points to compute and store gradients
240  ! we can merge the two loops because NGLLX == NGLLZ
241  do k = 1,ngllx
242  dux_dxi = dux_dxi + displs_poroelastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
243  duz_dxi = duz_dxi + displs_poroelastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
244  dux_dgamma = dux_dgamma + displs_poroelastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
245  duz_dgamma = duz_dgamma + displs_poroelastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
246 
247 
248  dwx_dxi = dwx_dxi + displw_poroelastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
249  dwz_dxi = dwz_dxi + displw_poroelastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
250  dwx_dgamma = dwx_dgamma + displw_poroelastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
251  dwz_dgamma = dwz_dgamma + displw_poroelastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
252  enddo
253 
254  xixl = xix(i,j,ispec)
255  xizl = xiz(i,j,ispec)
256  gammaxl = gammax(i,j,ispec)
257  gammazl = gammaz(i,j,ispec)
258  jacobianl = jacobian(i,j,ispec)
259 
260  ! derivatives of displacement
261  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
262  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
263 
264  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
265  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
266 
267  dwx_dxl = dwx_dxi*xixl + dwx_dgamma*gammaxl
268  dwx_dzl = dwx_dxi*xizl + dwx_dgamma*gammazl
269 
270  dwz_dxl = dwz_dxi*xixl + dwz_dgamma*gammaxl
271  dwz_dzl = dwz_dxi*xizl + dwz_dgamma*gammazl
272 
273  ! compute potential energy
274  potential_energy = potential_energy &
275  + ( lambdalplus2mul_g*dux_dxl**2 &
276  + lambdalplus2mul_g*duz_dzl**2 &
277  + two*lambdal_g*dux_dxl*duz_dzl + mu_g*(dux_dzl + duz_dxl)**2 &
278  + two*c_biot*dwx_dxl*dux_dxl + two*c_biot*dwz_dzl*duz_dzl &
279  + two*c_biot*(dwx_dxl*duz_dzl + dwz_dzl*dux_dxl) &
280  + m_biot*dwx_dxl**2 + m_biot*dwz_dzl**2 &
281  + two*m_biot*dwx_dxl*dwz_dzl )*wxgll(i)*wzgll(j)*jacobianl / two
282 
283  ! compute kinetic energy
284  if (phi > 0.0d0) then
285  kinetic_energy = kinetic_energy &
286  + ( rho_bar*(velocs_poroelastic(1,ibool(i,j,ispec))**2 &
287  + velocs_poroelastic(2,ibool(i,j,ispec))**2) &
288  + rho_f*tort/phi*(velocw_poroelastic(1,ibool(i,j,ispec))**2 &
289  + velocw_poroelastic(2,ibool(i,j,ispec))**2) &
290  + rho_f*(velocs_poroelastic(1,ibool(i,j,ispec))*velocw_poroelastic(1,ibool(i,j,ispec)) &
291  + velocs_poroelastic(2,ibool(i,j,ispec))*velocw_poroelastic(2,ibool(i,j,ispec))) &
292  )*wxgll(i)*wzgll(j)*jacobianl / two
293  else
294  kinetic_energy = kinetic_energy &
295  + rho_s*(velocs_poroelastic(1,ibool(i,j,ispec))**2 &
296  + velocs_poroelastic(2,ibool(i,j,ispec))**2)*wxgll(i)*wzgll(j)*jacobianl / two
297  endif
298  enddo
299  enddo
300 
301  !---
302  !--- acoustic spectral element
303  !---
304  else
305 
306  ! for the definition of potential energy in an acoustic fluid, see for instance
307  ! equation (23) of M. Maess et al., Journal of Sound and Vibration 296 (2006) 264-276
308 
309  ! in case of an acoustic medium, a potential Chi of (density * displacement) is used as in Chaljub and Valette,
310  ! Geophysical Journal International, vol. 158, p. 131-141 (2004) and *NOT* a velocity potential
311  ! as in Komatitsch and Tromp, Geophysical Journal International, vol. 150, p. 303-318 (2002).
312  ! This permits acoustic-elastic coupling based on a non-iterative time scheme.
313  ! Displacement is then: u = grad(Chi) / rho
314  ! Velocity is then: v = grad(Chi_dot) / rho (Chi_dot being the time derivative of Chi)
315  ! and pressure is: p = - Chi_dot_dot (Chi_dot_dot being the time second derivative of Chi).
316 
317  ! compute pressure in this element
318  call compute_pressure_one_element(ispec,pressure_element)
319 
320  ! compute velocity vector field in this element
321  call compute_vector_one_element(potential_dot_acoustic,potential_dot_gravitoacoustic, &
322  potential_dot_gravito,veloc_elastic,velocs_poroelastic,ispec,vector_field_element)
323 
324  ! get velocity and density in current spectral element
325  if (.not. assign_external_model) then
326  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec))
327  rhol = density(1,kmato(ispec))
328  kappal = lambdal_unrelaxed_elastic
329  cpl = sqrt(kappal/rhol)
330  endif
331 
332  ! double loop over GLL points
333  do j = 1,ngllz
334  do i = 1,ngllx
335 
336  !--- if external medium, get density of current grid point
337  if (assign_external_model) then
338  cpl = vpext(i,j,ispec)
339  rhol = rhoext(i,j,ispec)
340  endif
341 
342  jacobianl = jacobian(i,j,ispec)
343 
344  ! compute kinetic energy
345  kinetic_energy = kinetic_energy &
346  + rhol*(vector_field_element(1,i,j)**2 + vector_field_element(2,i,j)**2) *wxgll(i)*wzgll(j)*jacobianl / two
347 
348  ! compute potential energy
349  potential_energy = potential_energy &
350  + (pressure_element(i,j)**2)*wxgll(i)*wzgll(j)*jacobianl / (two * rhol * cpl**2)
351 
352  enddo
353  enddo
354 
355  endif
356  enddo
357 
358  end subroutine compute_energy

363 
364  subroutine compute_energy_fields()
365 
366  ! computes maximum, integrated energy and duration fields
367 
368  use constants, only: custom_real,ngllx,ngllz,ndim,two,zero
369 
370  use specfem_par, only: axisym,is_on_the_axis,nspec,ibool,deltat,veloc_elastic,potential_dot_acoustic,ispec_is_elastic, &
371  ispec_is_poroelastic,integrated_kinetic_energy_field,max_kinetic_energy_field, &
372  integrated_potential_energy_field,max_potential_energy_field,kinetic_effective_duration_field, &
373  potential_effective_duration_field,total_integrated_energy_field,max_total_energy_field, &
374  total_effective_duration_field,potential_dot_gravitoacoustic,potential_dot_gravito,velocs_poroelastic, &
375  poroelastcoef,vsext,vpext,rhoext,density,kmato,assign_external_model,jacobian,displ_elastic, &
376  hprime_xx,hprime_zz,hprimebar_xx,xix,xiz,gammax,gammaz
377 
378  implicit none
379 
380  ! local variables
381  integer :: ispec,i,j,k
382  real(kind=CUSTOM_REAL) :: cpl,csl
383 
384  ! Jacobian matrix and determinant
385  double precision :: xixl,xizl,gammaxl,gammazl,jacobianl
386  double precision :: rhol
387  double precision :: mul_unrelaxed_elastic,lambdal_unrelaxed_elastic,lambdaplus2mu_unrelaxed_elastic
388  double precision :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
389  double precision :: dux_dxl,duz_dxl,dux_dzl,duz_dzl
390 
391  ! vector field in an element
392  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ) :: vector_field_element
393  ! pressure in an element
394  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: pressure_element
395 
396  ! We save the value at the GLL point:
397  i=2
398  j=2
399 
400  ! loop over spectral elements
401  do ispec = 1,nspec
402 
403  !---
404  !--- elastic spectral element
405  !---
406  if (ispec_is_elastic(ispec)) then
407 
408  ! get relaxed elastic parameters of current spectral element
409  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec))
410  mul_unrelaxed_elastic = poroelastcoef(2,1,kmato(ispec))
411  lambdaplus2mu_unrelaxed_elastic = poroelastcoef(3,1,kmato(ispec))
412 
413  rhol = density(1,kmato(ispec))
414 
415  !--- if external medium, get elastic parameters of current grid point
416  if (assign_external_model) then
417  cpl = vpext(i,j,ispec)
418  csl = vsext(i,j,ispec)
419  rhol = rhoext(i,j,ispec)
420  mul_unrelaxed_elastic = rhol*csl*csl
421  lambdal_unrelaxed_elastic = rhol*cpl*cpl - two*mul_unrelaxed_elastic
422  lambdaplus2mu_unrelaxed_elastic = lambdal_unrelaxed_elastic + two*mul_unrelaxed_elastic
423  endif
424 
425  ! derivative along x and along z
426  dux_dxi = 0._custom_real
427  duz_dxi = 0._custom_real
428 
429  dux_dgamma = 0._custom_real
430  duz_dgamma = 0._custom_real
431 
432  ! first double loop over GLL points to compute and store gradients
433  ! we can merge the two loops because NGLLX == NGLLZ
434  if (axisym) then
435  if (is_on_the_axis(ispec)) then
436  do k = 1,ngllx
437  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
438  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprimebar_xx(i,k)
439  enddo
440  else
441  do k = 1,ngllx
442  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
443  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
444  enddo
445  endif
446  else
447  do k = 1,ngllx
448  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
449  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
450  enddo
451  endif
452  do k = 1,ngllx
453  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
454  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
455  enddo
456 
457  xixl = xix(i,j,ispec)
458  xizl = xiz(i,j,ispec)
459  gammaxl = gammax(i,j,ispec)
460  gammazl = gammaz(i,j,ispec)
461  jacobianl = jacobian(i,j,ispec)
462 
463  ! derivatives of displacement
464  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
465  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
466 
467  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
468  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
469 
470  ! compute total integrated energy
471  ! We record just one point per element (i=2, j=2)
472  integrated_kinetic_energy_field(ispec) = integrated_kinetic_energy_field(ispec) &
473  + rhol*(veloc_elastic(1,ibool(i,j,ispec))**2 + veloc_elastic(2,ibool(i,j,ispec))**2) * deltat / two
474 
475  if (max_kinetic_energy_field(ispec) < rhol*(veloc_elastic(1,ibool(i,j,ispec))**2 + veloc_elastic(2,ibool(i,j,ispec))**2) / &
476  two) then
477  max_kinetic_energy_field(ispec) = rhol*(veloc_elastic(1,ibool(i,j,ispec))**2 + veloc_elastic(2,ibool(i,j,ispec))**2) / two
478  endif
479 
480  if (max_kinetic_energy_field(ispec) > zero) then
481  kinetic_effective_duration_field(ispec) = two*integrated_kinetic_energy_field(ispec)/max_kinetic_energy_field(ispec)
482  endif
483 
484  integrated_potential_energy_field(ispec) = integrated_potential_energy_field(ispec) &
485  + (lambdaplus2mu_unrelaxed_elastic*dux_dxl**2 &
486  + lambdaplus2mu_unrelaxed_elastic*duz_dzl**2 &
487  + two*lambdal_unrelaxed_elastic*dux_dxl*duz_dzl &

574  rhol * (vector_field_element(1,i,j)**2 + vector_field_element(2,i,j)**2) / two
575  endif
576 
577  endif
578

580 
581  if (max_total_energy_field(ispec) > zero) then
582  total_effective_duration_field(ispec) = two*total_integrated_energy_field(ispec) / max_total_energy_field(ispec)
583  endif
584 
585  enddo
