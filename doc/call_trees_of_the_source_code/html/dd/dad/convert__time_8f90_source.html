44  subroutine convtime(timestamp,yr,mon,day,hr,minvalue)
45 
46 ! Originally written by Shawn Smith (ssmith AT coaps.fsu.edu)
47 ! Updated Spring 1999 for Y2K compliance by Anthony Arguez (anthony AT coaps.fsu.edu).
48 
49 ! This subroutine will convert a given year, month, day, hour, and
50 ! minutes to a minutes from 01 Jan 1980 00:00 time stamp.
51 
52  implicit none
53 
54  integer, parameter :: MAX_YEAR = 3000
55 
56  integer, intent(out) :: timestamp
57 
58  integer, intent(in) :: yr,mon,day,hr,minvalue
59 
60  integer :: year(1980:max_year),month(12),leap_mon(12)
61 
62  integer :: min_day,min_hr,iyr
63 
64 ! function to determine if year is a leap year
65  logical, external :: is_leap_year
66 
67  data month /0, 44640, 84960, 129600, 172800, 217440, 260640, &
68  305280, 349920, 393120, 437760, 480960/
69 
70  data leap_mon /0, 44640, 86400, 131040, 174240, 218880, 262080, &
71  306720, 351360, 394560, 439200, 482400/
72 
73  data min_day, min_hr /1440, 60/
74 
75 ! loop added by Dimitri Komatitsch to fill array "year()" automatically
76  year(:) = 0
77  do iyr = 1981,max_year
78  if (is_leap_year(iyr-1)) then
79  year(iyr) = year(iyr-1) + 366*24*60 ! number of minutes in a year if leap year
80  else
81  year(iyr) = year(iyr-1) + 365*24*60 ! number of minutes in a year if not leap year
82  endif
83  enddo
84 
85 ! Test values to see if they fit valid ranges
86  if (yr < 1980) stop 'Error in convtime: year out of range, must be >= 1980'
87 
88  if (mon < 1 .or. mon > 12) stop 'Error in convtime: month out of range (1-12)'
89 
90  if (mon == 2) then
91  if (is_leap_year(yr) .and. (day < 1 .or. day > 29)) then
92  stop 'Error in convtime: February day out of range (1-29)'
93  else if (.not. is_leap_year(yr) .and. (day < 1 .or. day > 28)) then
94  stop 'Error in convtime: February day out of range (1-28)'
95  endif
96  else if (mon == 4 .or. mon == 6 .or. mon == 9 .or. mon == 11) then
97  if (day < 1 .or. day > 30) stop 'Error in convtime: day out of range (1-30)'
98  else
99  if (day < 1 .or. day > 31) stop 'Error in convtime: day out of range (1-31)'
100  endif
101 
102  if (hr < 0 .or. hr > 23) stop 'Error in convtime: hour out of range (0-23)'
103 
104  if (minvalue < 0 .or. minvalue > 60) stop 'Error in convtime: minute out of range (0-60)'
105 
106 ! convert time (test if leap year)
107  if (is_leap_year(yr)) then
108  timestamp = year(yr)+leap_mon(mon)+((day-1)*min_day)+(hr*min_hr)+minvalue
109  else
110  timestamp = year(yr)+month(mon)+((day-1)*min_day)+(hr*min_hr)+minvalue
111  endif
112 
113  end subroutine convtime

119  subroutine invtime(timestamp,yr,mon,day,hr,minvalue)
120 
121 ! This subroutine will convert a minutes timestamp to a year/month
122 ! date. Based on the function convtime by Shawn Smith (COAPS).
123 !
124 ! Written the spring of 1995, several iterations.
125 ! James N. Stricherz (stricherz AT coaps.fsu.edu)
126 !
127 ! Updated for Y2K compliance in July 1999.
128 ! Shyam Lakshmin (lakshmin AT coaps.fsu.edu)
129 !
130 ! This code returns correct results for the range of 01 Jan 1980 00:00
131 ! through 31 Dec 2020 23:59. I know it does, because I tried each minute of that range.
132 
133  implicit none
134 
135  integer, parameter :: MAX_YEAR = 3000
136 
137  integer, intent(in) :: timestamp
138 
139  integer, intent(out) :: yr,mon,day,hr,minvalue
140 
141  integer :: year(1980:max_year),month(13),leap_mon(13)
142 
143  integer :: min_day,min_hr,itime,tmon,ttime,thour,iyr,imon,iday,ihour
144 
145 ! function to determine if year is a leap year
146  logical, external :: is_leap_year
147 
148  data month /0, 44640, 84960, 129600, 172800, 217440, 260640, &
149  305280, 349920, 393120, 437760, 480960, 525600/
150 
151  data leap_mon /0, 44640, 86400, 131040, 174240, 218880, 262080, &
152  306720, 351360, 394560, 439200, 482400, 527040/
153 
154  data min_day, min_hr /1440, 60/
155 
156 ! loop added by Dimitri Komatitsch to fill array "year()" automatically
157  year(:) = 0
158  do iyr = 1981,max_year
159  if (is_leap_year(iyr-1)) then
160  year(iyr) = year(iyr-1) + 366*24*60 ! number of minutes in a year if leap year
161  else
162  year(iyr) = year(iyr-1) + 365*24*60 ! number of minutes in a year if not leap year
163  endif
164  enddo
165 
166 ! OK, let us invert the effects of the years: subtract off the
167 ! number of minutes per year until it goes negative
168 ! iyr then gives the year that the time (in minutes) occurs
169  if (timestamp >= year(max_year)) stop 'year too high in invtime'
170 
171  iyr=1979
172  itime=timestamp
173 
174  10 iyr=iyr+1
175  ttime=itime-year(iyr)
176  if (ttime <= 0) then
177  if (iyr == 1980) iyr=iyr+1
178  iyr=iyr-1
179  itime=itime-year(iyr)
180  else
181  goto 10
182  endif
183 
184 ! assign the return variable
185  yr=iyr
186 
187 ! OK, the remaining time is less than one full year, so convert
188 ! by the same method as above into months
189  imon=0
190 
191 ! if not leap year
192  if (.not. is_leap_year(iyr)) then
193 
194 ! increment the month, and subtract off the minutes from the
195 ! remaining time for a non-leap year
196  20 imon=imon+1
197  tmon=itime-month(imon)
198  if (tmon > 0) then
199  goto 20
200  else if (tmon < 0) then
201  imon=imon-1
202  itime=itime-month(imon)
203  else
204  if (imon > 12) then
205  imon=imon-12
206  yr=yr+1
207  endif
208  mon=imon
209  day=1
210  hr=0
211  minvalue=0
212  return
213  endif
214 
215 ! if leap year
216  else
217 
218 ! same thing, same code, but for a leap year
219  30 imon=imon+1
220  tmon=itime-leap_mon(imon)
221  if (tmon > 0) then
222  goto 30
223  else if (tmon < 0) then
224  imon=imon-1
225  itime=itime-month(imon)
226  else
227  if (imon > 12) then
228  imon=imon-12
229  yr=yr+1
230  endif
231  mon=imon
232  day=1
233  hr=0
234  minvalue=0
235  return
236  endif
237  endif
238 
239 ! assign the return variable
240  mon=imon
241 
242 ! any remaining minutes will belong to day/hour/minutes
243 ! OK, let us get the days
244  iday=0
245  40 iday=iday+1
246  ttime=itime-min_day
247  if (ttime >= 0) then
248  itime=ttime
249  goto 40
250  endif
251 
252 ! assign the return variable
253  if (is_leap_year(iyr) .and. mon > 2) then
254  day=iday-1
255  else
256  day=iday
257  endif
258 
259 ! pick off the hours of the days...remember, hours can be 0, so we start at -1
260  ihour=-1
261  50 ihour=ihour+1
262  thour=itime-min_hr
263  if (thour >= 0) then
264  itime=thour
265  goto 50
266  endif
267 
268 ! assign the return variables
269  hr=ihour
270 
271 ! the remainder at this point is the minutes, so return them directly
272  minvalue=itime
273 
274  end subroutine invtime
