40

46  integer,intent(in) :: iglob
47 
48  real(kind=CUSTOM_REAL) :: displ_x,displ_z

50 ! local variables
51  real, parameter :: pigrec = 3.1415927
52 
53  real :: alpha,tho,A,c,delayed,delta_x
54  integer :: forcing_type,k,ngoce_time_step,n_models,kk,ll
55 
56  double precision, dimension(:), allocatable :: goce_time,distance
57  double precision, dimension(:,:), allocatable :: syn
58  double precision :: t,t_used,signal_x1,signal_x2,fracx,fract
59  double precision :: x,z
60 
61  double precision :: f0 = 6000.0
62 
63  forcing_type = 1
64 
65  ! length of a PML element along x-axis at the edge which will be forced
66  delta_x = 2.4
67  alpha = 2
68 
69  ! infrasounds / seismic
70  tho = 30.0
71 
72 ! gravity wave test function
73 ! tho = 600.0
74 ! xo = 500000.0
75 ! lambdo = 20000.0
76 
77 ! gravity wave test function
78 ! tho = 600.0
79 ! xo = 3000.0
80 ! lambdo = 120.0
81 
82 ! gravity wave /tsunami
83 ! tho = 600.0 ! *20
84 ! c = 200.0 ! /20
85  a = 1
86  x = coord(1,iglob)
87  z = coord(2,iglob)
88  delayed = 0
89 
90  ! speed of light
91  c = 300000000.
92 
93  if (forcing_type == 1) then !! First test function : same forcing for the whole boundary
94  !if (ispec_is_PML(ispec_acoustic)) then
95  ! displ_x = 0
96  ! displ_z = 0
97  !else
98  ! infrasounds / seismic
99  !displ_x = 0 !* Apo
100  !displ_z = A * (exp(-(alpha*(deltat*it-40-t0)/tho)**2) &
101  ! - exp(-(alpha*(deltat*it-70-t0)/tho)**2)) !* Apo
102  t_used = deltat*(it-1) - 0.0007d0
103  !sin(2.0d0*pigrec*f0*t_used)
104  if ((z < -1.5d0) .and. (z > -3.5d0)) then
105  displ_x = real(2.0d0 * f0*f0 * (2.0d0 * f0*f0 * t_used**2 - 1.0d0) * & exp(-f0*f0*t_used**2),kind=CUSTOM_REAL) !(z+2.5d0)**3
106  displ_z = 0._custom_real
107  else
108  displ_x = 0._custom_real
109  displ_z = 0._custom_real
110  endif
111  ! gravity wave test function
112  ! displ_x = 0 !* Apo
113  ! displ_z = A * ( exp(-(alpha*(x-(xo-lambdo/2))/lambdo)**2) - &
114  ! exp(-(alpha*(x-(xo+lambdo/2))/lambdo)**2) ) * &
115  ! (exp(-(alpha*(deltat*it+1000-t0)/tho)**2) &
116  ! - exp(-(alpha*(deltat*it-1300-t0)/tho)**2)) !* Apo
117 
118  ! gravity wave /tsunami
119  ! displ_x = 0 !* Apo
120  ! displ_z = A * (exp(-(alpha*(deltat*it-1000-t0)/tho)**2) &
121  ! - exp(-(alpha*(deltat*it-1600-t0)/tho)**2)) !* Apo
122  !endif
123  endif
124 
125  !! Second test function : moving forcing
126  if (forcing_type == 2) then
127  displ_x = 0._custom_real !* Apo
128  displ_z = real(dble(A) * (exp(-(alpha*(deltat*it-40-t0-(x-delayed)/c)/tho)**2) &
129  - exp(-(alpha*(deltat*it-70-t0-(x-delayed)/c)/tho)**2)),kind=custom_real) !* Apo
130  endif
131 
132  !! forcing external
133  if (forcing_type == 3) then
134  ngoce_time_step = 255
135  n_models = 28
136  t =it*deltat
137 
138  allocate(goce_time(ngoce_time_step))
139  allocate(distance(n_models))
140  allocate(syn(n_models,ngoce_time_step))
141 
142  open(1000,file='../../EXAMPLES/acoustic_forcing_bottom/distance.txt',form='formatted')
143  open(1001,file='../../EXAMPLES/acoustic_forcing_bottom/forcing_signals.txt',form='formatted')
144 
145  read(1001,*) goce_time(:)
146 
147  do k = 1,n_models
148  read(1001,*) syn(k,:)
149  read(1000,*) distance(k)
150  enddo
151 
152  close(1000)
153  close(1001)
154 
155  kk = 1
156  do while(x >= distance(kk) .and. kk /= n_models)
157  kk = kk+1
158  enddo
159 
160  ll = 1
161  do while(t >= goce_time(ll) .and. ll /= ngoce_time_step)
162  ll = ll+1
163  enddo
164 
165  if (x == 0 .and. it == 1) then
166  displ_z = real(syn(1,1),kind=custom_real)
167  else
168  if (x == 0) then
169  fract = (t-goce_time(ll-1))/(goce_time(ll)-goce_time(ll-1))
170  displ_z = real((syn(1,ll-1) + fract * (syn(1,ll)-syn(1,ll-1))),kind=custom_real)
171  else
172  if (it == 1) then
173  fracx = (x-distance(kk-1))/(distance(kk)-distance(kk-1))
174  displ_z = real((syn(kk-1,1) + fracx * (syn(kk,1)-syn(kk-1,1))),kind=custom_real)
175  else
176  ! interpolation in time
177  fract = (t-goce_time(ll-1))/(goce_time(ll)-goce_time(ll-1))
178  ! in x1 = distance(kk-1)
179  signal_x1 = syn(kk-1,ll-1) + fract * (syn(kk-1,ll)-syn(kk-1,ll-1))
180  ! in x2 = distance(kk)
181  signal_x2 = syn(kk,ll-1) + fract * (syn(kk,ll)-syn(kk,ll-1))
182  ! spatial interpolation
183  fracx = (x-distance(kk-1))/(distance(kk)-distance(kk-1))
184  displ_z = real((signal_x1 + fracx * (signal_x2 - signal_x1)),kind=custom_real)
185  endif
186  endif
187  endif
188 
189  displ_x = 0._custom_real
190 
191  endif
192 
193  if (abs(displ_x) < tinyval) displ_x = 0._custom_real
194  if (abs(displ_z) < tinyval) displ_z = 0._custom_real
195 
196  end subroutine acoustic_forcing_boundary
