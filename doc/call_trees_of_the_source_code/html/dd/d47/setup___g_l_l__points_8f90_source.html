35  subroutine setup_gll_points()
36 
37 ! computes shape functions and their derivatives for SEM grid
38 
39  use specfem_par
40 
41  implicit none
42 
43  ! local parameters
44  integer :: i,j,ier
45 
46  ! set up Gauss-Lobatto-Legendre points, weights and also derivation matrices (hprime_**,..)
47  call define_derivation_matrices()
48 
49  if (axisym) then
50  ! set up Gauss-Lobatto-Jacobi points, weights and also derivation matrices
51  call define_glj_derivation_matrix()
52  endif
53 
54  ! shape arrays
55  allocate(shape2d(ngnod,ngllx,ngllz),dershape2d(ndim,ngnod,ngllx,ngllz),stat=ier)
56  if (ier /= 0) stop 'Error allocating shape arrays'
57 
58  do j = 1,ngllz
59  do i = 1,ngllx
60  call define_shape_functions(shape2d(:,i,j),dershape2d(:,:,i,j),xigll(i),zigll(j),ngnod)
61  enddo
62  enddo
63 
64  ! allocate 1-D Lagrange interpolators and derivatives
65  ! for source and receivers
66  allocate(hxir(ngllx), &
67  hxis(ngllx), &
68  hpxir(ngllx), &
69  hpxis(ngllx), &
70  hgammar(ngllz), &
71  hgammas(ngllz), &
72  hpgammar(ngllz), &
73  hpgammas(ngllz),stat=ier)
74  if (ier /= 0) stop 'error allocating arrays for interpolators'
75 
76  end subroutine setup_gll_points
