34  subroutine read_parameter_file(myrank,imesher,BROADCAST_AFTER_READ)
35 
36 ! reads in DATA/Par_file
37 
38  use constants, only: imain
39  use shared_parameters
40 
41  implicit none
42 
43  integer, intent(in) :: myrank,imesher
44  logical, intent(in) :: BROADCAST_AFTER_READ
45 
46  ! initializes
47  call read_parameter_file_init()
48 
49  ! only master process reads in Par_file
50  if (myrank == 0) then
51  ! opens file Par_file
52  call open_parameter_file()
53 
54  ! reads only parameters (without receiver-line section, material tables or region definitions)
55  call read_parameter_file_only()
56 
57  ! reads receiver lines
58  call read_parameter_file_receiversets()
59 
60  ! reads material definitions
61  call read_material_table()
62 
63  ! mesher reads in internal region table for setting up mesh elements
64  if (imesher == 1 .and. (.not. read_external_mesh) ) then
65  ! internal meshing
66  ! user output
67  write(imain,*)
68  write(imain,*) 'Mesh from internal meshing:'
69  ! reads interface definitions from interface file (we need to have nxread & nzread value for checking regions)
70  call read_interfaces_file()
71 
72  ! internal meshing
73  nx = nxread
74  nz = nzread
75 
76  ! setup mesh array
77  ! multiply by 2 if elements have 9 nodes
78  if (ngnod == 9) then
79  nx = nx * 2
80  nz = nz * 2
81  nz_layer(:) = nz_layer(:) * 2
82  endif
83 
84  ! total number of elements
85  nelmnts = nxread * nzread
86 
87  ! reads material regions defined in Par_file
88  call read_regions()
89  endif
90 
91  ! closes file Par_file
92  call close_parameter_file()
93  endif
94 
95  ! master process broadcasts to all
96  ! note: this is only needed at the moment for the solver to setup a simulation run
97  if (broadcast_after_read) then
98  call bcast_all_singlei(simulation_type)
99  call bcast_all_singlei(noise_tomography)
100  call bcast_all_singlel(save_forward)
101 
102  call bcast_all_singlei(nproc)
103  call bcast_all_singlei(partitioning_method)
104  call bcast_all_singlei(ngnod)
105 
106  call bcast_all_singlei(nstep)
107  call bcast_all_singledp(dt)
108 
109  call bcast_all_singlei(time_stepping_scheme)
110  call bcast_all_singlel(axisym)
111  call bcast_all_singlel(p_sv)
112  call bcast_all_singlel(gpu_mode)
113  call bcast_all_singlei(setup_with_binary_database)
114 
115  call bcast_all_string(model)
116  call bcast_all_string(save_model)
117 
118  call bcast_all_singlel(attenuation_viscoelastic)
119  call bcast_all_singlel(attenuation_viscoacoustic)
120  call bcast_all_singlel(attenuation_poro_fluid_part)
121  call bcast_all_singledp(q0)
122  call bcast_all_singledp(freq0)
123  call bcast_all_singlei(n_sls)
124  call bcast_all_singledp(f0_attenuation)
125  call bcast_all_singlel(read_velocities_at_f0)
126 
127  call bcast_all_singlel(undo_attenuation)
128  call bcast_all_singlei(nt_dump_attenuation)
129 
130  call bcast_all_singlei(nsources)
131  call bcast_all_singlel(force_normal_to_surface)
132  call bcast_all_singlel(initialfield)
133  call bcast_all_singlel(add_bielak_conditions_bottom)
134  call bcast_all_singlel(add_bielak_conditions_right)
135  call bcast_all_singlel(add_bielak_conditions_top)
136  call bcast_all_singlel(add_bielak_conditions_left)
137  call bcast_all_singlel(acoustic_forcing)
138 
139  call bcast_all_singlei(seismotype)
140  call bcast_all_singlei(subsamp_seismos)
141  call bcast_all_singlel(use_trick_for_better_pressure)
142  call bcast_all_singlei(nstep_between_output_seismos)
143  call bcast_all_singlel(compute_integrated_energy_field)
144  call bcast_all_singledp(user_t0)
145  call bcast_all_singlel(save_ascii_seismograms)
146  call bcast_all_singlel(save_binary_seismograms_single)
147  call bcast_all_singlel(save_binary_seismograms_double)
148  call bcast_all_singlel(su_format)
149  call bcast_all_singlel(use_existing_stations)
150  call bcast_all_singlei(nreceiversets)
151  call bcast_all_singledp(anglerec)
152  call bcast_all_singlel(rec_normal_to_surface)
153 
154  call bcast_all_singlel(save_ascii_kernels)
155 
156  call bcast_all_singlel(pml_boundary_conditions)
157  call bcast_all_singlei(nelem_pml_thickness)
158  call bcast_all_singlel(rotate_pml_activate)
159  call bcast_all_singledp(rotate_pml_angle)
160 
161  call bcast_all_singlel(stacey_absorbing_conditions)
162  call bcast_all_singlel(add_periodic_conditions)
163  call bcast_all_singledp(periodic_horiz_dist)
164 
165  call bcast_all_singlei(nbmodels)
166  call bcast_all_string(tomography_file)
167  call bcast_all_singlel(read_external_mesh)
168 
169  if (.not. read_external_mesh) then
170  call bcast_all_singlel(absorbbottom)
171  call bcast_all_singlel(absorbright)
172  call bcast_all_singlel(absorbtop)
173  call bcast_all_singlel(absorbleft)
174  endif
175 
176  call bcast_all_singlei(nstep_between_output_info)
177  call bcast_all_singlel(output_grid_gnuplot)
178  call bcast_all_singlel(output_grid_ascii)
179  call bcast_all_singlel(output_energy)
180  call bcast_all_singlei(ntstep_between_output_energy)
181 
182  call bcast_all_singlei(nstep_between_output_images)
183  call bcast_all_singledp(cutsnaps)
184 
185  call bcast_all_singlel(output_color_image)
186  call bcast_all_singlei(imagetype_jpeg)
187  call bcast_all_singledp(factor_subsample_image)
188  call bcast_all_singlel(use_constant_max_amplitude)
189  call bcast_all_singledp(constant_max_amplitude_to_use)
190  call bcast_all_singledp(power_display_color)
191  call bcast_all_singlel(draw_sources_and_receivers)
192  call bcast_all_singlel(draw_water_in_blue)
193  call bcast_all_singlel(use_snapshot_number_in_filename)
194 
195  call bcast_all_singlel(output_postscript_snapshot)
196  call bcast_all_singlei(imagetype_postscript)
197  call bcast_all_singlel(meshvect)
198  call bcast_all_singlel(modelvect)
199  call bcast_all_singlel(boundvect)
200  call bcast_all_singlel(interpol)
201  call bcast_all_singlei(pointsdisp)
202  call bcast_all_singlei(subsamp_postscript)
203  call bcast_all_singledp(sizemax_arrows)
204  call bcast_all_singlel(us_letter)
205 
206  call bcast_all_singlel(output_wavefield_dumps)
207  call bcast_all_singlei(imagetype_wavefield_dumps)
208  call bcast_all_singlel(use_binary_for_wavefield_dumps)
209 
210  call bcast_all_singlei(number_of_simultaneous_runs)
211  call bcast_all_singlel(broadcast_same_mesh_and_model)
212  endif
213 
214  ! derive additional settings/flags based on input parameters
215  call read_parameter_file_derive_flags()
216 
217  end subroutine read_parameter_file

261  subroutine read_parameter_file_only()
262 
263 ! reads only parameters without receiver-line section and material tables
264 
265  use constants, only: imain
266  use shared_parameters
267 
268  implicit none
269 
270  ! local parameters
271  integer :: i,irange
272 
273  integer,external :: err_occurred
274 
275  !--------------------------------------------------------------------
276  !
277  ! simulation input paramters
278  !
279  !--------------------------------------------------------------------
280 
281  ! read file names and path for output
282  call read_value_string_p(title, 'solver.title')
283  if (err_occurred() /= 0) stop 'error reading parameter title in Par_file'
284 
285  ! read type of simulation
286  call read_value_integer_p(simulation_type, 'solver.SIMULATION_TYPE')
287  if (err_occurred() /= 0) stop 'error reading parameter SIMULATION_TYPE in Par_file'
288 
289  call read_value_integer_p(noise_tomography, 'solver.NOISE_TOMOGRAPHY')
290  if (err_occurred() /= 0) stop 'error reading parameter NOISE_TOMOGRAPHY in Par_file'
291 
292  call read_value_logical_p(save_forward, 'solver.SAVE_FORWARD')
293  if (err_occurred() /= 0) stop 'error reading parameter SAVE_FORWARD in Par_file'
294 
295  ! read info about partitioning
296  call read_value_integer_p(nproc, 'solver.NPROC')
297  if (err_occurred() /= 0) stop 'error reading parameter NPROC in Par_file'
298 
299  call read_value_integer_p(partitioning_method, 'mesher.partitioning_method')
300  if (err_occurred() /= 0) stop 'error reading parameter partitioning_method in Par_file'
301 
302  call read_value_integer_p(ngnod, 'mesher.ngnod')
303  if (err_occurred() /= 0) stop 'error reading parameter ngnod in Par_file'
304 
305  ! read time step parameters
306  call read_value_integer_p(nstep, 'solver.NSTEP')
307  if (err_occurred() /= 0) stop 'error reading parameter NSTEP in Par_file'
308 
309  call read_value_double_precision_p(dt, 'solver.DT')
310  if (err_occurred() /= 0) stop 'error reading parameter DT in Par_file'
311 
312  call read_value_integer_p(time_stepping_scheme, 'solver.time_stepping_scheme')
313  if (err_occurred() /= 0) stop 'error reading parameter time_stepping_scheme in Par_file'
314 
315  ! axisymmetric (2.5D) or Cartesian planar (2D) simulation
316  call read_value_logical_p(axisym, 'solver.AXISYM')
317  if (err_occurred() /= 0) stop 'error reading parameter AXISYM in Par_file'
318 
319  ! determine if body or surface (membrane) waves calculation
320  call read_value_logical_p(p_sv, 'solver.P_SV')
321  if (err_occurred() /= 0) stop 'error reading parameter P_SV in Par_file'
322 
323  call read_value_logical_p(gpu_mode, 'solver.GPU_MODE')
324  if (err_occurred() /= 0) stop 'error reading parameter GPU_MODE in Par_file'
325 
326  call read_value_integer_p(setup_with_binary_database, 'solver.setup_with_binary_database')
327  if (err_occurred() /= 0) stop 'error reading parameter setup_with_binary_database in Par_file'
328 
329  call read_value_string_p(model, 'mesher.MODEL')
330  if (err_occurred() /= 0) stop 'error reading parameter MODEL in Par_file'
331 
332  call read_value_string_p(save_model, 'mesher.SAVE_MODEL')
333  if (err_occurred() /= 0) stop 'error reading parameter SAVE_MODEL in Par_file'
334 
335  !--------------------------------------------------------------------
336  !
337  ! attenuation
338  !
339  !--------------------------------------------------------------------
340 
341  call read_value_logical_p(attenuation_viscoelastic, 'solver.ATTENUATION_VISCOELASTIC')
342  if (err_occurred() /= 0) stop 'error reading parameter ATTENUATION_VISCOELASTIC in Par_file'
343 
344  ! read viscous attenuation parameters (acoustic media)
345  call read_value_logical_p(attenuation_viscoacoustic, 'solver.ATTENUATION_VISCOACOUSTIC')
346  if (err_occurred() /= 0) stop 'error reading parameter ATTENUATION_VISCOACOUSTIC in Par_file'
347 
348  ! read viscous attenuation parameters (poroelastic media)
349  call read_value_logical_p(attenuation_poro_fluid_part, 'solver.ATTENUATION_PORO_FLUID_PART')
350  if (err_occurred() /= 0) stop 'error reading parameter ATTENUATION_PORO_FLUID_PART in Par_file'
351 
352  call read_value_double_precision_p(q0, 'solver.Q0')
353  if (err_occurred() /= 0) stop 'error reading parameter Q0 in Par_file'
354 
355  call read_value_double_precision_p(freq0, 'solver.freq0')
356  if (err_occurred() /= 0) stop 'error reading parameter freq0 in Par_file'
357 
358  ! read constants for attenuation
359  call read_value_integer_p(n_sls, 'solver.N_SLS')
360  if (err_occurred() /= 0) stop 'error reading parameter N_SLS in Par_file'
361 
362  call read_value_double_precision_p(f0_attenuation, 'solver.f0_attenuation')
363  if (err_occurred() /= 0) stop 'error reading parameter f0_attenuation in Par_file'
364 
365  call read_value_logical_p(read_velocities_at_f0, 'solver.READ_VELOCITIES_AT_f0')
366  if (err_occurred() /= 0) stop 'error reading parameter READ_VELOCITIES_AT_f0 in Par_file'
367 
368  call read_value_logical_p(undo_attenuation, 'solver.UNDO_ATTENUATION')
369  if (err_occurred() /= 0) stop 'error reading parameter UNDO_ATTENUATION in Par_file'
370 
371  call read_value_integer_p(nt_dump_attenuation, 'solver.NT_DUMP_ATTENUATION')
372  if (err_occurred() /= 0) stop 'error reading parameter NT_DUMP_ATTENUATION in Par_file'
373 
374  !--------------------------------------------------------------------
375  !
376  ! sources
377  !
378  !--------------------------------------------------------------------
379 
380  ! read source infos
381  call read_value_integer_p(nsources, 'solver.NSOURCES')
382  if (err_occurred() /= 0) stop 'error reading parameter NSOURCES in Par_file'
383 
384  call read_value_logical_p(force_normal_to_surface, 'solver.force_normal_to_surface')
385  if (err_occurred() /= 0) stop 'error reading parameter force_normal_to_surface in Par_file'
386 
387  call read_value_logical_p(initialfield, 'solver.initialfield')
388  if (err_occurred() /= 0) stop 'error reading parameter initialfield in Par_file'
389 
390  call read_value_logical_p(add_bielak_conditions_bottom, 'solver.add_Bielak_conditions_bottom')
391  if (err_occurred() /= 0) stop 'error reading parameter add_Bielak_conditions_bottom in Par_file'

393  call read_value_logical_p(add_bielak_conditions_right, 'solver.add_Bielak_conditions_right')

397  if (err_occurred() /= 0) stop 'error reading parameter add_Bielak_conditions_top in Par_file'
398 
399  call read_value_logical_p(add_bielak_conditions_left, 'solver.add_Bielak_conditions_left')
400  if (err_occurred() /= 0) stop 'error reading parameter add_Bielak_conditions_left in Par_file'
401 
402  ! read acoustic forcing flag
403  call read_value_logical_p(acoustic_forcing, 'solver.ACOUSTIC_FORCING')

440

443

488 
489  call read_value_logical_p(add_periodic_conditions, 'solver.ADD_PERIODIC_CONDITIONS')

491
