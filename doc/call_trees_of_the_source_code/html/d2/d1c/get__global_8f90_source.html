34 ! create a sorted version of the indirect addressing to reduce cache misses
35 
36  subroutine get_global()
37 
38  use constants, only: ngllx,ngllz,max_string_len,in_data_files
39 
40  use specfem_par, only: nspec,ibool,copy_ibool_ori,integer_mask_ibool,save_model,myrank
41 
42  implicit none
43 
44  ! local parameters
45  integer :: inumber,ispec,i,j
46  integer :: ier
47  character(len=MAX_STRING_LEN) :: outputname
48 
49  ! initializes temporary arrays
50  integer_mask_ibool(:) = -1
51  copy_ibool_ori(:,:,:) = ibool(:,:,:)
52 
53  inumber = 0
54 
55  ! reduce cache misses in all the elements
56  ! loop over spectral elements
57  do ispec = 1,nspec
58  do j = 1,ngllz
59  do i = 1,ngllx
60  if (integer_mask_ibool(copy_ibool_ori(i,j,ispec)) == -1) then
61  ! create a new point
62  inumber = inumber + 1
63  ibool(i,j,ispec) = inumber
64  integer_mask_ibool(copy_ibool_ori(i,j,ispec)) = inumber
65  else
66  ! use an existing point created previously
67  ibool(i,j,ispec) = integer_mask_ibool(copy_ibool_ori(i,j,ispec))
68  endif
69  enddo
70  enddo
71  enddo
72 
73  if (trim(save_model) == 'binary') then
74  write(outputname,'(a,i6.6,a)') trim(in_data_files)//'proc',myrank,'_NSPEC_ibool.bin'
75 
76  open(888,file=trim(outputname),status='unknown',form='unformatted',iostat=ier)
77  if (ier /= 0) stop 'Error opening smoothed kernel file'
78  write(888) nspec
79  write(888) ibool
80  close(888)
81  endif
82 
83  end subroutine get_global
