33 
34  subroutine get_mpi()
35 
36 ! sets up MPI arrays
37 
38  use constants, only: imain
39  use shared_parameters, only: nproc
40  use specfem_par
41 
42  implicit none
43 
44  ! local parameters
45  integer :: i,iinterface,imax_all,ier
46 
47  ! user output
48  if (myrank == 0) then
49  write(imain,*)
50  write(imain,*) 'Setting up MPI communication arrays'
51  write(imain,*)
52  call flush_imain()
53  endif
54 
55  ! initializes
56  ninterface_acoustic = 0
57  ninterface_elastic = 0
58  ninterface_poroelastic = 0
59 
60  max_nibool_interfaces_ext_mesh = 0
61  max_ibool_interfaces_size_ac = 0
62  max_ibool_interfaces_size_el = 0
63  max_ibool_interfaces_size_po = 0
64 
65  ! user output
66  call max_all_i(ninterface,imax_all)
67  ! user output
68  if (myrank == 0) then
69  write(imain,*) ' maximum number of MPI interfaces (for a single slice) = ',imax_all
70  write(imain,*)
71  call flush_imain()
72  endif
73 
74  ! preparing for MPI communications
75  call get_mpi_interfaces()
76 
77  if (nproc > 1) then
78  ! user output
79  if (myrank == 0) then
80  write(imain,*) ' master process:'
81  write(imain,*) ' number of MPI interfaces in acoustic domain = ',ninterface_acoustic
82  write(imain,*) ' number of MPI interfaces in elastic domain = ',ninterface_elastic
83  write(imain,*) ' number of MPI interfaces in poroelastic domain = ',ninterface_poroelastic
84  write(imain,*)
85  call flush_imain()
86  endif
87 
88  ! total MPI interfaces
89  max_nibool_interfaces_ext_mesh = maxval(nibool_interfaces_ext_mesh(:))
90 
91  ! domain buffers for MPI communications
92  max_ibool_interfaces_size_ac = maxval(nibool_interfaces_acoustic(:))
93  max_ibool_interfaces_size_el = ndim*maxval(nibool_interfaces_elastic(:))
94  max_ibool_interfaces_size_po = ndim*maxval(nibool_interfaces_poroelastic(:))
95 
96  if (acoustic_simulation) then
97  allocate(request_send_recv_acoustic(ninterface_acoustic*2),stat=ier)
98  if (ier /= 0) stop 'error in allocation of array request_send_recv_acoustic'
99  allocate(buffer_send_faces_vector_ac(max_ibool_interfaces_size_ac,ninterface_acoustic),stat=ier)
100  if (ier /= 0) stop 'error in allocation of array buffer_send_faces_vector_ac'
101  allocate(buffer_recv_faces_vector_ac(max_ibool_interfaces_size_ac,ninterface_acoustic),stat=ier)
102  if (ier /= 0) stop 'error in allocation of array buffer_recv_faces_vector_ac'
103  endif
104 
105  if (elastic_simulation) then
106  allocate(request_send_recv_elastic(ninterface_elastic*2),stat=ier)
107  if (ier /= 0) stop 'error in allocation of array request_send_recv_elastic'
108  allocate(buffer_send_faces_vector_el(max_ibool_interfaces_size_el,ninterface_elastic),stat=ier)
109  if (ier /= 0) stop 'error in allocation of array buffer_send_faces_vector_el'
110  allocate(buffer_recv_faces_vector_el(max_ibool_interfaces_size_el,ninterface_elastic),stat=ier)
111  if (ier /= 0) stop 'error in allocation of array buffer_recv_faces_vector_el'
112  endif
113 
114  if (poroelastic_simulation) then
115  allocate(request_send_recv_poro(ninterface_poroelastic*4),stat=ier)
116  if (ier /= 0) stop 'error in allocation of array request_send_recv_poro'
117  allocate(buffer_send_faces_vector_pos(max_ibool_interfaces_size_po,ninterface_poroelastic),stat=ier)
118  if (ier /= 0) stop 'error in allocation of array buffer_send_faces_vector_pos'
119  allocate(buffer_recv_faces_vector_pos(max_ibool_interfaces_size_po,ninterface_poroelastic),stat=ier)
120  if (ier /= 0) stop 'error in allocation of array buffer_recv_faces_vector_pos'
121  allocate(buffer_send_faces_vector_pow(max_ibool_interfaces_size_po,ninterface_poroelastic),stat=ier)
122  if (ier /= 0) stop 'error in allocation of array buffer_send_faces_vector_pow'
123  allocate(buffer_recv_faces_vector_pow(max_ibool_interfaces_size_po,ninterface_poroelastic),stat=ier)
124  if (ier /= 0) stop 'error in allocation of array buffer_recv_faces_vector_pow'
125  endif
126 
127  else
128  ! safety check
129  if (myrank /= 0) stop 'Invalid myrank for serial simulation'
130 
131  ! user output
132  write(imain,*) ' This is a single process simulation, no need for MPI communication'
133  write(imain,*)
134  call flush_imain()
135  endif ! end of test on whether there is more than one process (NPROC > 1)
136 
137  ! MPI interfaces arrays
138  if (ninterface > 0) then
139 
140  allocate(ibool_interfaces_ext_mesh(max_nibool_interfaces_ext_mesh,ninterface),stat=ier)
141  if (ier /= 0) stop 'error in allocation of array ibool_interfaces_ext_mesh'
142  ibool_interfaces_ext_mesh(:,:) = 0
143 
144  do iinterface = 1,ninterface
145  do i = 1,nibool_interfaces_ext_mesh(iinterface)
146  ibool_interfaces_ext_mesh(i,iinterface) = ibool_interfaces_ext_mesh_init(i,iinterface)
147  enddo
148  enddo
149  else
150  ! dummy array
151  allocate(ibool_interfaces_ext_mesh(1,1))
152  endif
153 
154  ! sets up inner and outer elements
155  call get_mpi_setup_inner_outer_elements()
156 
157  ! setups up phase_inner arrays for looping over inner/outer elements in different domains (acoustic/elastic/..)
158  ! note: must have flags ispec_is_elastic() etc. and flags ispec_is_inner() already set before calling this routine
159  call get_mpi_phase_domains()
160 
161  end subroutine get_mpi
162

166 
167  subroutine get_mpi_interfaces()
168 
169 ! sets up the MPI interface for communication between partitions
170 
171  use constants, only: custom_real,imain,ngllx,ngllz
172 
173  use shared_parameters, only: nproc
174 
175  use specfem_par, only: nspec,ibool,nglob,ninterface,myrank,coord,acoustic_simulation
176 
177  use specfem_par, only: ibool_interfaces_acoustic,ibool_interfaces_elastic, &
178  ibool_interfaces_poroelastic, &
179  nibool_interfaces_acoustic,nibool_interfaces_elastic, &
180  nibool_interfaces_poroelastic, &
181  ninterface_acoustic,ninterface_elastic,ninterface_poroelastic
182 
183  use specfem_par, only: buffer_send_faces_vector_ac,buffer_recv_faces_vector_ac,request_send_recv_acoustic
184 
185  implicit none
186 
187  !local parameters
188  double precision, dimension(:), allocatable :: xp,zp
189  double precision, dimension(:), allocatable :: work
190  integer, dimension(:), allocatable :: locval
191  integer, dimension(:), allocatable :: nibool_interfaces_true
192 
193  ! for MPI buffers
194  integer, dimension(:), allocatable :: reorder_interface,ind,ninseg,iwork
195  integer, dimension(:), allocatable :: ibool_dummy
196  logical, dimension(:), allocatable :: ifseg
197  integer :: iinterface,ilocnum
198  integer :: num_points1, num_points2
199 
200  ! assembly test
201  integer :: i,j,ispec,iglob,inum,idomain
202  integer :: max_nibool_interfaces,num_nibool,num_interface
203  real(kind=CUSTOM_REAL), dimension(:),allocatable :: test_flag_cr
204  logical,dimension(:),allocatable :: mask_ibool
205 
206  ! checks if anything to do
207  if (nproc <= 1) return
208 
209  ! gets global indices for points on MPI interfaces
210  ! (defined by my_interfaces) between different partitions
211  ! and stores them in ibool_interfaces*** & nibool_interfaces*** (number of total points)
212  call prepare_assemble_mpi()
213 
214  ! sorts ibool comm buffers lexicographically for all MPI interfaces
215  num_points1 = 0
216  num_points2 = 0
217  allocate(nibool_interfaces_true(ninterface))
218 
219  ! acoustic/elastic/poroelastic domains
220  do idomain = 1,3
221 
222  ! checks number of interface in this domain
223  num_interface = 0
224  if (idomain == 1) then
225  num_interface = ninterface_acoustic
226  else if (idomain == 2) then
227  num_interface = ninterface_elastic
228  else if (idomain == 3) then
229  num_interface = ninterface_poroelastic
230  endif
231  if (num_interface == 0) cycle
232 
233  ! loops over interfaces
234  do iinterface = 1, ninterface
235 
236  ! number of global points in this interface
237  num_nibool = 0
238  if (idomain == 1) then
239  num_nibool = nibool_interfaces_acoustic(iinterface)
240  else if (idomain == 2) then
241  num_nibool = nibool_interfaces_elastic(iinterface)
242  else if (idomain == 3) then
243  num_nibool = nibool_interfaces_poroelastic(iinterface)
244  endif
245  ! checks if anything to sort
246  if (num_nibool == 0) cycle
247 
248  allocate(xp(num_nibool))
249  allocate(zp(num_nibool))
250  allocate(locval(num_nibool))
251  allocate(ifseg(num_nibool))
252  allocate(reorder_interface(num_nibool))
253  allocate(ibool_dummy(num_nibool))
254  allocate(ind(num_nibool))
255  allocate(ninseg(num_nibool))
256  allocate(iwork(num_nibool))
257  allocate(work(num_nibool))
258 
259  ! works with a copy of ibool array
260  if (idomain == 1) then
261  ibool_dummy(:) = ibool_interfaces_acoustic(1:num_nibool,iinterface)
262  else if (idomain == 2) then
263  ibool_dummy(:) = ibool_interfaces_elastic(1:num_nibool,iinterface)
264  else if (idomain == 3) then
265  ibool_dummy(:) = ibool_interfaces_poroelastic(1:num_nibool,iinterface)
266  endif
267 
268  ! gets x,y,z coordinates of global points on MPI interface
269  do ilocnum = 1, num_nibool
270  iglob = ibool_dummy(ilocnum)
271  xp(ilocnum) = coord(1,iglob)
272  zp(ilocnum) = coord(2,iglob)
273  enddo
274 
275  ! sorts (lexicographically?) ibool_interfaces and updates value
276  ! of total number of points nibool_interfaces_true(iinterface)
277  call sort_array_coordinates(num_nibool,xp,zp, &
278  ibool_dummy, &
279  reorder_interface,locval,ifseg, &
280  nibool_interfaces_true(iinterface), &
281  ind,ninseg,iwork,work)
282 
283  ! checks that number of MPI points are still the same
284  num_points1 = num_points1 + num_nibool
285  num_points2 = num_points2 + nibool_interfaces_true(iinterface)
286  if (num_points1 /= num_points2) then
287  write(imain,*) 'Error sorting MPI interface points:',myrank
288  write(imain,*) ' domain:',idomain
289  write(imain,*) ' interface:',iinterface,num_points1,num_points2
290  call exit_mpi(myrank,'error sorting MPI interface')
291  endif
292 
293  ! stores new order of ibool array
294  if (idomain == 1) then
295  ibool_interfaces_acoustic(1:num_nibool,iinterface) = ibool_dummy(:)
296  else if (idomain == 2) then
297  ibool_interfaces_elastic(1:num_nibool,iinterface) = ibool_dummy(:)
298  else if (idomain == 3) then
299  ibool_interfaces_poroelastic(1:num_nibool,iinterface) = ibool_dummy(:)
300  endif
301 
302  ! cleanup temporary arrays
303  deallocate(xp)
304  deallocate(zp)
305  deallocate(locval)
306  deallocate(ifseg)
307  deallocate(reorder_interface)
308  deallocate(ibool_dummy)
309  deallocate(ind)
310  deallocate(ninseg)
311  deallocate(iwork)
312  deallocate(work)
313  enddo
314  enddo
315 
316  ! cleanup
317  deallocate(nibool_interfaces_true)
318 
319  ! outputs total number of MPI interface points
320  call sum_all_i(num_points2,num_points1)
321  if (myrank == 0) then
322  write(imain,*) ' total MPI interface points: ',num_points1
323  write(imain,*)
324  call flush_imain()
325  endif
326 
327  ! checks interfaces in acoustic domains
328  if (acoustic_simulation) then
329  if (myrank == 0) then
330  write(imain,*) ' checking acoustic interfaces:'
331  call flush_imain()
332  endif
333 
334  inum = 0
335  if (ninterface_acoustic > 0) then
336  ! checks with assembly of test fields
337  allocate(test_flag_cr(nglob))
338  test_flag_cr(:) = 0._custom_real
339 
340  ! sets identifier on all global nodes in this slice
341  do ispec = 1, nspec
342  ! sets flags on global points
343  do j = 1, ngllz
344  do i = 1, ngllx
345  ! global index
346  iglob = ibool(i,j,ispec)
347  ! sets identifier
348  test_flag_cr(iglob) = myrank + 1.0
349  enddo
350  enddo
351  enddo
352 
353  ! count global points on MPI interfaces to neighbors
354  allocate(mask_ibool(nglob))
355  mask_ibool(:) = .false.
356  do iinterface = 1, ninterface
357  do inum = 1,nibool_interfaces_acoustic(iinterface)
358  iglob = ibool_interfaces_acoustic(inum,iinterface)
359  if (.not. mask_ibool(iglob)) mask_ibool(iglob) = .true.
360  enddo
361  enddo
362  inum = count(mask_ibool(:) .eqv. .true.)
363  deallocate(mask_ibool)
364  endif
365 
366  ! note: this MPI reduction awaits information from all processes.
367  ! thus, avoid an MPI deadlock in case some of the paritions have no acoustic interface
368  call sum_all_i(inum,num_points1)
369  if (myrank == 0) then
370  write(imain,*) ' total number of global acoustic interface points: ',num_points1
371  call flush_imain()
372  endif
373 
374  ! checks if assembly works
375  inum = 0
376  if (ninterface_acoustic > 0) then
377  ! adds contributions from different partitions to flag arrays
378  max_nibool_interfaces = maxval(nibool_interfaces_acoustic(:))
379 
380  allocate(request_send_recv_acoustic(ninterface_acoustic*2))
381  allocate(buffer_send_faces_vector_ac(max_nibool_interfaces,ninterface_acoustic))
382  allocate(buffer_recv_faces_vector_ac(max_nibool_interfaces,ninterface_acoustic))
383 
384  ! CUSTOM_REAL arrays
385  call assemble_mpi_scalar_ac_blocking(test_flag_cr)
386 
387  ! checks number of interface points
388  inum = 0
389  do iglob = 1,nglob
390  ! only counts flags with MPI contributions
391  if (nint(test_flag_cr(iglob)) > myrank + 1 ) inum = inum + 1
392  enddo
393 
394  deallocate(request_send_recv_acoustic)
395  deallocate(buffer_send_faces_vector_ac)
396  deallocate(buffer_recv_faces_vector_ac)
397  deallocate(test_flag_cr)
398  endif
399 
400  ! note: this MPI reduction awaits information from all processes.
401  call sum_all_i(inum,num_points2)
402  if (myrank == 0) then
403  write(imain,*) ' total number of global points assembled by acoustic MPI interfaces:',num_points2
404  call flush_imain()
405  endif
406 
407  ! checks
408  if (num_points1 /= num_points2) then
409  stop 'Error acoustic MPI interfaces has invalid assembly'
410  else
411  if (myrank == 0) then
412  write(imain,*) ' interfaces okay'
413  write(imain,*)
414  call flush_imain()
415  endif
416  endif
417 
418  endif
419 
420  end subroutine get_mpi_interfaces

425 
426  subroutine get_mpi_setup_inner_outer_elements()
427 
428 ! sets up inner and outer elements for overlapping communication
429 
430  use constants, only: imain,ngllx,ngllz
431 
432  use specfem_par, only: nspec_inner,nspec_outer,ispec_is_inner, &
433  ninterface,nibool_interfaces_ext_mesh,ibool_interfaces_ext_mesh, &
434  ibool,copy_ibool_ori,integer_mask_ibool, &
435  myrank,nspec,nglob,nproc
436 
437  implicit none
438 
439  ! local parameters
440  integer :: ispec,i,j,iglob,iinterface
441  ! inner/outer elements in the case of an MPI simulation
442  integer :: ispec_inner,ispec_outer
443  integer :: num_ispec_outer, num_ispec_inner
444 
445  logical,dimension(:),allocatable :: iglob_is_inner
446  integer, dimension(:,:,:), allocatable :: ibool_outer,ibool_inner
447  integer, dimension(:), allocatable :: ispec_outer_to_glob, ispec_inner_to_glob
448 
449  real :: percentage_edge
450  integer :: nglob_outer,nglob_inner
451 
452  ! initializes flags on elements (default to all inner elements)
453  allocate(ispec_is_inner(nspec))
454  ispec_is_inner(:) = .true.
455 
456  ! initializes flags on global nodes
457  allocate(iglob_is_inner(nglob))
458  iglob_is_inner(:) = .true.
459 
460  if (nproc > 1) then
461  ! sets flag on shared global nodes on MPI interfaces
462  do iinterface = 1, ninterface
463  do i = 1, nibool_interfaces_ext_mesh(iinterface)
464  iglob = ibool_interfaces_ext_mesh(i,iinterface)
465  iglob_is_inner(iglob) = .false.
466  enddo
467  enddo
468  endif
469 
470  ! determines flags for inner elements (purely inside the partition) which contains an "outer" global node
471  do ispec = 1, nspec
472  do j = 1, ngllz
473  do i = 1, ngllx
474  iglob = ibool(i,j,ispec)
475  ispec_is_inner(ispec) = ( iglob_is_inner(iglob) .and. ispec_is_inner(ispec) )
476  enddo
477  enddo
478  enddo
479 
480  ! counts number of outer elements (in this slice)
481  nspec_inner = count(ispec_is_inner(:) .eqv. .true.)
482  nspec_outer = nspec - nspec_inner
483 
484  ! building of corresponding arrays between inner/outer elements and their global number
485  allocate(ispec_outer_to_glob(nspec_outer))
486  allocate(ispec_inner_to_glob(nspec_inner))
487 
488  num_ispec_outer = 0
489  num_ispec_inner = 0
490  do ispec = 1, nspec
491  if (ispec_is_inner(ispec)) then
492  num_ispec_inner = num_ispec_inner + 1
493  ispec_inner_to_glob(num_ispec_inner) = ispec
494  else
495  num_ispec_outer = num_ispec_outer + 1
496  ispec_outer_to_glob(num_ispec_outer) = ispec
497  endif
498  enddo
499 
500  ! user output
501  if (myrank == 0) then
502  write(imain,*) ' number of outer elements = ',nspec_outer
503  write(imain,*) ' number of inner elements = ',nspec_inner
504  write(imain,*)
505 
506  percentage_edge = 100.*nspec_inner/real(nspec)
507  write(imain,*) ' percentage of outer elements ',100. - percentage_edge,'%'
508  write(imain,*) ' percentage of inner elements ',percentage_edge,'%'
509  write(imain,*)
510  call flush_imain()
511  endif
512 
513  ! copies ibool into separate arrays for inner/outer elements
514  allocate(ibool_outer(ngllx,ngllz,nspec_outer))
515  allocate(ibool_inner(ngllx,ngllz,nspec_inner))
516 
517  ! loop over spectral elements
518  do ispec_outer = 1,nspec_outer
519  ! get global numbering for inner or outer elements
520  ispec = ispec_outer_to_glob(ispec_outer)
521  ibool_outer(:,:,ispec_outer) = ibool(:,:,ispec)
522  enddo
523 
524  ! loop over spectral elements
525  do ispec_inner = 1,nspec_inner
526  ! get global numbering for inner or outer elements
527  ispec = ispec_inner_to_glob(ispec_inner)
528  ibool_inner(:,:,ispec_inner) = ibool(:,:,ispec)
529  enddo
530 
531  ! reduces cache misses for outer elements
532  if (nspec_outer > 0) then
533  call get_global_indirect_addressing(nspec_outer,nglob,ibool_outer,copy_ibool_ori,integer_mask_ibool)
534 
535  ! the total number of points without multiples in this region is now known
536  nglob_outer = maxval(ibool_outer)
537  else
538  nglob_outer = 0
539  endif
540 
541  ! reduces cache misses for inner elements
542  if (nspec_inner > 0) then
543  call get_global_indirect_addressing(nspec_inner,nglob,ibool_inner,copy_ibool_ori,integer_mask_ibool)
544 
545  ! the total number of points without multiples in this region is now known
546  nglob_inner = maxval(ibool_inner)
547  else
548  nglob_inner = 0
549  endif
550 
551  ! user output
552  if (myrank == 0) then
553  write(imain,*) ' number of global nodes in outer elements = ',nglob_outer
554  write(imain,*) ' number of global nodes in inner elements = ',nglob_inner
555  write(imain,*)
556  call flush_imain()
557  endif
558 
559  ! frees temporary arrays
560  deallocate(ibool_inner,ibool_outer)
561  deallocate(ispec_inner_to_glob,ispec_outer_to_glob)
562  deallocate(iglob_is_inner)
563 
564  end subroutine get_mpi_setup_inner_outer_elements

572  subroutine get_mpi_phase_domains()
573 
574  use constants, only: imain
575  use specfem_par
576 
577  implicit none
578 
579  ! local parameters
580  integer :: ispec,ier
581  integer :: ispec_inner,ispec_outer
582  real :: percentage_edge
583 
584  ! user output
585  if (myrank == 0) then
586  write(imain,*) ' determining communication phases:'
587  call flush_imain()
588  endif
589 
590  ! elastic domains
591  nspec_inner_elastic = 0
592  nspec_outer_elastic = 0
593 
594  ! only if this slice contains elastic elements
595  if (any_elastic) then
596  ! counts inner and outer elements
597  do ispec = 1, nspec
598  if (ispec_is_elastic(ispec)) then
599  if (ispec_is_inner(ispec) .eqv. .true.) then
600  nspec_inner_elastic = nspec_inner_elastic + 1
601  else
602  nspec_outer_elastic = nspec_outer_elastic + 1
603  endif
604  endif
605  enddo
606 
607  ! stores indices of inner and outer elements
608  num_phase_ispec_elastic = max(nspec_inner_elastic,nspec_outer_elastic)
609  if (num_phase_ispec_elastic < 0 ) stop 'Error elastic simulation: num_phase_ispec_elastic is < zero'
610 
611  allocate( phase_ispec_inner_elastic(num_phase_ispec_elastic,2),stat=ier)
612  if (ier /= 0 ) stop 'Error allocating array phase_ispec_inner_elastic'
613  phase_ispec_inner_elastic(:,:) = 0
614 
615  ispec_inner = 0
616  ispec_outer = 0
617  do ispec = 1, nspec
618  if (ispec_is_elastic(ispec)) then
619  if (ispec_is_inner(ispec) .eqv. .true.) then
