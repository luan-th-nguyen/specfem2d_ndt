34 subroutine iterate_time()
35 
36 #ifdef USE_MPI
37  use mpi
38 #endif
39 
40  use constants, only: imain,noise_save_everywhere
41  use specfem_par
42  use specfem_par_gpu
43  use specfem_par_noise, only: noise_tomography
44 
45  implicit none
46 
47  ! local parameters
48 
49 ! time the code in order to compute element weights, which can then be used to define typical weights
50 ! for the SCOTCH domain decomposer (this is crucial, in particular in the presence of PML elements
51 ! and/or for mixed simulations, for instance fluid/solid and so on)
52  logical, parameter :: TIME_THE_COST_TO_COMPUTE_WEIGHTS_FOR_THE_DOMAIN_DECOMPOSER = .false.
53 
54  ! time
55  character(len=8) :: datein
56  character(len=10) :: timein
57  character(len=5) :: zone
58  integer, dimension(8) :: time_values
59  integer :: year,mon,day,hr,minutes,timestamp
60 
61  real :: start_time_of_time_loop,finish_time_of_time_loop,duration_of_time_loop_in_seconds
62 
63  if (myrank == 0) write(imain,400) ! Write = T i m e e v o l u t i o n l o o p =
64 !
65 !---- s t a r t t i m e i t e r a t i o n s
66 !
67 
68  ! count elapsed wall-clock time
69  call date_and_time(datein,timein,zone,time_values)
70  ! time_values(1): year
71  ! time_values(2): month of the year
72  ! time_values(3): day of the month
73  ! time_values(5): hour of the day
74  ! time_values(6): minutes of the hour
75  ! time_values(7): seconds of the minute
76  ! time_values(8): milliseconds of the second
77 
78  ! get timestamp in minutes of current date and time
79  year = time_values(1)
80  mon = time_values(2)
81  day = time_values(3)
82  hr = time_values(5)
83  minutes = time_values(6)
84  call convtime(timestamp,year,mon,day,hr,minutes)
85 
86  ! convert to seconds instead of minutes, to be more precise for 2D runs, which can be fast
87  timestamp_seconds_start = timestamp*60.d0 + time_values(7) + time_values(8)/1000.d0
88 
89 ! *********************************************************
90 ! ************* MAIN LOOP OVER THE TIME STEPS *************
91 ! *********************************************************
92 
93  ! synchronize all processes to make sure everybody is ready to start time loop
94  call synchronize_all()
95 
96  if (myrank == 0) then
97  write(imain,*)
98  write(imain,*) 'Starting time iteration loop ...'
99  write(imain,*)
100  call flush_imain()
101  endif
102 
103  ! initialize variables for writing seismograms
104  seismo_offset = 0
105  seismo_current = 0
106 
107  if (time_the_cost_to_compute_weights_for_the_domain_decomposer) then
108  if (nproc /= 1) call exit_mpi(myrank,'timing for element weights should be done in serial mode')
109  if (nstep < 1000) call exit_mpi(myrank,'timing for element weights should be done with at least 1000 time steps')
110  if (nstep > 10000) call exit_mpi(myrank,'timing for element weights does not need to be done with more than 10000 time steps')
111  if (nstep_between_output_info < nstep / 5) &
112  call exit_mpi(myrank,'timing for element weights should be done with NSTEP_BETWEEN_OUTPUT_INFO not smaller than NSTEP / 5')
113  if (simulation_type /= 1) call exit_mpi(myrank,'timing for element weights should be done with SIMULATION_TYPE = 1')
114  if (.not. p_sv) call exit_mpi(myrank,'timing for element weights should be done with P_SV set to true')
115  if (gpu_mode) call exit_mpi(myrank,'timing for element weights should be done with GPU_MODE turned off')
116  if (save_ascii_seismograms) &
117  call exit_mpi(myrank,'timing for element weights should be done with save_ASCII_seismograms turned off')
118  if (save_binary_seismograms_single) &
119  call exit_mpi(myrank,'timing for element weights should be done with save_binary_seismograms_single turned off')
120  if (save_binary_seismograms_double) &
121  call exit_mpi(myrank,'timing for element weights should be done with save_binary_seismograms_double turned off')
122  if (output_color_image) call exit_mpi(myrank,'timing for element weights should be done with output_color_image turned off')
123  if (output_postscript_snapshot) &
124  call exit_mpi(myrank,'timing for element weights should be done with output_postscript_snapshot turned off')
125  if (output_wavefield_dumps) &
126  call exit_mpi(myrank,'timing for element weights should be done with output_wavefield_dumps turned off')
127  if (output_energy) call exit_mpi(myrank,'timing for element weights should be done with OUTPUT_ENERGY turned off')
128  if (compute_integrated_energy_field) &
129  call exit_mpi(myrank,'timing for element weights should be done with COMPUTE_INTEGRATED_ENERGY_FIELD turned off')
130  endif
131 
132  call cpu_time(start_time_of_time_loop)
133 
134  do it = 1,nstep
135  ! compute current time
136  timeval = (it-1) * deltat
137 
138  ! display time step and max of norm of displacement
139  if (mod(it,nstep_between_output_info) == 0 .or. it == 5 .or. it == nstep) then
140  call check_stability()
141  endif
142 
143  do i_stage = 1, stage_time_scheme
144 
145  ! updates wavefields using Newmark time scheme
146  ! forward wavefield (acoustic/elastic/poroelastic)
147  call update_displ_newmark()
148  ! reconstructed/backward wavefields
149  if (simulation_type == 3) call update_displ_newmark_backward()
150 
151  ! acoustic domains
152  if (acoustic_simulation) then
153  if (.not. gpu_mode) then
154  call compute_forces_acoustic_main()
155  if (simulation_type == 3) call compute_forces_acoustic_main_backward()
156  else
157  ! on GPU
158  if (any_acoustic) call compute_forces_acoustic_gpu()
159  endif
160  endif
161 
162  ! gravitoacoustic domains
163  if (gravitoacoustic_simulation) then
164  if (.not. gpu_mode) then
165  call compute_forces_gravitoacoustic_main()
166  else
167  ! on GPU
168  if (any_gravitoacoustic) call exit_mpi(myrank,'gravitoacoustic not implemented in GPU MODE yet')
169  endif
170  endif
171 
172  ! elastic domains
173  if (elastic_simulation) then
174  if (.not. gpu_mode) then
175  call compute_forces_viscoelastic_main()
176  if (simulation_type == 3) call compute_forces_viscoelastic_main_backward()
177  else
178  ! on GPU
179  if (any_elastic) call compute_forces_viscoelastic_gpu()
180  endif
181  endif
182 
183  ! poroelastic domains
184  if (poroelastic_simulation) then
185  if (.not. gpu_mode) then
186  call compute_forces_poroelastic_main()
187  if (simulation_type == 3) call compute_forces_poroelastic_main_backward()
188  else
189  ! on GPU
190  if (any_poroelastic) call exit_mpi(myrank,'poroelastic not implemented in GPU MODE yet')
191  endif
192  endif
193 
194  enddo ! stage_time_scheme (LDDRK or RK)
195 
196  ! reads in lastframe for adjoint/kernels calculation
197  if (simulation_type == 3 .and. it == 1) then
198  call read_forward_arrays()
199  endif
200 
201  ! noise simulations
202  select case (noise_tomography)
203  case (1)
204  ! stores generating wavefield
205  call save_surface_movie_noise()
206  case (2)
207  ! stores complete wavefield for reconstruction
208  if (noise_save_everywhere) call save_surface_movie_noise()
209  case (3)
210  ! reconstructs forward wavefield based on complete wavefield storage
211  if (noise_save_everywhere) call read_wavefield_noise()
212  end select
213 
214  if (output_energy) then
215  call compute_and_output_energy()
216  endif
217 
218  ! computes integrated_energy_field
219  if (compute_integrated_energy_field) then
220  call it_compute_integrated_energy_field_and_output() ! compute the field int_0^t v^2 dt and write it on file if needed
221  endif
222 
223  ! loop on all the receivers to compute and store the seismograms
224  call write_seismograms()
225 
226  ! kernels calculation
227  if (simulation_type == 3) then
228  call compute_kernels()
229  endif
230 
231  ! display results at given time steps
232  call write_movie_output()
233 
234  enddo ! end of the main time loop
235 
236  call cpu_time(finish_time_of_time_loop)
237 
238  if (time_the_cost_to_compute_weights_for_the_domain_decomposer) then
239  if (myrank == 0) then
240  duration_of_time_loop_in_seconds = finish_time_of_time_loop - start_time_of_time_loop
241  write(imain,*)
242  write(imain,*) 'Total duration of the time loop in seconds = ',duration_of_time_loop_in_seconds,' s'
243  write(imain,*) 'Total number of time steps = ',nstep
244  write(imain,*) 'Average duration of a time step of the time loop = ',duration_of_time_loop_in_seconds / real(NSTEP),' s'
245  write(imain,*) 'Total number of spectral elements in the mesh = ',nspec
246  write(imain,*) ' of which ',nspec - count(ispec_is_pml),' are regular elements'
247  write(imain,*) ' and ',count(ispec_is_pml),' are PML elements.'
248  write(imain,*) 'Average duration of the calculation per spectral element = ', &
249  duration_of_time_loop_in_seconds / real(NSTEP * NSPEC),' s'
250  write(imain,*)
251  call flush_imain()
252  endif
253  endif
254 
255 ! *********************************************************
256 ! ************* END MAIN LOOP OVER THE TIME STEPS *********
257 ! *********************************************************!
258 
259  ! Transfer fields from GPU card to host for further analysis
260  if (gpu_mode) call it_transfer_from_gpu()
261 
262 
263  !---- formats
264  400 format(/1x,41('=')/,' = T i m e e v o l u t i o n l o o p ='/1x,41('=')/)
265 
266 end subroutine iterate_time

272 subroutine it_transfer_from_gpu()
273 
274 ! transfers fields on GPU back onto CPU
275 
276  use constants, only: two,four_thirds
277  use specfem_par
278  use specfem_par_gpu
279 
280  implicit none
281 
282  ! local parameters
283  integer :: i,j,ispec,iglob
284  real(kind=CUSTOM_REAL) :: rhol,mul,kappal
285 
286  ! Fields transfer for imaging
287  ! acoustic domains
288  if (any_acoustic ) then
289  call transfer_fields_ac_from_device(nglob_ab,potential_acoustic,potential_dot_acoustic, &
290  potential_dot_dot_acoustic,mesh_pointer)
291  endif
292 
293  ! elastic domains
294  if (any_elastic) then
295  call transfer_fields_el_from_device(ndim*nglob_ab,tmp_displ_2d,tmp_veloc_2d,tmp_accel_2d,mesh_pointer)
296 
297  if (p_sv) then
298  ! P-SV waves
299  displ_elastic(1,:) = tmp_displ_2d(1,:)
300  displ_elastic(2,:) = tmp_displ_2d(2,:)
301 
302  veloc_elastic(1,:) = tmp_veloc_2d(1,:)
303  veloc_elastic(2,:) = tmp_veloc_2d(2,:)
304 
305  accel_elastic(1,:) = tmp_accel_2d(1,:)
306  accel_elastic(2,:) = tmp_accel_2d(2,:)
307  else
308  ! SH waves
309  displ_elastic(1,:) = tmp_displ_2d(1,:)
310  veloc_elastic(1,:) = tmp_veloc_2d(1,:)
311  accel_elastic(1,:) = tmp_accel_2d(1,:)
312  endif
313  endif
314 
315  ! finishes kernel calculations
316  if (simulation_type == 3) then
317  ! Kernel transfer
318  ! acoustic domains
319  if (any_acoustic) then
320  call transfer_kernels_ac_to_host(mesh_pointer,rho_ac_kl,kappa_ac_kl,nspec_ab)
321 
322  rhop_ac_kl(:,:,:) = rho_ac_kl(:,:,:) + kappa_ac_kl(:,:,:)
323  alpha_ac_kl(:,:,:) = two * kappa_ac_kl(:,:,:)
324  endif
325 
326  ! elastic domains
327  if (any_elastic) then
328  call transfer_kernels_el_to_host(mesh_pointer,rho_kl,mu_kl,kappa_kl,nspec_ab)
329 
330  ! Multiply each kernel point with the local coefficient
331  do ispec = 1, nspec
332  if (ispec_is_elastic(ispec)) then
333  do j = 1, ngllz
334  do i = 1, ngllx
335  iglob = ibool(i,j,ispec)
336  if (.not. assign_external_model) then
337  mul = poroelastcoef(2,1,kmato(ispec))
338  kappal = poroelastcoef(3,1,kmato(ispec)) - four_thirds * mul
339  rhol = density(1,kmato(ispec))
340  else
341  rhol = rhoext(i,j,ispec)
342  mul = rhoext(i,j,ispec)*vsext(i,j,ispec)*vsext(i,j,ispec)
343  kappal = rhoext(i,j,ispec)*vpext(i,j,ispec)*vpext(i,j,ispec) - four_thirds * mul
344  endif
345 
346  rho_kl(i,j,ispec) = - rhol * rho_kl(i,j,ispec)
347  mu_kl(i,j,ispec) = - two * mul * mu_kl(i,j,ispec)
348  kappa_kl(i,j,ispec) = - kappal * kappa_kl(i,j,ispec)
349  enddo
350  enddo
351  endif
352  enddo
353  endif
354  endif
355 
356 end subroutine it_transfer_from_gpu
