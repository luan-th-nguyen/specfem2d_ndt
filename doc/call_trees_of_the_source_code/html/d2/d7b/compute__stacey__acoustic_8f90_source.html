34  subroutine compute_stacey_acoustic(potential_dot_dot_acoustic,potential_dot_acoustic)
35 
36 ! absorbing boundaries
37 ! for Stacey paraxial absorbing conditions (more precisely: Sommerfeld in the case of a fluid) we implement them here
38 
39  use constants, only: custom_real,ngllx,ngllz,zero,one,two,two_thirds,iedge1,iedge2,iedge3,iedge4
40 
41  use specfem_par, only: axisym,nglob,nelemabs,it,any_acoustic, &
42  assign_external_model,ibool,kmato,numabs,ispec_is_acoustic, &
43  codeabs,codeabs_corner, &
44  density,poroelastcoef,xix,xiz,gammax,gammaz,jacobian, &
45  vpext,rhoext, &
46  wxgll,wzgll, &
47  ibegin_edge1,iend_edge1,ibegin_edge3,iend_edge3, &
48  ibegin_edge4,iend_edge4,ibegin_edge2,iend_edge2, &
49  save_forward, &
50  ib_left,ib_right,ib_bottom,ib_top, &
51  b_absorb_acoustic_left,b_absorb_acoustic_right, &
52  b_absorb_acoustic_bottom,b_absorb_acoustic_top, &
53  stacey_absorbing_conditions
54 
55  implicit none
56 
57  real(kind=CUSTOM_REAL), dimension(nglob),intent(inout) :: potential_dot_dot_acoustic
58  real(kind=CUSTOM_REAL), dimension(nglob),intent(in) :: potential_dot_acoustic
59 
60  ! local parameters
61  integer :: ispecabs,ispec,i,j,iglob
62  integer :: ibegin,iend,jbegin,jend
63  real(kind=CUSTOM_REAL) :: weight,xxi,zxi,xgamma,zgamma,jacobian1D
64  ! material properties of the acoustic medium
65  real(kind=CUSTOM_REAL) :: mul_relaxed,lambdal_relaxed,kappal,cpl,rhol
66 
67  ! checks if anything to do
68  if (.not. stacey_absorbing_conditions) return
69  if (.not. any_acoustic) return
70 
71  do ispecabs = 1,nelemabs
72  ispec = numabs(ispecabs)
73 
74  ! Sommerfeld condition if acoustic
75  if (ispec_is_acoustic(ispec)) then
76 
77  ! get elastic parameters of current spectral element
78  lambdal_relaxed = poroelastcoef(1,1,kmato(ispec))
79  mul_relaxed = poroelastcoef(2,1,kmato(ispec))
80 
81  if (axisym) then ! CHECK kappa
82  kappal = lambdal_relaxed + two_thirds * mul_relaxed
83  else
84  kappal = lambdal_relaxed + mul_relaxed
85  endif
86 
87  rhol = density(1,kmato(ispec))
88 
89  cpl = sqrt(kappal/rhol)
90 
91  !--- left absorbing boundary
92  if (codeabs(iedge4,ispecabs)) then
93  i = 1
94  jbegin = ibegin_edge4(ispecabs)
95  jend = iend_edge4(ispecabs)
96  do j = jbegin,jend
97  iglob = ibool(i,j,ispec)
98  ! external velocity model
99  if (assign_external_model) then
100  cpl = vpext(i,j,ispec)
101  rhol = rhoext(i,j,ispec)
102  endif
103  xgamma = - xiz(i,j,ispec) * jacobian(i,j,ispec)
104  zgamma = + xix(i,j,ispec) * jacobian(i,j,ispec)
105  jacobian1d = sqrt(xgamma ** 2 + zgamma ** 2)
106  weight = jacobian1d * wzgll(j)
107 
108  ! adds absorbing boundary contribution
109  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) - potential_dot_acoustic(iglob) * weight/cpl/rhol
110 
111  if (save_forward) then
112  ! saves contribution
113  b_absorb_acoustic_left(j,ib_left(ispecabs),it) = potential_dot_acoustic(iglob) * weight/cpl/rhol
114  endif
115  enddo
116  endif ! end of left absorbing boundary
117 
118  !--- right absorbing boundary
119  if (codeabs(iedge2,ispecabs)) then
120  i = ngllx
121  jbegin = ibegin_edge2(ispecabs)
122  jend = iend_edge2(ispecabs)
123  do j = jbegin,jend
124  iglob = ibool(i,j,ispec)
125  ! external velocity model
126  if (assign_external_model) then
127  cpl = vpext(i,j,ispec)
128  rhol = rhoext(i,j,ispec)
129  endif
130  xgamma = - xiz(i,j,ispec) * jacobian(i,j,ispec)
131  zgamma = + xix(i,j,ispec) * jacobian(i,j,ispec)
132  jacobian1d = sqrt(xgamma ** 2 + zgamma ** 2)
133  weight = jacobian1d * wzgll(j)
134 
135  ! adds absorbing boundary contribution
136  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) - potential_dot_acoustic(iglob) * weight/cpl/rhol
137 
138  if (save_forward) then
139  ! saves contribution
140  b_absorb_acoustic_right(j,ib_right(ispecabs),it) = potential_dot_acoustic(iglob) * weight/cpl/rhol
141  endif
142  enddo
143  endif ! end of right absorbing boundary
144 
145  !--- bottom absorbing boundary
146  if (codeabs(iedge1,ispecabs)) then
147  j = 1
148  ibegin = ibegin_edge1(ispecabs)
149  iend = iend_edge1(ispecabs)
150  ! exclude corners to make sure there is no contradiction on the normal
151  if (codeabs_corner(1,ispecabs)) ibegin = 2
152  if (codeabs_corner(2,ispecabs)) iend = ngllx-1
153  do i = ibegin,iend
154  iglob = ibool(i,j,ispec)
155  ! external velocity model
156  if (assign_external_model) then
157  cpl = vpext(i,j,ispec)
158  rhol = rhoext(i,j,ispec)
159  endif
160  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
161  zxi = - gammax(i,j,ispec) * jacobian(i,j,ispec)
162  jacobian1d = sqrt(xxi ** 2 + zxi ** 2)
163  weight = jacobian1d * wxgll(i)
164 
165  ! adds absorbing boundary contribution
166  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) - potential_dot_acoustic(iglob) * weight/cpl/rhol
167 
168  if (save_forward) then
169  ! saves contribution
170  b_absorb_acoustic_bottom(i,ib_bottom(ispecabs),it) = potential_dot_acoustic(iglob) * weight/cpl/rhol
171  endif
172  enddo
173  endif ! end of bottom absorbing boundary
174 
175  !--- top absorbing boundary
176  if (codeabs(iedge3,ispecabs)) then
177  j = ngllz
178  ibegin = ibegin_edge3(ispecabs)
179  iend = iend_edge3(ispecabs)
180  ! exclude corners to make sure there is no contradiction on the normal
181  if (codeabs_corner(3,ispecabs)) ibegin = 2
182  if (codeabs_corner(4,ispecabs)) iend = ngllx-1
183  do i = ibegin,iend
184  iglob = ibool(i,j,ispec)
185  ! external velocity model
186  if (assign_external_model) then
187  cpl = vpext(i,j,ispec)
188  rhol = rhoext(i,j,ispec)
189  endif
190  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
191  zxi = - gammax(i,j,ispec) * jacobian(i,j,ispec)
192  jacobian1d = sqrt(xxi ** 2 + zxi ** 2)
193  weight = jacobian1d * wxgll(i)
194 
195  ! adds absorbing boundary contribution
196  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) - potential_dot_acoustic(iglob) * weight/cpl/rhol
197 
198  if (save_forward) then
199  ! saves contribution
200  b_absorb_acoustic_top(i,ib_top(ispecabs),it) = potential_dot_acoustic(iglob) * weight/cpl/rhol
201  endif
202  enddo
203  endif ! end of top absorbing boundary
204 
205  endif ! acoustic ispec
206  enddo
207 
208  end subroutine compute_stacey_acoustic

214  subroutine compute_stacey_acoustic_backward(b_potential_dot_dot_acoustic)
215 
216 ! absorbing boundaries
217 ! uses contributions stored in forward simulation
218 
219  use constants, only: custom_real,ngllx,ngllz,zero,one,two,iedge1,iedge2,iedge3,iedge4
220 
221  use specfem_par, only: nglob,nelemabs,it,nstep,any_acoustic, &
222  ibool,numabs,ispec_is_acoustic, &
223  codeabs,codeabs_corner, &
224  ibegin_edge1,iend_edge1,ibegin_edge3,iend_edge3, &
225  ibegin_edge4,iend_edge4,ibegin_edge2,iend_edge2, &
226  ib_left,ib_right,ib_bottom,ib_top, &
227  b_absorb_acoustic_left,b_absorb_acoustic_right, &
228  b_absorb_acoustic_bottom,b_absorb_acoustic_top, &
229  stacey_absorbing_conditions
230 
231  implicit none
232 
233  real(kind=CUSTOM_REAL), dimension(nglob),intent(inout) :: b_potential_dot_dot_acoustic
234 
235  ! local parameters
236  integer :: ispecabs,ispec,i,j,iglob
237  integer :: ibegin,iend,jbegin,jend
238  integer :: it_tmp
239 
240  ! checks if anything to do
241  if (.not. stacey_absorbing_conditions) return
242  if (.not. any_acoustic) return
243 
244  ! time increment step
245  it_tmp = nstep - it + 1
246 
247  do ispecabs = 1,nelemabs
248  ispec = numabs(ispecabs)
249 
250  ! Sommerfeld condition if acoustic
251  if (ispec_is_acoustic(ispec)) then
252  !--- left absorbing boundary
253  if (codeabs(iedge4,ispecabs)) then
254  i = 1
255  jbegin = ibegin_edge4(ispecabs)
256  jend = iend_edge4(ispecabs)
257  do j = jbegin,jend
258  iglob = ibool(i,j,ispec)
259  b_potential_dot_dot_acoustic(iglob) = b_potential_dot_dot_acoustic(iglob) - &
260  b_absorb_acoustic_left(j,ib_left(ispecabs),it_tmp)
261 
262  enddo
263  endif ! end of left absorbing boundary
264 
265  !--- right absorbing boundary
266  if (codeabs(iedge2,ispecabs)) then
267  i = ngllx
268  jbegin = ibegin_edge2(ispecabs)
269  jend = iend_edge2(ispecabs)
270  do j = jbegin,jend
271  iglob = ibool(i,j,ispec)
272  ! adds (previously) stored contribution
273  b_potential_dot_dot_acoustic(iglob) = b_potential_dot_dot_acoustic(iglob) - &
274  b_absorb_acoustic_right(j,ib_right(ispecabs),it_tmp)
275  enddo
276  endif ! end of right absorbing boundary
277 
278  !--- bottom absorbing boundary
279  if (codeabs(iedge1,ispecabs)) then
280  j = 1
281  ibegin = ibegin_edge1(ispecabs)
282  iend = iend_edge1(ispecabs)
283  ! exclude corners to make sure there is no contradiction on the normal
284  if (codeabs_corner(1,ispecabs)) ibegin = 2
285  if (codeabs_corner(2,ispecabs)) iend = ngllx-1
286  do i = ibegin,iend
287  iglob = ibool(i,j,ispec)
288  ! adds (previously) stored contribution
289  b_potential_dot_dot_acoustic(iglob) = b_potential_dot_dot_acoustic(iglob) - &
290  b_absorb_acoustic_bottom(i,ib_bottom(ispecabs),it_tmp)
291  enddo
292  endif ! end of bottom absorbing boundary
293 
294  !--- top absorbing boundary
295  if (codeabs(iedge3,ispecabs)) then
296  j = ngllz
297  ibegin = ibegin_edge3(ispecabs)
298  iend = iend_edge3(ispecabs)
299  ! exclude corners to make sure there is no contradiction on the normal
300  if (codeabs_corner(3,ispecabs)) ibegin = 2
301  if (codeabs_corner(4,ispecabs)) iend = ngllx-1
302  do i = ibegin,iend
303  iglob = ibool(i,j,ispec)
304  b_potential_dot_dot_acoustic(iglob) = b_potential_dot_dot_acoustic(iglob) - &
305  b_absorb_acoustic_top(i,ib_top(ispecabs),it_tmp)
306  enddo
307  endif ! end of top absorbing boundary
308  endif ! acoustic ispec
309  enddo
310 
311  end subroutine compute_stacey_acoustic_backward
