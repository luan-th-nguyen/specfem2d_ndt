34 ! XSMOOTH_SEM
35 !
36 ! USAGE
37 ! mpirun -np NPROC bin/xsmooth_sem SIGMA_H SIGMA_V KERNEL_NAME INPUT_DIR OUPUT_DIR GPU_MODE
38 !
39 !
40 ! COMMAND LINE ARGUMENTS
41 ! SIGMA_H - horizontal smoothing radius
42 ! SIGMA_V - vertical smoothing radius
43 ! KERNEL_NAME - kernel name, e.g. alpha_kernel
44 ! INPUT_DIR - directory from which kernels are read
45 ! OUTPUT_DIR - directory to which smoothed kernels are written
46 ! GPU_MODE - use GPUs to process smoothing (logical T F)
47 
48 ! DESCRIPTION
49 ! Smooths kernels by convolution with a Gaussian. Writes the resulting
50 ! smoothed kernels to OUTPUT_DIR.
51 !
52 ! Files written to OUTPUT_DIR have the suffix 'smooth' appended,
53 ! e.g. proc***alpha_kernel.bin becomes proc***alpha_kernel_smooth.bin
54 !
55 ! This program's primary use case is to smooth kernels. It can be used though on
56 ! any scalar field of dimension (NGLLX,NGLLZ,NSPEC).
57 !
58 ! This is a parallel program -- it must be invoked with mpirun or other
59 ! appropriate utility. Operations are performed in embarassingly-parallel
60 ! fashion.

482  subroutine smoothing_weights_vec(x0,z0,sigma_h2_inv,sigma_v2_inv,exp_val,xx_elem,zz_elem)
483 
484  use constants
485  implicit none
486 
487  real(kind=CUSTOM_REAL),dimension(NGLLX,NGLLZ),intent(out) :: exp_val
488  real(kind=CUSTOM_REAL),dimension(NGLLX,NGLLZ),intent(in) :: xx_elem, zz_elem
489  real(kind=CUSTOM_REAL),intent(in) :: x0,z0,sigma_h2_inv,sigma_v2_inv
490 
491 
492  ! local parameters
493  integer :: ii,jj
494  real(kind=CUSTOM_REAL) :: dist_h,dist_v
495 
496 
497  do jj = 1, ngllz
498  do ii = 1, ngllx
499  ! gets vertical and horizontal distance
500  call get_distance_square_vec(dist_h,dist_v,x0,z0,xx_elem(ii,jj),zz_elem(ii,jj))
501  ! Gaussian function
502  exp_val(ii,jj) = exp(- sigma_h2_inv*dist_h - sigma_v2_inv*dist_v)
503  enddo
504  enddo

506  end subroutine smoothing_weights_vec
