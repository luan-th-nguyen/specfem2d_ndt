35  subroutine read_forward_arrays()
36 
37 ! restores last time snapshot saved for backward/reconstruction of wavefields
38 ! note: this is done here after the Newmark time scheme, otherwise the indexing for sources
39 ! and adjoint sources will become more complicated
40 ! that is, index it for adjoint sources will match index NSTEP - 1 for backward/reconstructed wavefields
41 
42  use constants, only: output_files
43  use specfem_par
44  use specfem_par_gpu
45 
46  implicit none
47 
48  ! local parameters
49  integer :: ier
50  character(len=MAX_STRING_LEN) :: outputname
51 
52  ! acoustic medium
53  if (any_acoustic) then
54  write(outputname,'(a,i6.6,a)') 'lastframe_acoustic',myrank,'.bin'
55  open(unit=55,file=trim(output_files)//trim(outputname),status='old',action='read',form='unformatted',iostat=ier)
56  if (ier /= 0) call exit_mpi(myrank,'Error opening file '//trim(output_files)//'lastframe_acoustic**.bin')
57 
58  read(55) b_potential_acoustic
59  read(55) b_potential_dot_acoustic
60  read(55) b_potential_dot_dot_acoustic
61 
62  close(55)
63 
64  if (gpu_mode) then
65  ! transfers fields onto GPU
66  call transfer_b_fields_ac_to_device(nglob_ab,b_potential_acoustic, &
67  b_potential_dot_acoustic, &
68  b_potential_dot_dot_acoustic, &
69  mesh_pointer)
70  else
71  ! free surface for an acoustic medium
72  call enforce_acoustic_free_surface(b_potential_dot_dot_acoustic,b_potential_dot_acoustic,b_potential_acoustic)
73  endif
74  endif
75 
76  ! elastic medium
77  if (any_elastic) then
78  write(outputname,'(a,i6.6,a)') 'lastframe_elastic',myrank,'.bin'
79  open(unit=55,file=trim(output_files)//trim(outputname),status='old',action='read',form='unformatted',iostat=ier)
80  if (ier /= 0) call exit_mpi(myrank,'Error opening file '//trim(output_files)//'lastframe_elastic**.bin')
81 
82  read(55) b_displ_elastic
83  read(55) b_veloc_elastic
84  read(55) b_accel_elastic
85  close(55)
86 
87  !SH (membrane) waves
88  if (.not. p_sv) then
89  ! only index array(1,:) contains SH wavefield
90  b_displ_elastic(2,:) = 0._custom_real
91  b_veloc_elastic(2,:) = 0._custom_real
92  b_accel_elastic(2,:) = 0._custom_real
93  endif
94 
95  if (gpu_mode) then
96  ! prepares wavefields for transfering
97  if (p_sv) then
98  tmp_displ_2d(1,:) = b_displ_elastic(1,:)
99  tmp_displ_2d(2,:) = b_displ_elastic(2,:)
100  tmp_veloc_2d(1,:) = b_veloc_elastic(1,:)
101  tmp_veloc_2d(2,:) = b_veloc_elastic(2,:)
102  tmp_accel_2d(1,:) = b_accel_elastic(1,:)
103  tmp_accel_2d(2,:) = b_accel_elastic(2,:)
104  else
105  ! SH waves
106  tmp_displ_2d(1,:) = b_displ_elastic(1,:)
107  tmp_displ_2d(2,:) = 0._custom_real
108  tmp_veloc_2d(1,:) = b_veloc_elastic(1,:)
109  tmp_veloc_2d(2,:) = 0._custom_real
110  tmp_accel_2d(1,:) = b_accel_elastic(1,:)
111  tmp_accel_2d(2,:) = 0._custom_real
112  endif
113  call transfer_b_fields_to_device(ndim*nglob_ab,tmp_displ_2d,tmp_veloc_2d,tmp_accel_2d,mesh_pointer)
114  endif
115  endif
116 
117  ! poroelastic medium
118  if (any_poroelastic) then
119  write(outputname,'(a,i6.6,a)') 'lastframe_poroelastic_s',myrank,'.bin'
120  open(unit=55,file=trim(output_files)//trim(outputname),status='old',action='read',form='unformatted',iostat=ier)
121  if (ier /= 0) call exit_mpi(myrank,'Error opening file '//trim(output_files)//'lastframe_poroelastic_s**.bin')
122 
123  read(55) b_displs_poroelastic
124  read(55) b_velocs_poroelastic
125  read(55) b_accels_poroelastic
126  close(55)
127 
128  write(outputname,'(a,i6.6,a)') 'lastframe_poroelastic_w',myrank,'.bin'
129  open(unit=56,file=trim(output_files)//trim(outputname),status='old',action='read',form='unformatted',iostat=ier)
130  if (ier /= 0) call exit_mpi(myrank,'Error opening file '//trim(output_files)//'lastframe_poroelastic_w**.bin')
131 
132  read(56) b_displw_poroelastic
133  read(56) b_velocw_poroelastic
134  read(56) b_accelw_poroelastic
135  close(56)
136 
137  ! safety check
138  if (gpu_mode) then
139  stop 'GPU_MODE error: sorry, reading lastframe from poroelastic simulation not implemented yet'
140  endif
141  endif
142 
143  end subroutine read_forward_arrays

149  subroutine read_forward_arrays_undoatt()
150 
151 ! reads in saved wavefields
152 
153  use constants, only: iin_undo_att,max_string_len,output_files
154 
155  use specfem_par, only: myrank,iteration_on_subset,nsubset_iterations, &
156  any_acoustic,any_elastic,attenuation_viscoelastic, &
157  b_potential_acoustic,b_potential_dot_acoustic,b_potential_dot_dot_acoustic, &
158  b_displ_elastic,b_veloc_elastic,b_accel_elastic,b_e1,b_e11,b_e13
159 
160  implicit none
161 
162  ! local parameters
163  integer :: iteration_on_subset_tmp
164  integer :: ier
165  character(len=MAX_STRING_LEN) :: outputname
166 
167  ! current subset iteration
168  iteration_on_subset_tmp = nsubset_iterations - iteration_on_subset + 1
169 
170  ! reads in saved wavefield
171  write(outputname,'(a,i6.6,a,i6.6,a)') 'proc',myrank,'_save_frame_at',iteration_on_subset_tmp,'.bin'
172 
173  ! opens corresponding snapshot file for reading
174  open(unit=iin_undo_att,file=trim(output_files)//outputname, &
175  status='old',action='read',form='unformatted',iostat=ier)
176  if (ier /= 0 ) call exit_mpi(myrank,'Error opening file proc***_save_frame_at** for reading')
177 
178  if (any_acoustic) then
179  read(iin_undo_att) b_potential_dot_dot_acoustic
180  read(iin_undo_att) b_potential_dot_acoustic
181  read(iin_undo_att) b_potential_acoustic
182  endif
183 
184  if (any_elastic) then
185  read(iin_undo_att) b_accel_elastic
186  read(iin_undo_att) b_veloc_elastic
187  read(iin_undo_att) b_displ_elastic
188 
189  if (attenuation_viscoelastic) then
190  read(iin_undo_att) b_e1
191  read(iin_undo_att) b_e11
192  read(iin_undo_att) b_e13
193  endif
194  endif
195 
196  close(iin_undo_att)
197 
198  end subroutine read_forward_arrays_undoatt
