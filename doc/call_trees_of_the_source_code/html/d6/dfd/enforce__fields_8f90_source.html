57  subroutine build_forced()
58 ! ----------------------------------------------------------------------------------------
59 ! This subroutine build the logical array: is_forced which gives for each GLL
60 ! point if we impose the displacement on this point or not.
61 ! For example: iglob_is_forced(iglob) = .false.
62 ! forced has already been already initialized to .false.
63 ! ----------------------------------------------------------------------------------------
64 
65  use specfem_par, only: coord,nglob,ibool,iglob_is_forced,myrank,nelem_acforcing,codeacforcing,numacforcing,ibool, &
66  pml_boundary_conditions,ispec_is_pml,read_external_mesh
67  use enforce_par
68  use constants, only: tinyval,imain,ngllx,ngllz,iedge1,iedge2,iedge3,iedge4
69 
70  implicit none
71 
72  !local variables
73  integer :: inum,ispec,i,j,iglob
74 
75  if (read_external_mesh) then
76 
77  ! loop on all the forced edges
78  do inum = 1,nelem_acforcing
79  ispec = numacforcing(inum)
80  !--- left acoustic forcing boundary
81  if (codeacforcing(iedge4,inum)) then
82  i = 1
83  do j = 1,ngllz
84  iglob = ibool(i,j,ispec)
85  if (pml_boundary_conditions) then
86  if (.not. ispec_is_pml(ispec)) then
87  iglob_is_forced(iglob) = .true.
88  nforced = nforced + 1
89  endif
90  else
91  iglob_is_forced(iglob) = .true.
92  nforced = nforced + 1
93  endif
94  enddo
95  endif ! end of left acoustic forcing boundary
96  !--- right acoustic forcing boundary
97  if (codeacforcing(iedge2,inum)) then
98  i = ngllx
99  do j = 1,ngllz
100  iglob = ibool(i,j,ispec)
101  if (pml_boundary_conditions) then
102  if (.not. ispec_is_pml(ispec)) then
103  iglob_is_forced(iglob) = .true.
104  nforced = nforced + 1
105  endif
106  else
107  iglob_is_forced(iglob) = .true.
108  nforced = nforced + 1
109  endif
110  enddo
111  endif ! end of right acoustic forcing boundary
112  !--- bottom acoustic forcing boundary
113  if (codeacforcing(iedge1,inum)) then
114  j = 1
115  do i = 1,ngllx
116  iglob = ibool(i,j,ispec)
117  if (pml_boundary_conditions) then
118  if (.not. ispec_is_pml(ispec)) then
119  iglob_is_forced(iglob) = .true.
120  nforced = nforced + 1
121  endif
122  else
123  iglob_is_forced(iglob) = .true.
124  nforced = nforced + 1
125  endif
126  enddo
127  endif ! end of bottom acoustic forcing boundary
128  !--- top acoustic forcing boundary
129  if (codeacforcing(iedge3,inum)) then
130  j = ngllz
131  do i = 1,ngllx
132  iglob = ibool(i,j,ispec)
133  if (pml_boundary_conditions) then
134  if (.not. ispec_is_pml(ispec)) then
135  iglob_is_forced(iglob) = .true.
136  nforced = nforced + 1
137  endif
138  else
139  iglob_is_forced(iglob) = .true.
140  nforced = nforced + 1
141  endif
142  enddo
143  endif ! end of top acoustic forcing boundary
144  enddo
145 
146  else
147  ! internal mesher:
148  ! Loop on all the GLL points
149  do iglob = 1,nglob
150  ! forced points along a specified x-line
151  if (abs(coord(1,iglob) - xforced) < tinyval) then
152  iglob_is_forced(iglob) = .true.
153  nforced = nforced + 1
154  endif
155  enddo
156 
157  endif
158 
159  ! user output
160  call sum_all_i(nforced, nforced_sum)
161  if (myrank == 0) then
162  if (nforced_sum == 0) then
163  call exit_mpi(myrank,'No forced integration point found!')
164  else
165  write(imain,*) "Number of GLL points forced:",nforced," over ",nglob
166  call flush_imain()
167  endif
168  endif
169 
170  end subroutine build_forced

176  subroutine enforce_fields(iglob,it)
177 ! ----------------------------------------------------------------------------------------
178 ! This subroutine impose the fields at given GLL points and at a given time steps
179 ! ----------------------------------------------------------------------------------------
180 
181  use specfem_par, only: coord,displ_elastic,veloc_elastic,accel_elastic,deltat,deltatover2, &
182  deltatsquareover2
183  use enforce_par
184  use constants, only: tinyval,custom_real,two,pi
185 
186  implicit none
187 
188  ! Inputs
189  integer, intent(in) :: iglob,it
190 
191  ! Local variables
192  real(kind=CUSTOM_REAL) :: f0 = 0.5236d6 ! frequency
193  real(kind=CUSTOM_REAL) :: h = 1.0d-3 ! half width of the plate
194  real(kind=CUSTOM_REAL) :: cp = 5960.0d0 ! Compressional waves velocity
195  real(kind=CUSTOM_REAL) :: cs = 3260.d0 ! Shear waves velocity
196  real(kind=CUSTOM_REAL), dimension(2) :: accelOld,velocOld
197  real(kind=CUSTOM_REAL) :: factor,x,z,facx,facz,tval,tval_old
198  real(kind=CUSTOM_REAL) :: omegaj
199  real(kind=CUSTOM_REAL) :: time_dependence_x,time_dependence_x_old,time_dependence_z,time_dependence_z_old
200  complex :: ux,uz
201  logical :: antisym = .true.
202  integer :: Nc = 10
203 
204  factor = 1.0d0
205  omegaj = two*pi*f0
206 
207  ! gets global node location
208  x = coord(1,iglob)
209  z = coord(2,iglob)
210 
211  ! timing
212  tval = (it-1)*deltat
213  tval_old = (it-2)*deltat
214 
215  if (tval < two*nc*pi/omegaj) then
216  time_dependence_x = omegaj**2*cos(omegaj*tval/nc)*sin(omegaj*tval)/nc**2 + &
217  2*omegaj**2*sin(omegaj*tval/nc)*cos(omegaj*tval)/nc - &
218  (1.0d0 - cos(omegaj*tval/nc))*omegaj**2*sin(omegaj*tval)
219  time_dependence_z = omegaj**2*cos(omegaj*tval/nc)*cos(omegaj*tval)/nc**2 - &
220  2*omegaj**2*sin(omegaj*tval/nc)*sin(omegaj*tval)/nc - &
221  (1.0d0 - cos(omegaj*tval/nc))*omegaj**2*cos(omegaj*tval)
222  else
223  time_dependence_x = 0.0d0
224  time_dependence_z = 0.0d0
225  endif
226  if (tval_old < two*nc*pi/omegaj) then
227  time_dependence_x_old = omegaj**2*cos(omegaj*tval_old/nc)*sin(omegaj*tval_old)/nc**2 + &
228  2*omegaj**2*sin(omegaj*tval_old/nc)*cos(omegaj*tval_old)/nc - &
229  (1.0d0 - cos(omegaj*tval_old/nc))*omegaj**2*sin(omegaj*tval_old)
230 
231  time_dependence_z_old = omegaj**2*cos(omegaj*tval_old/nc)*cos(omegaj*tval_old)/nc**2 - &
232  2*omegaj**2*sin(omegaj*tval_old/nc)*sin(omegaj*tval_old)/nc - &
233  (1.0d0 - cos(omegaj*tval_old/nc))*omegaj**2*cos(omegaj*tval_old)
234  else
235  time_dependence_x_old = 0.0d0
236  time_dependence_z_old = 0.0d0
237  endif
238 
239  call calculateuxuz(ux,uz,z,cp,cs,h,omegaj,antisym)
240 
241  if ((abs(real(ux)) < tinyval) .and. (abs(aimag(ux)) > tinyval)) then ! if ux is imaginary
242  !print *,"Ux imaginary"
243  if (abs(aimag(uz)) > tinyval) then ! ... and uz not real
244  print *,"Problem!! ux is imaginary while uz is not real (ux:",ux," uz:",uz,")"
245  endif
246  facx = aimag(ux)
247  facz = real(uz)
248  else if ((abs(real(ux)) > tinyval) .and. (abs(aimag(ux)) < tinyval)) then ! if ux is real
249  !print *,"Ux real"
250  if (abs(real(uz)) > tinyval) then ! ... and uz not imaginary
251  print *,"Problem!! ux is real while uz is not imaginary (ux:",ux," uz:",uz,")"
252  endif
253  facx = real(ux)
254  facz = aimag(uz)
255  else
256  facx = real(ux)
257  facz = aimag(uz)
258  endif
259 
260  !print *,z,facz
261 
262  if (abs(z + 0.0d-3) < 1.0d-3) then
263  if (it == 1) then
264  ! We initialize the variables
265  displ_elastic(1,iglob) = factor*facx*0.0d0
266  displ_elastic(2,iglob) = factor*facz*0.0d0
267  veloc_elastic(1,iglob) = factor*facx*0.0d0
268  veloc_elastic(2,iglob) = factor*facz*0.0d0
269  accel_elastic(1,iglob) = factor*facx*0.0d0
270  accel_elastic(2,iglob) = factor*facz*(omegaj**2/nc**2)
271  else
272  ! We set what we want
273  accel_elastic(1,iglob) = factor*facx*time_dependence_x
274  accelold(1) = factor*facx*time_dependence_x_old
275  accel_elastic(2,iglob) = factor*facz*time_dependence_z
276  accelold(2) = factor*facz*time_dependence_z_old
277  ! Do not change anything below: we compute numerically the velocity and displacement
278  velocold(1) = veloc_elastic(1,iglob)
279  veloc_elastic(1,iglob) = veloc_elastic(1,iglob) + deltatover2*(accelold(1) + accel_elastic(1,iglob))
280  displ_elastic(1,iglob) = displ_elastic(1,iglob) + deltat*velocold(1) + deltatsquareover2*accelold(1)
281  velocold(2) = veloc_elastic(2,iglob)
282  veloc_elastic(2,iglob) = veloc_elastic(2,iglob) + deltatover2*(accelold(2) + accel_elastic(2,iglob))
283  displ_elastic(2,iglob) = displ_elastic(2,iglob) + deltat*velocold(2) + deltatsquareover2*accelold(2)
284  endif
285  else
286  displ_elastic(:,iglob) = 0.0d0
287  veloc_elastic(:,iglob) = 0.0d0
288  accel_elastic(:,iglob) = 0.0d0
289  endif
290 
291  end subroutine enforce_fields

297  subroutine calculateuxuz(ux,uz,zi,cp,cs,h,omegaj,antisym)
298  ! ----------------------------------------------------------------------------------------
299  ! See eq. (3) (4) Hora & Cervena 2012
300  ! ----------------------------------------------------------------------------------------
301 
302  use constants, only: custom_real,two,pi
303 
304  implicit none
305 
306  ! Inputs
307  logical, intent(in) :: antisym
308  real(kind=CUSTOM_REAL), intent(in) :: zi,cp,cs,h,omegaj
309 
310  ! Outputs
311  complex, intent(out) :: ux,uz
312 
313  ! Local variables
314  real(kind=CUSTOM_REAL) :: cphase
315  complex :: alpha,beta,k,k2,alpha2,beta2
316  complex :: jj = (0.0,1.0)
317 
318  call calculatecphase(cphase,omegaj,antisym,h)
319  k2 = (omegaj/cphase)**2
320  alpha2 = (omegaj/cp)**2 - k2
321  beta2 = (omegaj/cs)**2 - k2
322  beta = sqrt(beta2)
323  alpha = sqrt(alpha2)
324  k = sqrt(k2)
325  if (.not. antisym) then
326  ux = jj * beta * (cos(beta*zi)/cos(beta*h) - two*k2*cos(alpha*zi)/((k2 - beta2)*cos(alpha*h)))
327  uz = - k * (sin(beta*zi)/cos(beta*h) + two*alpha*beta*sin(alpha*zi)/((k2 - beta2)*cos(alpha*h)))
328  else
329  if (.false.) then ! Hora
330  ux = - jj * beta * (sin(beta*zi)/sin(beta*h) - two*k2*sin(alpha*zi)/((k2 - beta2 )*sin(alpha*h)))
331  uz = - k * (cos(beta*zi)/sin(beta*h) + two*alpha*beta*cos(alpha*zi)/((k2 - beta2 )*sin(alpha*h)))
332  else ! Alleyne
333  ux = jj * beta * (sin(beta*zi)/sin(beta*h) - two*k2*sin(alpha*zi)/((k2 - beta2)*sin(alpha*h)))
334  uz = - k * (cos(beta*zi)/sin(beta*h) + two*alpha*beta*cos(alpha*zi)/((k2 - beta2)*sin(alpha*h)))
335  endif
336  endif
337 
338  end subroutine calculateuxuz

344  subroutine calculatecphase(cphase,omegaj,antisym,h) !,cp,cs,h)
345  ! ----------------------------------------------------------------------------------------
346  ! This is not trivial !! We use here an approximate value
347  ! do something clever here

352 
353  implicit none

355  ! Inputs
356  logical, intent(in) :: antisym

361 
362  ! Local variables
363  real(kind=CUSTOM_REAL) :: calculated1 = 0.5d6 * 1.5d-3
364  real(kind=CUSTOM_REAL) :: calculated2 = 2.087d6 * 1.0d-3
365  real(kind=CUSTOM_REAL) :: calculated3 = 0.5236d6 * 1.0d-3
366  real(kind=CUSTOM_REAL) :: calculated4 = 2.3113d6 * 1.0d-3

374  else
375  cphase = 5300.0d0 ! S0 at fd = 0.75 Mhz.mm
376  endif
377  else if (abs(freq*h - calculated2) < tinyval) then ! warning this line need double precision
378  if (antisym) then
379  cphase = 5234.0d0 ! A1 at fd = 2.087 Mhz.mm
380  else
381  cphase = 5883.0d0 ! S1 at fd = 2.087 Mhz.mm
382  endif
383  else if (abs(freq*h - calculated3) < tinyval) then ! warning this line need double precision
384  if (antisym) then
385  cphase = 1937.0d0 ! A0 at fd = 0.5236 Mhz.mm
386  else
387  cphase = 5445.0d0 ! S0 at fd = 0.5236 Mhz.mm
388  endif
389  else if (abs(freq*h - calculated4) < tinyval) then ! warning this line need double precision
390  if (antisym) then
391  cphase = 4595.0d0 ! A1 at fd = 2.3113 Mhz.mm
392  else
393  cphase = 5780.0d0 ! S1 at fd = 2.3113 Mhz.mm
394  endif
395  else
396  print *,"f*d = ",freq*h,"Hz*m",calculated3
397  call exit_mpi(myrank,'Phase speed not calculated for this f*d product')
398  endif
399 
400  end subroutine calculatecphase
