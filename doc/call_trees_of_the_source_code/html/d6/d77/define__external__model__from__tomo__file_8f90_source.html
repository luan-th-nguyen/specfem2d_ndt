77  integer function searchinf(length, array, value,delta)
78  ! Given a a sorted array and a value, returns the index of the element that
79  ! is closest to, but less than, the given value.
80  ! Uses a binary search algorithm.
81  ! Never returns the last value !
82  !
83  ! Behaviour with vec=(-5,-2,8,12,40,45,60,62,80,95) :
84  ! value = 42.5 --> searchInf = 40
85  ! value = -2 --> searchInf = -2
86  ! value = -5 --> searchInf = -5
87  ! value = -95 --> searchInf = -5
88  ! value = 100 --> searchInf = 80
89  ! value = -8 --> searchInf = -5
90  !
91  ! TODO : This function could be rewriten using a binary search algorithm
92  ! for a better efficiency
93 
94  implicit none
95 
96  integer, intent(in) :: length
97  double precision, dimension(length), intent(in) :: array
98  double precision, intent(in) :: value
99  double precision, intent(in), optional :: delta
100  double precision :: d
101  integer :: index_closest = 1
102 
103  if (present(delta)) then
104  d = delta
105  else
106  d = 1e-9
107  endif
108 
109  if (length <= 0) then
110  stop "Incorrect length in searchInf"
111  else if (length == 1) then
112  searchinf = array(1)
113  else
114  if (array(1) > array(2)) stop "searchInf needs an increasing array"
115  if (value < array(1)) then
116  searchinf = 1
117  return
118  else if (value >= array(length)) then
119  searchinf = length - 1
120  return
121  endif
122  index_closest = 1
123  do while (array(index_closest) < value)
124  index_closest = index_closest+1
125  enddo
126  if (abs(array(index_closest)-value) > d) then
127  searchinf = index_closest - 1
128  else
129  searchinf = index_closest
130  endif
131 
132  endif
133  return
134  end function searchinf

140  double precision function interpolate(x_len, x_array, y_len, y_array, f, x, y, delta)
141  ! This function uses bilinear interpolation to estimate the value
142  ! of a function f at point (x,y)
143  ! f is assumed to be sampled on a regular grid, with the grid x values specified
144  ! by x_array and the grid y values specified by y_array
145  ! This function works even if (x,y) is outside the definition domain of f
146  ! (case 1,2,3,4,5,6,7,8 below). It that case we just use border values
147  ! y
148  ! ^
149  ! case 3 | case 6 | case 2
150  ! ------------------------
151  ! | |
152  ! case 7 | case 9 | case 8
153  ! | |
154  ! ------------------------> x
155  ! case 1 | case 5 | case 4
156  ! Reference: http://en.wikipedia.org/wiki/Bilinear_interpolation
157 
158  implicit none
159 
160  integer, intent(in) :: x_len, y_len
161  double precision, dimension(x_len), intent(in) :: x_array
162  double precision, dimension(y_len), intent(in) :: y_array
163  double precision, dimension(x_len, y_len), intent(in) :: f
164  double precision, intent(in) :: x,y
165  double precision, intent(in), optional :: delta
166 
167  ! Local variables
168  double precision :: denom, x1, x2, y1, y2, minx, maxx, miny, maxy
169  integer :: i,j
170  double precision :: d
171 
172  if (present(delta)) then
173  d = delta
174  else
175  d = 1e-9
176  endif
177 
178  minx = minval(x_array)
179  maxx = maxval(x_array)
180  miny = minval(y_array)
181  maxy = maxval(y_array)
182  ! y
183  ! ^
184  ! case 3 | case 6 | case 2
185  ! ------------------------
186  ! | |
187  ! case 7 | case 9 | case 8
188  ! | |
189  ! ------------------------> x
190  ! case 1 | case 5 | case 4
191 
192  if (x <= minx .and. y <= miny) then ! case 1
193  !print *,"case1"
194  interpolate = f(1,1)
195  else if (x > maxx .and. y >= maxy) then ! case 2
196  !print *,"case2"
197  interpolate = f(x_len,y_len)
198  else if (x <= minx .and. y >= maxy) then ! case 3
199  !print *,"case3"
200  interpolate = f(1,y_len)
201  else if (x >= maxx .and. y <= miny) then ! case 4
202  !print *,"case4"
203  interpolate = f(x_len,1)
204  else if (x >= minx .and. x <= maxx .and. y <= miny) then ! case 5
205  !print *,"case5"
206  y1 = y_array(1)
207  y2 = y_array(2)
208  ! y = y1
209  i = searchinf(x_len, x_array, x) ! Test also if the array is increasing
210  j = 1
211  if (i < x_len) then
212  x1 = x_array(i)
213  x2 = x_array(i+1)
214  else
215  x1 = x_array(i-1)
216  x2 = x_array(i)
217  endif
218  denom = (x2 - x1)*(y2 - y1)
219  interpolate = (f(i,j)*(x2-x)*(y2-y1) + f(i+1,j)*(x-x1)*(y2-y1))/denom
220  else if (x >= minx .and. x <= maxx .and. y >= maxy) then ! case 6
221  !print *,"case6"
222  y1 = y_array(y_len-1)
223  y2 = y_array(y_len)
224  ! y = y2
225  i = searchinf(x_len, x_array, x) ! Test also if the array is increasing
226  j = y_len - 1
227  if (i < x_len) then
228  x1 = x_array(i)
229  x2 = x_array(i+1)
230  else
231  x1 = x_array(i-1)
232  x2 = x_array(i)
233  endif
234  denom = (x2 - x1)*(y2 - y1)
235  interpolate = (f(i,j+1)*(x2-x)*(y2-y1) + f(i+1, j+1)*(x-x1)*(y2-y1))/denom
236  else if (x <= minx .and. y >= miny .and. y <= maxy) then ! case 7
237  !print *,"case7"
238  x1 = x_array(1)
239  x2 = x_array(2)
240  ! x = x1
241  i = 1
242  j = searchinf(y_len, y_array, y) ! Test also if the array is increasing
243  if (j < y_len) then
244  y1 = y_array(j)
245  y2 = y_array(j+1)
246  else
247  y1 = y_array(j-1)
248  y2 = y_array(j)
249  endif
250  denom = (x2 - x1)*(y2 - y1)
251  interpolate = (f(i,j)*(x2-x1)*(y2-y) + f(i,j+1)*(x2-x1)*(y-y1))/denom
252  else if (x >= maxx .and. y >= miny .and. y <= maxy) then ! case 8
253  !print *,"case8"
254  x1 = x_array(x_len-1)
255  x2 = x_array(x_len)
256  ! x = x2
257  i = x_len - 1
258  j = searchinf(y_len, y_array, y) ! Test also if the array is increasing
259  if (j < y_len) then
260  y1 = y_array(j)
261  y2 = y_array(j+1)
262  else
263  y1 = y_array(j-1)
264  y2 = y_array(j)
265  endif
266  denom = (x2 - x1)*(y2 - y1)
267  interpolate = (f(i+1,j)*(x-x1)*(y2-y) + f(i+1, j+1)*(x-x1)*(y-y1))/denom
268  else ! case 9 ! The point is exactly on the area defined
269  !print *,"case9"
270  i = searchinf(x_len, x_array, x) ! Test also if the array is increasing
271  j = searchinf(y_len, y_array, y) ! Test also if the array is increasing
272 
273  if (i < x_len) then
274  x1 = x_array(i)
275  x2 = x_array(i+1)
276  else
277  x1 = x_array(i-1)
278  x2 = x_array(i)
279  endif
280  if (j < y_len) then
281  y1 = y_array(j)
282  y2 = y_array(j+1)
283  else
284  y1 = y_array(j-1)
285  y2 = y_array(j)
286  endif
287  denom = (x2 - x1)*(y2 - y1)
288  interpolate = (f(i,j)*(x2-x)*(y2-y) + f(i+1,j)*(x-x1)*(y2-y) + &
289  f(i,j+1)*(x2-x)*(y-y1) + f(i+1, j+1)*(x-x1)*(y-y1))/denom
290  endif
291 
292  end function interpolate
293 
294 end module interpolation

300  subroutine define_external_model_from_tomo_file()
301 
302 ! ----------------------------------------------------------------------------------------
303 ! Read a tomo file and loop over all GLL points to set the values of vp,vs and rho
304 ! ----------------------------------------------------------------------------------------
305 
306  use specfem_par, only: tomo_material,coord,nspec,ibool,kmato,rhoext,vpext,vsext, &
307  qkappa_attenuation,qmu_attenuation,anisotropy, &
308  qkappa_attenuationext,qmu_attenuationext,poroelastcoef,density, &
309  c11ext,c13ext,c15ext,c33ext,c35ext,c55ext,c12ext,c23ext,c25ext,c22ext,gravityext,nsqext
310 
311  use specfem_par, only: myrank,tomography_file
312 
313  use model_tomography_par
314  use interpolation
315 
316  use constants, only: ngllx,ngllz,tinyval,imain
317 
318  implicit none
319 
320  ! local parameters
321 
322  integer :: i,j,ispec,iglob
323  double precision :: xmesh,zmesh
324 
325  ! user output
326  if (myrank == 0) then
327  write(imain,*) ' tomographic model file: ',trim(tomography_file)
328  call flush_imain()
329  endif
330 
331  ! Read external tomo file TOMOGRAPHY_FILE
332  call read_tomo_file()
333 
334  ! remove gravity
335  ! leave these arrays here even if you do not assign them to use them because they need to be cleared
336  gravityext(:,:,:) = 0.d0
337  nsqext(:,:,:) = 0.d0
338 
339  ! default no attenuation
340  qkappa_attenuationext(:,:,:) = 9999.d0
341  qmu_attenuationext(:,:,:) = 9999.d0
342 
343  ! default no anisotropy
344  c11ext(:,:,:) = 0.d0
345  c13ext(:,:,:) = 0.d0
346  c15ext(:,:,:) = 0.d0
347  c33ext(:,:,:) = 0.d0
348  c35ext(:,:,:) = 0.d0
349  c55ext(:,:,:) = 0.d0
350  c12ext(:,:,:) = 0.d0
351  c23ext(:,:,:) = 0.d0
352  c25ext(:,:,:) = 0.d0
353  c22ext(:,:,:) = 0.d0
354 
355  ! loop on all the elements of the mesh, and inside each element loop on all the GLL points
356  do ispec = 1,nspec
357  do j = 1,ngllz
358  do i = 1,ngllx
359  iglob = ibool(i,j,ispec)
360  if (kmato(ispec) == tomo_material) then
361  ! If the material has been set to < 0 on the Par_file
362  xmesh = coord(1,iglob)
363  zmesh = coord(2,iglob)
364 
365  rhoext(i,j,ispec) = interpolate(nx, x_tomo, nz, z_tomo, rho_tomo, xmesh, zmesh,tinyval)
366  vpext(i,j,ispec) = interpolate(nx, x_tomo, nz, z_tomo, vp_tomo, xmesh, zmesh,tinyval)
367  vsext(i,j,ispec) = interpolate(nx, x_tomo, nz, z_tomo, vs_tomo, xmesh, zmesh,tinyval)
368 
369  !! ABAB : The 3 following lines are important, otherwise PMLs won't work. TODO check that
370  !! (we assign these values several times: indeed for each kmato(ispec) it can exist a lot of rhoext(i,j,ispec) )
371  density(1,kmato(ispec)) = rhoext(i,j,ispec)
372  poroelastcoef(3,1,kmato(ispec)) = rhoext(i,j,ispec) * vpext(i,j,ispec) * vpext(i,j,ispec)
373  poroelastcoef(2,1,kmato(ispec)) = rhoext(i,j,ispec) * vsext(i,j,ispec) * vsext(i,j,ispec)
374 
375  !! ABAB : I do the same with anisotropy and attenuation even if I don't use them (for the future) :
376  anisotropy(1,kmato(ispec)) = c11ext(i,j,ispec)
377  anisotropy(2,kmato(ispec)) = c13ext(i,j,ispec)
378  anisotropy(3,kmato(ispec)) = c15ext(i,j,ispec)
379  anisotropy(4,kmato(ispec)) = c33ext(i,j,ispec)
380  anisotropy(5,kmato(ispec)) = c35ext(i,j,ispec)
381  anisotropy(6,kmato(ispec)) = c55ext(i,j,ispec)
382  anisotropy(7,kmato(ispec)) = c12ext(i,j,ispec)
383  anisotropy(8,kmato(ispec)) = c23ext(i,j,ispec)
384  anisotropy(9,kmato(ispec)) = c25ext(i,j,ispec)
385  anisotropy(10,kmato(ispec)) = c22ext(i,j,ispec)
386 
387  qkappa_attenuation(kmato(ispec)) = qkappa_attenuationext(i,j,ispec)
388  qmu_attenuation(kmato(ispec)) = qmu_attenuationext(i,j,ispec)
389  else
390  ! Internal model
391  rhoext(i,j,ispec) = density(1,kmato(ispec))
392  vpext(i,j,ispec) = sqrt(poroelastcoef(3,1,kmato(ispec))/rhoext(i,j,ispec))
393  vsext(i,j,ispec) = sqrt(poroelastcoef(2,1,kmato(ispec))/rhoext(i,j,ispec))
394 
395  qkappa_attenuationext(i,j,ispec) = qkappa_attenuation(kmato(ispec))
396  qmu_attenuationext(i,j,ispec) = qmu_attenuation(kmato(ispec))
397 
398  c11ext(i,j,ispec) = anisotropy(1,kmato(ispec))
399  c13ext(i,j,ispec) = anisotropy(2,kmato(ispec))
400  c15ext(i,j,ispec) = anisotropy(3,kmato(ispec))
401  c33ext(i,j,ispec) = anisotropy(4,kmato(ispec))
402  c35ext(i,j,ispec) = anisotropy(5,kmato(ispec))
403  c55ext(i,j,ispec) = anisotropy(6,kmato(ispec))
404  c12ext(i,j,ispec) = anisotropy(7,kmato(ispec))
405  c23ext(i,j,ispec) = anisotropy(8,kmato(ispec))
406  c25ext(i,j,ispec) = anisotropy(9,kmato(ispec))
407  c22ext(i,j,ispec) = anisotropy(10,kmato(ispec))
408  endif
409  enddo
410  enddo
411  enddo
412 
413  end subroutine define_external_model_from_tomo_file

418 
419  subroutine read_tomo_file()
420 
421 ! ----------------------------------------------------------------------------------------
422 ! This subroutine reads the external ASCII tomo file TOMOGRAPHY_FILE (path to which is
423 ! given in the Par_file).
424 ! This file format is not very clever however it is the one used in specfem3D hence
425 ! we chose to implement it here as well
426 ! The external tomographic model is represented by a grid of points with assigned material
427 ! properties and homogeneous resolution along each spatial direction x and z. The xyz file
428 ! TOMOGRAPHY_FILE that describe the tomography should be located in the TOMOGRAPHY_PATH
429 ! directory, set in the Par_file. The format of the file, as read from
430 ! define_external_model_from_xyz_file.f90 looks like :
431 !
432 ! ORIGIN_X ORIGIN_Z END_X END_Z
433 ! SPACING_X SPACING_Z
434 ! NX NZ
435 ! VP_MIN VP_MAX VS_MIN VS_MAX RHO_MIN RHO_MAX
436 ! x(1) z(1) vp vs rho
437 ! x(2) z(1) vp vs rho
438 ! ...
439 ! x(NX) z(1) vp vs rho
440 ! x(1) z(2) vp vs rho
441 ! x(2) z(2) vp vs rho
442 ! ...
443 ! x(NX) z(2) vp vs rho
444 ! x(1) z(3) vp vs rho
445 ! ...
446 ! ...
447 ! x(NX) z(NZ) vp vs rho
448 !
449 ! Where :
450 ! _x and z must be increasing
451 ! _ORIGIN_X, END_X are, respectively, the coordinates of the initial and final tomographic
452 ! grid points along the x direction (in meters)
453 ! _ORIGIN_Z, END_Z are, respectively, the coordinates of the initial and final tomographic
454 ! grid points along the z direction (in meters)
455 ! _SPACING_X, SPACING_Z are the spacing between the tomographic grid points along the x
456 ! and z directions, respectively (in meters)
457 ! _NX, NZ are the number of grid points along the spatial directions x and z,
458 ! respectively; NX is given by [(END_X - ORIGIN_X)/SPACING_X]+1; NZ is the same as NX, but
459 ! for z direction.
460 ! _VP_MIN, VP_MAX, VS_MIN, VS_MAX, RHO_MIN, RHO_MAX are the minimum and maximum values of
461 ! the wave speed vp and vs (in m.s-1) and of the density rho (in kg.m-3); these values
462 ! could be the actual limits of the tomographic parameters in the grid or the minimum
463 ! and maximum values to which we force the cut of velocity and density in the model.
464 ! _After these first four lines, in the file file_name the tomographic grid points are
465 ! listed with the corresponding values of vp, vs and rho, scanning the grid along the x
466 ! coordinate (from ORIGIN_X to END_X with step of SPACING_X) for each given z (from ORIGIN_Z
467 ! to END_Z, with step of SPACING_Z).
468 ! ----------------------------------------------------------------------------------------
469 
470  use specfem_par, only: myrank,tomography_file
471 
472  use model_tomography_par
473  use constants, only: iin,imain
474 
475  implicit none
476 
477  ! local parameters
478  integer :: ier,irecord,i,j
479  character(len=150) :: string_read
480 
481  double precision, dimension(:), allocatable :: x_tomography,z_tomography,vp_tomography,vs_tomography,rho_tomography
482 
483  ! opens file for reading
484  open(unit=iin,file=trim(tomography_file),status='old',action='read',iostat=ier)
485  if (ier /= 0) then
486  print *,'Error: could not open tomography file: ',trim(tomography_file)
487  print *,'Please check your settings in Par_file ...'
488  call exit_mpi(myrank,'Error reading tomography file')
489  endif
490 
491  ! --------------------------------------------------------------------------------------
492  ! header infos
493  ! --------------------------------------------------------------------------------------
494  ! reads in model dimensions
495  ! format: #origin_x #origin_z #end_x #end_z
496  call tomo_read_next_line(iin,string_read)
497  read(string_read,*) origin_x, origin_z, end_x, end_z
498 
499  ! --------------------------------------------------------------------------------------
500  ! model increments
501  ! format: #dx #dz
502  ! --------------------------------------------------------------------------------------
503  call tomo_read_next_line(iin,string_read)
504  read(string_read,*) spacing_x, spacing_z
505 
506  ! --------------------------------------------------------------------------------------
507  ! reads in models entries
508  ! format: #nx #nz
509  ! --------------------------------------------------------------------------------------
510  call tomo_read_next_line(iin,string_read)
511  read(string_read,*) nx,nz
512 
513  ! --------------------------------------------------------------------------------------
514  ! reads in models min/max statistics
515  ! format: #vp_min #vp_max #vs_min #vs_max #density_min #density_max
516  ! --------------------------------------------------------------------------------------
517  call tomo_read_next_line(iin,string_read)
518  read(string_read,*) vp_min,vp_max,vs_min,vs_max,rho_min,rho_max
519 
520  ! Determines total maximum number of element records
521  nrecord = int(nx*nz)
522 
523  ! allocate models parameter records
524  allocate(x_tomography(nrecord),z_tomography(nrecord),vp_tomography(nrecord),vs_tomography(nrecord), &
525  rho_tomography(nrecord),stat=ier)
526  allocate(x_tomo(nx),z_tomo(nz),vp_tomo(nx,nz),vs_tomo(nx,nz),rho_tomo(nx,nz),stat=ier)
527  if (ier /= 0) call exit_mpi(myrank,'not enough memory to allocate tomo arrays')
528 
529  ! Checks the number of records for points definition while storing them
530  irecord = 0
531  do while (irecord < nrecord)
532  call tomo_read_next_line(iin,string_read)
533  read(string_read,*) x_tomography(irecord+1),z_tomography(irecord+1), &
534  vp_tomography(irecord+1),vs_tomography(irecord+1),rho_tomography(irecord+1)
535 
536  if (irecord < nx) x_tomo(irecord+1) = x_tomography(irecord+1)
537 
538  irecord = irecord + 1
539  enddo
540 
541  z_tomo = z_tomography(::nx)
542 
543  do i = 1,nx
544  do j = 1,nz
545  vp_tomo(i,j) = vp_tomography(nx*(j-1)+i)
546  vs_tomo(i,j) = vs_tomography(nx*(j-1)+i)
547  rho_tomo(i,j) = rho_tomography(nx*(j-1)+i)
548  enddo
549  enddo
550 
551  if (irecord /= nrecord .and. myrank == 0) then
552  print *, 'Error: ',trim(tomography_file),' has invalid number of records'
553  print *, ' number of grid points specified (= NX*NZ):',nrecord
554  print *, ' number of file lines for grid points :',irecord
555  stop 'Error in tomography data file for the grid points definition'
556  endif
557 
558  if (myrank == 0) then
559  write(imain,*) ' Number of grid points = NX*NZ:',nrecord
560  write(imain,*)
561  call flush_imain()
562  endif
563 
564  ! closes file
565  close(iin)
