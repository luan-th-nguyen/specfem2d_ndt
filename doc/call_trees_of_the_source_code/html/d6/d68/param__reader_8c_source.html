50  code, for which Fortran is not very well suited.
51 
52  But Fortran-C mixes are pretty common these days, so I would not expect
53  any problems on that account. There are no wrapper functions used: just
54  the C routine called directly from a Fortran routine. Also, regarding
55  the use of C, I assumed this would not be a problem since there are
56  already six C files that make up part of the build (though they all are
57  related to the pyre-framework).

127 void
128 FC_FUNC_(param_read,PARAM_READ)(char * string_read, int * string_read_len, char * name, int * name_len, int * ierr)
129 {
130  char * namecopy;
131  char * blank;
132  char * namecopy2;
133  int status;
134  regex_t compiled_pattern;
135  char line[LINE_MAX];
136  int regret;
137  regmatch_t parameter[3];
138  char * keyword;
139  char * value;
140  size_t value_len;
141 
142  // Trim the keyword name we're looking for.
143  namecopy = strndup(name, *name_len);
144  blank = strchr(namecopy, ' ');
145  if (blank != NULL) {
146  namecopy[blank - namecopy] = '\0';
147  }
148  // Then get rid of any dot-terminated prefix.
149  namecopy2 = strchr(namecopy, '.');
150  if (namecopy2 != NULL) {
151  namecopy2 += 1;
152  } else {
153  namecopy2 = namecopy;
154  }
155  /* Regular expression for parsing lines from param file.
156  ** Good luck reading this regular expression. Basically, the lines of
157  ** the parameter file should be of the form 'parameter = value',
158  ** optionally followed by a #-delimited comment.
159  ** 'value' can be any number of space- or tab-separated words. Blank
160  ** lines, lines containing only white space and lines whose first non-
161  ** whitespace character is '#' are ignored. White space is generally
162  ** ignored. As you will see later in the code, if both parameter and
163  ** value are not specified the line is ignored.
164  */
165  char pattern[] = "^[ \t]*([^# \t]+)[ \t]*=[ \t]*([^# \t]+([ \t]+[^# \t]+)*)";
166 
167  // Compile the regular expression.
168  status = regcomp(&compiled_pattern, pattern, REG_EXTENDED);
169  if (status != 0) {
170  printf("regcomp returned error %d\n", status);
171  }
172  // Position the open file to the beginning.
173  if (fseek(fid, 0, SEEK_SET) != 0) {
174  printf("Can't seek to beginning of parameter file\n");
175  *ierr = 1;
176  regfree(&compiled_pattern);
177  return;
178  }
179  // Read every line in the file.
180  while (fgets(line, LINE_MAX, fid) != NULL) {
181  // Get rid of the ending newline.
182  int linelen = strlen(line);
183  if (line[linelen-1] == '\n') {
184  line[linelen-1] = '\0';
185  }
186  /* Test if line matches the regular expression pattern, if so
187  ** return position of keyword and value */
188  regret = regexec(&compiled_pattern, line, 3, parameter, 0);
189  // If no match, check the next line.
190  if (regret == REG_NOMATCH) {
191  continue;
192  }
193  // If any error, bail out with an error message.
194  if (regret != 0) {
195  printf("regexec returned error %d\n", regret);
196  *ierr = 1;
197  regfree(&compiled_pattern);
198  return;
199  }
200  // printf("Line read = %s\n", line);
201  // If we have a match, extract the keyword from the line.
202  keyword = strndup(line+parameter[1].rm_so, parameter[1].rm_eo-parameter[1].rm_so);
203  //printf("keyword: %s\n", keyword);
204 
205  // If the keyword is not the one we're looking for, check the next line.
206  if (strcmp(keyword, namecopy2) != 0) {
207  free(keyword);
208  continue;
209  }
210  free(keyword);
211  regfree(&compiled_pattern);
212  // If it matches, extract the value from the line.
213  value = strndup(line+parameter[2].rm_so, parameter[2].rm_eo-parameter[2].rm_so);
214  //printf("value: %s\n", value);
215 
216  // Clear out the return string with blanks, copy the value into it, and return.
217  memset(string_read, ' ', *string_read_len);
218 
219  value_len = strlen(value);
220  if (value_len > (size_t)*string_read_len)
221  value_len = *string_read_len;
222  strncpy(string_read, value, value_len);
223  // printf("'%s'='%s'\n", namecopy2, value);
224 
225  free(value);
226  free(namecopy);
227  *ierr = 0;
228  return;
229  }
230  // If no keyword matches, print out error and die.
231  printf("No match in parameter file for keyword '%s'\n", namecopy);
232  free(namecopy);
233  regfree(&compiled_pattern);
234  *ierr = 1;
235  return;
236 }
