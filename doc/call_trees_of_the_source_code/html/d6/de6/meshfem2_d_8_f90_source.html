330 ! in case of an acoustic medium, a potential Chi of (density * displacement) is used as in Chaljub and Valette,
331 ! Geophysical Journal International, vol. 158, p. 131-141 (2004) and *NOT* a velocity potential
332 ! as in Komatitsch and Tromp, Geophysical Journal International, vol. 150, p. 303-318 (2002).
333 ! This permits acoustic-elastic coupling based on a non-iterative time scheme.
334 ! Displacement is then: u = grad(Chi) / rho
335 ! Velocity is then: v = grad(Chi_dot) / rho (Chi_dot being the time derivative of Chi)
336 ! and pressure is: p = - Chi_dot_dot (Chi_dot_dot being the time second derivative of Chi).
337 ! The source in an acoustic element is a pressure source.
338 ! First-order acoustic-acoustic discontinuities are also handled automatically
339 ! because pressure is continuous at such an interface, therefore Chi_dot_dot
340 ! is continuous, therefore Chi is also continuous, which is consistent with
341 ! the spectral-element basis functions and with the assembling process.
342 ! This is the reason why a simple displacement potential u = grad(Chi) would
343 ! not work because it would be discontinuous at such an interface and would
344 ! therefore not be consistent with the basis functions.

559  if (ier /= 0) stop 'Error allocating array ispec_of_axial_elements'
560

564  enddo
565

567  ! no elements on the symmetry axis
568  allocate(ispec_of_axial_elements(nelem_on_the_axis),stat=ier)
569  if (ier /= 0) stop 'Error allocating array ispec_of_axial_elements'
570  endif
571 
572  endif ! of if (read_external_mesh) then

576  write(imain,*)

583  endif
584

586  write(imain,*)

589  write(imain,*) ' Min and max value of Z in the grid = ',minval(nodes_coords(2,:)),maxval(nodes_coords(2,:))
590  write(imain,*)
591

593  if (output_grid_gnuplot .and. .not. read_external_mesh) call save_gnuplot_file(ngnod,nx,nz,grid_point_x,grid_point_z)
594 
595  ! partitioning
596  call decompose_mesh()
597 
598  ! setting absorbing boundaries by elements instead of edges
599  if (any_abs) then
600  call merge_abs_boundaries(nbmodels, phi_read, num_material, ngnod)
601  endif
602 
603  ! setting acoustic forcing boundaries by elements instead of edges
604  if (acoustic_forcing) then
605  call merge_acoustic_forcing_boundaries(ngnod)
606  endif
607 
608  ! generate the databases for the solver
609  call save_databases()
610 
611  ! print position of the source
612  do i_source= 1,nsources
613  write(imain,*)
614  write(imain,*) 'Position (x,z) of the source = ',xs(i_source),zs(i_source)
615  write(imain,*)
616  enddo
617 
618  !--- compute position of the receivers and write the STATIONS file
619  if (.not. use_existing_stations) then
620 
621 !! DK DK for now we cannot use both record_at_surface_same_vertical and read_external_mesh
622 !! DK DK because we need to know splines to define the shape of the surface of the model
623  if (any(record_at_surface_same_vertical) .and. read_external_mesh) &
624  stop 'for now we cannot use both record_at_surface_same_vertical and read_external_mesh'
625

627 !! DK DK therefore let us allocate dummy arrays just to be able to call the "save_stations_file" subroutine
628  if (read_external_mesh) then
629  npoints_interface_top = 1
630  max_npoints_interface = 1
631  allocate(xinterface_top(1))
632  allocate(zinterface_top(1))
633  allocate(coefs_interface_top(1))

637  xinterface_top,zinterface_top,coefs_interface_top, &
638  npoints_interface_top,max_npoints_interface)
639  endif
640 
641  ! user output
642  if (nproc == 1) then
643  write(imain,*)
644  write(imain,*) 'This will be a serial simulation'
645  write(imain,*)
646  else
647  write(imain,*)
648  write(imain,*) 'This will be a parallel simulation on ',nproc,' processor cores'
649  write(imain,*)
650  endif
651 
652  ! frees memory
653  if (allocated(nz_layer)) deallocate(nz_layer)
654  if (allocated(elmnts)) deallocate(elmnts)
655 
656  ! mesher works only for single process
657  endif ! myrank == 0
658 
659  ! close main output file
660  if (myrank == 0 .and. imain /= istandard_output) close(imain)
661 
662  ! slave processes wait
663  call synchronize_all()
664 
665  ! MPI finish
666  call finalize_mpi()
