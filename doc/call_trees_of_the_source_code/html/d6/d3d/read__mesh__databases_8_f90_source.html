274  subroutine read_mesh_databases()
275 
276 #ifdef USE_MPI
277  use mpi
278 #endif
279 
280  use constants, only: iin,add_a_small_crack_in_the_medium,nb_points_to_add_to_npgeo
281  use specfem_par
282 
283  implicit none
284 
285  ! local parameters
286  ! for small crack case
287  integer :: npgeo_ori
288 
289  ! add a small crack (discontinuity) in the medium manually
290  npgeo_ori = npgeo
291  if (add_a_small_crack_in_the_medium) npgeo = npgeo + nb_points_to_add_to_npgeo
292 
293  ! continues reading database files
294  ! note: we opened the database file in read_mesh_for_init() and will continue reading from its current position
295  ! thus, the ordering here must be consistent with the order in save_databases.f90
296 
297  ! reads in source infos
298  call read_mesh_databases_sources()
299 
300  ! sets source parameters
301  call set_source_parameters()
302 
303  ! reads the spectral macrobloc nodal coordinates
304  ! and basic properties of the spectral elements
305  call read_mesh_databases_coorg_elem()
306 
307  ! reads attenuation information (needs source timefunction type from read sources before)
308  call read_mesh_databases_attenuation()
309 
310  ! material properties
311  call read_mesh_databases_mato()
312 
313  ! add a small crack (discontinuity) in the medium manually
314  if (add_a_small_crack_in_the_medium) call add_manual_crack(npgeo_ori)
315 
316  ! determines if each spectral element is elastic, poroelastic, or acoustic
317  call get_simulation_domains()
318 
319  ! reads interfaces data
320  call read_mesh_databases_interfaces()
321 
322  ! reads absorbing boundary data
323  call read_mesh_databases_absorbing()
324 
325  ! reads acoustic forcing boundary data
326  call read_mesh_databases_acoustic_forcing()
327 
328  ! reads acoustic free surface data
329  call read_mesh_databases_free_surf()
330 
331  ! reads coupled edges
332  call read_mesh_databases_coupled()
333 
334  ! reads tangential detection curve
335  call read_mesh_databases_tangential()
336 
337  ! reads axial elements data
338  call read_mesh_databases_axial_elements()
339 
340  ! end of reading
341  ! closes input Database file
342  close(iin)
343 
344  end subroutine read_mesh_databases

438  subroutine read_mesh_databases_coorg_elem()
439 
440 ! reads the spectral macrobloc nodal coordinates
441 
442  use constants, only: imain,iin
443  use specfem_par
444  use specfem_par_movie
445 
446  implicit none
447 
448  ! local parameters
449  integer :: ipoin,ip,ier
450  double precision, dimension(NDIM) :: coorgread
451  integer :: nspec_all,nelem_acforcing_all,nelem_acoustic_surface_all
452 
453  ! safety check
454  if (ndim /= 2) stop 'Invalid NDIM value to read coordinates, please check...'
455 
456  ! allocates nodal coordinates
457  allocate(coorg(ndim,npgeo),stat=ier)
458  if (ier /= 0) stop 'Error allocating coorg array'
459 
460  ! initializes
461  coorg(:,:) = 0.d0
462 
463  ! reads the spectral macrobloc nodal coordinates
464 
465  ! reads in values
466  ipoin = 0
467  do ip = 1,npgeo
468  ! reads coordinates
469  read(iin) ipoin,coorgread(1),coorgread(2)
470 
471  ! checks index
472  if (ipoin < 1 .or. ipoin > npgeo) call exit_mpi(myrank,'Wrong control point number')
473 
474  ! saves coordinate array
475  coorg(:,ipoin) = coorgread(:)
476  enddo
477 
478  !---- read the basic properties of the spectral elements
479  read(iin) numat,ngnod,nspec,pointsdisp,plot_lowerleft_corner_only
480 
481  read(iin) nelemabs,nelem_acforcing,nelem_acoustic_surface,num_fluid_solid_edges, &
482  num_fluid_poro_edges,num_solid_poro_edges,nnodes_tangential_curve, &
483  nelem_on_the_axis
484 
485  ! collects numbers
486  call sum_all_i(nspec,nspec_all)
487  call sum_all_i(nelem_acforcing,nelem_acforcing_all)
488  call sum_all_i(nelem_acoustic_surface,nelem_acoustic_surface_all)
489 
490  ! user output
491  if (myrank == 0) then
492  ! print element group main parameters
493  write(imain,107)
494  write(imain,207) nspec_all,ngnod,ngllx,ngllz,ngllx*ngllz,pointsdisp,numat, &
495  nelem_acforcing_all,nelem_acoustic_surface_all
496  call flush_imain()
497  endif
498 
499  ! allocates mesh arrays
500  ! elements
501  allocate(kmato(nspec))
502  allocate(knods(ngnod,nspec))
503 
504  ! material
505  allocate(density(2,numat))
506  allocate(anisotropy(10,numat)) ! don't forget c22 value (it is used for AXISYM simulations only)
507 
508  allocate(porosity(numat), &
509  tortuosity(numat), &
510  permeability(3,numat),stat=ier)
511  if (ier /= 0) stop 'Error allocating porosity,.. arrays'
512 
513  allocate(poroelastcoef(4,3,numat),stat=ier)
514  if (ier /= 0) stop 'Error allocating poroelastcoef arrays'
515 
516  allocate(qkappa_attenuation(numat), &
517  qmu_attenuation(numat),stat=ier)
518  if (ier /= 0) stop 'Error allocating attenuation arrays'
519 
520 
521  ! output formats
522 107 format(/5x,'-- Spectral Elements --',//)
523 
524 207 format(5x,'Number of spectral elements . . . . . . . . . (nspec) =',i7,/5x, &
525  'Number of control nodes per element . . . . . (ngnod) =',i7,/5x, &
526  'Number of points in X-direction . . . . . . . (NGLLX) =',i7,/5x, &
527  'Number of points in Y-direction . . . . . . . (NGLLZ) =',i7,/5x, &
528  'Number of points per element. . . . . . (NGLLX*NGLLZ) =',i7,/5x, &
529  'Number of points for display . . . . . . (pointsdisp) =',i7,/5x, &
530  'Number of element material sets . . . . . . . (numat) =',i7,/5x, &
531  'Number of acoustic forcing elements (nelem_acforcing) =',i7,/5x, &
532  'Number of acoustic free surf (nelem_acoustic_surface) =',i7)
533 
534  end subroutine read_mesh_databases_coorg_elem

563 
564 ! reads spectral macrobloc data
565 
566  use constants, only: iin
567

569

572  ! local parameters

574  integer :: ier
575  integer, dimension(:), allocatable :: knods_read
576 
577  ! reads and sets the material properties
578  call read_materials(f0_source(1))
579 
580  ! add support for using PML in MPI mode with external mesh
581  allocate(region_cpml(nspec),stat=ier)
582  if (ier /= 0) stop 'Error allocating region_CPML array'
583 
584  ! initializes

586  knods(:,:) = 0
587  region_cpml(:) = 0
588 
589  ! temporary read array
590  allocate(knods_read(ngnod),stat=ier)
591  if (ier /= 0) stop 'Error allocating temporary knods array'
592 
593  ! reads spectral macrobloc data
594 
595  ! reads in values
596  n = 0
597  do ispec = 1,nspec
598  ! format: #element_id #material_id #node_id1 #node_id2 #...
599  read(iin) n,kmato_read,(knods_read(k),k = 1,ngnod),pml_read
600

604

607  enddo
608 
609  ! frees temporary array
