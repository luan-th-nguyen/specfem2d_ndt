153  subroutine prepare_timerun_constants()
154 
155  use constants, only: half,zero
156  use specfem_par
157 
158  implicit none
159 
160  ! local parameters
161  integer :: ier
162 
163  ! defines coefficients of the Newmark time scheme
164  deltatover2 = half * deltat
165  deltatsquareover2 = half * deltat * deltat
166 
167  ! define coefficients of the Newmark time scheme for the backward wavefield
168  if (simulation_type == 3) then
169  if (undo_attenuation) then
170  ! moves forward
171  b_deltat = deltat
172  b_deltatover2 = deltatover2
173  b_deltatsquareover2 = deltatsquareover2
174  else
175  ! reconstructed wavefield moves backward in time from last snapshot
176  b_deltat = - deltat
177  b_deltatover2 = half * b_deltat
178  b_deltatsquareover2 = half * b_deltat * b_deltat
179  endif
180  else
181  ! will not be used, but initialized
182  b_deltat = 0._custom_real
183  b_deltatover2 = 0._custom_real
184  b_deltatsquareover2 = 0._custom_real
185  endif
186 
187  ! seismograms
188  ! allocate seismogram arrays
189  allocate(sisux(nstep_between_output_seismos/subsamp_seismos,nrecloc), &
190  sisuz(nstep_between_output_seismos/subsamp_seismos,nrecloc), &
191  siscurl(nstep_between_output_seismos/subsamp_seismos,nrecloc),stat=ier)
192  if (ier /= 0) stop 'Error allocating seismogram arrays'
193 
194  sisux(:,:) = zero ! double precision zero
195  sisuz(:,:) = zero
196  siscurl(:,:) = zero
197 
198  ! synchronizes all processes
199  call synchronize_all()
200 
201  end subroutine prepare_timerun_constants

272  dershape2d_display(ndim,ngnod,pointsdisp,pointsdisp),stat=ier)
273  if (ier /= 0) stop 'Error allocating shape arrays for display'
274 
275  ! computes shape functions and their derivatives for regular interpolated display grid
276  do j = 1,pointsdisp
277  do i = 1,pointsdisp
278  xirec = 2.d0*dble(i-1)/dble(pointsdisp-1) - 1.d0
279  gammarec = 2.d0*dble(j-1)/dble(pointsdisp-1) - 1.d0
280  call define_shape_functions(shape2d_display(:,i,j),dershape2d_display(:,:,i,j),xirec,gammarec,ngnod)
281  enddo
282  enddo
283 
284  ! for postscript snapshots
285  ! arrays for display images as snapshot postscript images
286  allocate(flagrange(ngllx,pointsdisp))
287  if (axisym) then
288  allocate(flagrange_glj(nglj,pointsdisp))
289  else
290  allocate(flagrange_glj(1,1))
291  endif
292 
293  ! compute Lagrange interpolants on a regular interpolated grid in (xi,gamma)
294  ! for display (assumes NGLLX = NGLLZ)
295  do j = 1,ngllx
296  do i = 1,pointsdisp
297  xirec = 2.d0*dble(i-1)/dble(pointsdisp-1) - 1.d0
298  flagrange(j,i) = hgll(j-1,xirec,xigll,ngllx)
299  if (axisym) flagrange_glj(j,i) = hglj(j-1,xirec,xiglj,nglj)
300  enddo
301  enddo
302 
303  allocate(xinterp(pointsdisp,pointsdisp))
304  allocate(zinterp(pointsdisp,pointsdisp))
305  allocate(uxinterp(pointsdisp,pointsdisp))
306  allocate(uzinterp(pointsdisp,pointsdisp))
307 
308  ! to display the whole vector field (it needs to be computed from the potential in acoustic elements,
309  ! thus it does not exist as a whole in case of simulations that contain some acoustic elements
310  ! and it thus needs to be computed specifically for display purposes)
311  allocate(vector_field_display(ndim,nglob))
312 
313  ! when periodic boundary conditions are on, some global degrees of freedom are going to be removed,
314  ! thus we need to set this array to zero otherwise some of its locations may contain random values
315  ! if the memory is not cleaned
316  vector_field_display(:,:) = 0.d0
317 
318  ! check the mesh, stability and number of points per wavelength
319  call check_grid()
320 
321  ! synchronizes all processes
322  call synchronize_all()
323 
324  end subroutine prepare_timerun_postscripts
