33 
34  subroutine compute_curl_one_element(ispec,curl_element)
35 
36 ! compute curl in (poro)elastic elements (for rotational study)
37 
38  use constants, only: custom_real,ngllx,ngllz,zero
39 
40  use specfem_par, only: displ_elastic,displs_poroelastic,ispec_is_elastic,ispec_is_poroelastic, &
41  xix,xiz,gammax,gammaz,ibool,hprime_xx,hprime_zz,myrank
42 
43  implicit none
44 
45  integer,intent(in) :: ispec
46 
47  ! curl in an element
48  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(out) :: curl_element
49 
50  ! local parameters
51  ! jacobian
52  real(kind=CUSTOM_REAL) :: xixl,xizl,gammaxl,gammazl
53 
54  ! spatial derivatives
55  real(kind=CUSTOM_REAL) :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
56  real(kind=CUSTOM_REAL) :: duz_dxl,dux_dzl
57  integer :: i,j,k
58 
59  ! initializes
60  curl_element(:,:) = 0._custom_real
61 
62  if (ispec_is_elastic(ispec)) then
63  ! elastic element
64  do j = 1,ngllz
65  do i = 1,ngllx
66  ! derivative along x and along z
67  dux_dxi = zero
68  duz_dxi = zero
69 
70  dux_dgamma = zero
71  duz_dgamma = zero
72 
73  ! first double loop over GLL points to compute and store gradients
74  ! we can merge the two loops because NGLLX == NGLLZ
75  do k = 1,ngllx
76  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
77  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
78  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
79  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
80  enddo
81 
82  xixl = xix(i,j,ispec)
83  xizl = xiz(i,j,ispec)
84  gammaxl = gammax(i,j,ispec)
85  gammazl = gammaz(i,j,ispec)
86 
87  ! derivatives of displacement
88  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
89  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
90 
91  ! store pressure
92  curl_element(i,j) = - 0.5d0 * (dux_dzl - duz_dxl)
93  enddo
94  enddo
95 
96  else if (ispec_is_poroelastic(ispec)) then
97  ! poro-elastic element
98  do j = 1,ngllz
99  do i = 1,ngllx
100  ! derivative along x and along z
101  dux_dxi = zero
102  duz_dxi = zero
103 
104  dux_dgamma = zero
105  duz_dgamma = zero
106 
107  ! first double loop over GLL points to compute and store gradients
108  ! we can merge the two loops because NGLLX == NGLLZ
109  do k = 1,ngllx
110  dux_dxi = dux_dxi + displs_poroelastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
111  duz_dxi = duz_dxi + displs_poroelastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
112  dux_dgamma = dux_dgamma + displs_poroelastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
113  duz_dgamma = duz_dgamma + displs_poroelastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
114  enddo
115 
116  xixl = xix(i,j,ispec)
117  xizl = xiz(i,j,ispec)
118  gammaxl = gammax(i,j,ispec)
119  gammazl = gammaz(i,j,ispec)
120 
121  ! derivatives of displacement
122  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
123  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
124 
125  ! store pressure
126  curl_element(i,j) = - 0.5d0 * (dux_dzl - duz_dxl)
127  enddo
128  enddo
129 
130  else
131  ! safety stop
132  call exit_mpi(myrank,'no curl in acoustic')
133 
134  endif ! end of test if acoustic or elastic element
135 
136  end subroutine compute_curl_one_element
