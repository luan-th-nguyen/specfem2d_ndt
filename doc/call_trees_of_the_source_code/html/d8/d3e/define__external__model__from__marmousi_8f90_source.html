34  subroutine define_external_model_from_marmousi(coord,ibool,rho,vp,vs,QKappa_attenuation,Qmu_attenuation,gravity,Nsq, &
35  c11,c13,c15,c33,c35,c55,c12,c23,c25,nspec,nglob)
36 
37 ! reads in external model using a marmousi format which defines a compaction gradient
38 
39 
40  use constants, only: custom_real,ngllx,ngllz,ndim,imain
41 
42  use specfem_par, only: poroelastcoef,density,kmato,myrank
43 
44  implicit none
45 
46  integer, intent(in) :: nspec,nglob
47 
48  double precision, dimension(NDIM,nglob), intent(in) :: coord
49 
50  integer, dimension(NGLLX,NGLLZ,nspec), intent(in) :: ibool
51 
52  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec), intent(out) :: rho,vp,vs
53  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec), intent(out) :: QKappa_attenuation,Qmu_attenuation
54  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec), intent(out) :: gravity,Nsq
55  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec), intent(out) :: c11,c15,c13,c33,c35,c55,c12,c23,c25
56 
57  ! local parameters
58  integer :: i,j,ispec,iglob,imat
59 
60  double precision :: x,z,rho0,vp0,vs0,comp_grad
61 
62 ! dummy routine here, just to demonstrate how the model can be assigned
63 ! and how such a routine can be written
64 
65  ! user output
66  if (myrank == 0) then
67  write(imain,*) ' marmousi model: ','using compaction gradient'
68  call flush_imain()
69  endif
70 
71 ! remove gravity
72 ! leave these arrays here even if you do not assign them to use them because they need to be cleared
73  gravity(:,:,:) = 0.d0
74  nsq(:,:,:) = 0.d0
75 
76  ! default no attenuation
77  qkappa_attenuation(:,:,:) = 9999.d0
78  qmu_attenuation(:,:,:) = 9999.d0
79 
80 ! loop on all the elements of the mesh, and inside each element loop on all the GLL points
81  do ispec = 1,nspec
82  do j = 1,ngllz
83  do i = 1,ngllx
84 
85  iglob = ibool(i,j,ispec)
86  imat = kmato(ispec)
87 
88  ! initial model values
89  rho0 = density(1,imat)
90  vp0 = sqrt(poroelastcoef(3,1,imat)/rho0)
91  vs0 = sqrt(poroelastcoef(2,1,imat)/rho0)
92  comp_grad = poroelastcoef(4,1,imat)
93 
94  x = coord(1,iglob)
95  z = coord(2,iglob)
96 
97  ! updates model values using depth and compaction gradient information
98  rho(i,j,ispec) = rho0
99  vp(i,j,ispec) = vp0 + comp_grad*z
100 
101  ! assumes Poisson solids
102  vs(i,j,ispec) = vp(i,j,ispec) / sqrt(3.d0)
103 
104  qkappa_attenuation(i,j,ispec) = 9999. ! this means no attenuation
105  qmu_attenuation(i,j,ispec) = 9999. ! this means no attenuation
106 
107  c11(i,j,ispec) = 0.d0 ! this means no anisotropy
108  c13(i,j,ispec) = 0.d0
109  c15(i,j,ispec) = 0.d0
110  c33(i,j,ispec) = 0.d0
111  c35(i,j,ispec) = 0.d0
112  c55(i,j,ispec) = 0.d0
113  c12(i,j,ispec) = 0.d0
114  c23(i,j,ispec) = 0.d0
115  c25(i,j,ispec) = 0.d0
116 
117  enddo
118  enddo
119  enddo
120 
121  end subroutine define_external_model_from_marmousi
