35  subroutine determine_abs_surface()
36 
37 ! determines absorbing boundary elements
38 
39  use constants, only: ibottom,iright,itop,ileft
40 
41  use part_unstruct_par, only: nelemabs,abs_surface,elmnts,nxread,nzread
42 
43  use shared_parameters, only: ngnod,absorbbottom,absorbleft,absorbright,absorbtop
44 
45  implicit none
46 
47  ! local parameters
48  integer :: ix,iz
49  integer :: inumelem
50 
51  !
52  !--- definition of absorbing boundaries
53  !
54  nelemabs = 0
55  if (absorbbottom) nelemabs = nelemabs + nxread
56  if (absorbtop) nelemabs = nelemabs + nxread
57  if (absorbleft) nelemabs = nelemabs + nzread
58  if (absorbright) nelemabs = nelemabs + nzread
59 
60  allocate(abs_surface(5,nelemabs))
61 
62  ! generate the list of absorbing elements
63  if (nelemabs > 0) then
64  nelemabs = 0
65  do iz = 1,nzread
66  do ix = 1,nxread
67  inumelem = (iz-1)*nxread + ix
68  if (absorbbottom .and. iz == 1) then
69  nelemabs = nelemabs + 1
70  abs_surface(1,nelemabs) = inumelem-1
71  abs_surface(2,nelemabs) = 2
72  abs_surface(3,nelemabs) = elmnts(0+ngnod*(inumelem-1))
73  abs_surface(4,nelemabs) = elmnts(1+ngnod*(inumelem-1))
74  abs_surface(5,nelemabs) = ibottom
75  !is_abs_surf(inumelem) = .true.
76  endif
77  if (absorbright .and. ix == nxread) then
78  nelemabs = nelemabs + 1
79  abs_surface(1,nelemabs) = inumelem-1
80  abs_surface(2,nelemabs) = 2
81  abs_surface(3,nelemabs) = elmnts(1+ngnod*(inumelem-1))
82  abs_surface(4,nelemabs) = elmnts(2+ngnod*(inumelem-1))
83  abs_surface(5,nelemabs) = iright
84  !is_abs_surf(inumelem) = .true.
85  endif
86  if (absorbtop .and. iz == nzread) then
87  nelemabs = nelemabs + 1
88  abs_surface(1,nelemabs) = inumelem-1
89  abs_surface(2,nelemabs) = 2
90  abs_surface(3,nelemabs) = elmnts(3+ngnod*(inumelem-1))
91  abs_surface(4,nelemabs) = elmnts(2+ngnod*(inumelem-1))
92  abs_surface(5,nelemabs) = itop
93  !is_abs_surf(inumelem) = .true.
94  endif
95  if (absorbleft .and. ix == 1) then
96  nelemabs = nelemabs + 1
97  abs_surface(1,nelemabs) = inumelem-1
98  abs_surface(2,nelemabs) = 2
99  abs_surface(3,nelemabs) = elmnts(0+ngnod*(inumelem-1))
100  abs_surface(4,nelemabs) = elmnts(3+ngnod*(inumelem-1))
101  abs_surface(5,nelemabs) = ileft
102  !is_abs_surf(inumelem) = .true.
103  endif
104  enddo
105  enddo
106  endif
107 
108  end subroutine determine_abs_surface
