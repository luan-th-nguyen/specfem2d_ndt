34  subroutine setup_mesh()
35 
36 ! creates mesh related properties, local to global mesh numbering and node locations
37 
38  use constants, only: imain
39  use specfem_par
40 
41  implicit none
42 
43  if (setup_with_binary_database == 2) return
44 
45  ! user output
46  if (myrank == 0) then
47  write(imain,*)
48  write(imain,*) 'Setting up mesh'
49  call flush_imain()
50  endif
51 
52  ! generate the global numbering
53  call setup_mesh_numbering()
54 
55  ! sets point coordinates
56  call setup_mesh_coordinates()
57 
58  ! sets material properties on node points
59  call setup_mesh_properties()
60 
61  ! for periodic edges
62  call setup_mesh_periodic_edges()
63 
64  ! for acoustic forcing
65  call setup_mesh_acoustic_forcing_edges()
66 
67  ! reads in external models and re-assigns material properties
68  call setup_mesh_external_models()
69 
70  ! checks domain flags

72 
73  ! sets domain flags
74  call setup_mesh_domains()
75 
76  ! sets up domain coupling, i.e. edge detection for domain coupling

78 
79  ! sets up MPI arrays and interfaces

81 
82  ! user output

84  write(imain,*) 'All mesh setup done successfully'
85  call flush_imain()
86  endif
87

97  subroutine setup_mesh_numbering()
98 
99  use constants, only: imain,fast_numbering
100  use specfem_par
101 
102  implicit none
103 
104  ! local parameters
105  integer :: ier
106  ! to count the number of degrees of freedom
107  integer :: nspec_acoustic_total,nspec_total,nglob_total
108  integer :: nb_acoustic_DOFs,nb_elastic_DOFs
109  double precision :: ratio_1DOF,ratio_2DOFs
110 
111  ! "slow and clean" or "quick and dirty" version
112  if (fast_numbering) then
113  call createnum_fast()
114  else
115  call createnum_slow()
116  endif
117 
118  ! gets total numbers for all slices
119  call sum_all_i(nspec_acoustic,nspec_acoustic_total)
120  call sum_all_i(nspec,nspec_total)
121  call sum_all_i(nglob,nglob_total)
122 
123  if (myrank == 0) then
124  write(imain,*) 'Mesh numbering:'
125  write(imain,*) ' Total number of elements: ',nspec_total
126  write(imain,*)
127  write(imain,*) ' Total number of acoustic elements = ',nspec_acoustic_total
128  write(imain,*) ' Total number of elastic/visco/poro elements = ',nspec_total - nspec_acoustic_total
129  write(imain,*)
130 #ifdef USE_MPI
131  write(imain,*) 'Approximate total number of grid points in the mesh'
132  write(imain,*) '(with a few duplicates coming from MPI buffers): ',nglob_total
133 #else
134  write(imain,*) 'Exact total number of grid points in the mesh: ',nglob_total
135 #endif
136 
137  ! percentage of elements with 2 degrees of freedom per point
138  ratio_2dofs = (nspec_total - nspec_acoustic_total) / dble(nspec_total)
139  ratio_1dof = nspec_acoustic_total / dble(nspec_total)
140 
141  nb_acoustic_dofs = nint(nglob_total*ratio_1dof)
142 
143  ! elastic elements have two degrees of freedom per point
144  nb_elastic_dofs = nint(nglob_total*ratio_2dofs*2)
145 
146  if (p_sv) then
147  write(imain,*)
148  write(imain,*) 'Approximate number of acoustic degrees of freedom in the mesh: ',nb_acoustic_dofs
149  write(imain,*) 'Approximate number of elastic degrees of freedom in the mesh: ',nb_elastic_dofs
150  write(imain,*) ' (there are 2 degrees of freedom per point for elastic elements)'
151  write(imain,*)
152  write(imain,*) 'Approximate total number of degrees of freedom in the mesh'
153  write(imain,*) '(sum of the two values above): ',nb_acoustic_dofs + nb_elastic_dofs
154  write(imain,*)
155  write(imain,*) ' (for simplicity viscoelastic or poroelastic elements, if any,'
156  write(imain,*) ' are counted as elastic in the above three estimates;'
157  write(imain,*) ' in reality they have more degrees of freedom)'
158  write(imain,*)
159  endif
160  call flush_imain()
161  endif
162 
163  ! allocate temporary arrays
164  allocate(integer_mask_ibool(nglob),stat=ier)
165  if (ier /= 0 ) stop 'error allocating integer_mask_ibool'
166  allocate(copy_ibool_ori(ngllx,ngllz,nspec),stat=ier)
167  if (ier /= 0 ) stop 'error allocating copy_ibool_ori'
168 
169  ! reduce cache misses by sorting the global numbering in the order in which it is accessed in the time loop.
170  ! this speeds up the calculations significantly on modern processors
171  call get_global()
172 
173  ! synchronizes all processes
174  call synchronize_all()
175 
176  end subroutine setup_mesh_numbering

182  subroutine setup_mesh_coordinates()
183 
184  use constants, only: zero
185  use specfem_par
186 
187  implicit none
188 
189  ! local parameters
190  ! Jacobian matrix and determinant
191  double precision :: xixl,xizl,gammaxl,gammazl,jacobianl
192  double precision :: xi,gamma,x,z
193 
194  integer :: i,j,ispec,iglob,ier
195 
196  ! to help locate elements with a negative Jacobian using OpenDX
197  logical :: found_a_negative_jacobian
198 
199  ! allocate other global arrays
200  allocate(coord(ndim,nglob),stat=ier)
201  if (ier /= 0) stop 'Error allocating coord array'
202 
203  ! sets the coordinates of the points of the global grid
204  found_a_negative_jacobian = .false.
205  do ispec = 1,nspec
206  do j = 1,ngllz
207  do i = 1,ngllx
208  if (axisym) then
209  if (is_on_the_axis(ispec)) then
210  xi = xiglj(i)
211  else
212  xi = xigll(i)
213  endif
214  else
215  xi = xigll(i)
216  endif
217  gamma = zigll(j)
218 
219  call recompute_jacobian(xi,gamma,x,z,xixl,xizl,gammaxl,gammazl, &
220  jacobianl,coorg,knods,ispec,ngnod,nspec,npgeo, &
221  .false.)
222 
223  if (jacobianl <= zero) found_a_negative_jacobian = .true.
224 
225  ! coordinates of global nodes
226  iglob = ibool(i,j,ispec)
227  coord(1,iglob) = x
228  coord(2,iglob) = z
229 
230  xix(i,j,ispec) = real(xixl,kind=custom_real)
231  xiz(i,j,ispec) = real(xizl,kind=custom_real)
232  gammax(i,j,ispec) = real(gammaxl,kind=custom_real)
233  gammaz(i,j,ispec) = real(gammazl,kind=custom_real)
234  jacobian(i,j,ispec) = real(jacobianl,kind=custom_real)
235 
236  enddo
237  enddo
238  enddo
239 
240 ! create an OpenDX file containing all the negative elements displayed in red, if any
241 ! this allows users to locate problems in a mesh based on the OpenDX file created at the second iteration
242 ! do not create OpenDX files if no negative Jacobian has been found, or if we are running in parallel
243 ! (because writing OpenDX routines is much easier in serial)
244  if (found_a_negative_jacobian .and. nproc == 1) then
245  call save_opendx_jacobian(nspec,npgeo,ngnod,knods,coorg,xigll,zigll,axisym,is_on_the_axis,xiglj)
246  endif
247 
248  ! stop the code at the first negative element found, because such a mesh cannot be computed
249  if (found_a_negative_jacobian) then
250  do ispec = 1,nspec
251  do j = 1,ngllz
252  do i = 1,ngllx
253  if (axisym) then
254  if (is_on_the_axis(ispec)) then
255  xi = xiglj(i)
256  else
257  xi = xigll(i)
258  endif
259  else
260  xi = xigll(i)
261  endif
262  gamma = zigll(j)
263 
264  call recompute_jacobian(xi,gamma,x,z,xixl,xizl,gammaxl,gammazl, &
265  jacobianl,coorg,knods,ispec,ngnod,nspec,npgeo, &
266  .true.)
267  enddo
268  enddo
269  enddo
270  endif
271 
272  ! synchronizes all processes
273  call synchronize_all()
274 
275  end subroutine setup_mesh_coordinates

356  subroutine setup_mesh_periodic_edges()
357 
358  use constants, only: imain,ngllx,ngllz,hugeval
359  use specfem_par
360 
361  implicit none
362 
363  ! local parameters
364  integer :: ispec,i,j,iglob,iglob2,ier
365  double precision :: xmaxval,xminval,ymaxval,yminval,xtol,xtypdist
366  integer :: counter
367 
368 ! allocate an array to make sure that an acoustic free surface is not enforced on periodic edges
369  allocate(this_ibool_is_a_periodic_edge(nglob),stat=ier)
370  if (ier /= 0) stop 'Error allocating periodic edge array'
371 
372  this_ibool_is_a_periodic_edge(:) = .false.
373 
374 ! periodic conditions: detect common points between left and right edges and replace one of them with the other
375  if (add_periodic_conditions) then
376  ! user output
377  if (myrank == 0) then
378  write(imain,*)
379  write(imain,*) 'implementing periodic boundary conditions'
380  write(imain,*) 'in the horizontal direction with a periodicity distance of ',periodic_horiz_dist,' m'
381  if (periodic_horiz_dist <= 0.d0) stop 'PERIODIC_HORIZ_DIST should be greater than zero when using ADD_PERIODIC_CONDITIONS'
382  write(imain,*)
383  write(imain,*) '*****************************************************************'
384  write(imain,*) '*****************************************************************'
385  write(imain,*) '**** BEWARE: because of periodic conditions, values computed ****'
386  write(imain,*) '**** by check_grid() below will not be reliable ****'
387  write(imain,*) '*****************************************************************'
388  write(imain,*) '*****************************************************************'
389  write(imain,*)
390  endif
391 
392  ! set up a local geometric tolerance
393  xtypdist = +hugeval
394 
395  do ispec = 1,nspec
396 
397  xminval = +hugeval
398  yminval = +hugeval
399  xmaxval = -hugeval
400  ymaxval = -hugeval
401 
402  ! only loop on the four corners of each element to get a typical size
403  do j = 1,ngllz,ngllz-1
404  do i = 1,ngllx,ngllx-1
405  iglob = ibool(i,j,ispec)
406  xmaxval = max(coord(1,iglob),xmaxval)
407  xminval = min(coord(1,iglob),xminval)
408  ymaxval = max(coord(2,iglob),ymaxval)
409  yminval = min(coord(2,iglob),yminval)
410  enddo
411  enddo
412 
413  ! compute the minimum typical "size" of an element in the mesh
414  xtypdist = min(xtypdist,xmaxval-xminval)
415  xtypdist = min(xtypdist,ymaxval-yminval)
416 
417  enddo
418 
419  ! define a tolerance, small with respect to the minimum size
420  xtol = 1.d-4 * xtypdist
421 
422 ! detect the points that are on the same horizontal line (i.e. at the same height Z)
423 ! and that have a value of the horizontal coordinate X that differs by exactly the periodicity length;
424 ! if so, make them all have the same global number, which will then implement periodic boundary conditions automatically.
425 ! We select the smallest value of iglob and assign it to all the points that are the same due to periodicity,
426 ! this way the maximum value of the ibool() array will remain as small as possible.
427 !
428 ! *** IMPORTANT: this simple algorithm will be slow for large meshes because it has a cost of NGLOB^2 / 2
429 ! (where NGLOB is the number of points per MPI slice, not of the whole mesh though). This could be
430 ! reduced to O(NGLOB log(NGLOB)) by using a quicksort algorithm on the coordinates of the points to detect the multiples
431 ! (as implemented in routine createnum_fast() elsewhere in the code). This could be done one day if needed instead
432 ! of the very simple double loop below.
433  if (myrank == 0) then
434  write(imain,*) 'start detecting points for periodic boundary conditions '// &
435  '(the current algorithm can be slow and could be improved)...'
436  endif
437 
438  counter = 0
439  do iglob = 1,nglob-1
440  do iglob2 = iglob + 1,nglob
441  ! check if the two points have the exact same Z coordinate
442  if (abs(coord(2,iglob2) - coord(2,iglob)) < xtol) then
443  ! if so, check if their X coordinate differs by exactly the periodicity distance
444  if (abs(abs(coord(1,iglob2) - coord(1,iglob)) - periodic_horiz_dist) < xtol) then
445  ! if so, they are the same point, thus replace the highest value of ibool with the lowest
446  ! to make them the same global point and thus implement periodicity automatically
447  counter = counter + 1
448  this_ibool_is_a_periodic_edge(iglob) = .true.
449  this_ibool_is_a_periodic_edge(iglob2) = .true.
450  do ispec = 1,nspec
451  do j = 1,ngllz
452  do i = 1,ngllx
453  if (ibool(i,j,ispec) == iglob2) ibool(i,j,ispec) = iglob
454  enddo
455  enddo
456  enddo
457  endif
458  endif
459  enddo
460  enddo
461 
462  if (myrank == 0) write(imain,*) 'done detecting points for periodic boundary conditions.'
463 
464  if (counter > 0) write(imain,*) 'implemented periodic conditions on ',counter,' grid points on proc ',myrank
465 
466  endif ! of if (ADD_PERIODIC_CONDITIONS)
467 
468  end subroutine setup_mesh_periodic_edges

474  subroutine setup_mesh_acoustic_forcing_edges()
475 
476 ! acoustic forcing edge detection
477 
478  use constants, only: imain,ibottom,iright,itop,ileft
479  use specfem_par
480 
481  implicit none
482 
483  ! local parameters
484  integer :: ipoin1D
485 
486  ! acoustic forcing edge detection
487  ! the elements forming an edge are already known (computed in meshfem2D),
488  ! the common nodes forming the edge are computed here
489  if (acoustic_forcing) then
490 
491  ! user output
492  if (myrank == 0) then
493  write(imain,*)
494  write(imain,*) 'Acoustic forcing simulation'
495  write(imain,*)
496  write(imain,*) 'Beginning of acoustic forcing edge detection'
497  call flush_imain()
498  endif
499 
500  ! define i and j points for each edge
501  do ipoin1d = 1,ngllx
502 
503  ivalue(ipoin1d,ibottom) = ngllx - ipoin1d + 1
504  ivalue_inverse(ipoin1d,ibottom) = ipoin1d
505  jvalue(ipoin1d,ibottom) = ngllz
506  jvalue_inverse(ipoin1d,ibottom) = ngllz
507 
508  ivalue(ipoin1d,iright) = 1
509  ivalue_inverse(ipoin1d,iright) = 1
510  jvalue(ipoin1d,iright) = ngllz - ipoin1d + 1
511  jvalue_inverse(ipoin1d,iright) = ipoin1d
512 
513  ivalue(ipoin1d,itop) = ipoin1d
514  ivalue_inverse(ipoin1d,itop) = ngllx - ipoin1d + 1
515  jvalue(ipoin1d,itop) = 1
516  jvalue_inverse(ipoin1d,itop) = 1
517 
518  ivalue(ipoin1d,ileft) = ngllx
519  ivalue_inverse(ipoin1d,ileft) = ngllx
520  jvalue(ipoin1d,ileft) = ipoin1d
521  jvalue_inverse(ipoin1d,ileft) = ngllz - ipoin1d + 1
522 
523  enddo
524 
525  endif ! if (ACOUSTIC_FORCING)
526 
527  ! synchronizes all processes
528  call synchronize_all()
529 
530  end subroutine setup_mesh_acoustic_forcing_edges

609  subroutine setup_mesh_basic_check()
610 
611 ! basic checks on mesh parameters
612 
613  use specfem_par
614 
615  implicit none
616 
617  ! local parameters
618  integer :: ispec
619 
620  ! performs basic checks on parameters read
621  ! mutually exclusive domain flags (element can only belong to a single domain)
622  do ispec = 1,nspec
623  ! exclusive domain flags
624  if (ispec_is_acoustic(ispec) .and. ispec_is_elastic(ispec)) &
625  stop 'Error invalid domain element found! element is acoustic and elastic, please check...'
626 
627  if (ispec_is_acoustic(ispec) .and. ispec_is_poroelastic(ispec)) &
628  stop 'Error invalid domain element found! element is acoustic and poroelastic, please check...'
629 
630  if (ispec_is_acoustic(ispec) .and. ispec_is_gravitoacoustic(ispec)) &
631  stop 'Error invalid domain element found! element is acoustic and gravitoacoustic, please check...'
632 
633  if (ispec_is_elastic(ispec) .and. ispec_is_poroelastic(ispec)) &
634  stop 'Error invalid domain element found! element is elastic and poroelastic, please check...'
635 
636  if (ispec_is_elastic(ispec) .and. ispec_is_gravitoacoustic(ispec)) &
637  stop 'Error invalid domain element found! element is elastic and gravitoacoustic, please check...'
638 
639  ! un-assigned element
640  if ((.not. ispec_is_acoustic(ispec)) .and. &

665

667

673  call get_simulation_domain_counts()
674

676  call sum_all_i(nspec_acoustic,nspec_acoustic_all)
677  call sum_all_i(nspec_elastic,nspec_elastic_all)
678  call sum_all_i(nspec_poroelastic,nspec_poroelastic_all)
679  call sum_all_i(nspec_gravitoacoustic,nspec_gravitoacoustic_all)

686  write(imain,*) ' total number of poroelastic elements = ',nspec_poroelastic_all
687  write(imain,*) ' total number of gravitoacoustic elements = ',nspec_gravitoacoustic_all
688  endif
689

691

694  if (myrank == 0) then
695  if (nspec_total /= nspec_in_domains) then
696  write(imain,*) 'Error invalid total number of elements from domains ',nspec_in_domains,'instead of',nspec_total
697  call exit_mpi(myrank,'Invalid total number of domain elements')
698  endif
699  endif
700

706  ! (sets global flag for all slices)
707  call any_all_l(any_elastic, elastic_simulation)
708  call any_all_l(any_acoustic, acoustic_simulation)
709  call any_all_l(any_poroelastic, poroelastic_simulation)
710  call any_all_l(any_gravitoacoustic, gravitoacoustic_simulation)

714  if (attenuation_viscoelastic) call exit_mpi(myrank,'currently cannot have attenuation if poroelastic simulation only')
715  endif

718  if (poroelastic_simulation) then
719  if (attenuation_poro_fluid_part .and. time_stepping_scheme /= 1) &
720  stop 'RK and LDDRK time scheme not supported for poroelastic simulations with attenuation'
721  endif

725 ! call exit_MPI(myrank,'Cannot put absorbing boundaries if anisotropic materials along edges')
726 
727  if (attenuation_viscoelastic .and. all_anisotropic) then
728  call exit_mpi(myrank,'Cannot turn attenuation on in anisotropic materials')
729  endif
730 
731  ! synchronizes all processes
732  call synchronize_all()
