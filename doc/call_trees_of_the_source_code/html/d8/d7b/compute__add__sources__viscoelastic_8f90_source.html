58 
59  ! element containing source

61 
62  ! source element is elastic

64

66  stf_used = source_time_function(i_source,it,i_stage)
67

75  iglob = ibool(i,j,ispec)

77  sourcearrays(i_source,1,i,j) * stf_used
78  accel_elastic(2,iglob) = accel_elastic(2,iglob) + &
79  sourcearrays(i_source,2,i,j) * stf_used
80  enddo
81  enddo
82  else
83  ! SH (membrane) calculation
84  do j = 1,ngllz
85  do i = 1,ngllx
86  iglob = ibool(i,j,ispec)
87 
88  accel_elastic(1,iglob) = accel_elastic(1,iglob) + &
89  sourcearrays(i_source,1,i,j) * stf_used
90 
91  ! daniel debug source contribution

93  !write(1234,*) it, dble(sourcearrays(i_source,1,i,j) * source_time_function(i_source,it,i_stage)), &

96 
97  enddo

111  subroutine compute_add_sources_viscoelastic_moving_source(accel_elastic,it,i_stage)
112 
113  use constants, only: custom_real,ngllx,ngllz,ndim,source_is_moving,tinyval,nglj,imain
114 
115  use specfem_par, only: p_sv,ispec_is_elastic,nglob_elastic, &
116  nsources,source_time_function, &
117  islice_selected_source,ispec_selected_source,sourcearrays, &
118  ibool,coord,nspec,nglob,xigll,zigll,z_source,nproc, & !These 3 lines are for moving src
119  xi_source,gamma_source,coorg,knods,ngnod,npgeo,iglob_source,x_source,z_source,deltat,t0,myrank, &
120  time_stepping_scheme,hxis_store,hgammas_store,tshift_src,source_type,ispec_is_acoustic, &
121  hxis,hpxis,hgammas,hpgammas,anglesource,ispec_is_poroelastic,mxx,mxz,mzz,gammax,gammaz,xix,xiz, &
122  axisym,xiglj,is_on_the_axis,initialfield
123  implicit none
124 
125  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_elastic) :: accel_elastic
126  integer :: it, i_stage
127 
128  !local variable
129  integer :: i_source,i,j,iglob,ispec
130  real(kind=CUSTOM_REAL) :: stf_used
131  double precision :: hlagrange
132  double precision :: xminSource,vSource,timeval,t_used
133  ! single source array
134  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ) :: sourcearray
135 
136  ! checks if anything to do
137  if (.not. source_is_moving) return
138 
139  xminsource = -5000.0d0 !m
140  vsource = 2150.0d0 !1425.0d0 !1250.0 !m/s
141 
142  if (time_stepping_scheme == 1) then
143  ! Newmark
144  timeval = (it-1)*deltat
145  else
146  call exit_mpi(myrank,'Not implemented!')
147  endif
148 
149  ! moves and re-locates sources along x-axis
150 
151  do i_source = 1,nsources
152  if (abs(source_time_function(i_source,it,i_stage)) > tinyval) then
153  t_used = (timeval-t0-tshift_src(i_source))
154 
155  x_source(i_source) = xminsource + vsource*t_used !timeval?
156 
157  ! collocated force source
158  if (source_type(i_source) == 1) then
159  call locate_source(ibool,coord,nspec,nglob,xigll,zigll, &
160  x_source(i_source),z_source(i_source), &
161  ispec_selected_source(i_source),islice_selected_source(i_source), &
162  nproc,myrank,xi_source(i_source),gamma_source(i_source),coorg,knods,ngnod,npgeo, &
163  iglob_source(i_source),.true.)
164 
165  else if (source_type(i_source) == 2) then
166  ! moment-tensor source
167  call locate_source(ibool,coord,nspec,nglob,xigll,zigll, &
168  x_source(i_source),z_source(i_source), &
169  ispec_selected_source(i_source),islice_selected_source(i_source), &
170  nproc,myrank,xi_source(i_source),gamma_source(i_source),coorg,knods,ngnod,npgeo, &
171  iglob_source(i_source),.false.)
172 
173  else if (.not. initialfield) then
174 
175  call exit_mpi(myrank,'incorrect source type')
176 
177  endif
178 
179  ispec = ispec_selected_source(i_source)
180 
181  ! Lagrange interpolators
182  if (axisym) then
183  if (is_on_the_axis(ispec)) then
184  call lagrange_any(xi_source(i_source),nglj,xiglj,hxis,hpxis)
185  !do j = 1,NGLJ ! ABAB same result with that loop, this is good
186  ! hxis(j) = hglj(j-1,xi_source(i_source),xiglj,NGLJ)
187  !enddo
188  else
189  call lagrange_any(xi_source(i_source),ngllx,xigll,hxis,hpxis)
190  endif
191  else
192  call lagrange_any(xi_source(i_source),ngllx,xigll,hxis,hpxis)
193  endif
194  call lagrange_any(gamma_source(i_source),ngllz,zigll,hgammas,hpgammas)
195 
196  if (mod(it,10) == 0) then
197  ! write(IMAIN,*) "myrank:",myrank
198  ! user output
199  if (myrank == islice_selected_source(i_source)) then
200  iglob = ibool(2,2,ispec_selected_source(i_source))
201  !write(IMAIN,*) 'xcoord: ',coord(1,iglob)
202  write(imain,*) 'it??: ',it,'xcoord: ',coord(1,iglob)," iglob",iglob
203  !'source carried by proc',myrank," source x:",x_source(i_source)," ispec:",ispec_selected_source(i_source)
204 
205  !call flush_IMAIN()
206  endif
207 
208  endif
209 
210  ! stores Lagrangians for source
211  hxis_store(i_source,:) = hxis(:)
212  hgammas_store(i_source,:) = hgammas(:)
213 
214  sourcearray(:,:,:) = 0._custom_real
215 
216  ! computes source arrays
217  select case (source_type(i_source))
218  case (1)
219  ! collocated force source
220  do j = 1,ngllz
221  do i = 1,ngllx
222  hlagrange = hxis_store(i_source,i) * hgammas_store(i_source,j)
223 
224  ! source element is acoustic
225  if (ispec_is_acoustic(ispec)) then
226  sourcearray(:,i,j) = real(hlagrange,kind=custom_real)
227  endif
228 
229  ! source element is elastic
230  if (ispec_is_elastic(ispec)) then
231  if (p_sv) then
232  ! P_SV case
233  sourcearray(1,i,j) = real(- sin(anglesource(i_source)) * hlagrange,kind=custom_real)
234  sourcearray(2,i,j) = real(cos(anglesource(i_source)) * hlagrange,kind=custom_real)
235  else
236  ! SH case (membrane)
237  sourcearray(:,i,j) = real(hlagrange,kind=custom_real)
238  endif
239  endif
240 
241  ! source element is poroelastic
242  if (ispec_is_poroelastic(ispec)) then
243  sourcearray(1,i,j) = real(- sin(anglesource(i_source)) * hlagrange,kind=custom_real)
244  sourcearray(2,i,j) = real(cos(anglesource(i_source)) * hlagrange,kind=custom_real)
245  endif
246 
247  enddo
248  enddo
249 
250  case (2)
251  ! moment-tensor source
252  call compute_arrays_source(ispec,xi_source(i_source),gamma_source(i_source),sourcearray, &
253  mxx(i_source),mzz(i_source),mxz(i_source),xix,xiz,gammax,gammaz,xigll,zigll,nspec)
254  ! checks source
255  if (ispec_is_acoustic(ispec)) then
256  call exit_mpi(myrank,'cannot have moment tensor source in acoustic element')
257  endif
258 
259  ! checks wave type
260  if (ispec_is_elastic(ispec)) then
261  if (.not. p_sv ) call exit_mpi(myrank,'cannot have moment tensor source in SH (membrane) waves calculation')
262  endif
263 
264  end select
265 
266  ! stores sourcearray for all sources
267  sourcearrays(i_source,:,:,:) = sourcearray(:,:,:)
268 
269  endif
270  enddo
271 
272  ! --- add the source
273  do i_source = 1,nsources
274 
275  ! if this processor core carries the source
276  if (myrank == islice_selected_source(i_source)) then
277 
278  ! element containing source
279  ispec = ispec_selected_source(i_source)
280 
281  ! source element is elastic
282  if (ispec_is_elastic(ispec)) then
283 
284  ! source time function
285  stf_used = source_time_function(i_source,it,i_stage)
286 
287  ! adds source term
288  ! note: we use sourcearrays for both collocated forces and moment tensors
289  ! (see setup in setup_source_interpolation() routine)
290  if (p_sv) then
291  ! P-SV calculation
292  do j = 1,ngllz
293  do i = 1,ngllx
294  iglob = ibool(i,j,ispec)
295  accel_elastic(1,iglob) = accel_elastic(1,iglob) + &
296  sourcearrays(i_source,1,i,j) * stf_used
297  accel_elastic(2,iglob) = accel_elastic(2,iglob) + &
298  sourcearrays(i_source,2,i,j) * stf_used
299  enddo
300  enddo
301  else
302  ! SH (membrane) calculation
303  do j = 1,ngllz
304  do i = 1,ngllx
305  iglob = ibool(i,j,ispec)
306 
307  accel_elastic(1,iglob) = accel_elastic(1,iglob) + &
308  sourcearrays(i_source,1,i,j) * stf_used
309 
310  ! daniel debug source contribution
311  !if (iglob == 37905) &
312  !write(1234,*) it, dble(sourcearrays(i_source,1,i,j) * source_time_function(i_source,it,i_stage)), &
313  ! accel_elastic(1,iglob),source_time_function(i_source,it,i_stage),sourcearrays(i_source,1,i,j)
314 
315 
316  enddo
317  enddo
318  endif
319 
320  endif ! source element is elastic
321  endif ! if this processor core carries the source
322  enddo ! do i_source= 1,NSOURCES
323 
324  end subroutine compute_add_sources_viscoelastic_moving_source

361  accel_elastic(2,iglob) = accel_elastic(2,iglob) + real(xir_store_loc(irec_local,i)*gammar_store_loc(irec_local,j)* & source_adjoint(irec_local,it_tmp,2),kind=custom_real)
362  enddo
363  enddo

370  enddo
371  enddo
372  endif
373  endif ! if element is elastic

375  enddo ! irec_local = 1,nrecloc
