34  subroutine define_derivation_matrices()
35 
36  use constants, only: gaussalpha,gaussbeta,ngllx,ngllz,zero,custom_real
37 
38  use specfem_par, only: xigll,zigll,wxgll,wzgll,hprime_xx,hprime_zz,hprimewgll_xx,hprimewgll_zz
39 
40  implicit none
41 
42  ! temporary arrays
43  ! note: wxgll,wzgll from specfem_par are defined as CUSTOM_REAL
44  ! the subroutine zwgljd() works with double precision
45  double precision, dimension(NGLLX) :: wxgll_dble
46  double precision, dimension(NGLLZ) :: wzgll_dble
47 
48  ! function for calculating derivatives of Lagrange polynomials
49  double precision, external :: lagrange_deriv_GLL
50 
51  integer :: i1,i2,k1,k2
52 
53  ! set up coordinates of the Gauss-Lobatto-Legendre points
54  call zwgljd(xigll,wxgll_dble,ngllx,gaussalpha,gaussbeta)
55  call zwgljd(zigll,wzgll_dble,ngllz,gaussalpha,gaussbeta)
56 
57  ! converts to CUSTOM_REAL
58  wxgll(:) = real(wxgll_dble(:),kind=custom_real)
59  wzgll(:) = real(wzgll_dble(:),kind=custom_real)
60 
61  ! if number of points is odd, the middle abscissa is exactly zero
62  if (mod(ngllx,2) /= 0) xigll((ngllx-1)/2+1) = zero
63  if (mod(ngllz,2) /= 0) zigll((ngllz-1)/2+1) = zero
64 
65 ! calculate derivatives of the Lagrange polynomials
66 ! and precalculate some products in double precision
67 ! hprime(i,j) = h'_j(xigll_i) by definition of the derivation matrix
68  do i1 = 1,ngllx
69  do i2 = 1,ngllx
70  hprime_xx(i2,i1) = real(lagrange_deriv_GLL(i1-1,i2-1,xigll,NGLLX),kind=custom_real)
71  hprimewgll_xx(i2,i1) = wxgll(i2) * hprime_xx(i2,i1)
72  enddo
73  enddo
74 
75  do k1 = 1,ngllz
76  do k2 = 1,ngllz
77  hprime_zz(k2,k1) = real(lagrange_deriv_GLL(k1-1,k2-1,zigll,NGLLZ),kind=custom_real)
78  hprimewgll_zz(k2,k1) = wzgll(k2) * hprime_zz(k2,k1)
79  enddo
80  enddo
81 
82  end subroutine define_derivation_matrices

92  subroutine define_glj_derivation_matrix()
93 
94  use constants, only: nglj,custom_real
95 
96  use specfem_par, only: xiglj,wxglj,hprimebar_xx,hprimebarwglj_xx
97 
98  implicit none
99 
100  double precision, parameter :: alphaGLJ = 0.d0,betaglj = 1.d0
101 
102  ! note: wxglj from specfem_par are defined as CUSTOM_REAL
103  ! the subroutine zwgljd() works with double precision
104  double precision, dimension(NGLJ) :: wxglj_dble
105 
106  ! function for calculating derivatives of GLJ polynomials
107  double precision, external :: poly_deriv_GLJ
108 
109  integer i1,i2
110 
111  ! set up coordinates of the Gauss-Lobatto-Jacobi points
112  call zwgljd(xiglj,wxglj_dble,nglj,alphaglj,betaglj)
113 
114  ! converts to CUSTOM_REAL
115  wxglj(:) = real(wxglj_dble(:),kind=custom_real)
116 
117 ! calculate derivatives of the GLJ quadrature polynomials
118 ! and precalculate some products in double precision
119 ! hprimeBar(i,j) = hBar'_j(xiglj_i) by definition of the derivation matrix
120  do i1 = 1,nglj
121  do i2 = 1,nglj
122  hprimebar_xx(i2,i1) = real(poly_deriv_GLJ(i1-1,i2-1,xiglj,NGLJ),kind=custom_real)
123  hprimebarwglj_xx(i2,i1) = wxglj(i2) * hprimebar_xx(i2,i1)
124  enddo
125  enddo
126 
127  end subroutine define_glj_derivation_matrix
