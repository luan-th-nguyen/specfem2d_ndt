34  subroutine save_gnuplot_file(ngnod,nx,nz,x,z)
35 
36 ! creates a Gnuplot file that displays the grid
37 
38  use constants, only: imain,output_files
39 
40  implicit none
41 
42  integer :: ngnod,nx,nz
43  double precision, dimension(0:nx,0:nz) :: x,z
44 
45  ! local parameters
46  integer :: ier,istepx,istepz,ili,icol
47 
48  ! user output
49  write(imain,*)
50  write(imain,*) 'Saving the grid in Gnuplot format...'
51  write(imain,*)
52 
53  open(unit=20,file=trim(output_files)//'gridfile.gnu',status='unknown',iostat=ier)
54  if (ier /= 0 ) then
55  print *,'Error opening gnuplot file for writing: ',trim(output_files)//'gridfile.gnu'
56  print *,'Please make sure directory ',trim(output_files),' exists...'
57  stop 'Error saving gnuplot file'
58  endif
59 
60  ! draw horizontal lines of the grid
61  write(imain,*) 'drawing horizontal lines of the grid'
62  istepx = 1
63  if (ngnod == 4) then
64  istepz = 1
65  else
66  istepz = 2
67  endif
68  do ili=0,nz,istepz
69  do icol=0,nx-istepx,istepx
70  write(20,*) sngl(x(icol,ili)),sngl(z(icol,ili))
71  write(20,*) sngl(x(icol+istepx,ili)),sngl(z(icol+istepx,ili))
72  write(20,10)
73  enddo
74  enddo
75 
76  ! draw vertical lines of the grid
77  write(imain,*) 'drawing vertical lines of the grid'
78  if (ngnod == 4) then
79  istepx = 1
80  else
81  istepx = 2
82  endif
83  istepz = 1
84  do icol=0,nx,istepx
85  do ili=0,nz-istepz,istepz
86  write(20,*) sngl(x(icol,ili)),sngl(z(icol,ili))
87  write(20,*) sngl(x(icol,ili+istepz)),sngl(z(icol,ili+istepz))
88  write(20,10)
89  enddo
90  enddo
91 
92 10 format('')
93 
94  close(20)
95 
96  ! create a Gnuplot script to display the grid
97  open(unit=20,file=trim(output_files)//'plot_gridfile.gnu',status='unknown',iostat=ier)
98  if (ier /= 0 ) stop 'Error saving plotgnu file'
99 
100  write(20,*) '#set term wxt'
101  write(20,*) 'set term postscript landscape monochrome solid "Helvetica" 22'
102  write(20,*) 'set output "',trim(output_files)//'gridfile.ps"'
103  write(20,*) '#set xrange [',sngl(minval(x)),':',sngl(maxval(x)),']'
104  write(20,*) '#set yrange [',sngl(minval(z)),':',sngl(maxval(z)),']'
105  ! use same unit length on both X and Y axes
106  write(20,*) 'set size ratio -1'
107  write(20,*) 'set loadpath "'//trim(output_files)//'"'
108  write(20,*) 'plot "gridfile.gnu" title "Macrobloc mesh" w l'
109  write(20,*) 'pause -1 "Hit any key..."'
110  close(20)
111 
112  write(imain,*) 'Grid saved in Gnuplot format...'
113  write(imain,*)
114 
115  end subroutine save_gnuplot_file
