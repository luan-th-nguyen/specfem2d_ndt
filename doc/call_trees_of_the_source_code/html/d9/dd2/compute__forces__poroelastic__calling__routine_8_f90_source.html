34  subroutine compute_forces_poroelastic_main()
35 
36  use constants, only: use_poro_viscous_damping,alpha_lddrk,beta_lddrk
37  use specfem_par
38 
39  implicit none
40 
41  ! non-blocking MPI
42  ! iphase: iphase = 1 is for computing outer elements (on MPI interface),
43  ! iphase = 2 is for computing inner elements
44  integer :: iphase
45 
46  ! checks if anything to do in this slice
47  if (.not. any_poroelastic) return
48 
49  ! implement viscous attenuation for poroelastic media
50  if (attenuation_poro_fluid_part) call compute_attenuation_poro_fluid_part()
51 
52  ! distinguishes two runs: for elements on MPI interfaces, and elements within the partitions
53  do iphase = 1,2
54 
55  ! main solver for the poroelastic elements: first the solid (u_s) then the fluid (w)
56  call compute_forces_poro_solid(displs_poroelastic,displw_poroelastic,accels_poroelastic,epsilondev_s,iphase)
57  call compute_forces_poro_fluid(displs_poroelastic,displw_poroelastic,accelw_poroelastic,epsilondev_w,iphase)
58 
59  ! compute additional contributions
60  if (iphase == 1) then
61  ! viscous damping
62  if (use_poro_viscous_damping) call compute_forces_poro_viscous_damping()
63 
64  ! Stacey absorbing boundary
65  if (anyabs) then
66  call compute_stacey_poro_fluid(f0_source(1))
67  call compute_stacey_poro_solid(f0_source(1))
68  endif
69 
70  ! add coupling with the acoustic side
71  if (coupled_acoustic_poro) then
72  call compute_coupling_poro_ac(potential_dot_dot_acoustic,accels_poroelastic,accelw_poroelastic,1)
73  endif
74 
75  ! add coupling with the elastic side
76  if (coupled_elastic_poro) then
77  call compute_coupling_poro_viscoelastic(displ_elastic,displs_poroelastic,displw_poroelastic,accels_poroelastic)
78  endif
79 
80  ! add source
81  if (.not. initialfield) then
82  if (simulation_type == 1) then
83  ! forward wavefield
84  call compute_add_sources_poro(accels_poroelastic,accelw_poroelastic,it,i_stage)
85  else if (simulation_type == 3) then
86  ! adjoint wavefield
87  call compute_add_sources_poro_adjoint()
88  endif
89  endif
90  endif
91 
92 #ifdef USE_MPI
93  ! assembling accels_proelastic & accelw_poroelastic for poroelastic elements
94  if (nproc > 1 .and. ninterface_poroelastic > 0) then
95  if (iphase == 1) then
96  call assemble_mpi_vector_po_s(accels_poroelastic,accelw_poroelastic)
97  else
98  call assemble_mpi_vector_po_w(accels_poroelastic,accelw_poroelastic)
99  endif
100  endif
101 #endif
102 
103  enddo ! iphase
104 
105  ! multiply by the inverse of the mass matrix
106  ! solid
107  accels_poroelastic(1,:) = accels_poroelastic(1,:) * rmass_s_inverse_poroelastic(:)
108  accels_poroelastic(2,:) = accels_poroelastic(2,:) * rmass_s_inverse_poroelastic(:)
109  ! fluid
110  accelw_poroelastic(1,:) = accelw_poroelastic(1,:) * rmass_w_inverse_poroelastic(:)
111  accelw_poroelastic(2,:) = accelw_poroelastic(2,:) * rmass_w_inverse_poroelastic(:)
112 
113  ! update velocity
114  select case (time_stepping_scheme)
115  case (1)
116  ! Newmark
117  call update_veloc_poroelastic_newmark()
118  case (2)
119  ! LDDRK
120  call update_veloc_poroelastic_lddrk()
121  case (3)
122  ! Runge-Kutta
123  call update_veloc_poroelastic_rk()
124  case default
125  stop 'Time stepping scheme not implemented yet for poroelastic case'
126  end select
127 
128  ! imposes continuity for stabilization
129  if (coupled_elastic_poro) then
130  call compute_coupling_poro_viscoelastic_for_stabilization(veloc_elastic,accel_elastic, &
131  velocs_poroelastic,accels_poroelastic, &
132  velocw_poroelastic,accelw_poroelastic,deltatover2)
133  endif
134 
135  end subroutine compute_forces_poroelastic_main

142  subroutine compute_forces_poroelastic_main_backward()
143 
144  use constants, only: use_poro_viscous_damping,alpha_lddrk,beta_lddrk
145  use specfem_par
146 
147  implicit none
148 
149  ! non-blocking MPI
150  ! iphase: iphase = 1 is for computing outer elements (on MPI interface),
151  ! iphase = 2 is for computing inner elements
152  integer :: iphase
153 
154  ! safety check
155  if (simulation_type /= 3) return
156 
157  ! checks if anything to do in this slice
158  if (.not. any_poroelastic) return
159 
160  ! implement viscous attenuation for poroelastic media
161  if (attenuation_poro_fluid_part) stop 'ATTENUATION_PORO_FLUID_PART not implemented yet for backward/kernel simulations'
162 
163  ! distinguishes two runs: for elements on MPI interfaces, and elements within the partitions
164  do iphase = 1,2
165 
166  ! main solver for the poroelastic elements: first the solid (u_s) then the fluid (w)
167  call compute_forces_poro_solid(b_displs_poroelastic,b_displw_poroelastic,b_accels_poroelastic,b_epsilondev_s,iphase)
168  call compute_forces_poro_fluid(b_displs_poroelastic,b_displw_poroelastic,b_accelw_poroelastic,b_epsilondev_w,iphase)
169 
170  ! compute additional contributions
171  if (iphase == 1) then
172  ! viscous damping
173  if (use_poro_viscous_damping) call compute_forces_poro_viscous_damping_backward()
174 
175  ! Stacey absorbing boundary
176  if (anyabs) then
177  call compute_stacey_poro_fluid_backward()
178  call compute_stacey_poro_solid_backward()
179  endif
180 
181  ! add coupling with the acoustic side
182  if (coupled_acoustic_poro) then
183  call compute_coupling_poro_ac(b_potential_dot_dot_acoustic,b_accels_poroelastic,b_accelw_poroelastic,3)
184  endif
185 
186  ! add coupling with the elastic side
187  if (coupled_elastic_poro) then
188  call compute_coupling_poro_viscoelastic(b_displ_elastic,b_displs_poroelastic,b_displw_poroelastic,b_accels_poroelastic)
189  endif
190 
191  ! add source
192  if (.not. initialfield) then
193  ! backward wavefield
194  call compute_add_sources_poro(b_accels_poroelastic,b_accelw_poroelastic,nstep-it+1,stage_time_scheme-i_stage+1)
195  endif
196  endif
197 
198 #ifdef USE_MPI
199  ! assembling accels_proelastic & accelw_poroelastic for poroelastic elements
200  if (nproc > 1 .and. ninterface_poroelastic > 0) then
201  if (iphase == 1) then
202  call assemble_mpi_vector_po_s(b_accels_poroelastic,b_accelw_poroelastic)
203  else
204  call assemble_mpi_vector_po_w(b_accels_poroelastic,b_accelw_poroelastic)
205  endif
206  endif
207 #endif
208 
209  enddo ! iphase
210 
211  ! multiply by the inverse of the mass matrix
212  ! solid
213  b_accels_poroelastic(1,:) = b_accels_poroelastic(1,:) * rmass_s_inverse_poroelastic(:)
214  b_accels_poroelastic(2,:) = b_accels_poroelastic(2,:) * rmass_s_inverse_poroelastic(:)
215  ! fluid
216  b_accelw_poroelastic(1,:) = b_accelw_poroelastic(1,:) * rmass_w_inverse_poroelastic(:)
217  b_accelw_poroelastic(2,:) = b_accelw_poroelastic(2,:) * rmass_w_inverse_poroelastic(:)
218 
219  ! update velocity
220  select case (time_stepping_scheme)
221  case (1)
222  ! Newmark
223  call update_veloc_poroelastic_newmark_backward()
224  case (2)
225  ! LDDRK
226  if (simulation_type == 3) stop 'LDDRK scheme for poroelastic kernel simulation not implemented yet'
227  case (3)
228  ! Runge-Kutta
229  if (simulation_type == 3) stop 'RK scheme for poroelastic kernel simulation not implemented yet'
230  case default
231  stop 'Time stepping scheme not implemented yet for poroelastic case'
232  end select
233 
234  ! imposes continuity for stabilization
235  if (coupled_elastic_poro) then
236  call compute_coupling_poro_viscoelastic_for_stabilization(b_veloc_elastic,b_accel_elastic, &
237  b_velocs_poroelastic,b_accels_poroelastic, &
238  b_velocw_poroelastic,b_accelw_poroelastic,b_deltatover2)
239  endif
240 
241  end subroutine compute_forces_poroelastic_main_backward
