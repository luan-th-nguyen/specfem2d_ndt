35  subroutine initialize_simulation()
36 
37 #ifdef USE_MPI
38  use mpi
39 #endif
40  use constants, only: imain,istandard_output,size_real,nstage,output_files
41  use specfem_par
42  use specfem_par_movie, only: cutsnaps
43 
44  implicit none
45 
46  ! local parameters
47  integer :: ier
48 
49 !***********************************************************************
50 !
51 ! i n i t i a l i z a t i o n p h a s e
52 !
53 !***********************************************************************
54 
55  output_files = trim(output_files) ! TODO OUTPUT_FILES!
56 
57  ! number of MPI processes
58  call world_size(nproc)
59 
60  ! myrank is the rank of each process, between 0 and NPROC-1.
61  ! as usual in MPI, process 0 is in charge of coordinating everything
62  ! and also takes care of the main output
63  call world_rank(myrank)
64 
65  ! check process setup
66  if (nproc < 1) stop 'should have NPROC >= 1'
67 
68  ! checks rank to make sure that myrank is zero for serial version
69 #ifdef USE_MPI
70  if (myrank >= nproc) stop 'Error: invalid MPI rank'
71 #else
72  ! serial version: checks rank is initialized
73  if (myrank /= 0) stop 'Error: process should have myrank zero for serial version'
74 #endif
75 
76  ! determine if we write to file instead of standard output
77  if (myrank == 0 .and. imain /= istandard_output) then
78  ! sets main output file name
79  ! opens for simulation output
80  open(imain,file=trim(output_files)//'output_solver.txt',status='unknown',action='write',iostat=ier)
81  if (ier /= 0 ) call exit_mpi(myrank,'Error opening file '//trim(output_files)//'output_solver.txt')
82  endif
83 
84  ! user output
85  if (myrank == 0) then
86  write(imain,*)
87 #ifdef USE_MPI
88  write(imain,*) '**********************************************'
89  write(imain,*) '**** Specfem 2-D Solver - MPI version ****'
90  write(imain,*) '**********************************************'
91  write(imain,*)
92  write(imain,*) 'There are ',nproc,' MPI processes'
93  write(imain,*) 'Processes are numbered from 0 to ',nproc-1
94  write(imain,*)
95  write(imain,*) 'There is a total of ',nproc,' slices'
96 #else
97  write(imain,*) '**********************************************'
98  write(imain,*) '**** Specfem 2-D Solver - serial version ****'
99  write(imain,*) '**********************************************'
100 #endif
101  write(imain,*)
102  write(imain,*) 'NDIM = ',ndim
103  write(imain,*)
104  write(imain,*) 'NGLLX = ',ngllx
105  write(imain,*) 'NGLLZ = ',ngllz
106  write(imain,*)
107  ! write information about precision used for floating-point operations
108  if (custom_real == size_real) then
109  write(imain,*) 'using single precision for the calculations'
110  else
111  write(imain,*) 'using double precision for the calculations'
112  endif
113  write(imain,*)
114  write(imain,*) 'smallest and largest possible floating-point numbers are: ', &
115  tiny(1._custom_real),huge(1._custom_real)
116  call flush_imain()
117  endif
118 
119  ! starts reading in Database file (header info, simulation flags, number of elements
120  call read_mesh_for_init()
121 
122  ! checks flags
123  call initialize_simulation_check()
124 
125  ! initializes GPU cards
126  if (gpu_mode) call initialize_gpu()
127 
128  ! ----------- initialization and determining parameters
129  ! allocates mesh
130  ! local to global indexing
131  allocate(ibool(ngllx,ngllz,nspec),stat=ier)
132  if (ier /= 0) stop 'Error allocating ibool array'
133 
134  ! mesh arrays
135  allocate(xix(ngllx,ngllz,nspec), &
136  xiz(ngllx,ngllz,nspec), &
137  gammax(ngllx,ngllz,nspec), &
138  gammaz(ngllx,ngllz,nspec), &
139  jacobian(ngllx,ngllz,nspec),stat=ier)
140  if (ier /= 0) stop 'Error allocating mesh arrays for databases'
141 
142  ! domain flags
143  allocate(ispec_is_elastic(nspec), &
144  ispec_is_acoustic(nspec), &
145  ispec_is_gravitoacoustic(nspec), &
146  ispec_is_poroelastic(nspec),stat=ier)
147  if (ier /= 0) stop 'Error allocating domain flag arrays'
148 
149  ispec_is_elastic(:) = .false.
150  ispec_is_acoustic(:) = .false.
151  ispec_is_gravitoacoustic(:) = .false.
152  ispec_is_poroelastic(:) = .false.
153 
154  ! element property flags
155  allocate(ispec_is_anisotropic(nspec), &
156  ispec_is_pml(nspec), stat=ier)
157  if (ier /= 0) stop 'Error allocating element property flag arrays'
158 
159  ispec_is_anisotropic(:) = .false.
160  ispec_is_pml(:) = .false.
161 
162  ! initializes domain interfaces
163  ninterface_acoustic = 0
164  ninterface_elastic = 0
165  ninterface_poroelastic = 0
166 
167  ! time scheme
168  ! defines number of stages of chosen time stepping scheme
169  if (time_stepping_scheme == 1) then
170  ! Newmark
171  stage_time_scheme = 1
172  else if (time_stepping_scheme == 2) then
173  ! LDDRK
174  stage_time_scheme = nstage ! 6
175  else if (time_stepping_scheme == 3) then
176  ! Runge-Kutta
177  stage_time_scheme = 4
178  endif
179 
180  ! converts percentage
181  cutsnaps = cutsnaps / 100.d0
182 
183  ! sets model flag
184  if (trim(model) == 'default') then
185  assign_external_model = .false.
186  else
187  assign_external_model = .true.
188  endif
189 
190  ! make sure NSTEP is a multiple of subsamp_seismos
191  ! if not, increase it a little bit, to the next multiple
192  if (mod(nstep,subsamp_seismos) /= 0) then
193  nstep = (nstep/subsamp_seismos + 1)*subsamp_seismos
194 
195  ! user output
196  if (myrank == 0) then
197  write(imain,*)
198  write(imain,*) 'NSTEP is not a multiple of subsamp_seismos'
199  write(imain,*) 'thus increasing it automatically to the next multiple, which is ',nstep
200  write(imain,*)
201  endif
202  endif
203 
204  ! output seismograms at least once at the end of the simulation
205  nstep_between_output_seismos = min(nstep,nstep_between_output_seismos)
206 
207  ! make sure NSTEP_BETWEEN_OUTPUT_SEISMOS is a multiple of subsamp_seismos
208  if (mod(nstep_between_output_seismos,subsamp_seismos) /= 0) then
209  if (myrank == 0) then
210  write(imain,*) 'Invalid number of NSTEP_BETWEEN_OUTPUT_SEISMOS = ',nstep_between_output_seismos
211  write(imain,*) 'Must be a multiple of subsamp_seismos = ',subsamp_seismos
212  endif
213  stop 'Error: NSTEP_BETWEEN_OUTPUT_SEISMOS must be a multiple of subsamp_seismos'
214  endif
215 
216  end subroutine initialize_simulation

223  subroutine initialize_simulation_check()
224 
225  use constants, only: imain,istandard_output,use_enforce_fields
226  use specfem_par
227  use specfem_par_movie
228 
229  implicit none
230 
231  ! synchronizes processes
232  call synchronize_all()
233 
234  ! number of processes
235  if (nproc_read_from_database < 1) stop 'should have nproc_read_from_database >= 1'
236 
237  ! check that the code is running with the requested nb of processes
238  if (nproc /= nproc_read_from_database) then
239  if (myrank == 0) then
240  write(imain,*)
241  write(imain,*) 'Error: number of processors supposed to run on: ',nproc_read_from_database
242  write(imain,*) 'Error: number of MPI processors actually run on: ',nproc
243  write(imain,*)
244  if (imain /= istandard_output) then
245  print *
246  print *, 'Error specfem3D: number of processors supposed to run on: ',nproc_read_from_database
247  print *, 'Error specfem3D: number of MPI processors actually run on: ',nproc
248  print *
249  endif
250  endif
251  call exit_mpi(myrank,'wrong number of MPI processes, must always have NPROC == nproc_read_from_database')
252  endif
253 
254  ! time scheme
255  if (simulation_type == 3 .and. (time_stepping_scheme == 2 .or. time_stepping_scheme == 3)) &
256  stop 'RK and LDDRK time scheme not supported for adjoint inversion'
257 
258  ! standard RK scheme
259  if (time_stepping_scheme == 3) then
260  if (nproc > 1) &
261  stop 'MPI support for standard Runge-Kutta scheme is not implemented yet'
262 
263  if (pml_boundary_conditions) &
264  stop 'PML boundary conditions not implemented with standard Runge Kutta scheme yet'
265  endif
266 
267  ! Bielak parameter setup
268  if (add_bielak_conditions .and. .not. initialfield) &
269  stop 'need to have an initial field to add Bielak plane wave conditions'
270 
271  ! seismogram output
272  if (seismotype < 1 .or. seismotype > 6) &
273  stop 'seismotype should be 1(=displ), 2(=veloc), 3(=accel), 4(=pressure), 5(=curl of displ) or 6(=the fluid potential)'
274 
275  if (su_format .and. .not. save_binary_seismograms_single) &
276  stop 'SU_FORMAT seismograms are single precision and thus require save_binary_seismograms_single set to .true.'
277 
278  if (save_forward .and. (seismotype /= 1 .and. seismotype /= 6)) then
279  ! user warning
280  if (myrank == 0) then
281  write(imain,*)
282  write(imain,*) '***** WARNING *****'
283  write(imain,*) 'SAVE_FORWARD simulation: uses seismotype = ',seismotype
284  write(imain,*) ' Note that seismograms usually must be in displacement/potential for (poro)elastic/acoustic domains'
285  write(imain,*) ' and seismotype should be 1 (elastic/poroelastic adjoint source) or 6 (acoustic adjoint source)'
286  write(imain,*) '*******************'
287  write(imain,*)
288  endif
289  ! safety stop
290  ! stop
291  endif
292 
293  ! image type
294  if (imagetype_postscript < 1 .or. imagetype_postscript > 4) stop 'Wrong type for PostScript snapshots'
295 
296  ! checks attenuation setting
297  if (simulation_type == 1 .and. save_forward .and. (attenuation_poro_fluid_part)) then
298  print *, '*************** Error ***************'
299  stop 'Anisotropy & Viscous damping are not presently implemented for adjoint calculations'
300  endif
301 
302  if (simulation_type == 1 .and. save_forward .and. attenuation_poro_fluid_part .and. (.not. undo_attenuation)) then
303  print *, '*************** Error ***************'
304  stop 'attenuation is only implemented for adjoint calculations with UNDO_ATTENUATION'
305  endif
306 
307  if ((.not. p_sv) .and. attenuation_viscoelastic) then
308  print *, '*************** WARNING ***************'
309  print *, 'Attenuation and anisotropy are not implemented for surface (membrane) waves calculation'
310  print *, '*************** WARNING ***************'
311  stop 'Please set P_SV flag to .true. for simulations with attenuation and anisotropy'
312  endif
313 
314  ! enforcing wavefield at global degrees-of-freedom
315 #ifdef FORCE_VECTORIZATION
316  if (myrank == 0) then
317  write(imain,*)
318  write(imain,*) ' using array vectorization: FORCE_VECTORIZATION'
319  write(imain,*)
320  endif
321  ! safety stop
322  if (use_enforce_fields) &
323  stop 'USE_ENFORCE_FIELDS is not supported yet when using FORCE_VECTORIZATION'
324 #endif
325 
326  if (use_enforce_fields .and. simulation_type /= 1) &
327  stop 'USE_ENFORCE_FIELDS is not supported yet for adjoint/kernel simulations'
328 
329 
330  ! synchronizes processes
331  call synchronize_all()
332 
333  end subroutine initialize_simulation_check

339  subroutine initialize_gpu()
340 
341 ! initialization for GPU cards
342 
343  use constants, only: imain
344  use specfem_par
345 
346  implicit none
347 
348  ! local parameters
349  integer :: ncuda_devices,ncuda_devices_min,ncuda_devices_max
350 
351  ! GPU_MODE now defined in Par_file
352  if (myrank == 0) then
353  write(imain,*)
354  write(imain,*) "GPU_MODE Active."
355  call flush_imain()
356  endif
357 
358  ! check for GPU runs
359  if (ngllx /= 5 .or. ngllz /= 5 ) &
360  stop 'GPU mode can only be used if NGLLX == NGLLZ == 5'
361  if (custom_real /= 4 ) &
362  stop 'GPU mode runs only with CUSTOM_REAL == 4'
363  if (undo_attenuation) &
364  stop 'for undo_attenuation, GPU_MODE is not supported yet'
365  if (pml_boundary_conditions) &
366  stop 'for PML boundaries, GPU_MODE is not supported yet'
367 
368  ! initializes GPU and outputs info to files for all processes
369  call initialize_cuda_device(myrank,ncuda_devices)
370 
371  ! synchronizes all processes
372  call synchronize_all()
373 
374  ! collects min/max of local devices found for statistics
375 #ifdef USE_MPI
376  call min_all_i(ncuda_devices,ncuda_devices_min)
377  call max_all_i(ncuda_devices,ncuda_devices_max)
378 #else
379  ncuda_devices_min = ncuda_devices
380  ncuda_devices_max = ncuda_devices
381 #endif
382 
383  if (myrank == 0) then
384  write(imain,*) "GPU number of devices per node: min =",ncuda_devices_min
385  write(imain,*) " max =",ncuda_devices_max
386  write(imain,*)
387  call flush_imain()
388  endif
389 
390  ! synchronizes processes
391  call synchronize_all()
392 
393  end subroutine initialize_gpu
