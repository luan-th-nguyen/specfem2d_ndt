34 ! for viscoelastic solver
35 
36  subroutine compute_coupling_viscoelastic_ac()
37 
38  use constants, only: custom_real,ngllx,ngllz,nglj,zero,one,two, &
39  iright,ileft,ibottom,itop,alpha_lddrk,beta_lddrk
40 
41  use specfem_par, only: simulation_type,num_fluid_solid_edges, &
42  ibool,wxgll,wzgll,xix,xiz,gammax,gammaz,jacobian,ivalue,jvalue,ivalue_inverse,jvalue_inverse, &
43  potential_acoustic,potential_dot_acoustic,potential_dot_dot_acoustic, &
44  accel_elastic,fluid_solid_acoustic_ispec, &
45  fluid_solid_acoustic_iedge,fluid_solid_elastic_ispec,fluid_solid_elastic_iedge, &
46  potential_acoustic_adj_coupling, &
47  axisym,coord,is_on_the_axis,xiglj,wxglj, &
48  rmemory_sfb_potential_ddot_acoustic,timeval,deltat, &
49  rmemory_sfb_potential_ddot_acoustic_lddrk,i_stage,time_stepping_scheme
50  ! PML arrays
51  use specfem_par, only: pml_boundary_conditions,nspec_pml,ispec_is_pml,spec_to_pml,region_cpml, &
52  k_x_store,k_z_store,d_x_store,d_z_store,alpha_x_store,alpha_z_store,potential_acoustic_old
53 
54  implicit none
55 
56  !local variable
57  integer :: inum,ispec_acoustic,ispec_elastic,iedge_acoustic,iedge_elastic,ipoin1D,i,j,iglob,ii2,jj2, &
58  ispec_PML,CPML_region_local,singularity_type
59  real(kind=CUSTOM_REAL) :: pressure,xxi,zxi,xgamma,zgamma,jacobian1D,nx,nz,weight
60  real(kind=CUSTOM_REAL), dimension(NGLJ,NGLLZ) :: r_xiplus1
61  double precision :: kappa_x,kappa_z,d_x,d_z,alpha_x,alpha_z,beta_x,beta_z, &
62  A0,A1,A2,A3,A4,bb_1,coef0_1,coef1_1,coef2_1,bb_2,coef0_2,coef1_2,coef2_2
63 
64  ! loop on all the coupling edges
65  do inum = 1,num_fluid_solid_edges
66 
67  ! get the edge of the acoustic element
68  ispec_acoustic = fluid_solid_acoustic_ispec(inum)
69  iedge_acoustic = fluid_solid_acoustic_iedge(inum)
70 
71  ! get the corresponding edge of the elastic element
72  ispec_elastic = fluid_solid_elastic_ispec(inum)
73  iedge_elastic = fluid_solid_elastic_iedge(inum)
74 
75  ! implement 1D coupling along the edge
76  do ipoin1d = 1,ngllx
77 
78  ! get point values for the acoustic side, which matches our side in the inverse direction
79  i = ivalue_inverse(ipoin1d,iedge_acoustic)
80  j = jvalue_inverse(ipoin1d,iedge_acoustic)
81  iglob = ibool(i,j,ispec_acoustic)
82 
83  ! compute pressure on the fluid/solid edge
84  pressure = - potential_dot_dot_acoustic(iglob)
85 
86  if (simulation_type == 3) then
87  ! new definition of adjoint displacement and adjoint potential
88  ! adjoint definition: pressure^\dagger = potential^\dagger
89  pressure = potential_acoustic_adj_coupling(iglob)
90  endif
91 
92  ! PML
93  ! (overwrites pressure value if needed)
94  if (pml_boundary_conditions) then
95  if (ispec_is_pml(ispec_acoustic) .and. nspec_pml > 0) then
96  ispec_pml = spec_to_pml(ispec_acoustic)
97 
98  cpml_region_local = region_cpml(ispec_acoustic)
99  kappa_x = k_x_store(i,j,ispec_pml)
100  kappa_z = k_z_store(i,j,ispec_pml)
101  d_x = d_x_store(i,j,ispec_pml)
102  d_z = d_z_store(i,j,ispec_pml)
103  alpha_x = alpha_x_store(i,j,ispec_pml)
104  alpha_z = alpha_z_store(i,j,ispec_pml)
105  beta_x = alpha_x + d_x / kappa_x
106  beta_z = alpha_z + d_z / kappa_z
107 
108  call l_parameter_computation(timeval,deltat,kappa_x,beta_x,alpha_x,kappa_z,beta_z,alpha_z, &
109  cpml_region_local,a0,a1,a2,a3,a4,singularity_type, &
110  bb_1,coef0_1,coef1_1,coef2_1,bb_2,coef0_2,coef1_2,coef2_2)
111 
112  if (time_stepping_scheme == 1) then
113  ! Newmark
114  rmemory_sfb_potential_ddot_acoustic(1,i,j,inum) = &
115  coef0_1 * rmemory_sfb_potential_ddot_acoustic(1,i,j,inum) + &
116  coef1_1 * potential_acoustic(iglob) + coef2_1 * potential_acoustic_old(iglob)
117  else if (time_stepping_scheme == 2) then
118  ! LDDRK
119  rmemory_sfb_potential_ddot_acoustic_lddrk(1,i,j,inum) = &
120  alpha_lddrk(i_stage) * rmemory_sfb_potential_ddot_acoustic_lddrk(1,i,j,inum) + &
121  deltat * (-bb_1 * rmemory_sfb_potential_ddot_acoustic(1,i,j,inum) + potential_acoustic(iglob))
122  rmemory_sfb_potential_ddot_acoustic(1,i,j,inum) = rmemory_sfb_potential_ddot_acoustic(1,i,j,inum) + &
123  beta_lddrk(i_stage) * rmemory_sfb_potential_ddot_acoustic_lddrk(1,i,j,inum)
124  endif
125 
126  pressure = - (a0 * potential_dot_dot_acoustic(iglob) + a1 * potential_dot_acoustic(iglob) + &
127  a2 * potential_acoustic(iglob) + a3 * rmemory_sfb_potential_ddot_acoustic(1,i,j,inum))
128  endif
129  endif
130 
131  ! get point values for the elastic side
132  ii2 = ivalue(ipoin1d,iedge_elastic)
133  jj2 = jvalue(ipoin1d,iedge_elastic)
134  iglob = ibool(ii2,jj2,ispec_elastic)
135 
136  ! compute the 1D Jacobian and the normal to the edge: for their expression see for instance
137  ! O. C. Zienkiewicz and R. L. Taylor, The Finite Element Method for Solid and Structural Mechanics,
138  ! Sixth Edition, electronic version, www.amazon.com, p. 204 and Figure 7.7(a),
139  ! or Y. K. Cheung, S. H. Lo and A. Y. T. Leung, Finite Element Implementation,
140  ! Blackwell Science, page 110, equation (4.60).
141 
142  if (axisym) then
143 ! axial elements are always rotated by the mesher to make sure it is their i == 1 edge that is on the axis
144 ! and thus the case of an edge located along j == constant does not need to be considered here
145  if (is_on_the_axis(ispec_acoustic) .and. i == 1) then
146  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
147  r_xiplus1(i,j) = xxi
148  else if (is_on_the_axis(ispec_acoustic)) then
149  r_xiplus1(i,j) = coord(1,ibool(i,j,ispec_acoustic))/(xiglj(i)+one)
150  endif
151  endif
152 
153  if (iedge_acoustic == itop ) then
154 
155  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
156  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
157  jacobian1d = sqrt(xxi**2 + zxi**2)
158  nx = - zxi / jacobian1d
159  nz = + xxi / jacobian1d
160  if (axisym) then
161  if (is_on_the_axis(ispec_acoustic)) then
162  weight = jacobian1d * wxglj(i) * r_xiplus1(i,j)
163  else
164  weight = jacobian1d * wxgll(i) * coord(1,ibool(i,j,ispec_acoustic))
165  endif
166  else
167  weight = jacobian1d * wxgll(i)
168  endif
169 
170  else if (iedge_acoustic == ibottom ) then
171 
172  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
173  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
174  jacobian1d = sqrt(xxi**2 + zxi**2)
175  nx = + zxi / jacobian1d
176  nz = - xxi / jacobian1d
177  if (axisym) then
178  if (is_on_the_axis(ispec_acoustic)) then
179  weight = jacobian1d * wxglj(i) * r_xiplus1(i,j)
180  else
181  weight = jacobian1d * wxgll(i) * coord(1,ibool(i,j,ispec_acoustic))
182  endif
183  else
184  weight = jacobian1d * wxgll(i)
185  endif
186 
187  else if (iedge_acoustic == ileft) then
188 
189  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
190  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
191  jacobian1d = sqrt(xgamma**2 + zgamma**2)
192  nx = - zgamma / jacobian1d
193  nz = + xgamma / jacobian1d
194  if (axisym) then
195  if (is_on_the_axis(ispec_acoustic)) then
196  stop 'error: rotated element detected on the symmetry axis, this should not happen'
197  else
198  weight = jacobian1d * wzgll(j) * coord(1,ibool(i,j,ispec_acoustic))
199  endif
200  else
201  weight = jacobian1d * wzgll(j)
202  endif
203 
204  else if (iedge_acoustic == iright) then
205 
206  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
207  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
208  jacobian1d = sqrt(xgamma**2 + zgamma**2)
209  nx = + zgamma / jacobian1d
210  nz = - xgamma / jacobian1d
211  if (axisym) then
212  if (is_on_the_axis(ispec_acoustic)) then
213  stop 'error: rotated element detected on the symmetry axis, this should not happen'
214  else
215  weight = jacobian1d * wzgll(j) * coord(1,ibool(i,j,ispec_acoustic))
216  endif
217  else
218  weight = jacobian1d * wzgll(j)
219  endif
220 
221  endif
222 
223 !! DK DK QUENTIN visco begin ici comme tu disais il faudrait coupler la composante tangentielle
224 !! DK DK QUENTIN en plus de la composante normale je suppose, mais c'est peut-etre deja bon
225 !! DK DK QUENTIN i.e. deja le cas avec les deux equations ci-dessous, j'ai pas encore verifie
226 !! DK DK QUENTIN (de maniere sure on ajoute un truc aux deux composantes du vecteur acceleration
227 !! DK DK QUENTIN cot'e solide, donc on ajoute bien aussi a la composante tangentielle, mais je ne
228 !! DK DK QUENTIN sais pas si le fait de ne mettre que "pressure" dans le truc a ajouter est suffisant
229 !! DK DK QUENTIN dans le cas viscoacoustique comme il l'est dans le cas acoustique non dissipatif)
230  accel_elastic(1,iglob) = accel_elastic(1,iglob) + weight*nx*pressure
231  accel_elastic(2,iglob) = accel_elastic(2,iglob) + weight*nz*pressure
232 
233  enddo
234  enddo
235 
236  end subroutine compute_coupling_viscoelastic_ac

241  subroutine compute_coupling_viscoelastic_ac_backward()
242 
243  use constants, only: custom_real,ngllx,ngllz,nglj,zero,one,two,iright,ileft,ibottom,itop
244 
245  use specfem_par, only: num_fluid_solid_edges,ibool,wxgll,wzgll,xix,xiz,gammax,gammaz, &
246  jacobian,ivalue,jvalue,ivalue_inverse,jvalue_inverse, &
247  b_potential_dot_dot_acoustic,b_accel_elastic,fluid_solid_acoustic_ispec, &
248  fluid_solid_acoustic_iedge,fluid_solid_elastic_ispec,fluid_solid_elastic_iedge, &
249  axisym,coord,is_on_the_axis,xiglj,wxglj
250  implicit none
251 
252  !local variable
253  integer :: inum,ispec_acoustic,ispec_elastic,iedge_acoustic,iedge_elastic,ipoin1D,i,j,iglob,ii2,jj2
254  real(kind=CUSTOM_REAL) :: b_pressure,xxi,zxi,xgamma,zgamma,jacobian1D,nx,nz,weight
255  real(kind=CUSTOM_REAL), dimension(NGLJ,NGLLZ) :: r_xiplus1
256 
257  ! loop on all the coupling edges
258  do inum = 1,num_fluid_solid_edges
259 
260  ! get the edge of the acoustic element
261  ispec_acoustic = fluid_solid_acoustic_ispec(inum)
262  iedge_acoustic = fluid_solid_acoustic_iedge(inum)
263 
264  ! get the corresponding edge of the elastic element
265  ispec_elastic = fluid_solid_elastic_ispec(inum)
266  iedge_elastic = fluid_solid_elastic_iedge(inum)
267 
268  ! implement 1D coupling along the edge
269  do ipoin1d = 1,ngllx
270 
271  ! get point values for the acoustic side, which matches our side in the inverse direction
272  i = ivalue_inverse(ipoin1d,iedge_acoustic)
273  j = jvalue_inverse(ipoin1d,iedge_acoustic)
274  iglob = ibool(i,j,ispec_acoustic)
275 
276  b_pressure = - b_potential_dot_dot_acoustic(iglob)
277 
278  ! get point values for the elastic side
279  ii2 = ivalue(ipoin1d,iedge_elastic)
280  jj2 = jvalue(ipoin1d,iedge_elastic)
281  iglob = ibool(ii2,jj2,ispec_elastic)
282 
283  ! compute the 1D Jacobian and the normal to the edge: for their expression see for instance
284  ! O. C. Zienkiewicz and R. L. Taylor, The Finite Element Method for Solid and Structural Mechanics,
285  ! Sixth Edition, electronic version, www.amazon.com, p. 204 and Figure 7.7(a),
286  ! or Y. K. Cheung, S. H. Lo and A. Y. T. Leung, Finite Element Implementation,
287  ! Blackwell Science, page 110, equation (4.60).
288 
289  if (axisym) then
290  if (is_on_the_axis(ispec_acoustic) .and. i == 1) then
291  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
292  r_xiplus1(i,j) = xxi
293  else if (is_on_the_axis(ispec_acoustic)) then
294  r_xiplus1(i,j) = coord(1,ibool(i,j,ispec_acoustic))/(xiglj(i)+one)
295  endif
296  endif
297 
298  if (iedge_acoustic == itop ) then
299  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
300  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
301  jacobian1d = sqrt(xxi**2 + zxi**2)
302  nx = - zxi / jacobian1d
303  nz = + xxi / jacobian1d
304  if (axisym) then
305  if (is_on_the_axis(ispec_acoustic)) then
306  weight = jacobian1d * wxglj(i) * r_xiplus1(i,j)
307  else
308  weight = jacobian1d * wxgll(i) * coord(1,ibool(i,j,ispec_acoustic))
309  endif
310  else
311  weight = jacobian1d * wxgll(i)
312  endif
313  else if (iedge_acoustic == ibottom ) then
314  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
315  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
316  jacobian1d = sqrt(xxi**2 + zxi**2)
317  nx = + zxi / jacobian1d
318  nz = - xxi / jacobian1d
319  if (axisym) then
320  if (is_on_the_axis(ispec_acoustic)) then
321  weight = jacobian1d * wxglj(i) * r_xiplus1(i,j)
322  else
323  weight = jacobian1d * wxgll(i) * coord(1,ibool(i,j,ispec_acoustic))
324  endif
325  else
326  weight = jacobian1d * wxgll(i)
327  endif
328  else if (iedge_acoustic == ileft ) then
329  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
330  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
331  jacobian1d = sqrt(xgamma**2 + zgamma**2)
332  nx = - zgamma / jacobian1d
333  nz = + xgamma / jacobian1d
334  if (axisym) then
335  if (is_on_the_axis(ispec_acoustic)) then
336  stop 'error: rotated element detected on the symmetry axis, this should not happen'
337  else
338  weight = jacobian1d * wzgll(j) * coord(1,ibool(i,j,ispec_acoustic))
339  endif
340  else
341  weight = jacobian1d * wzgll(j)
342  endif
343  else if (iedge_acoustic == iright) then
344  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
345  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
346  jacobian1d = sqrt(xgamma**2 + zgamma**2)
347  nx = + zgamma / jacobian1d
348  nz = - xgamma / jacobian1d
349  if (axisym) then
350  if (is_on_the_axis(ispec_acoustic)) then
351  stop 'error: rotated element detected on the symmetry axis, this should not happen'
352  else
353  weight = jacobian1d * wzgll(j) * coord(1,ibool(i,j,ispec_acoustic))
354  endif
355  else
356  weight = jacobian1d * wzgll(j)
357  endif
358  endif
359 
360  b_accel_elastic(1,iglob) = b_accel_elastic(1,iglob) + weight*nx*b_pressure
361  b_accel_elastic(2,iglob) = b_accel_elastic(2,iglob) + weight*nz*b_pressure
362 
363  enddo
364  enddo
365 
366  end subroutine compute_coupling_viscoelastic_ac_backward
