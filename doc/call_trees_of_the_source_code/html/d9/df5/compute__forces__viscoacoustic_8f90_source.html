34  subroutine compute_forces_acoustic(potential_dot_dot_acoustic,potential_dot_acoustic,potential_acoustic, &
35  PML_BOUNDARY_CONDITIONS,potential_acoustic_old,iphase,e1_fluid)
36 
37 ! compute forces in the acoustic elements in forward simulation and in adjoint simulation in adjoint inversion
38 
39  use constants, only: custom_real,ngllx,ngllz,nglj,cpml_x_only,cpml_z_only,iright,ileft,ibottom,itop, &
40  zero,one,two,two_thirds,iedge1,iedge2,iedge3,iedge4, &
41  alpha_lddrk,beta_lddrk,c_lddrk
42 
43  use specfem_par, only: nglob, &
44  assign_external_model,ibool,kmato,ispec_is_acoustic, &
45  density,rhoext, &
46  xix,xiz,gammax,gammaz,jacobian, &
47  hprime_xx,hprimewgll_xx, &
48  hprime_zz,hprimewgll_zz,wxgll,wzgll, &
49  axisym,is_on_the_axis,coord,hprimebar_xx,hprimebarwglj_xx,xiglj,wxglj,attenuation_viscoacoustic, &
50  nspec_att, n_sls
51 
52  ! overlapping communication
53  use specfem_par, only: nspec_inner_acoustic,nspec_outer_acoustic,phase_ispec_inner_acoustic
54 
55  ! PML arrays
56  use specfem_par, only: ispec_is_pml
57 
58  implicit none
59 
60  real(kind=CUSTOM_REAL), dimension(nglob),intent(inout) :: potential_dot_dot_acoustic
61  real(kind=CUSTOM_REAL), dimension(nglob),intent(in) :: potential_dot_acoustic,potential_acoustic
62 
63  logical,intent(in) :: PML_BOUNDARY_CONDITIONS
64  real(kind=CUSTOM_REAL), dimension(nglob) :: potential_acoustic_old
65 
66  integer,intent(in) :: iphase
67 
68  ! local parameters
69  integer :: ispec,i,j,k,iglob
70 
71  ! spatial derivatives
72  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: dux_dxi,dux_dgamma
73  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: dux_dxl,dux_dzl
74 
75  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: potential_elem
76  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: tempx1,tempx2,tempx3
77  real(kind=CUSTOM_REAL), dimension(NGLJ,NGLLZ) :: r_xiplus1
78 
79  ! Jacobian matrix and determinant
80  real(kind=CUSTOM_REAL), dimension(6,NGLLX,NGLLZ) :: deriv
81  real(kind=CUSTOM_REAL) :: xixl,xizl,gammaxl,gammazl,jacobianl
82 
83  real(kind=CUSTOM_REAL) :: rhol,fac
84  real(kind=CUSTOM_REAL) :: temp1l,temp2l
85 
86  ! local PML parameters
87  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: potential_dot_dot_acoustic_PML
88 
89  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec_ATT,N_SLS) :: e1_fluid
90 
91  integer :: num_elements,ispec_p
92 
93  integer :: i_sls
94 
95  ! choses inner/outer elements
96  if (iphase == 1) then
97  num_elements = nspec_outer_acoustic
98  else
99  num_elements = nspec_inner_acoustic
100  endif
101 
102  ! loop over spectral elements
103  do ispec_p = 1,num_elements
104 
105  ! returns element id from stored element list
106  ispec = phase_ispec_inner_acoustic(ispec_p,iphase)
107 
108  ! only for acoustic spectral elements
109  if (.not. ispec_is_acoustic(ispec)) cycle
110 
111  ! gets local potential for element
112  rhol = density(1,kmato(ispec))
113  do j = 1,ngllz
114  do i = 1,ngllx
115  iglob = ibool(i,j,ispec)
116  potential_elem(i,j) = potential_acoustic(iglob)
117 
118  ! stores local array for element xi/gamma/jacobian (for better performance)
119  deriv(1,i,j) = xix(i,j,ispec)
120  deriv(2,i,j) = xiz(i,j,ispec)
121  deriv(3,i,j) = gammax(i,j,ispec)
122  deriv(4,i,j) = gammaz(i,j,ispec)
123  deriv(5,i,j) = jacobian(i,j,ispec)
124  ! if external density model
125  if (assign_external_model) then
126  rhol = rhoext(i,j,ispec)
127  endif
128  deriv(6,i,j) = jacobian(i,j,ispec) / rhol
129  enddo
130  enddo
131 
132  ! first double loop over GLL points to compute and store gradients
133  call mxm_2comp_singlea(dux_dxi,dux_dgamma,potential_elem,hprime_xx,hprime_zz)
134 
135  ! AXISYM case overwrites dux_dxi
136  if (axisym) then
137  if (is_on_the_axis(ispec)) then
138  do j = 1,ngllz
139  do i = 1,ngllx
140  ! derivative along x and along z
141  dux_dxi(i,j) = 0._custom_real
142  do k = 1,ngllx
143  dux_dxi(i,j) = dux_dxi(i,j) + potential_elem(k,j) * hprimebar_xx(i,k)
144  enddo
145  enddo
146  enddo
147  endif
148  endif
149 
150  ! gets derivatives of ux and uz with respect to x and z
151  do j = 1,ngllz
152  do i = 1,ngllx
153  xixl = deriv(1,i,j)
154  xizl = deriv(2,i,j)
155  gammaxl = deriv(3,i,j)
156  gammazl = deriv(4,i,j)
157 
158  ! derivatives of potential
159  dux_dxl(i,j) = dux_dxi(i,j) * xixl + dux_dgamma(i,j) * gammaxl
160  dux_dzl(i,j) = dux_dxi(i,j) * xizl + dux_dgamma(i,j) * gammazl
161  enddo
162  enddo
163 
164  ! AXISYM case overwrite dux_dxl
165  if (axisym) then
166  if (is_on_the_axis(ispec)) then
167  ! dchi/dr=rho * u_r=0 on the axis
168  ! i == 1
169  do j = 1,ngllz
170  dux_dxl(1,j) = 0._custom_real
171  enddo
172  endif
173  endif
174 
175  ! derivative along x and along zbb
176  if (pml_boundary_conditions) then
177  call pml_compute_memory_variables_acoustic(ispec,nglob,potential_acoustic_old,dux_dxl,dux_dzl)
178  endif
179 
180  ! first double loop to compute gradient
181  if (axisym) then
182  ! AXISYM case
183  if (is_on_the_axis(ispec)) then
184  do j = 1,ngllz
185  do i = 1,ngllx
186  xixl = deriv(1,i,j)
187  xizl = deriv(2,i,j)
188  gammaxl = deriv(3,i,j)
189  gammazl = deriv(4,i,j)
190  jacobianl = deriv(5,i,j)
191  fac = deriv(6,i,j) ! jacobian/rho
192 
193  if (i == 1) then
194  ! dchi/dr=rho * u_r=0 on the axis
195  dux_dxl(i,j) = 0._custom_real
196  r_xiplus1(i,j) = gammaz(i,j,ispec) * jacobianl
197  else
198  r_xiplus1(i,j) = coord(1,ibool(i,j,ispec))/(xiglj(i) + one)
199  endif
200  tempx1(i,j) = r_xiplus1(i,j) * fac * (xixl * dux_dxl(i,j) + xizl * dux_dzl(i,j))
201  tempx2(i,j) = r_xiplus1(i,j) * fac * (gammaxl * dux_dxl(i,j) + gammazl * dux_dzl(i,j))
202  enddo
203  enddo
204  else
205  do j = 1,ngllz
206  do i = 1,ngllx
207  xixl = deriv(1,i,j)
208  xizl = deriv(2,i,j)
209  gammaxl = deriv(3,i,j)
210  gammazl = deriv(4,i,j)
211  jacobianl = deriv(5,i,j)
212  fac = deriv(6,i,j) ! jacobian/rho
213 
214  tempx1(i,j) = coord(1,ibool(i,j,ispec)) * fac * (xixl * dux_dxl(i,j) + xizl * dux_dzl(i,j))
215  tempx2(i,j) = coord(1,ibool(i,j,ispec)) * fac * (gammaxl * dux_dxl(i,j) + gammazl * dux_dzl(i,j))
216  enddo
217  enddo
218  endif
219  else
220  ! default case
221  do j = 1,ngllz
222  do i = 1,ngllx
223  xixl = deriv(1,i,j)
224  xizl = deriv(2,i,j)
225  gammaxl = deriv(3,i,j)
226  gammazl = deriv(4,i,j)
227  jacobianl = deriv(5,i,j)
228  fac = deriv(6,i,j) ! jacobian/rho
229 
230  tempx1(i,j) = fac * (xixl * dux_dxl(i,j) + xizl * dux_dzl(i,j))
231  tempx2(i,j) = fac * (gammaxl * dux_dxl(i,j) + gammazl * dux_dzl(i,j))
232  enddo
233  enddo
234  endif
235 
236  ! first double loop over GLL points to compute and store gradients
237  if (pml_boundary_conditions) then
238  ! calculates contribution from each C-PML element to update acceleration
239  call pml_compute_accel_contribution_acoustic(ispec,nglob, &
240  potential_acoustic,potential_acoustic_old,potential_dot_acoustic, &
241  potential_dot_dot_acoustic_pml,r_xiplus1)
242  endif
243 
244 !! DK DK QUENTIN visco begin
245  if (attenuation_viscoacoustic) then
246  ! attenuation is implemented following the memory variable formulation of
247  ! Carcione et al., Wave propagation simulation in a linear viscoacoustic medium,
248  ! Geophysical Journal, vol. 93, p. 393-407 (1988)
249 
250  tempx3 = 0._custom_real
251  do j = 1,ngllz
252  do i = 1,ngllx
253  ! loop on all the standard linear solids
254  do i_sls = 1,n_sls
255  tempx3(i,j) = tempx3(i,j) + e1_fluid(i,j,ispec,i_sls)
256  enddo
257  tempx3(i,j) = jacobian(i,j,ispec) * tempx3(i,j)
258  enddo
259  enddo
260 
261  endif
262 !! DK DK QUENTIN visco end
263 
264 !
265 ! second double-loop over GLL to compute all the terms
266 !
267 
268  ! along x direction and z direction
269  ! and assemble the contributions
270  if (axisym) then
271  ! axisymmetric case
272  if (is_on_the_axis(ispec)) then
273  do j = 1,ngllz
274  do i = 1,ngllx
275  ! assembles the contributions
276  temp1l = 0._custom_real
277  temp2l = 0._custom_real
278  do k = 1,ngllx
279  temp1l = temp1l + tempx1(k,j) * hprimebarwglj_xx(k,i)
280  temp2l = temp2l + tempx2(i,k) * hprimewgll_zz(k,j)
281  enddo
282  ! sums contributions from each element to the global values
283  iglob = ibool(i,j,ispec)
284  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) &
285  - (wzgll(j) * temp1l + wxglj(i) * temp2l)
286  enddo
287  enddo
288  else
289  do j = 1,ngllz
290  do i = 1,ngllx
291  ! assembles the contributions
292  temp1l = 0._custom_real
293  temp2l = 0._custom_real
294  do k = 1,ngllx
295  temp1l = temp1l + tempx1(k,j) * hprimewgll_xx(k,i)
296  temp2l = temp2l + tempx2(i,k) * hprimewgll_zz(k,j)
297  enddo
298  ! sums contributions from each element to the global values
299  iglob = ibool(i,j,ispec)
300  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) &
301  - (wzgll(j) * temp1l + wxgll(i) * temp2l)
302  enddo
303  enddo
304  endif
305  else
306  ! default case
307  do j = 1,ngllz
308  do i = 1,ngllx
309  ! assembles the contributions
310  temp1l = 0._custom_real
311  temp2l = 0._custom_real
312  do k = 1,ngllx
313  temp1l = temp1l + tempx1(k,j) * hprimewgll_xx(k,i)
314  temp2l = temp2l + tempx2(i,k) * hprimewgll_zz(k,j)
315  enddo
316  ! sums contributions from each element to the global values
317  iglob = ibool(i,j,ispec)
318  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) &
319  - (wzgll(j) * temp1l + wxgll(i) * temp2l)
320  enddo
321  enddo
322  endif
323 
324  ! PML contribution
325  if (pml_boundary_conditions) then
326  if (ispec_is_pml(ispec)) then
327  do j = 1,ngllz
328  do i = 1,ngllx
329  iglob = ibool(i,j,ispec)
330  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) - potential_dot_dot_acoustic_pml(i,j)
331  enddo
332  enddo
333  endif
334  endif
335 
336  if (attenuation_viscoacoustic) then
337  do j = 1,ngllz
338  do i = 1,ngllx
339  ! sums contributions from each element to the global values
340  iglob = ibool(i,j,ispec)
341  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) + wzgll(j) * wxgll(i) * tempx3(i,j)
342  enddo
343  enddo
344  endif
345 
346  enddo ! end of loop over all spectral elements

352  subroutine mxm_2comp_singlea(x,z,A,B,C)
353 
354 ! matrix x matrix multiplication, merging 2 loops for x = A^t B^t and z = A C^t
355 !
356 ! index notation:
357 ! general matrix multiplication: uij = (A B)ij = Aik Bkj
358 ! here: xij = (A^t B^t)ij = Akj Bik = (B A)ij
359 ! zij = (A C^t)ij = Aik Cjk
360 !
361 ! original loops:
362 !
363 ! do j = 1,NGLLZ
364 ! do i = 1,NGLLX
365 ! ! derivative along x and along z
366 ! dux_dxi(i,j) = 0._CUSTOM_REAL
367 ! dux_dgamma(i,j) = 0._CUSTOM_REAL
368 !
369 ! ! first double loop over GLL points to compute and store gradients
370 ! ! we can merge the two loops because NGLLX == NGLLZ
371 ! do k = 1,NGLLX
372 ! dux_dxi(i,j) = dux_dxi(i,j) + potential_elem(k,j) * hprime_xx(i,k)
373 ! dux_dgamma(i,j) = dux_dgamma(i,j) + potential_elem(i,k) * hprime_zz(j,k)
374 ! enddo
375 ! enddo
376 ! enddo
377 
378  use constants, only: ngllx,ngllz,custom_real
379 
380  implicit none
381 
382  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(out) :: x,z
383  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(in) :: A,B,C
384 
385  ! local parameters
386  integer :: i,j,k
387 
388  select case(ngllx)
389  case (5)
390  do j = 1,5
391  do i = 1,5
392  ! loop unrolling
393  x(i,j) = a(1,j) * b(i,1) + a(2,j) * b(i,2) + a(3,j) * b(i,3) + a(4,j) * b(i,4) + a(5,j) * b(i,5)
394  z(i,j) = a(i,1) * c(j,1) + a(i,2) * c(j,2) + a(i,3) * c(j,3) + a(i,4) * c(j,4) + a(i,5) * c(j,5)
395  enddo
396  enddo
397 
398  case default
399  do j = 1,ngllz
400  do i = 1,ngllx
401  x(i,j) = 0._custom_real
402  z(i,j) = 0._custom_real
403  do k = 1,ngllx
404  x(i,j) = x(i,j) + a(k,j) * b(i,k)
405  z(i,j) = z(i,j) + a(i,k) * c(j,k)
406  enddo
407  enddo
408  enddo
409  end select
410 
411  end subroutine mxm_2comp_singlea
412 
413  end subroutine compute_forces_acoustic
