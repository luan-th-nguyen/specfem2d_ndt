38 /* added code to force Flush-To-Zero (FTZ) on Intel processors */
39 /* otherwise Gradual Underflow can be extremely slow. With Intel */
40 /* ifort one can use compiler option -ftz, but no such option exists */
41 /* in gcc and gfortran therefore we call an assembler routine directly here. */
42 /* Very precise description available at */
43 /* http://software.intel.com/en-us/articles/x87-and-sse-floating-point-assists-in-ia-32-flush-to-zero-ftz-and-denormals-are-zero-daz/ */
44 /* and at http://www.rz.uni-karlsruhe.de/rz/docs/VTune/reference/vc148.htm */
45 /* from http://software.intel.com/en-us/articles/x87-and-sse-floating-point-assists-in-ia-32-flush-to-zero-ftz-and-denormals-are-zero-daz : */
46 
47 /* Flush-To-Zero (FTZ) mode */
48 
49 /* FTZ is a method of bypassing IEEE 754 methods of dealing with */
50 /* invalid floating-point numbers due to underflows. This mode is much */
51 /* faster. Two conditions must be met for FTZ processing to occur: */
52 
53 /* * The FTZ bit (bit 15) in the MXCSR register must be masked (value = 1). */
54 /* * The underflow exception (bit 11) needs to be masked (value = 1). */
55 
56 /* This routine is not strictly necessary for SPECFEM, thus if it does not compile on your system
57  (since it calls some low-level system routines) just suppress all the lines below (i.e. make it an empty file)
58  and comment out the call to force_ftz() in the main SPECFEM program */
59 
60 #include "config.h"
61 
62 #define FTZ_BIT 15
63 #define UNDERFLOW_EXCEPTION_MASK 11
64 
65 #ifdef __GNUC__ // only use these features on gnu compiler
66 #ifdef HAVE_XMMINTRIN
67  #define FORCE_FTZ
68  #include <xmmintrin.h>
69 #elif HAVE_EMMINTRIN
70  #include <emmintrin.h>
71  #define FORCE_FTZ
72 #endif
73 #endif // __GNUC__

75 void
76 FC_FUNC_(force_ftz,FORCE_FTZ)()
77 {
78 #ifdef __GNUC__
79 #ifdef FORCE_FTZ
80  unsigned int x;
81 
82  /* force FTZ by setting bits 11 and 15 to one */
83  x = _mm_getcsr();
84  x |= (1 << FTZ_BIT);
85  x |= (1 << UNDERFLOW_EXCEPTION_MASK);
86  _mm_setcsr(x);
87 #endif
88 #endif // __GNUC__
89 }
