45 extern "C"
46 void FC_FUNC_(initialize_cuda_device,
47  INITIALIZE_CUDA_DEVICE)(int* myrank_f,int* ncuda_devices) {
48  TRACE("initialize_cuda_device");
49 
50  int device;
51  int device_count;
52 
53  // Gets rank number of MPI process
54  int myrank = *myrank_f;
55 
56  // note: from here on we use the runtime API ...
57 
58  // Gets number of GPU devices
59  device_count = 0;
60  cudaGetDeviceCount(&device_count);
61  // Do not check if command failed with `exit_on_cuda_error` since it calls cudaDevice()/ThreadSynchronize():
62  // If multiple MPI tasks access multiple GPUs per node, they will try to synchronize
63  // GPU 0 and depending on the order of the calls, an error will be raised
64  // when setting the device number. If MPS is enabled, some GPUs will silently not be used.
65  //
66  // being verbose and catches error from first call to CUDA runtime function, without synchronize call
67  cudaError_t err = cudaGetLastError();
68  if (err != cudaSuccess){
69  fprintf(stderr,"Error after cudaGetDeviceCount: %s\n", cudaGetErrorString(err));
70  exit_on_error("CUDA runtime error: cudaGetDeviceCount failed\n\nplease check if driver and runtime libraries work together\nor on titan enable environment: CRAY_CUDA_PROXY=1 to use single GPU with multiple MPI processes\n\nexiting...\n");
71  }
72 
73  // returns device count to fortran
74  if (device_count == 0) exit_on_error("CUDA runtime error: there is no device supporting CUDA\n");
75  *ncuda_devices = device_count;
76 
77  // Sets the active device
78  if (device_count >= 1) {
79  // generalized for more GPUs per node
80  // note: without previous context release, cudaSetDevice will complain with the cuda error
81  // "setting the device when a process is active is not allowed"
82 
83  // releases previous contexts
84 #if CUDA_VERSION < 4000
85  cudaThreadExit();
86 #else
87  cudaDeviceReset();
88 #endif
89 
90  //printf("rank %d: cuda device count = %d sets device = %d \n",myrank,device_count,myrank % device_count);
91  //MPI_Barrier(MPI_COMM_WORLD);
92 
93  // sets active device
94 #ifdef CUDA_DEVICE_ID
95  // uses fixed device id when compile with e.g.: -DCUDA_DEVICE_ID=1
96  device = CUDA_DEVICE_ID;
97  if (myrank == 0) printf("setting cuda devices with id = %d for all processes by -DCUDA_DEVICE_ID\n\n",device);
98 
99  cudaSetDevice( device );
100  exit_on_cuda_error("cudaSetDevice has invalid device");
101 
102  // double check that device was properly selected
103  cudaGetDevice(&device);
104  if (device != CUDA_DEVICE_ID) {
105  printf("error rank: %d devices: %d \n",myrank,device_count);
106  printf(" cudaSetDevice()=%d\n cudaGetDevice()=%d\n",CUDA_DEVICE_ID,device);
107  exit_on_error("CUDA set/get device error: device id conflict \n");
108  }
109 #else
110  // device changes for different mpi processes according to number of device per node
111  // (assumes that number of devices per node is the same for different compute nodes)
112  device = myrank % device_count;
113 
114  cudaSetDevice( device );
115  exit_on_cuda_error("cudaSetDevice has invalid device");
116 
117  // double check that device was properly selected
118  cudaGetDevice(&device);
119  if (device != (myrank % device_count)) {
120  printf("error rank: %d devices: %d \n",myrank,device_count);
121  printf(" cudaSetDevice()=%d\n cudaGetDevice()=%d\n",myrank%device_count,device);
122  exit_on_error("CUDA set/get device error: device id conflict \n");
123  }
124 #endif
125  }
126 
127  // returns a handle to the active device
128  cudaGetDevice(&device);
129 
130  // get device properties
131  struct cudaDeviceProp deviceProp;
132  cudaGetDeviceProperties(&deviceProp,device);
133 
134  // exit if the machine has no CUDA-enabled device
135  if (deviceProp.major == 9999 && deviceProp.minor == 9999){
136  fprintf(stderr,"No CUDA-enabled device found, exiting...\n\n");
137  exit_on_error("CUDA runtime error: there is no CUDA-enabled device found\n");
138  }
139 
140  // outputs device infos to file
141  char filename[BUFSIZ];
142  FILE* fp;
143  int do_output_info;
144 
145  // by default, only master process outputs device infos to avoid file cluttering
146  do_output_info = 0;
147  if (myrank == 0) {
148  do_output_info = 1;
149  sprintf(filename,OUTPUT_FILES_PATH"/gpu_device_info.txt");
150  }
151  // debugging
152  if (DEBUG) {
153  do_output_info = 1;
154  sprintf(filename,OUTPUT_FILES_PATH"/gpu_device_info_proc_%06d.txt",myrank);
155  }
156 
157  // output to file
158  if (do_output_info) {
159  fp = fopen(filename,"w");
160  if (fp != NULL){
161  // display device properties
162  fprintf(fp,"Device Name = %s\n",deviceProp.name);
163  fprintf(fp,"memory:\n");
164  fprintf(fp," totalGlobalMem (in MB): %f\n",(unsigned long) deviceProp.totalGlobalMem / (1024.f * 1024.f));
165  fprintf(fp," totalGlobalMem (in GB): %f\n",(unsigned long) deviceProp.totalGlobalMem / (1024.f * 1024.f * 1024.f));
166  fprintf(fp," totalConstMem (in bytes): %lu\n",(unsigned long) deviceProp.totalConstMem);
167  fprintf(fp," Maximum 1D texture size (in bytes): %lu\n",(unsigned long) deviceProp.maxTexture1D);
168  fprintf(fp," sharedMemPerBlock (in bytes): %lu\n",(unsigned long) deviceProp.sharedMemPerBlock);
169  fprintf(fp," regsPerBlock (in bytes): %lu\n",(unsigned long) deviceProp.regsPerBlock);
170  fprintf(fp,"blocks:\n");
171  fprintf(fp," Maximum number of threads per block: %d\n",deviceProp.maxThreadsPerBlock);
172  fprintf(fp," Maximum size of each dimension of a block: %d x %d x %d\n",
173  deviceProp.maxThreadsDim[0],deviceProp.maxThreadsDim[1],deviceProp.maxThreadsDim[2]);
174  fprintf(fp," Maximum sizes of each dimension of a grid: %d x %d x %d\n",
175  deviceProp.maxGridSize[0],deviceProp.maxGridSize[1],deviceProp.maxGridSize[2]);
176  fprintf(fp,"features:\n");
177  fprintf(fp," Compute capability of the device = %d.%d\n", deviceProp.major, deviceProp.minor);
178  fprintf(fp," multiProcessorCount: %d\n",deviceProp.multiProcessorCount);
179  if (deviceProp.canMapHostMemory){
180  fprintf(fp," canMapHostMemory: TRUE\n");
181  }else{
182  fprintf(fp," canMapHostMemory: FALSE\n");
183  }
184  if (deviceProp.deviceOverlap){
185  fprintf(fp," deviceOverlap: TRUE\n");
186  }else{
187  fprintf(fp," deviceOverlap: FALSE\n");
188  }
189  if (deviceProp.concurrentKernels){
190  fprintf(fp," concurrentKernels: TRUE\n");
191  }else{
192  fprintf(fp," concurrentKernels: FALSE\n");
193  }
194  // outputs initial memory infos via cudaMemGetInfo()
195  double free_db,used_db,total_db;
196  get_free_memory(&free_db,&used_db,&total_db);
197  fprintf(fp,"memory usage:\n");
198  fprintf(fp," rank %d: GPU memory usage: used = %f MB, free = %f MB, total = %f MB\n",myrank,
199  used_db/1024.0/1024.0, free_db/1024.0/1024.0, total_db/1024.0/1024.0);
200 
201  // closes output file
202  fclose(fp);
203  }
204  }
205 
206  // make sure that the device has compute capability >= 1.3
207  if (deviceProp.major < 1){
208  fprintf(stderr,"Compute capability major number should be at least 1, exiting...\n\n");
209  exit_on_error("CUDA Compute capability major number should be at least 1\n");
210  }
211  if (deviceProp.major == 1 && deviceProp.minor < 3){
212  fprintf(stderr,"Compute capability should be at least 1.3, exiting...\n");
213  exit_on_error("CUDA Compute capability major number should be at least 1.3\n");
214  }
215 
216  // we use pinned memory for asynchronous copy
217  if (! deviceProp.canMapHostMemory){
218  fprintf(stderr,"Device capability should allow to map host memory, exiting...\n");
219  exit_on_error("CUDA Device capability canMapHostMemory should be TRUE\n");
220  }
221 
222  // checks kernel optimization setting
223 #ifdef USE_LAUNCH_BOUNDS
224  // see: mesh_constants_cuda.h
225  // performance statistics: main kernel Kernel_2_**_impl():
226  // shared memory per block = 6200 for Kepler: total = 49152 -> limits active blocks to 7
227  // registers per thread = 72 (limited by LAUNCH_MIN_BLOCKS 7)
228  // registers per block = 9216 total = 65536 (limited by LAUNCH_MIN_BLOCKS 7)
229 
230  // shared memory
231  if (deviceProp.sharedMemPerBlock > 49152 && LAUNCH_MIN_BLOCKS <= 7) {
232  if (myrank == 0) {
233  printf("GPU non-optimal settings: your setting of using LAUNCH_MIN_BLOCK %i is too low and limits the register usage\n",
234  LAUNCH_MIN_BLOCKS);
235  }
236  }
237 
238  // registers
239  if (deviceProp.regsPerBlock > 65536 && LAUNCH_MIN_BLOCKS <= 7) {
240  if (myrank == 0) {
241  printf("GPU non-optimal settings: your setting of using LAUNCH_MIN_BLOCK %i is too low and limits the register usage\n",
242  LAUNCH_MIN_BLOCKS);
243  }
244  }
245 #endif
246 
247 }
