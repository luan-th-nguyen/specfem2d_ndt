44  double precision function endw1(n,alpha,beta)
45 
46  implicit none
47 
48  integer n
49  double precision alpha,beta
50 
51  double precision, parameter :: zero=0.d0,one=1.d0,two=2.d0,three=3.d0,four=4.d0
52  double precision apb,f1,fint1,fint2,f2,di,abn,abnn,a1,a2,a3,f3
53  double precision, external :: gammaf
54  integer i
55 
56  f3 = zero
57  apb = alpha+beta
58  if (n == 0) then
59  endw1 = zero
60  return
61  endif
62  f1 = gammaf(alpha+two)*gammaf(beta+one)/gammaf(apb+three)
63  f1 = f1*(apb+two)*two**(apb+two)/two
64  if (n == 1) then
65  endw1 = f1
66  return
67  endif
68  fint1 = gammaf(alpha+two)*gammaf(beta+one)/gammaf(apb+three)
69  fint1 = fint1*two**(apb+two)
70  fint2 = gammaf(alpha+two)*gammaf(beta+two)/gammaf(apb+four)
71  fint2 = fint2*two**(apb+three)
72  f2 = (-two*(beta+two)*fint1 + (apb+four)*fint2) * (apb+three)/four
73  if (n == 2) then
74  endw1 = f2
75  return
76  endif
77  do i=3,n
78  di = dble(i-1)
79  abn = alpha+beta+di
80  abnn = abn+di
81  a1 = -(two*(di+alpha)*(di+beta))/(abn*abnn*(abnn+one))
82  a2 = (two*(alpha-beta))/(abnn*(abnn+two))
83  a3 = (two*(abn+one))/((abnn+two)*(abnn+one))
84  f3 = -(a2*f2+a1*f1)/a3
85  f1 = f2
86  f2 = f3
87  enddo
88  endw1 = f3
89 
90  end function endw1

178  subroutine jacg (xjac,np,alpha,beta)
179 
180 !=======================================================================
181 !
182 ! computes np Gauss points, which are the zeros of the
183 ! Jacobi polynomial with parameters alpha and beta
184 !
185 ! .alpha = beta = 0.0 -> Legendre points
186 ! .alpha = beta = -0.5 -> Chebyshev points
187 !
188 !=======================================================================
189 
190  implicit none
191 
192  integer np
193  double precision alpha,beta
194  double precision xjac(np)
195 
196  ! local parameters
197  integer k,j,i,jmin,jm,n
198  double precision xlast,dth,x,x1,x2,recsum,delx,xmin,swap
199  double precision p,pd,pm1,pdm1,pm2,pdm2
200 
201  integer, parameter :: K_MAX_ITER = 10
202  double precision, parameter :: zero = 0.d0, eps = 1.0d-12
203 
204  pm1 = zero
205  pm2 = zero
206  pdm1 = zero
207  pdm2 = zero
208 
209  xlast = 0.d0
210  n = np-1
211  dth = 4.d0*datan(1.d0)/(2.d0*dble(n)+2.d0)
212  p = 0.d0
213  pd = 0.d0
214 
215  do j=1,np
216  if (j == 1) then
217  x = dcos((2.d0*(dble(j)-1.d0)+1.d0)*dth)
218  else
219  x1 = dcos((2.d0*(dble(j)-1.d0)+1.d0)*dth)
220  x2 = xlast
221  x = (x1+x2)/2.d0
222  endif
223 
224  do k=1,k_max_iter
225  call jacobf (p,pd,pm1,pdm1,pm2,pdm2,np,alpha,beta,x)
226  recsum = 0.d0
227  jm = j-1
228  do i=1,jm
229  recsum = recsum+1.d0/(x-xjac(np-i+1))
230  enddo
231  delx = -p/(pd-recsum*p)
232  x = x+delx
233 
234  ! exits loop if increment too small
235  if (abs(delx) < eps) exit
236 
237  enddo
238 
239  ! checks bounds
240  if (np-j+1 < 1 .or. np-j+1 > np) stop 'error np-j+1-index in jacg'
241 
242  xjac(np-j+1) = x
243  xlast = x
244  enddo
245 
246  jmin = 0
247 
248  ! orders xjac array in increasing values
249  do i=1,np
250  xmin = 2.d0
251  jmin = i
252 
253  ! looks for index with minimum value
254  do j=i,np
255  ! note: some compilers (cray) might be too aggressive in optimizing this loop,
256  ! thus we need this temporary array value x to store and compare values
257  x = xjac(j)
258 
259  if (x < xmin) then
260  xmin = x
261  jmin = j
262  endif
263  enddo
264 
265  ! checks bounds
266  if (jmin < 1 .or. jmin > np) stop 'error j-index in jacg'
267 
268  if (jmin /= i) then
269  swap = xjac(i)
270  xjac(i) = xjac(jmin)
271  xjac(jmin) = swap
272  endif
273 
274  enddo
275 
276  end subroutine jacg

282  subroutine jacobf (poly,pder,polym1,pderm1,polym2,pderm2,n,alp,bet,x)
283 
284 !=======================================================================
285 !
286 ! Computes the Jacobi polynomial of degree n and its derivative at x
287 !
288 !=======================================================================
289 
290  implicit none
291 
292  double precision poly,pder,polym1,pderm1,polym2,pderm2,alp,bet,x
293  integer n
294 
295  double precision apb,polyl,pderl,dk,a1,a2,b3,a3,a4,polyn,pdern,psave,pdsave
296  integer k
297 
298  apb = alp+bet
299  poly = 1.d0
300  pder = 0.d0
301  psave = 0.d0
302  pdsave = 0.d0
303 
304  if (n == 0) return
305 
306  polyl = poly
307  pderl = pder
308  poly = (alp-bet+(apb+2.d0)*x)/2.d0
309  pder = (apb+2.d0)/2.d0
310  if (n == 1) return
311 
312  do k=2,n
313  dk = dble(k)
314  a1 = 2.d0*dk*(dk+apb)*(2.d0*dk+apb-2.d0)
315  a2 = (2.d0*dk+apb-1.d0)*(alp**2-bet**2)
316  b3 = (2.d0*dk+apb-2.d0)
317  a3 = b3*(b3+1.d0)*(b3+2.d0)
318  a4 = 2.d0*(dk+alp-1.d0)*(dk+bet-1.d0)*(2.d0*dk+apb)
319  polyn = ((a2+a3*x)*poly-a4*polyl)/a1
320  pdern = ((a2+a3*x)*pder-a4*pderl+a3*poly)/a1
321  psave = polyl
322  pdsave = pderl
323  polyl = poly
324  poly = polyn
325  pderl = pder
326  pder = pdern
327  enddo
328 
329  polym1 = polyl
330  pderm1 = pderl
331  polym2 = psave
332  pderm2 = pdsave
333 
334  end subroutine jacobf

340  double precision function pndleg (Z,N)
341 
342 !------------------------------------------------------------------------
343 !
344 ! Compute the derivative of the Nth order Legendre polynomial at Z.
345 ! Based on the recursion formula for the Legendre polynomials.
346 !
347 !------------------------------------------------------------------------
348  implicit none
349 
350  double precision z
351  integer n
352 
353  double precision P1,P2,P1D,P2D,P3D,DBLE_K,P3
354  integer k
355 
356  p1 = 1.d0
357  p2 = z
358  p1d = 0.d0
359  p2d = 1.d0
360  p3d = 1.d0
361 
362  do k = 1, n-1
363  dble_k = dble(k)
364  p3 = ((2.d0*dble_k+1.d0)*z*p2 - dble_k*p1)/(dble_k+1.d0)
365  p3d = ((2.d0*dble_k+1.d0)*p2 + (2.d0*dble_k+1.d0)*z*p2d - dble_k*p1d) / (dble_k+1.d0)
366  p1 = p2
367  p2 = p3
368  p1d = p2d
369  p2d = p3d
370  enddo
371 
372  pndleg = p3d
373 
374  end function pndleg

415  double precision function pnormj (n,alpha,beta)
416 
417  implicit none
418 
419  double precision alpha,beta
420  integer n
421 
422  double precision one,two,dn,const,prod,dindx,frac
423  double precision, external :: gammaf
424  integer i
425 
426  one = 1.d0
427  two = 2.d0
428  dn = dble(n)
429  const = alpha+beta+one
430 
431  if (n <= 1) then
432  prod = gammaf(dn+alpha)*gammaf(dn+beta)
433  prod = prod/(gammaf(dn)*gammaf(dn+alpha+beta))
434  pnormj = prod * two**const/(two*dn+const)
435  return
436  endif
437 
438  prod = gammaf(alpha+one)*gammaf(beta+one)
439  prod = prod/(two*(one+const)*gammaf(const+one))
440  prod = prod*(one+alpha)*(two+alpha)
441  prod = prod*(one+beta)*(two+beta)
442 
443  do i=3,n
444  dindx = dble(i)
445  frac = (dindx+alpha)*(dindx+beta)/(dindx*(dindx+alpha+beta))
446  prod = prod*frac
447  enddo
448 
449  pnormj = prod * two**const/(two*dn+const)
450 
451  end function pnormj

457  subroutine zwgjd(z,w,np,alpha,beta)
458 
459 !=======================================================================
460 !
461 ! Z w g j d : Generate np Gauss-Jacobi points and weights
462 ! associated with Jacobi polynomial of degree n = np-1
463 !
464 ! Note : Coefficients alpha and beta must be greater than -1.
465 ! ----
466 !=======================================================================
467 
468  implicit none
469 
470  double precision, parameter :: zero=0.d0,one=1.d0,two=2.d0
471 
472  integer np
473  double precision z(np),w(np)
474  double precision alpha,beta
475 
476  ! local parameters
477  integer n,np1,np2,i
478  double precision p,pd,pm1,pdm1,pm2,pdm2
479  double precision apb,dnp1,dnp2,fac1,fac2,fac3,fnorm,rcoef
480  double precision, external :: gammaf,pnormj
481 
482  pd = zero
483  pm1 = zero
484  pm2 = zero
485  pdm1 = zero
486  pdm2 = zero
487 
488  n = np-1
489  apb = alpha+beta
490  p = zero
491  pdm1 = zero
492 
493  if (np <= 0) stop 'minimum number of Gauss points is 1'
494 
495  if ((alpha <= -one) .or. (beta <= -one)) stop 'alpha and beta must be greater than -1'
496 
497  if (np == 1) then
498  z(1) = (beta-alpha)/(apb+two)
499  w(1) = gammaf(alpha+one)*gammaf(beta+one)/gammaf(apb+two) * two**(apb+one)
500  return
501  endif
502 
503  call jacg(z,np,alpha,beta)
504 
505  np1 = n+1
506  np2 = n+2
507  dnp1 = dble(np1)
508  dnp2 = dble(np2)
509  fac1 = dnp1+alpha+beta+one
510  fac2 = fac1+dnp1
511  fac3 = fac2+one
512  fnorm = pnormj(np1,alpha,beta)
513  rcoef = (fnorm*fac2*fac3)/(two*fac1*dnp2)
514  do i=1,np
515  call jacobf(p,pd,pm1,pdm1,pm2,pdm2,np2,alpha,beta,z(i))
516  w(i) = -rcoef/(p*pdm1)
517  enddo
518 
519  end subroutine zwgjd

525  subroutine zwgljd(z,w,np,alpha,beta)
526 
527 !=======================================================================
528 !
529 ! Z w g l j d : Generate np Gauss-Lobatto-Jacobi points and the
530 ! ----------- weights associated with Jacobi polynomials of degree
531 ! n = np-1.
532 !
533 ! Note : alpha and beta coefficients must be greater than -1.
534 ! Legendre polynomials are special case of Jacobi polynomials
535 ! just by setting alpha and beta to 0.
536 !
537 !=======================================================================
538 
539  implicit none
540 
541  double precision, parameter :: zero=0.d0,one=1.d0,two=2.d0,tol_zero=1.d-30
542 
543  integer np
544  double precision alpha,beta
545  double precision z(np), w(np)
546 
547  ! local parameters
548  integer n,nm1,i
549  double precision p,pd,pm1,pdm1,pm2,pdm2
550  double precision alpg,betg
551  double precision, external :: endw1,endw2
552 
553  p = zero
554  pm1 = zero
555  pm2 = zero
556  pdm1 = zero
557  pdm2 = zero
558 
559  n = np-1
560  nm1 = n-1
561  pd = zero
562 
563  if (np <= 1) stop 'minimum number of Gauss-Lobatto points is 2'
564 
565 ! with spectral elements, use at least 3 points
566  if (np <= 2) stop 'minimum number of Gauss-Lobatto points for the SEM is 3'
567 
568  if ((alpha <= -one) .or. (beta <= -one)) stop 'alpha and beta must be greater than -1'
569 
570  if (nm1 > 0) then
571  alpg = alpha+one
572  betg = beta+one
573  call zwgjd(z(2:n),w(2:n),nm1,alpg,betg)
574  endif
575 
576 ! start and end point at exactly -1 and 1
577  z(1) = - one
578  z(np) = one
579 
580 ! note: Jacobi polynomials with (alpha,beta) equal to zero become Legendre polynomials.
581 ! for Legendre polynomials, if number of points is odd, the middle abscissa is exactly zero
582  if (abs(alpha) < tol_zero .and. abs(beta) < tol_zero) then
583  if (mod(np,2) /= 0) z((np-1)/2+1) = zero
584  endif
585 
586 ! weights
587  do i=2,np-1
588  w(i) = w(i)/(one-z(i)**2)
589  enddo
590 
591  call jacobf(p,pd,pm1,pdm1,pm2,pdm2,n,alpha,beta,z(1))
592  w(1) = endw1(n,alpha,beta)/(two*pd)
593 
594  call jacobf(p,pd,pm1,pdm1,pm2,pdm2,n,alpha,beta,z(np))
595  w(np) = endw2(n,alpha,beta)/(two*pd)
596 
597  end subroutine zwgljd
