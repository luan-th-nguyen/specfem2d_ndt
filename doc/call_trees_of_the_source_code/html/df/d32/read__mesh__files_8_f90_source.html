64  subroutine read_mesh_nodes_coords_from_interfaces()
65 
66  use constants, only: imain,iin_interfaces,dont_ignore_junk,max_string_len,mygroup,in_data_files
67 
68  use part_unstruct_par, only: nodes_coords,max_npoints_interface,number_of_interfaces, &
69  npoints_interface_top,xinterface_top,zinterface_top,coefs_interface_top, &
70  nx,nz,nxread,nzread,nz_layer,number_of_layers,nnodes,grid_point_x,grid_point_z
71 
72  use source_file_par, only: source_surf,xs,zs
73 
74  use shared_parameters, only: ngnod,interfacesfile,nsources,xmin_param,xmax_param,number_of_simultaneous_runs
75 
76  implicit none
77 
78  ! local parameters
79  integer :: ilayer,ipoint_current
80  integer :: num_node
81 
82  integer :: npoints_interface_bottom
83  double precision, dimension(:), allocatable :: xinterface_bottom,zinterface_bottom,coefs_interface_bottom
84 
85  ! to compute the coordinate transformation
86  integer :: ioffset
87  double precision :: gamma,absx,a00,a01,bot0,top0
88  double precision :: tang1,tangN
89 
90  integer :: i_source
91  integer :: ix,iz
92  integer :: i,j
93 
94  ! external functions
95  integer, external :: num_4, num_9
96  double precision, external :: value_spline
97  character(len=MAX_STRING_LEN) :: interfaces_filename,path_to_add
98 
99  interfaces_filename = trim(in_data_files)//trim(interfacesfile)
100 
101  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
102  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
103  interfaces_filename = path_to_add(1:len_trim(path_to_add))//interfaces_filename(1:len_trim(interfaces_filename))
104  endif
105 
106  ! get interface data from external file
107  write(imain,*) 'Reading interface data from file: '//trim(in_data_files)//interfacesfile(1:len_trim(interfacesfile))
108  open(unit=iin_interfaces,file=trim(interfaces_filename),status='old')
109 
110  ! allocate arrays for the grid
111  allocate(grid_point_x(0:nx,0:nz))
112  allocate(grid_point_z(0:nx,0:nz))
113 
114  grid_point_x(:,:) = 0.d0
115  grid_point_z(:,:) = 0.d0
116 
117  ! top interface arrays (required also for station locations)
118  allocate(xinterface_top(max_npoints_interface))
119  allocate(zinterface_top(max_npoints_interface))
120  allocate(coefs_interface_top(max_npoints_interface))
121 
122  ! temporary arrays
123  allocate(xinterface_bottom(max_npoints_interface))
124  allocate(zinterface_bottom(max_npoints_interface))
125  allocate(coefs_interface_bottom(max_npoints_interface))
126 
127  ! read number of interfaces
128  call read_value_integer(iin_interfaces,dont_ignore_junk,number_of_interfaces)
129 
130  ! read bottom interface
131  call read_value_integer(iin_interfaces,dont_ignore_junk,npoints_interface_bottom)
132 
133  ! loop on all the points describing this interface
134  do ipoint_current = 1,npoints_interface_bottom
135  call read_two_interface_points(iin_interfaces,dont_ignore_junk, &
136  xinterface_bottom(ipoint_current),zinterface_bottom(ipoint_current))
137  enddo
138 
139  ! loop on all the layers
140  do ilayer = 1,number_of_layers
141 
142  ! read top interface
143  call read_value_integer(iin_interfaces,dont_ignore_junk,npoints_interface_top)
144 
145  ! loop on all the points describing this interface
146  do ipoint_current = 1,npoints_interface_top
147  call read_two_interface_points(iin_interfaces,dont_ignore_junk, &
148  xinterface_top(ipoint_current),zinterface_top(ipoint_current))
149  enddo
150 
151  ! compute the spline for the bottom interface, impose the tangent on both edges
152  tang1 = (zinterface_bottom(2)-zinterface_bottom(1)) / (xinterface_bottom(2)-xinterface_bottom(1))
153  tangn = (zinterface_bottom(npoints_interface_bottom)-zinterface_bottom(npoints_interface_bottom-1)) / &
154  (xinterface_bottom(npoints_interface_bottom)-xinterface_bottom(npoints_interface_bottom-1))
155 
156  call spline_construction(xinterface_bottom,zinterface_bottom,npoints_interface_bottom, &
157  tang1,tangn,coefs_interface_bottom)
158 
159  ! compute the spline for the top interface, impose the tangent on both edges
160  tang1 = (zinterface_top(2)-zinterface_top(1)) / (xinterface_top(2)-xinterface_top(1))
161  tangn = (zinterface_top(npoints_interface_top)-zinterface_top(npoints_interface_top-1)) / &
162  (xinterface_top(npoints_interface_top)-xinterface_top(npoints_interface_top-1))
163 
164  call spline_construction(xinterface_top,zinterface_top,npoints_interface_top, &
165  tang1,tangn,coefs_interface_top)
166 
167  ! check if we are in the last layer, which contains topography,
168  ! and modify the position of the source accordingly if it is located exactly at the surface
169  do i_source= 1,nsources
170  if (source_surf(i_source) .and. ilayer == number_of_layers) then
171  ! user output
172  write(imain,*) 'source ', i_source
173  write(imain,*) ' target (input) z: ', zs(i_source)
174 
175  zs(i_source) = value_spline(xs(i_source),xinterface_top,zinterface_top, &
176  coefs_interface_top,npoints_interface_top)
177 
178  write(imain,*) ' surface (actual) z: ', zs(i_source)
179  endif
180  enddo
181 
182  ! compute the offset of this layer in terms of number of spectral elements below along Z
183  if (ilayer > 1) then
184  ioffset = sum(nz_layer(1:ilayer-1))
185  else
186  ioffset = 0
187  endif
188 
189  !--- definition of the mesh
190 
191  do ix = 0,nx
192 
193  ! evenly spaced points along X
194  absx = xmin_param + (xmax_param - xmin_param) * dble(ix) / dble(nx)
195 
196  ! value of the bottom and top splines
197  bot0 = value_spline(absx,xinterface_bottom,zinterface_bottom,coefs_interface_bottom,npoints_interface_bottom)
198  top0 = value_spline(absx,xinterface_top,zinterface_top,coefs_interface_top,npoints_interface_top)
199 
200  do iz = 0,nz_layer(ilayer)
201 
202  ! linear interpolation between bottom and top
203  gamma = dble(iz) / dble(nz_layer(ilayer))
204  a00 = 1.d0 - gamma
205  a01 = gamma
206 
207  ! coordinates of the grid points
208  grid_point_x(ix,iz + ioffset) = absx
209  grid_point_z(ix,iz + ioffset) = a00*bot0 + a01*top0
210 
211  enddo
212 
213  enddo
214 
215  ! the top interface becomes the bottom interface before switching to the next layer
216  npoints_interface_bottom = npoints_interface_top
217  xinterface_bottom(:) = xinterface_top(:)
218  zinterface_bottom(:) = zinterface_top(:)
219 
220  enddo
221 
222  close(iin_interfaces)
223 
224  deallocate(xinterface_bottom,zinterface_bottom,coefs_interface_bottom)
225 
226  ! sets node coordinates
227  nnodes = (nz+1)*(nx+1)
228 
229  allocate(nodes_coords(2,nnodes))
230  nodes_coords(:,:) = 0.d0
231 
232  if (ngnod == 4) then
233  do j = 0, nz
234  do i = 0, nx
235  num_node = num_4(i,j,nxread)
236  nodes_coords(1, num_node) = grid_point_x(i,j)
237  nodes_coords(2, num_node) = grid_point_z(i,j)
238  enddo
239  enddo
240  else if (ngnod == 9) then
241  do j = 0, nz
242  do i = 0, nx
243  num_node = num_9(i,j,nxread,nzread)
244  nodes_coords(1, num_node) = grid_point_x(i,j)
245  nodes_coords(2, num_node) = grid_point_z(i,j)
246  enddo
247  enddo
248  else
249  stop 'ngnod should be either 4 or 9'
250  endif
251 
252  end subroutine read_mesh_nodes_coords_from_interfaces
