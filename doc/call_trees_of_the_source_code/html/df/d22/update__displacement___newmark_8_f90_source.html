35 !
36 ! Newmark time scheme
37 !
38 !------------------------------------------------------------------------------------------------
39 !
40 ! explicit Newmark time scheme with acoustic & elastic domains:
41 ! (see e.g. Hughes, 1987; Chaljub et al., 2003)
42 !
43 ! chi(t+delta_t) = chi(t) + delta_t chi_dot(t) + 1/2 delta_t**2 chi_dot_dot(t)
44 ! chi_dot(t+delta_t) = chi_dot(t) + 1/2 delta_t chi_dot_dot(t) + 1/2 delta_t chi_dot_dot(t+delta_t)
45 ! chi_dot_dot(t+delta_t) = 1/M_acoustic( -K_acoustic chi(t+delta) + B_acoustic u(t+delta_t) + f(t+delta_t) )
46 !
47 ! u(t+delta_t) = u(t) + delta_t v(t) + 1/2 delta_t**2 a(t)
48 ! v(t+delta_t) = v(t) + 1/2 delta_t a(t) + 1/2 delta_t a(t+delta_t)
49 ! a(t+delta_t) = 1/M_elastic ( -K_elastic u(t+delta) + B_elastic chi_dot_dot(t+delta_t) + f( t+delta_t) )
50 !
51 ! where
52 ! chi, chi_dot, chi_dot_dot are acoustic (fluid) potentials ( dotted with respect to time)
53 ! u, v, a are displacement,velocity & acceleration
54 ! M is mass matrix, K stiffness matrix and B boundary term for acoustic/elastic domains
55 ! f denotes a source term (acoustic/elastic)
56 !
57 ! note that this stage calculates the predictor terms
58 !
59 ! for
60 ! potential chi_dot(t+delta) requires + 1/2 delta_t chi_dot_dot(t+delta_t)
61 ! at a later stage (corrector) once where chi_dot_dot(t+delta) is calculated
62 ! and similar,
63 ! velocity v(t+delta_t) requires + 1/2 delta_t a(t+delta_t)
64 ! at a later stage once where a(t+delta) is calculated
65 ! also:
66 ! boundary term B_elastic requires chi_dot_dot(t+delta)
67 ! thus chi_dot_dot has to be updated first before the elastic boundary term is considered
68

406  subroutine update_displacement_newmark_acoustic(deltat,deltatover2,deltatsquareover2, &
407  potential_dot_dot_acoustic,potential_dot_acoustic, &
408  potential_acoustic, &
409  PML_BOUNDARY_CONDITIONS,potential_acoustic_old)
410 
411  use specfem_par, only: nglob_acoustic,custom_real
412 
413  implicit none
414 
415  double precision,intent(in) :: deltat,deltatover2,deltatsquareover2
416  real(kind=CUSTOM_REAL), dimension(nglob_acoustic),intent(inout) :: potential_acoustic,potential_dot_acoustic, &
417  potential_dot_dot_acoustic
418 
419  logical,intent(in) :: PML_BOUNDARY_CONDITIONS
420  real(kind=CUSTOM_REAL), dimension(nglob_acoustic),intent(inout) :: potential_acoustic_old
421 
422  ! local parameters
423 #ifdef FORCE_VECTORIZATION
424  integer :: i
425 #endif
426 
427  ! PML simulations
428  if (pml_boundary_conditions) then
429 
430  ! note: todo - for elastic, there is an additional factor 1/TWO to the default deltasquareover2 for the acceleration term
431  ! find explanations where?
432 #ifdef FORCE_VECTORIZATION
433  do i = 1,nglob_acoustic
434  potential_acoustic_old(i) = potential_acoustic(i) + deltatsquareover2 * potential_dot_dot_acoustic(i)
435  enddo
436 #else
437  potential_acoustic_old(:) = potential_acoustic(:) + deltatsquareover2 * potential_dot_dot_acoustic(:)
438 #endif
439 
440  endif ! PML_BOUNDARY_CONDITIONS
441 
442 
443 #ifdef FORCE_VECTORIZATION
444  do i = 1,nglob_acoustic
445  potential_acoustic(i) = potential_acoustic(i) + deltat * potential_dot_acoustic(i) &
446  + deltatsquareover2 * potential_dot_dot_acoustic(i)
447  potential_dot_acoustic(i) = potential_dot_acoustic(i) + deltatover2 * potential_dot_dot_acoustic(i)
448  potential_dot_dot_acoustic(i) = 0._custom_real
449  enddo
450 #else
451  potential_acoustic(:) = potential_acoustic(:) + deltat * potential_dot_acoustic &
452  + deltatsquareover2 * potential_dot_dot_acoustic(:)
453  potential_dot_acoustic(:) = potential_dot_acoustic(:) + deltatover2 * potential_dot_dot_acoustic(:)
454  potential_dot_dot_acoustic(:) = 0._custom_real
455 #endif
456 
457  end subroutine update_displacement_newmark_acoustic

463  subroutine update_displacement_newmark_elastic(deltat,deltatover2,deltatsquareover2, &
464  accel_elastic,veloc_elastic, &
465  displ_elastic,displ_elastic_old, &
466  PML_BOUNDARY_CONDITIONS)
467 
468  use constants, only: custom_real,ndim,two
469 
470  use specfem_par, only: nglob_elastic,attenuation_viscoelastic
471 
472 #ifndef FORCE_VECTORIZATION
473  use constants, only: use_enforce_fields
474  use specfem_par, only: iglob_is_forced,it
475 #endif
476 
477  implicit none
478 
479  double precision,intent(in) :: deltat,deltatover2,deltatsquareover2
480  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_elastic),intent(inout) :: accel_elastic,veloc_elastic, &
481  displ_elastic,displ_elastic_old
482 
483  logical,intent(in) :: PML_BOUNDARY_CONDITIONS
484 
485  ! local parameters
486 #ifdef FORCE_VECTORIZATION
487  integer :: i
488 #else
489  integer :: iglob
490 #endif
491 
492  ! attenuation/PML simulations
493  if (pml_boundary_conditions .or. attenuation_viscoelastic) then
494 
495  ! note: todo - there is an additional factor 1/TWO to the default deltasquareover2 for the acceleration term
496  ! find explanations where?
497 #ifdef FORCE_VECTORIZATION
498  do i = 1,ndim*nglob_elastic
499  displ_elastic_old(i,1) = displ_elastic(i,1) + deltatsquareover2/two * accel_elastic(i,1)
500  enddo
501 #else
502  displ_elastic_old(:,:) = displ_elastic(:,:) + deltatsquareover2/two * accel_elastic(:,:)
503 #endif
504 
505  endif ! PML
506 
507 #ifdef FORCE_VECTORIZATION
508  !! DK DK this allows for full vectorization by using a trick to see the 2D array as a 1D array
509  !! DK DK whose dimension is the product of the two dimensions, the second dimension being equal to 1
510  ! vectorized updates
511  do i = 1,ndim*nglob_elastic
512  displ_elastic(i,1) = displ_elastic(i,1) + deltat * veloc_elastic(i,1) + deltatsquareover2 * accel_elastic(i,1)
513  veloc_elastic(i,1) = veloc_elastic(i,1) + deltatover2 * accel_elastic(i,1)
514  accel_elastic(i,1) = 0._custom_real
515  enddo
516 #else
517  ! non-vectorized updates
518  if (use_enforce_fields) then
519  do iglob = 1,nglob_elastic
520  if (iglob_is_forced(iglob)) then
521  call enforce_fields(iglob,it)
522  else
523  ! big loop over all the global points (not elements) in the mesh to update
524  ! the displacement and velocity vectors and clear the acceleration vector
525  displ_elastic(:,iglob) = displ_elastic(:,iglob) &
526  + deltat * veloc_elastic(:,iglob) + deltatsquareover2 * accel_elastic(:,iglob)
527  veloc_elastic(:,iglob) = veloc_elastic(:,iglob) + deltatover2 * accel_elastic(:,iglob) ! Predictor
528  accel_elastic(:,iglob) = 0._custom_real
529  endif
530  enddo
531  else
532  displ_elastic(:,:) = displ_elastic(:,:) + deltat * veloc_elastic(:,:) + deltatsquareover2 * accel_elastic(:,:)
533  veloc_elastic(:,:) = veloc_elastic(:,:) + deltatover2 * accel_elastic(:,:)
534  accel_elastic(:,:) = 0._custom_real
535  endif
536 
537 #endif
538 
539  end subroutine update_displacement_newmark_elastic

548  velocw_poroelastic,displw_poroelastic)

582  accels_poroelastic(:,:) = 0._custom_real
583 #endif
584

589  + deltatsquareover2*accelw_poroelastic(i,1)

591  accelw_poroelastic(i,1) = 0._custom_real
592  enddo
593 #else
594  displw_poroelastic(:,:) = displw_poroelastic(:,:) + deltat*velocw_poroelastic(:,:) &
595  + deltatsquareover2*accelw_poroelastic(:,:)
596  velocw_poroelastic(:,:) = velocw_poroelastic(:,:) + deltatover2*accelw_poroelastic(:,:)
597  accelw_poroelastic(:,:) = 0._custom_real
598 #endif
