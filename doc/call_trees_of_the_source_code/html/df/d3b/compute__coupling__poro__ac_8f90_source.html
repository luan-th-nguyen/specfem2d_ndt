34 ! for poro solver
35 
36  subroutine compute_coupling_poro_ac(potential_dot_dot_acoustic,accels_poroelastic,accelw_poroelastic,FORWARD_OR_ADJOINT)
37 
38  use constants, only: custom_real,ndim,ngllx,ngllz,nglj,cpml_x_only,cpml_z_only,iright,ileft,ibottom,itop,one
39 
40  use specfem_par, only: num_fluid_poro_edges,ibool,wxgll,wzgll,xix,xiz, &
41  gammax,gammaz,jacobian,ivalue,jvalue,ivalue_inverse,jvalue_inverse, &
42  fluid_poro_acoustic_ispec,fluid_poro_acoustic_iedge, &
43  fluid_poro_poroelastic_ispec,fluid_poro_poroelastic_iedge, &
44  porosity,tortuosity,density,kmato, &
45  potential_acoustic_adj_coupling, &
46  nglob_acoustic,nglob_poroelastic, &
47  simulation_type
48 
49  implicit none
50 
51  real(kind=CUSTOM_REAL), dimension(nglob_acoustic), intent(in) :: potential_dot_dot_acoustic
52  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_poroelastic), intent(inout) :: accels_poroelastic,accelw_poroelastic
53 
54  ! FORWARD_OR_ADJOINT == 1 for forward wavefield, FORWARD_OR_ADJOINT == 3 for backward/reconstruction wavefield
55  integer,intent(in) :: FORWARD_OR_ADJOINT
56 
57  !local variable
58  integer :: inum,ispec_acoustic,iedge_acoustic,ispec_poroelastic,iedge_poroelastic, &
59  ipoin1D,i,j,ii2,jj2,iglob
60  real(kind=CUSTOM_REAL) :: pressure,xxi,zxi,xgamma,zgamma,jacobian1D,nx,nz,weight
61  real(kind=CUSTOM_REAL) :: fac_s,fac_w
62  double precision :: phi,tort,rho_f,rho_s,rho_bar
63  integer :: material
64 
65  ! loop on all the coupling edges
66  do inum = 1,num_fluid_poro_edges
67  ! get the edge of the acoustic element
68  ispec_acoustic = fluid_poro_acoustic_ispec(inum)
69  iedge_acoustic = fluid_poro_acoustic_iedge(inum)
70  ! get the corresponding edge of the poroelastic element
71  ispec_poroelastic = fluid_poro_poroelastic_ispec(inum)
72  iedge_poroelastic = fluid_poro_poroelastic_iedge(inum)
73 
74  ! implement 1D coupling along the edge
75  do ipoin1d = 1,ngllx
76  ! get point values for the acoustic side, which matches our side in the inverse direction
77  i = ivalue_inverse(ipoin1d,iedge_acoustic)
78  j = jvalue_inverse(ipoin1d,iedge_acoustic)
79  iglob = ibool(i,j,ispec_acoustic)
80 
81  ! get poroelastic parameters
82  material = kmato(ispec_poroelastic)
83  phi = porosity(material)
84  tort = tortuosity(material)
85  rho_f = density(2,material)
86  rho_s = density(1,material)
87  rho_bar = (1.d0-phi)*rho_s + phi*rho_f
88 
89  fac_s = real((1.d0 - phi/tort),kind=custom_real)
90  fac_w = real((1.d0 - rho_f/rho_bar),kind=custom_real)
91 
92  ! compute pressure on the fluid/porous medium edge
93  pressure = - potential_dot_dot_acoustic(iglob)
94 
95  if (simulation_type == 3 .and. forward_or_adjoint == 1) then
96  ! new definition of adjoint displacement and adjoint potential
97  ! adjoint definition: pressure^\dagger = potential^\dagger
98  pressure = potential_acoustic_adj_coupling(iglob)
99  endif
100 
101  ! get point values for the poroelastic side
102  ii2 = ivalue(ipoin1d,iedge_poroelastic)
103  jj2 = jvalue(ipoin1d,iedge_poroelastic)
104  iglob = ibool(ii2,jj2,ispec_poroelastic)
105 
106  ! compute the 1D Jacobian and the normal to the edge: for their expression see for instance
107  ! O. C. Zienkiewicz and R. L. Taylor, The Finite Element Method for Solid and Structural Mechanics,
108  ! Sixth Edition, electronic version, www.amazon.com, p. 204 and Figure 7.7(a),
109  ! or Y. K. Cheung, S. H. Lo and A. Y. T. Leung, Finite Element Implementation,
110  ! Blackwell Science, page 110, equation (4.60).
111  if (iedge_acoustic == itop) then
112  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
113  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
114  jacobian1d = sqrt(xxi**2 + zxi**2)
115  nx = - zxi / jacobian1d
116  nz = + xxi / jacobian1d
117  weight = jacobian1d * wxgll(i)
118  else if (iedge_acoustic == ibottom) then
119  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
120  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
121  jacobian1d = sqrt(xxi**2 + zxi**2)
122  nx = + zxi / jacobian1d
123  nz = - xxi / jacobian1d
124  weight = jacobian1d * wxgll(i)
125  else if (iedge_acoustic == ileft) then
126  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
127  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
128  jacobian1d = sqrt(xgamma**2 + zgamma**2)
129  nx = - zgamma / jacobian1d
130  nz = + xgamma / jacobian1d
131  weight = jacobian1d * wzgll(j)
132  else if (iedge_acoustic == iright) then
133  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
134  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
135  jacobian1d = sqrt(xgamma**2 + zgamma**2)
136  nx = + zgamma / jacobian1d
137  nz = - xgamma / jacobian1d
138  weight = jacobian1d * wzgll(j)
139  endif
140 
141  ! contribution to the solid phase
142  accels_poroelastic(1,iglob) = accels_poroelastic(1,iglob) + weight*nx*pressure * fac_s
143  accels_poroelastic(2,iglob) = accels_poroelastic(2,iglob) + weight*nz*pressure * fac_s
144  ! contribution to the fluid phase
145  accelw_poroelastic(1,iglob) = accelw_poroelastic(1,iglob) + weight*nx*pressure * fac_w
146  accelw_poroelastic(2,iglob) = accelw_poroelastic(2,iglob) + weight*nz*pressure * fac_w
147  enddo
148  enddo
149 
150  end subroutine compute_coupling_poro_ac
