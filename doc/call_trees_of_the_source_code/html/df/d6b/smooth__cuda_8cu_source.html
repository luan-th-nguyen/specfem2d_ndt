56 __global__ void process_smooth(realw_const_p xstore_me,realw_const_p zstore_me,realw_const_p xstore_other,realw_const_p zstore_other, realw_const_p data_other, const realw sigma_h2_inv, const realw sigma_v2_inv, const int iker, const int nspec_me, const int nspec_other, const realw v_criterion, const realw h_criterion, realw_const_p jacobian, realw_p sum_data_smooth, realw_p normalisation,realw_const_p wgll_sq){
57 
58  int ispec = blockIdx.x + gridDim.x*blockIdx.y;
59  int igll = threadIdx.x;
60  int gll_other;
61  realw x_me, z_me, x_other, z_other, coef, normalisation_slice;
62  realw dat;
63  __shared__ int sh_test[NGLL2];
64  __shared__ realw sh_x_other[NGLL2];
65  __shared__ realw sh_z_other[NGLL2];
66  __shared__ realw sh_jacobian[NGLL2];
67  __shared__ realw sh_wgll_sq[NGLL2];
68  __shared__ realw sh_data[NGLL2];
69 
70  int n_loop = nspec_other/NGLL2 + 1;
71  x_me = xstore_me[NGLL2*ispec + igll ];
72  z_me = zstore_me[NGLL2*ispec + igll ];
73  sh_wgll_sq[igll]=wgll_sq[igll];
74  __syncthreads();
75 
76  dat=0;
77  normalisation_slice=0;
78  //We test 32 spectral elements at a time
79  for (int i=0;i<n_loop;i++)
80  {
81  if (NGLL2*i + threadIdx.x < nspec_other){
82  x_other = xstore_other[i*NGLL2*NGLL2 + threadIdx.x*NGLL2 ];
83  z_other = zstore_other[i*NGLL2*NGLL2 + threadIdx.x*NGLL2 ];
84  }
85  sh_test[threadIdx.x] = ( NGLL2*i + threadIdx.x >= nspec_other || (x_me-x_other)*(x_me-x_other) > h_criterion || (z_me-z_other)*(z_me-z_other) > v_criterion ) ? 1 : 0 ;
86  __syncthreads();
87 
88  //loop over each spectral element tested
89  for (int k=0;k<NGLL2;k++)
90  {
91  if (sh_test[k]) continue ;
92 
93  //Load data from other slice to shared memory
94  sh_x_other[igll] = xstore_other[i*NGLL2*NGLL2 + k*NGLL2 + igll ];
95  sh_z_other[igll] = zstore_other[i*NGLL2*NGLL2 + k*NGLL2 + igll ];
96  sh_data[igll] = data_other[i*NGLL2*NGLL2 + k*NGLL2 + igll ];
97  sh_jacobian[igll] = jacobian[i*NGLL2*NGLL2 + k*NGLL2 + igll ];
98  __syncthreads();
99 
100  for (int j=0;j<NGLL2;j++){
101 
102  gll_other = (igll + j) % NGLL2;
103 
104  x_other = sh_x_other[gll_other];
105  z_other = sh_z_other[gll_other];
106  coef = expf(- sigma_h2_inv*(x_me-x_other)*(x_me-x_other) - sigma_v2_inv*(z_me-z_other)*(z_me-z_other))*sh_jacobian[gll_other]*sh_wgll_sq[gll_other];
107  normalisation_slice = normalisation_slice + coef;
108  dat += sh_data[gll_other]*coef;
109  } //loop on each gll_other
110  } //loop on each spec_other tested
111  } //loop on each serie of 32 spec_other
112 
113  sum_data_smooth[NGLL2*nspec_me*iker+NGLL2*ispec + igll] += dat;
114  normalisation[NGLL2*ispec + igll] += normalisation_slice;
115 }

119 __global__ void normalize_data(realw_p data_smooth, realw_const_p normalisation,int nker, int nspec_me){
120  int ispec = blockIdx.x + gridDim.x*blockIdx.y;
121  realw norm = normalisation[NGLL2*ispec + threadIdx.x];
122  for (int j=0;j<nker;j++) data_smooth[NGLL2*nspec_me*j + NGLL2*ispec + threadIdx.x] /= norm/nker;
123 }

127 extern "C"
128 void FC_FUNC_(prepare_arrays_gpu,
129  PREPARE_arrays_GPU)(long * Container,
130  realw * xstore_me,
131  realw * zstore_me,
132  realw * sigma_h2_inv,
133  realw * sigma_v2_inv,
134  realw * h_criterion,
135  realw * v_criterion,
136  int * nspec_me,
137  int * nker,
138  realw * wgll_sq){
139 
140  Smooth_data* sp = (Smooth_data*)malloc( sizeof(Smooth_data) );
141  *Container = (long)sp;
142 
143  copy_todevice_realw((void**)&sp->x_me,xstore_me, NGLL2*(*nspec_me));
144  copy_todevice_realw((void**)&sp->z_me,zstore_me, NGLL2*(*nspec_me));
145  copy_todevice_realw((void**)&sp->wgll_sq,wgll_sq, NGLL2);
146 
147  sp->sigma_h2_inv= *sigma_h2_inv;
148  sp->sigma_v2_inv= *sigma_v2_inv;
149  sp->h_criterion = *h_criterion;
150  sp->v_criterion = *v_criterion;
151  sp->nspec_me = *nspec_me;
152  sp->nker = *nker;
153 
154  print_CUDA_error_if_any(cudaMalloc((void**)&sp->data_smooth,NGLL2*(*nspec_me)*(*nker)*sizeof(realw)),2000);
155  print_CUDA_error_if_any(cudaMemset(sp->data_smooth,0,NGLL2*(*nspec_me)*(*nker)*sizeof(realw)),2001);
156 
157  print_CUDA_error_if_any(cudaMalloc((void**)&sp->normalisation,NGLL2*(*nspec_me)*sizeof(realw)),2002);
158  print_CUDA_error_if_any(cudaMemset(sp->normalisation,0,NGLL2*(*nspec_me)*sizeof(realw)),2003);
159 }

163 extern "C"
164 void FC_FUNC_(compute_smooth,
165  COMPUTE_SMOOTH)(long * smooth_pointer,
166  realw * jacobian,
167  realw * xstore_other,
168  realw * zstore_other,
169  realw * data_other,
170  const int * nspec_other){
171  realw * x_other;
172  realw * z_other;
173  realw * d_data_other;
174  realw * d_jacobian;
175 
176  Smooth_data * sp = (Smooth_data*)*smooth_pointer;
177 
178  copy_todevice_realw((void**)&x_other,xstore_other,NGLL2*(*nspec_other));
179  copy_todevice_realw((void**)&z_other,zstore_other,NGLL2*(*nspec_other));
180  copy_todevice_realw((void**)&d_jacobian,jacobian,NGLL2*(*nspec_other));
181 
182  dim3 grid(sp->nspec_me,1);
183  dim3 threads(NGLL2,1,1);
184 
185  for (int i=0;i<sp->nker;i++)
186  {
187  copy_todevice_realw((void**)&d_data_other,&data_other[NGLL2*(*nspec_other)*i],NGLL2*(*nspec_other));
188  process_smooth<<<grid,threads>>>(sp->x_me,
189  sp->z_me,
190  x_other,
191  z_other,
192  d_data_other,
193  sp->sigma_h2_inv,
194  sp->sigma_v2_inv,
195  i,
196  sp->nspec_me,
197  *nspec_other,
198  sp->v_criterion,
199  sp->h_criterion,
200  d_jacobian,
201  sp->data_smooth,
202  sp->normalisation,
203  sp->wgll_sq);
204  cudaFree(d_data_other);
205  }
206 
207  synchronize_cuda();
208  cudaFree(x_other);
209  cudaFree(z_other);
210  cudaFree(d_jacobian);
211 }

216 void FC_FUNC_(get_smooth,

218 
219  Smooth_data * sp = (Smooth_data*)*smooth_pointer;
220  dim3 grid(sp->nspec_me,1);
221  dim3 threads(NGLL2,1,1);
222 
223  normalize_data<<<grid,threads>>>(sp->data_smooth,sp->normalisation,sp->nker,sp->nspec_me);

226 
227  cudaFree(sp->x_me);
228  cudaFree(sp->z_me);
229  cudaFree(sp->data_smooth);

231  cudaFree(sp->normalisation);
