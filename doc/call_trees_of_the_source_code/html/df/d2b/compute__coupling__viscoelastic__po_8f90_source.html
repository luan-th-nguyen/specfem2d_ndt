33 
34 ! for viscoelastic solver
35 
36  subroutine compute_coupling_viscoelastic_po()
37 
38  use constants, only: custom_real,iright,ileft,ibottom,itop,ngllx,ngllz,two,zero
39 
40  use specfem_par, only: simulation_type,num_solid_poro_edges, &
41  ibool,wxgll,wzgll,xix,xiz,gammax,gammaz,jacobian,ivalue,jvalue,ivalue_inverse,jvalue_inverse, &
42  hprime_xx,hprime_zz, &
43  solid_poro_elastic_ispec,solid_poro_elastic_iedge, &
44  solid_poro_poroelastic_ispec,solid_poro_poroelastic_iedge, &
45  kmato,poroelastcoef, &
46  assign_external_model,c11ext,c13ext,c15ext,c33ext,c35ext,c55ext,c12ext,c23ext,c25ext,anisotropy, &
47  displ_elastic,b_displ_elastic,displs_poroelastic,displw_poroelastic, &
48  b_displs_poroelastic,b_displw_poroelastic, &
49  accel_elastic,b_accel_elastic
50 
51  implicit none
52 
53  !local variables
54  integer :: inum,ispec_elastic,iedge_elastic,ispec_poroelastic,iedge_poroelastic, &
55  i,j,k,ipoin1D,iglob,ii2,jj2
56 
57  double precision :: phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar
58  double precision :: D_biot,H_biot,C_biot,M_biot
59 
60  double precision :: c11,c13,c15,c33,c35,c55,c12,c23,c25
61  double precision :: mul_unrelaxed_elastic,lambdal_unrelaxed_elastic,lambdaplus2mu_unrelaxed_elastic
62 
63  real(kind=CUSTOM_REAL) :: mu_G,lambdal_G,lambdalplus2mul_G
64 
65  real(kind=CUSTOM_REAL) :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
66  real(kind=CUSTOM_REAL) :: dwx_dxi,dwx_dgamma,dwz_dxi,dwz_dgamma
67  real(kind=CUSTOM_REAL) :: dux_dxl,duz_dxl,dux_dzl,duz_dzl
68  real(kind=CUSTOM_REAL) :: dwx_dxl,dwz_dxl,dwx_dzl,dwz_dzl
69  real(kind=CUSTOM_REAL) :: b_dux_dxi,b_dux_dgamma,b_duz_dxi,b_duz_dgamma
70  real(kind=CUSTOM_REAL) :: b_dux_dxl,b_duz_dxl,b_dux_dzl,b_duz_dzl
71  real(kind=CUSTOM_REAL) :: b_dwx_dxi,b_dwx_dgamma,b_dwz_dxi,b_dwz_dgamma
72  real(kind=CUSTOM_REAL) :: b_dwx_dxl,b_dwz_dxl,b_dwx_dzl,b_dwz_dzl
73  real(kind=CUSTOM_REAL) :: sigma_xx,sigma_xz,sigma_zz
74  real(kind=CUSTOM_REAL) :: b_sigma_xx,b_sigma_xz,b_sigma_zz
75  real(kind=CUSTOM_REAL) :: xxi,zxi,xgamma,zgamma,jacobian1D,nx,nz,weight
76  real(kind=CUSTOM_REAL) :: xixl,xizl,gammaxl,gammazl
77 
78  ! loop on all the coupling edges
79  do inum = 1,num_solid_poro_edges
80 
81  ! get the edge of the elastic element
82  ispec_elastic = solid_poro_elastic_ispec(inum)
83  iedge_elastic = solid_poro_elastic_iedge(inum)
84 
85  ! get the corresponding edge of the poroelastic element
86  ispec_poroelastic = solid_poro_poroelastic_ispec(inum)
87  iedge_poroelastic = solid_poro_poroelastic_iedge(inum)
88 
89  ! implement 1D coupling along the edge
90  do ipoin1d = 1,ngllx
91 
92  ! get point values for the poroelastic side, which matches our side in the inverse direction
93  i = ivalue_inverse(ipoin1d,iedge_poroelastic)
94  j = jvalue_inverse(ipoin1d,iedge_poroelastic)
95  iglob = ibool(i,j,ispec_poroelastic)
96 
97  ! gets poroelastic material
98  call get_poroelastic_material(ispec_poroelastic,phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f, &
99  mu_fr,kappa_fr,rho_bar)
100 
101  ! Biot coefficients for the input phi
102  call get_poroelastic_biot_coeff(phi,kappa_s,kappa_f,kappa_fr,mu_fr,d_biot,h_biot,c_biot,m_biot)
103 
104  mu_g = mu_fr
105  lambdal_g = h_biot - 2._custom_real*mu_fr
106  lambdalplus2mul_g = lambdal_g + two*mu_g
107 
108  ! derivative along x and along z for u_s and w
109  dux_dxi = zero
110  duz_dxi = zero
111 
112  dux_dgamma = zero
113  duz_dgamma = zero
114 
115  dwx_dxi = zero
116  dwz_dxi = zero
117 
118  dwx_dgamma = zero
119  dwz_dgamma = zero
120 
121  if (simulation_type == 3) then
122  b_dux_dxi = zero
123  b_duz_dxi = zero
124 
125  b_dux_dgamma = zero
126  b_duz_dgamma = zero
127 
128  b_dwx_dxi = zero
129  b_dwz_dxi = zero
130 
131  b_dwx_dgamma = zero
132  b_dwz_dgamma = zero
133  endif
134 
135  ! first double loop over GLL points to compute and store gradients
136  ! we can merge the two loops because NGLLX == NGLLZ
137  do k = 1,ngllx
138  dux_dxi = dux_dxi + displs_poroelastic(1,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
139  duz_dxi = duz_dxi + displs_poroelastic(2,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
140  dux_dgamma = dux_dgamma + displs_poroelastic(1,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
141  duz_dgamma = duz_dgamma + displs_poroelastic(2,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
142 
143  dwx_dxi = dwx_dxi + displw_poroelastic(1,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
144  dwz_dxi = dwz_dxi + displw_poroelastic(2,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
145  dwx_dgamma = dwx_dgamma + displw_poroelastic(1,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
146  dwz_dgamma = dwz_dgamma + displw_poroelastic(2,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
147  if (simulation_type == 3) then
148  b_dux_dxi = b_dux_dxi + b_displs_poroelastic(1,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
149  b_duz_dxi = b_duz_dxi + b_displs_poroelastic(2,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
150  b_dux_dgamma = b_dux_dgamma + b_displs_poroelastic(1,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
151  b_duz_dgamma = b_duz_dgamma + b_displs_poroelastic(2,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
152 
153  b_dwx_dxi = b_dwx_dxi + b_displw_poroelastic(1,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
154  b_dwz_dxi = b_dwz_dxi + b_displw_poroelastic(2,ibool(k,j,ispec_poroelastic))*hprime_xx(i,k)
155  b_dwx_dgamma = b_dwx_dgamma + b_displw_poroelastic(1,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
156  b_dwz_dgamma = b_dwz_dgamma + b_displw_poroelastic(2,ibool(i,k,ispec_poroelastic))*hprime_zz(j,k)
157  endif
158  enddo
159 
160  xixl = xix(i,j,ispec_poroelastic)
161  xizl = xiz(i,j,ispec_poroelastic)
162  gammaxl = gammax(i,j,ispec_poroelastic)
163  gammazl = gammaz(i,j,ispec_poroelastic)
164 
165  ! derivatives of displacement
166  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
167  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
168 
169  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
170  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
171 
172  dwx_dxl = dwx_dxi*xixl + dwx_dgamma*gammaxl
173  dwx_dzl = dwx_dxi*xizl + dwx_dgamma*gammazl
174 
175  dwz_dxl = dwz_dxi*xixl + dwz_dgamma*gammaxl
176  dwz_dzl = dwz_dxi*xizl + dwz_dgamma*gammazl
177 
178  if (simulation_type == 3) then
179  b_dux_dxl = b_dux_dxi*xixl + b_dux_dgamma*gammaxl
180  b_dux_dzl = b_dux_dxi*xizl + b_dux_dgamma*gammazl
181 
182  b_duz_dxl = b_duz_dxi*xixl + b_duz_dgamma*gammaxl
183  b_duz_dzl = b_duz_dxi*xizl + b_duz_dgamma*gammazl
184 
185  b_dwx_dxl = b_dwx_dxi*xixl + b_dwx_dgamma*gammaxl
186  b_dwx_dzl = b_dwx_dxi*xizl + b_dwx_dgamma*gammazl
187 
188  b_dwz_dxl = b_dwz_dxi*xixl + b_dwz_dgamma*gammaxl
189  b_dwz_dzl = b_dwz_dxi*xizl + b_dwz_dgamma*gammazl
190  endif
191  ! compute stress tensor (include attenuation or anisotropy if needed)
192 
193  ! no attenuation
194  sigma_xx = lambdalplus2mul_g*dux_dxl + lambdal_g*duz_dzl + c_biot*(dwx_dxl + dwz_dzl)
195  sigma_xz = mu_g*(duz_dxl + dux_dzl)
196  sigma_zz = lambdalplus2mul_g*duz_dzl + lambdal_g*dux_dxl + c_biot*(dwx_dxl + dwz_dzl)
197 
198  if (simulation_type == 3) then
199  b_sigma_xx = lambdalplus2mul_g*b_dux_dxl + lambdal_g*b_duz_dzl + c_biot*(b_dwx_dxl + b_dwz_dzl)
200  b_sigma_xz = mu_g*(b_duz_dxl + b_dux_dzl)
201  b_sigma_zz = lambdalplus2mul_g*b_duz_dzl + lambdal_g*b_dux_dxl + c_biot*(b_dwx_dxl + b_dwz_dzl)
202  endif
203  ! get point values for the elastic domain, which matches our side in the inverse direction
204  ii2 = ivalue(ipoin1d,iedge_elastic)
205  jj2 = jvalue(ipoin1d,iedge_elastic)
206  iglob = ibool(ii2,jj2,ispec_elastic)
207 
208  ! get elastic properties
209  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec_elastic))
210  mul_unrelaxed_elastic = poroelastcoef(2,1,kmato(ispec_elastic))
211  lambdaplus2mu_unrelaxed_elastic = poroelastcoef(3,1,kmato(ispec_elastic))
212 
213  ! derivative along x and along z for u_s and w
214  dux_dxi = zero
215  duz_dxi = zero
216 
217  dux_dgamma = zero
218  duz_dgamma = zero
219 
220  if (simulation_type == 3) then
221  b_dux_dxi = zero
222  b_duz_dxi = zero
223 
224  b_dux_dgamma = zero
225  b_duz_dgamma = zero
226  endif
227 
228  ! first double loop over GLL points to compute and store gradients
229  ! we can merge the two loops because NGLLX == NGLLZ
230  do k = 1,ngllx
231  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,jj2,ispec_elastic))*hprime_xx(ii2,k)
232  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,jj2,ispec_elastic))*hprime_xx(ii2,k)
233  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(ii2,k,ispec_elastic))*hprime_zz(jj2,k)
234  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(ii2,k,ispec_elastic))*hprime_zz(jj2,k)
235 
236  if (simulation_type == 3) then
237  b_dux_dxi = b_dux_dxi + b_displ_elastic(1,ibool(k,jj2,ispec_elastic))*hprime_xx(ii2,k)
238  b_duz_dxi = b_duz_dxi + b_displ_elastic(2,ibool(k,jj2,ispec_elastic))*hprime_xx(ii2,k)
239  b_dux_dgamma = b_dux_dgamma + b_displ_elastic(1,ibool(ii2,k,ispec_elastic))*hprime_zz(jj2,k)
240  b_duz_dgamma = b_duz_dgamma + b_displ_elastic(2,ibool(ii2,k,ispec_elastic))*hprime_zz(jj2,k)
241  endif
242  enddo
243 
244  xixl = xix(ii2,jj2,ispec_elastic)
245  xizl = xiz(ii2,jj2,ispec_elastic)
246  gammaxl = gammax(ii2,jj2,ispec_elastic)
247  gammazl = gammaz(ii2,jj2,ispec_elastic)
248 
249  ! derivatives of displacement
250  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
251  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
252 
253  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
254  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
255 
256  if (simulation_type == 3) then
257  b_dux_dxl = b_dux_dxi*xixl + b_dux_dgamma*gammaxl
258  b_dux_dzl = b_dux_dxi*xizl + b_dux_dgamma*gammazl
259 
260  b_duz_dxl = b_duz_dxi*xixl + b_duz_dgamma*gammaxl
261  b_duz_dzl = b_duz_dxi*xizl + b_duz_dgamma*gammazl
262  endif
263  ! compute stress tensor
264  ! full anisotropy
265  if (kmato(ispec_elastic) == 2) then
266  ! implement anisotropy in 2D
267  if (assign_external_model) then
268  c11 = c11ext(ii2,jj2,ispec_elastic)
269  c13 = c13ext(ii2,jj2,ispec_elastic)
270  c15 = c15ext(ii2,jj2,ispec_elastic)
271  c33 = c33ext(ii2,jj2,ispec_elastic)
272  c35 = c35ext(ii2,jj2,ispec_elastic)
273  c55 = c55ext(ii2,jj2,ispec_elastic)
274  c12 = c12ext(ii2,jj2,ispec_elastic)
275  c23 = c23ext(ii2,jj2,ispec_elastic)
276  c25 = c25ext(ii2,jj2,ispec_elastic)
277  else
278  c11 = anisotropy(1,kmato(ispec_elastic))
279  c13 = anisotropy(2,kmato(ispec_elastic))
280  c15 = anisotropy(3,kmato(ispec_elastic))
281  c33 = anisotropy(4,kmato(ispec_elastic))
282  c35 = anisotropy(5,kmato(ispec_elastic))
283  c55 = anisotropy(6,kmato(ispec_elastic))
284  c12 = anisotropy(7,kmato(ispec_elastic))
285  c23 = anisotropy(8,kmato(ispec_elastic))
286  c25 = anisotropy(9,kmato(ispec_elastic))
287  endif
288 
289  sigma_xx = sigma_xx + c11*dux_dxl + c15*(duz_dxl + dux_dzl) + c13*duz_dzl
290  sigma_zz = sigma_zz + c13*dux_dxl + c35*(duz_dxl + dux_dzl) + c33*duz_dzl
291  sigma_xz = sigma_xz + c15*dux_dxl + c55*(duz_dxl + dux_dzl) + c35*duz_dzl
292  else
293  ! no attenuation
294  sigma_xx = sigma_xx + lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl
295  sigma_xz = sigma_xz + mul_unrelaxed_elastic*(duz_dxl + dux_dzl)
296  sigma_zz = sigma_zz + lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl
297  endif
298 
299  if (simulation_type == 3) then
300  b_sigma_xx = b_sigma_xx + lambdaplus2mu_unrelaxed_elastic*b_dux_dxl + lambdal_unrelaxed_elastic*b_duz_dzl
301  b_sigma_xz = b_sigma_xz + mul_unrelaxed_elastic*(b_duz_dxl + b_dux_dzl)
302  b_sigma_zz = b_sigma_zz + lambdaplus2mu_unrelaxed_elastic*b_duz_dzl + lambdal_unrelaxed_elastic*b_dux_dxl
303  endif
304 
305  ! compute the 1D Jacobian and the normal to the edge: for their expression see for instance
306  ! O. C. Zienkiewicz and R. L. Taylor, The Finite Element Method for Solid and Structural Mechanics,
307  ! Sixth Edition, electronic version, www.amazon.com, p. 204 and Figure 7.7(a),
308  ! or Y. K. Cheung, S. H. Lo and A. Y. T. Leung, Finite Element Implementation,
309  ! Blackwell Science, page 110, equation (4.60).
310  if (iedge_poroelastic == itop) then
311  xxi = + gammaz(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
312  zxi = - gammax(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
313  jacobian1d = sqrt(xxi**2 + zxi**2)
314  nx = - zxi / jacobian1d
315  nz = + xxi / jacobian1d
316  weight = jacobian1d * wxgll(i)
317  else if (iedge_poroelastic == ibottom) then
318  xxi = + gammaz(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
319  zxi = - gammax(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
320  jacobian1d = sqrt(xxi**2 + zxi**2)
321  nx = + zxi / jacobian1d
322  nz = - xxi / jacobian1d
323  weight = jacobian1d * wxgll(i)
324  else if (iedge_poroelastic == ileft) then
325  xgamma = - xiz(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
326  zgamma = + xix(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
327  jacobian1d = sqrt(xgamma**2 + zgamma**2)
328  nx = - zgamma / jacobian1d
329  nz = + xgamma / jacobian1d
330  weight = jacobian1d * wzgll(j)
331  else if (iedge_poroelastic == iright) then
332  xgamma = - xiz(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
333  zgamma = + xix(i,j,ispec_poroelastic) * jacobian(i,j,ispec_poroelastic)
334  jacobian1d = sqrt(xgamma**2 + zgamma**2)
335  nx = + zgamma / jacobian1d
336  nz = - xgamma / jacobian1d
337  weight = jacobian1d * wzgll(j)
338  endif
339 
340  accel_elastic(1,iglob) = accel_elastic(1,iglob) - weight * (sigma_xx*nx + sigma_xz*nz)/2.d0
341  accel_elastic(2,iglob) = accel_elastic(2,iglob) - weight * (sigma_xz*nx + sigma_zz*nz)/2.d0
342 
343  if (simulation_type == 3) then
344  b_accel_elastic(1,iglob) = b_accel_elastic(1,iglob) - weight * (b_sigma_xx*nx + b_sigma_xz*nz)/2.d0
345  b_accel_elastic(2,iglob) = b_accel_elastic(2,iglob) - weight * (b_sigma_xz*nx + b_sigma_zz*nz)/2.d0
346  endif !if (SIMULATION_TYPE == 3) then
347 
348  enddo
349  enddo
350 
351  end subroutine compute_coupling_viscoelastic_po

358 
359  subroutine compute_coupling_viscoelastic_po_backward()
360 
361  use constants, only: custom_real,iright,ileft,ibottom,itop,ngllx,ngllz,two,zero
362 
363  use specfem_par, only: simulation_type,num_solid_poro_edges, &
364  ibool,wxgll,wzgll,xix,xiz,gammax,gammaz,jacobian,ivalue,jvalue,ivalue_inverse,jvalue_inverse, &
365  hprime_xx,hprime_zz, &
366  solid_poro_elastic_ispec,solid_poro_elastic_iedge, &
367  solid_poro_poroelastic_ispec,solid_poro_poroelastic_iedge, &
368  kmato,poroelastcoef, &
369  assign_external_model,c11ext,c13ext,c15ext,c33ext,c35ext,c55ext,c12ext,c23ext,c25ext,anisotropy, &
370  displ_elastic,b_displ_elastic,displs_poroelastic,displw_poroelastic, &
371  b_displs_poroelastic,b_displw_poroelastic, &
372  accel_elastic,b_accel_elastic
373 
374  implicit none
375 
376  !local variables
377  integer :: inum,ispec_elastic,iedge_elastic,ispec_poroelastic,iedge_poroelastic, &
378  i,j,k,ipoin1D,iglob,ii2,jj2
379 
380  double precision :: phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar
381  double precision :: D_biot,H_biot,C_biot,M_biot
382 
383  double precision :: c11,c13,c15,c33,c35,c55,c12,c23,c25
384  double precision :: mul_unrelaxed_elastic,lambdal_unrelaxed_elastic,lambdaplus2mu_unrelaxed_elastic
385 
386  real(kind=CUSTOM_REAL) :: mu_G,lambdal_G,lambdalplus2mul_G
387 
388  real(kind=CUSTOM_REAL) :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
389  real(kind=CUSTOM_REAL) :: dwx_dxi,dwx_dgamma,dwz_dxi,dwz_dgamma
390  real(kind=CUSTOM_REAL) :: dux_dxl,duz_dxl,dux_dzl,duz_dzl
