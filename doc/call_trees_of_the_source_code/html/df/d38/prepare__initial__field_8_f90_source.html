34 
35  subroutine prepare_initial_field(cploc,csloc)
36 
37  use constants, only: imain,pi,smallvaltol
38 
39  use specfem_par, only: myrank,any_acoustic,any_poroelastic,over_critical_angle, &
40  nsources,source_type,anglesource,x_source,z_source,f0_source,t0, &
41  nglob,numat,poroelastcoef,density,coord, &
42  anglesource_refl,c_inc,c_refl,time_offset, &
43  a_plane, b_plane, c_plane, &
44  accel_elastic,veloc_elastic,displ_elastic,myrank
45 
46  implicit none
47 
48  double precision,intent(out) :: cploc, csloc
49 
50  ! local parameters
51  integer :: numat_local,i
52  double precision :: denst,lambdaplus2mu,mu,p,x0_source,z0_source
53  double precision :: PP,PS,SP,SS,anglesource_abs
54  double precision :: xmax, xmin, zmax, zmin,x,z,t
55  double precision :: xmax_glob, xmin_glob, zmax_glob, zmin_glob
56 
57  double precision, external :: ricker_Bielak_displ,ricker_Bielak_veloc,ricker_Bielak_accel
58 
59  ! user output
60  if (myrank == 0) then
61  write(imain,*)
62  !! DK DK reading of an initial field from an external file has been suppressed
63  !! DK DK and replaced with the implementation of an analytical plane wave
64  !! DK DK write(IMAIN,*) 'Reading initial fields from external file...'
65  write(imain,*) 'Implementing an analytical initial plane wave...'
66  write(imain,*)
67  call flush_imain()
68  endif
69 
70  ! safety check
71  if (any_acoustic .or. any_poroelastic) &
72  call exit_mpi(myrank,'initial field currently implemented for purely elastic simulation only')
73 
74  !=======================================================================
75  !
76  ! Calculation of the initial field for a plane wave
77  !
78  !=======================================================================
79 
80  if (myrank == 0) then
81  write(imain,*) 'Number of grid points: ',nglob
82  write(imain,*)
83  write(imain,*) '*** calculation of the initial plane wave ***'
84  write(imain,*)
85  write(imain,*) 'To change the initial plane wave, change source_type in DATA/SOURCE'
86  write(imain,*) 'and use 1 or 4 for a plane P wave, 2 or 5 for a plane SV wave, 3 for a Rayleigh wave'
87  write(imain,*)
88 
89  ! only implemented for one source
90  if (nsources > 1) call exit_mpi(myrank,'calculation of the initial wave is only implemented for one source')
91  if (source_type(1) == 1 .or. source_type(1) == 4) then
92  write(imain,*) 'initial P wave of', anglesource(1)*180.d0/pi, 'degrees introduced.'
93  else if (source_type(1) == 2 .or. source_type(1) == 5) then
94  write(imain,*) 'initial SV wave of', anglesource(1)*180.d0/pi, ' degrees introduced.'
95  else if (source_type(1) == 3) then
96  write(imain,*) 'Rayleigh wave introduced.'
97  else
98  call exit_mpi(myrank, &
99  'Unrecognized source_type: should be 1 or 4 for plane P waves, 2 or 5 for plane SV waves, 3 for Rayleigh wave')
100  endif
101  endif
102 
103  ! allow negative anglesource(1): incidence from the right side of the domain
104  ! anglesource has been converted from degrees to radians before
105  anglesource_abs = abs(anglesource(1))
106  if (anglesource_abs > pi/2.d0 .and. source_type(1) /= 3) &
107  call exit_mpi(myrank,"incorrect anglesource: must have 0 <= anglesource < 90")
108 
109  ! only implemented for homogeneous media therefore only one material supported
110  numat_local = numat
111  if (numat /= 1) then
112  if (myrank == 0) then
113  write(imain,*)
114  write(imain,*) 'It is not possible to have several materials with a plane wave, thus using the first material.'
115  write(imain,*) 'This is not a homogenous model, it contains ',numat,' materials'
116  write(imain,*) 'but the plane wave initial and boundary fields'
117  write(imain,*) 'are computed by analytical formulas for a homogenous model.'
118  write(imain,*) 'Thus use at your own risk!'
119  write(imain,*)
120  endif
121  numat_local = 1
122  endif
123 
124  mu = poroelastcoef(2,1,numat_local)
125  lambdaplus2mu = poroelastcoef(3,1,numat_local)
126  denst = density(1,numat_local)
127 
128  cploc = sqrt(lambdaplus2mu/denst)
129  csloc = sqrt(mu/denst)
130 
131  ! P wave case
132  if (source_type(1) == 1 .or. source_type(1) == 4) then
133 
134  p = sin(anglesource_abs)/cploc
135  c_inc = cploc
136  c_refl = csloc
137 
138  anglesource_refl = asin(p*c_refl)
139 
140  ! from formulas (5.27) and (5.28) p 134 in Aki & Richards (2002)
141  pp = (- cos(2.d0*anglesource_refl)**2/csloc**3 &
142  + 4.d0*p**2*cos(anglesource_abs)*cos(anglesource_refl)/cploc) / &
143  ( cos(2.d0*anglesource_refl)**2/csloc**3 &
144  + 4.d0*p**2*cos(anglesource_abs)*cos(anglesource_refl)/cploc)
145 
146  ps = 4.d0*p*cos(anglesource_abs)*cos(2.d0*anglesource_refl) / &
147  (csloc**2*(cos(2.d0*anglesource_refl)**2/csloc**3 &
148  +4.d0*p**2*cos(anglesource_abs)*cos(anglesource_refl)/cploc))
149 
150  if (myrank == 0) then
151  write(imain,*) 'reflected convert plane wave angle: ', anglesource_refl*180.d0/pi
152  endif
153 
154  ! from Table 5.1 p141 in Aki & Richards (1980)
155  ! we put the opposite sign on z coefficients because z axis is oriented from bottom to top
156  a_plane(1) = sin(anglesource_abs); a_plane(2) = cos(anglesource_abs)
157  b_plane(1) = pp * sin(anglesource_abs); b_plane(2) = - pp * cos(anglesource_abs)
158  c_plane(1) = ps * cos(anglesource_refl); c_plane(2) = ps * sin(anglesource_refl)
159 
160  ! SV wave case
161  else if (source_type(1) == 2 .or. source_type(1) == 5) then
162 
163  p=sin(anglesource_abs)/csloc
164  c_inc = csloc
165  c_refl = cploc
166 
167  ! if this coefficient is greater than 1, we are beyond the critical SV wave angle and there cannot be a converted P wave
168  if (p*c_refl <= 1.d0) then
169  anglesource_refl = asin(p*c_refl)
170 
171  ! from formulas (5.30) and (5.31) p 140 in Aki & Richards (1980)
172  ss = (cos(2.d0*anglesource_abs)**2/csloc**3 &
173  - 4.d0*p**2*cos(anglesource_abs)*cos(anglesource_refl)/cploc) / &
174  (cos(2.d0*anglesource_abs)**2/csloc**3 &
175  + 4.d0*p**2*cos(anglesource_abs)*cos(anglesource_refl)/cploc)
176  sp = 4.d0*p*cos(anglesource_abs)*cos(2*anglesource_abs) / &
177  (cploc*csloc*(cos(2.d0*anglesource_abs)**2/csloc**3&
178  +4.d0*p**2*cos(anglesource_refl)*cos(anglesource_abs)/cploc))
179 
180  if (myrank == 0) then
181  write(imain,*) 'reflected convert plane wave angle: ', anglesource_refl*180.d0/pi
182  endif
183 
184  ! SV45 degree incident plane wave is a particular case
185  else if (anglesource_abs > pi/4.d0-1.0d-11 .and. anglesource_abs < pi/4.d0+1.0d-11) then
186  anglesource_refl = 0.d0
187  ss = -1.0d0
188  sp = 0.d0
189  else
190  over_critical_angle=.true.
191  anglesource_refl = 0.d0
192  ss = 0.0d0
193  sp = 0.d0
194  endif
195 
196  ! from Table 5.1 p141 in Aki & Richards (1980)
197  ! we put the opposite sign on z coefficients because z axis is oriented from bottom to top
198  a_plane(1) = cos(anglesource_abs); a_plane(2) = - sin(anglesource_abs)
199  b_plane(1) = ss * cos(anglesource_abs); b_plane(2) = ss * sin(anglesource_abs)
200  c_plane(1) = sp * sin(anglesource_refl); c_plane(2) = - sp * cos(anglesource_refl)
201 
202  ! Rayleigh case
203  else if (source_type(1) == 3) then
204  over_critical_angle=.true.
205  a_plane(1)=0.d0; a_plane(2)=0.d0
206  b_plane(1)=0.d0; b_plane(2)=0.d0
207  c_plane(1)=0.d0; c_plane(2)=0.d0
208  endif
209 
210  ! correct A_plane, B_plane and C_plane according to incident direction
211  if (anglesource(1) < 0.) then
212  a_plane(1)=-a_plane(1)
213  b_plane(1)=-b_plane(1)
214  c_plane(1)=-c_plane(1)
215  endif
216 
217 ! to suppress the reflected and converted plane wave fields
218  if (source_type(1) == 4 .or. source_type(1) == 5) then
219  b_plane(:) = 0
220  c_plane(:) = 0
221  endif
222 
223  ! get minimum and maximum values of mesh coordinates
224  xmin = minval(coord(1,:))
225  zmin = minval(coord(2,:))
226  xmax = maxval(coord(1,:))
227  zmax = maxval(coord(2,:))
228 
229  ! collects min/max on all slices
230  call min_all_all_dp(xmin, xmin_glob)
231  call max_all_all_dp(xmax, xmax_glob)
232  call min_all_all_dp(zmin, zmin_glob)
233  call max_all_all_dp(zmax, zmax_glob)
234 
235  xmin = xmin_glob
236  zmin = zmin_glob
237  xmax = xmax_glob
238  zmax = zmax_glob
239 
240  ! check if zs = zmax (free surface)
241  if (myrank == 0 .and. abs(z_source(1)-zmax) > smallvaltol) then
242  print *, 'It is sometimes easier to set zs in SOURCE = zmax in interfacefile to keep track of the initial wavefront'
243  endif
244 
245  ! add -t0 to match the actual (analytical) traveltime of plane waves
246  time_offset = 0.d0 - t0
247 
248  ! to correctly center the initial plane wave in the mesh
249  x0_source = x_source(1)
250  z0_source = z_source(1)
251 
252  if (myrank == 0) then
253  write(imain,*)
254  write(imain,*) 'You can modify the location of the initial plane wave by changing xs and zs in DATA/SOURCE.'
255  write(imain,*) ' for instance: xs=',x_source(1),' zs=',z_source(1), ' (zs can/should be the height of the free surface)'
256  write(imain,*)
257  call flush_imain()
258  endif
259 
260  if (.not. over_critical_angle) then
261 
262  do i = 1,nglob
263 
264  x = coord(1,i)
265  z = coord(2,i)
266 
267  ! z is from bottom to top therefore we take -z to make parallel with Aki & Richards
268 
269  z = z0_source - z
270  if (anglesource(1) >= 0.) then
271  x = x - x0_source
272  else
273  x = x0_source -x
274  endif
275 
276  t = 0.d0 + time_offset
277 
278  ! formulas for the initial displacement for a plane wave from Aki & Richards (1980)
279  displ_elastic(1,i) = &
280  a_plane(1) * ricker_bielak_displ(t - sin(anglesource_abs)*x/c_inc + cos(anglesource_abs)*z/c_inc,f0_source(1)) &
281  + b_plane(1) * ricker_bielak_displ(t - sin(anglesource_abs)*x/c_inc - cos(anglesource_abs)*z/c_inc,f0_source(1)) &
282  + c_plane(1) * ricker_bielak_displ(t - sin(anglesource_refl)*x/c_refl - cos(anglesource_refl)*z/c_refl,f0_source(1))
283  displ_elastic(2,i) = &
284  a_plane(2) * ricker_bielak_displ(t - sin(anglesource_abs)*x/c_inc + cos(anglesource_abs)*z/c_inc,f0_source(1)) &
285  + b_plane(2) * ricker_bielak_displ(t - sin(anglesource_abs)*x/c_inc - cos(anglesource_abs)*z/c_inc,f0_source(1)) &
286  + c_plane(2) * ricker_bielak_displ(t - sin(anglesource_refl)*x/c_refl - cos(anglesource_refl)*z/c_refl,f0_source(1))
287 
288  ! formulas for the initial velocity for a plane wave (first derivative in time of the displacement)
289  veloc_elastic(1,i) = &
290  a_plane(1) * ricker_bielak_veloc(t - sin(anglesource_abs)*x/c_inc + cos(anglesource_abs)*z/c_inc,f0_source(1)) &
291  + b_plane(1) * ricker_bielak_veloc(t - sin(anglesource_abs)*x/c_inc - cos(anglesource_abs)*z/c_inc,f0_source(1)) &
292  + c_plane(1) * ricker_bielak_veloc(t - sin(anglesource_refl)*x/c_refl - cos(anglesource_refl)*z/c_refl,f0_source(1))
293  veloc_elastic(2,i) = &
294  a_plane(2) * ricker_bielak_veloc(t - sin(anglesource_abs)*x/c_inc + cos(anglesource_abs)*z/c_inc,f0_source(1)) &
295  + b_plane(2) * ricker_bielak_veloc(t - sin(anglesource_abs)*x/c_inc - cos(anglesource_abs)*z/c_inc,f0_source(1)) &
296  + c_plane(2) * ricker_bielak_veloc(t - sin(anglesource_refl)*x/c_refl - cos(anglesource_refl)*z/c_refl,f0_source(1))
297 
298  ! formulas for the initial acceleration for a plane wave (second derivative in time of the displacement)
299  accel_elastic(1,i) = &
300  a_plane(1) * ricker_bielak_accel(t - sin(anglesource_abs)*x/c_inc + cos(anglesource_abs)*z/c_inc,f0_source(1)) &
301  + b_plane(1) * ricker_bielak_accel(t - sin(anglesource_abs)*x/c_inc - cos(anglesource_abs)*z/c_inc,f0_source(1)) &
302  + c_plane(1) * ricker_bielak_accel(t - sin(anglesource_refl)*x/c_refl - cos(anglesource_refl)*z/c_refl,f0_source(1))
303  accel_elastic(2,i) = &
304  a_plane(2) * ricker_bielak_accel(t - sin(anglesource_abs)*x/c_inc + cos(anglesource_abs)*z/c_inc,f0_source(1)) &
305  + b_plane(2) * ricker_bielak_accel(t - sin(anglesource_abs)*x/c_inc - cos(anglesource_abs)*z/c_inc,f0_source(1)) &
306  + c_plane(2) * ricker_bielak_accel(t - sin(anglesource_refl)*x/c_refl - cos(anglesource_refl)*z/c_refl,f0_source(1))
307 
308  enddo
309 
310 endif
311 
312 end subroutine prepare_initial_field

318  subroutine prepare_initial_field_paco()
319 
320  use constants, only: imain,iedge1,iedge2,iedge4,ngllx,ngllz,pi
321 
322  use specfem_par, only: myrank,nelemabs,left_bound,right_bound,bot_bound, &
323  numabs,codeabs,ibool, &
324  source_type,c_inc,c_refl, &
325  count_bottom,count_left,count_right
326 
327  implicit none
328 
329  ! local parameters
330  integer :: ispecabs,ispec,i,j,iglob,ibegin,iend
331 
332  if (myrank == 0) then
333  if (source_type(1) /= 3 ) &
334  write(imain,*) 'You are beyond the critical angle ( > ',asin(c_inc/c_refl)*180d0/pi,')'
335 
336  write(imain,*) '*************'
337  write(imain,*) 'We have to compute the initial field in the frequency domain'
338  write(imain,*) 'and then convert it to the time domain (can be long... be patient...)'
339  write(imain,*) '*************'
340  call flush_imain()
341  endif
342 
343  count_bottom=0
344  count_left=0
345  count_right=0
346  do ispecabs= 1,nelemabs
347  ispec=numabs(ispecabs)
348  if (codeabs(iedge4,ispecabs)) then
349  i = 1
350  do j = 1,ngllz
351  count_left=count_left+1
352  iglob = ibool(i,j,ispec)
353  left_bound(count_left)=iglob
354  enddo
355  endif
356  if (codeabs(iedge2,ispecabs)) then
357  i = ngllx
358  do j = 1,ngllz
359  count_right=count_right+1
360  iglob = ibool(i,j,ispec)
361  right_bound(count_right)=iglob
362  enddo
363  endif
364  if (codeabs(iedge1,ispecabs)) then
365  j = 1
366 !! DK DK not needed ! exclude corners to make sure there is no contradiction regarding the normal
367  ibegin = 1
368  iend = ngllx
369 !! DK DK not needed if (codeabs(IEDGE4,ispecabs)) ibegin = 2
370 !! DK DK not needed if (codeabs(IEDGE2,ispecabs)) iend = NGLLX-1
371  do i = ibegin,iend
372  count_bottom=count_bottom+1
373  iglob = ibool(i,j,ispec)
374  bot_bound(count_bottom)=iglob
375  enddo
376  endif
377  enddo
378 
379  end subroutine prepare_initial_field_paco
