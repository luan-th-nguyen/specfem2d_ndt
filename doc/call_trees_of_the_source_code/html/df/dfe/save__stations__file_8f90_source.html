34  subroutine save_stations_file(nreceiversets,nrec_line,xdeb,zdeb,xfin,zfin,record_at_surface_same_vertical, &
35  xinterface_top,zinterface_top,coefs_interface_top, &
36  npoints_interface_top,max_npoints_interface)
37 
38  use constants, only: iout,imain,max_string_len,mygroup,in_data_files
39  use shared_parameters, only: number_of_simultaneous_runs
40 
41  implicit none
42 
43  integer :: nreceiversets
44  integer, dimension(nreceiversets) :: nrec_line
45  double precision, dimension(nreceiversets) :: xdeb,zdeb,xfin,zfin
46  logical, dimension(nreceiversets) :: record_at_surface_same_vertical
47 
48  integer :: max_npoints_interface
49  double precision, dimension(max_npoints_interface) :: xinterface_top, &
50  zinterface_top,coefs_interface_top
51  integer :: npoints_interface_top
52 
53  !local parameters
54  integer :: ireceiverlines,irec,irec_global_number,ios
55  integer :: nrec_total
56  double precision :: xrec,zrec
57  double precision, external :: value_spline
58  character(len=MAX_STRING_LEN) :: stations_filename,path_to_add
59 
60  stations_filename = trim(in_data_files)//'STATIONS'
61 
62  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
63  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
64  stations_filename = path_to_add(1:len_trim(path_to_add))//stations_filename(1:len_trim(stations_filename))
65  endif
66 
67  ! user output
68  write(imain,*)
69  write(imain,*) 'writing the '//trim(stations_filename)//' file'
70  write(imain,*)
71 
72  ! total number of receivers in all the receiver lines
73  nrec_total = sum(nrec_line(:))
74 
75  write(imain,*)
76  write(imain,*) 'There are ',nrec_total,' receivers'
77 
78  write(imain,*)
79  write(imain,*) 'Target positions (x,z) of the ',nrec_total,' receivers'
80  write(imain,*)
81 
82  open(unit=iout,file=trim(stations_filename),status='unknown',iostat=ios)
83  if (ios /= 0 ) stop 'error saving STATIONS file'
84 
85  irec_global_number = 0
86 
87  ! loop on all the receiver lines
88  do ireceiverlines = 1,nreceiversets
89 
90  ! loop on all the receivers of this receiver line
91  do irec = 1,nrec_line(ireceiverlines)
92 
93  ! compute global receiver number
94  irec_global_number = irec_global_number + 1
95 
96  ! compute coordinates of the receiver
97  if (nrec_line(ireceiverlines) > 1) then
98  xrec = xdeb(ireceiverlines) + dble(irec-1)*(xfin(ireceiverlines) &
99  -xdeb(ireceiverlines))/dble(nrec_line(ireceiverlines)-1)
100  zrec = zdeb(ireceiverlines) + dble(irec-1)*(zfin(ireceiverlines) &
101  -zdeb(ireceiverlines))/dble(nrec_line(ireceiverlines)-1)
102  else
103  xrec = xdeb(ireceiverlines)
104  zrec = zdeb(ireceiverlines)
105  endif
106 
107  ! modify position of receiver if we must record exactly at the surface
108  if (record_at_surface_same_vertical(ireceiverlines)) &
109  zrec = value_spline(xrec,xinterface_top,zinterface_top,coefs_interface_top,npoints_interface_top)
110 
111  ! display position of the receiver
112  write(imain,*) 'Receiver ',irec_global_number,' = ',xrec,zrec
113 
114  write(iout,"('S',i4.4,' AA ',f20.7,1x,f20.7,' 0.0 0.0')") irec_global_number,xrec,zrec
115 
116  enddo
117  enddo
118 
119  close(iout)
120 
121  end subroutine save_stations_file
