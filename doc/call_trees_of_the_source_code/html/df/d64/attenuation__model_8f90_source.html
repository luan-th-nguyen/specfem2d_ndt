34  subroutine attenuation_model(QKappa_att,QMu_att,f0_attenuation,N_SLS, &
35  tau_epsilon_nu1_sent,inv_tau_sigma_nu1_sent,phi_nu1_sent,Mu_nu1_sent, &
36  tau_epsilon_nu2_sent,inv_tau_sigma_nu2_sent,phi_nu2_sent,Mu_nu2_sent)
37 
38 ! define the attenuation constants
39 
40  use constants, only: custom_real,one
41 
42  implicit none
43 
44  double precision,intent(in) :: QKappa_att,QMu_att
45  double precision,intent(in) :: f0_attenuation
46 
47  integer,intent(in) :: N_SLS
48  real(kind=CUSTOM_REAL), dimension(N_SLS),intent(out) :: inv_tau_sigma_nu1_sent,phi_nu1_sent
49  real(kind=CUSTOM_REAL), dimension(N_SLS),intent(out) :: inv_tau_sigma_nu2_sent,phi_nu2_sent
50  real(kind=CUSTOM_REAL), dimension(N_SLS),intent(out) :: tau_epsilon_nu1_sent,tau_epsilon_nu2_sent
51  real(kind=CUSTOM_REAL),intent(out) :: Mu_nu1_sent,Mu_nu2_sent
52 
53  ! local parameters
54  double precision, dimension(N_SLS) :: tau_epsilon_nu1_d,tau_epsilon_nu2_d
55  double precision, dimension(N_SLS) :: tau_sigma_nu1,tau_sigma_nu2
56  double precision :: f_min_attenuation, f_max_attenuation
57 
58  ! safety check
59  if (n_sls < 1) stop 'Invalid N_SLS value, must be at least 1'
60 
61 ! attenuation constants for standard linear solids
62 ! nu1 is the dilatation/incompressibility mode (QKappa)
63 ! nu2 is the shear mode (Qmu)
64 ! array index (1) is the first standard linear solid, (2) is the second etc.
65 
66 ! use a wide bandwidth (always OK when using three or more Standard Linear Solids, can be a bit inaccurate if using only two)
67  f_min_attenuation = f0_attenuation / 10.d0
68  f_max_attenuation = f0_attenuation * 10.d0
69 
70  call compute_attenuation_coeffs(n_sls,qkappa_att,f0_attenuation,f_min_attenuation,f_max_attenuation, &
71  tau_epsilon_nu1_d,tau_sigma_nu1)
72 
73  call compute_attenuation_coeffs(n_sls,qmu_att,f0_attenuation,f_min_attenuation,f_max_attenuation, &
74  tau_epsilon_nu2_d,tau_sigma_nu2)
75 
76 ! print *
77 ! print *,'N_SLS, QKappa_att, QMu_att = ',N_SLS, QKappa_att, QMu_att
78 ! print *,'f0_attenuation,f_min_attenuation,f_max_attenuation = ',f0_attenuation,f_min_attenuation,f_max_attenuation
79 ! print *,'tau_epsilon_nu1 = ',tau_epsilon_nu1_d
80 ! print *,'tau_sigma_nu1 = ',tau_sigma_nu1
81 ! print *,'tau_epsilon_nu2 = ',tau_epsilon_nu2_d
82 ! print *,'tau_sigma_nu2 = ',tau_sigma_nu2
83 ! print *
84 
85  if (any(tau_sigma_nu1 < 0.d0) .or. any(tau_sigma_nu2 < 0.d0) .or. &
86  any(tau_epsilon_nu1_d < 0.d0) .or. any(tau_epsilon_nu2_d < 0.d0)) &
87  stop 'error: negative relaxation time found for a viscoelastic material'
88 
89 ! in the old formulation of Carcione 1993, which is based on Liu et al. 1976, the 1/N factor is missing
90 ! and thus this does not apply; it only applies to the right formula with 1/N included
91  if (minval(tau_epsilon_nu1_d / tau_sigma_nu1) < 0.999d0 .or. &
92  minval(tau_epsilon_nu2_d / tau_sigma_nu2) < 0.999d0) then
93  print *
94  print *,'*******************************************************************************'
95  print *,'*******************************************************************************'
96  print *,'*******************************************************************************'
97  print *,'*******************************************************************************'
98  print *,'*******************************************************************************'
99  print *,'minval(tau_epsilon_nu1 / tau_sigma_nu1) = ',minval(tau_epsilon_nu1_d / tau_sigma_nu1)
100  print *,'minval(tau_epsilon_nu2 / tau_sigma_nu2) = ',minval(tau_epsilon_nu2_d / tau_sigma_nu2)
101  print *,'WARNING: tau_epsilon should never be smaller than tau_sigma for viscoelasticity'
102  print *,'*******************************************************************************'
103  print *,'*******************************************************************************'
104  print *,'*******************************************************************************'
105  print *,'*******************************************************************************'
106  print *,'*******************************************************************************'
107  endif

109 ! When implementing viscoelasticity according to the Carcione 1993 paper, attenuation is
110 ! non-causal rather than causal i.e. wave speed up instead of slowing down
111 ! when attenuation is turned on. We fixed that issue (which is not incorrect but non traditional)
112 ! by taking the unrelaxed state (infinite frequency) as a reference instead of the relaxed state (zero frequency)
113 ! and also using equations in Carcione's 2007 book.
114 ! See file doc/old_problem_attenuation_reference_Specfem2D_fixed_by_Xie_Zhinan.pdf
115 ! and doc/how_we_modified_Carcione_1993_to_make_it_causal_and_include_the_missing_1_over_L_factor.pdf
116 
117 ! See also J. M. Carcione, Seismic modeling in viscoelastic media, Geophysics,
118 ! vol. 58(1), p. 110-120 (1993) for two memory-variable mechanisms (page 112).
119 
120 ! and J. M. Carcione, D. Kosloff and R. Kosloff, Wave propagation simulation
121 ! in a linear viscoelastic medium, Geophysical Journal International,
122 ! vol. 95, p. 597-611 (1988) for two memory-variable mechanisms (page 604).
123 
124 !
125 !--- other constants computed from the parameters above, do not modify
126 !
127 
128  ! SLS parameters
129  ! for Qkappa (set by tau**1,phi**1,Mu_nu1**) and Qmu (set by tau**2,phi**2,Mu_nu2**)
130  tau_epsilon_nu1_sent(:) = real(tau_epsilon_nu1_d(:),kind=custom_real)
131  tau_epsilon_nu2_sent(:) = real(tau_epsilon_nu2_d(:),kind=custom_real)
132 
133  inv_tau_sigma_nu1_sent(:) = real(1.d0 / tau_sigma_nu1(:),kind=custom_real)
134  inv_tau_sigma_nu2_sent(:) = real(1.d0 / tau_sigma_nu2(:),kind=custom_real)
135 
136  ! use the right formula with 1/N included
137  phi_nu1_sent(:) = real((1.d0 - tau_epsilon_nu1_d(:)/tau_sigma_nu1(:)) / tau_sigma_nu1(:) & / sum(tau_epsilon_nu1_d/tau_sigma_nu1),kind=custom_real)
138  phi_nu2_sent(:) = real((1.d0 - tau_epsilon_nu2_d(:)/tau_sigma_nu2(:)) / tau_sigma_nu2(:) & / sum(tau_epsilon_nu2_d/tau_sigma_nu2),kind=custom_real)
139 
140  mu_nu1_sent = real(sum(tau_epsilon_nu1_d/tau_sigma_nu1) / dble(N_SLS),kind=custom_real)
141  mu_nu2_sent = real(sum(tau_epsilon_nu2_d/tau_sigma_nu2) / dble(N_SLS),kind=custom_real)
142 
143  if (mu_nu1_sent < 1. .or. mu_nu2_sent < 1.) &
144  stop 'error in Zener viscoelasticity: must have Mu_nu1 and Mu_nu2 both greater than one'
145 
146  end subroutine attenuation_model

252  subroutine shift_velocities_from_f0(vp,vs,rho, &
253  f0_attenuation,N_SLS, &
254  tau_epsilon_nu1,tau_epsilon_nu2,inv_tau_sigma_nu1,inv_tau_sigma_nu2)
255 
256 ! From Emmanuel Chaljub, ISTerre, OSU Grenoble, France:
257 
258 ! shift (i.e. change) velocities read from the input file to take average physical dispersion into account,
259 ! i.e. if needed change the reference frequency at which these velocities are defined internally in the code:
260 
261 ! by default, the velocity values that are read in the Par_file of the code are supposed to be the unrelaxed values,
262 ! i.e. the velocities at infinite frequency.
263 ! We may want to change this and impose that the values read are those for a given frequency (here f0_attenuation).
264 !
265 ! The unrelaxed values are then defined from the reference values read at frequency f0_attenuation as follows:
266 !
267 ! mu_unrelaxed = mu (w_ref) * [ ( 1 + (1/N) Sum_k ak ) / (1 + (1/N) Sum_k ak/(1+1/(w_ref*tau_sigma_k)**2) ) ]
268 !
269 ! where w_ref = 2*PI*f0_attenuation
270 ! tau_k = tau_epsilon_k is the strain relaxation time of the k-th SLS mechanism
271 ! ak = tau_k/tau_sigma_k - 1
272 ! The ak are the solutions of the linear system:
273 ! (1/N) Sum_k { [ w*tau_k*Q(w) - (w*tau_k)^2 ] / [ 1 + (w*tau_k)^2 ] } ak = 1
274 !
275 ! To see how to obtain these formulas, see for instance equations (8), (9) and (10) of
276 ! P. Moczo, E. Bystricky, J. Kristek, J. M. Carcione and M. Bouchon,
277 ! Hybrid modeling of P-SV seismic motion at inhomogeneous viscoelastic topographic structures,
278 ! Bulletin of the Seismological Society of Americal, vol. 87, p. 1305-1323 (1997).
279 !
280 ! See also some of the PDF files in the "doc/" directory of the code.
281 !
282 ! The above formulas are for a Zener-body approximation of a Q(f) model, which is what SPECFEM uses.
283 ! If one wants to use exact formulas for a truly constant Q model instead (i.e., not approximated by a set of Zener bodies),
284 ! the expression of the scaling can be found for instance in equation (49) of Komatitsch and Tromp,
285 ! Spectral-Element Simulations of Global Seismic Wave Propagation-I. Validation, Geophys. J. Int. vol. 149 p. 390-412 (2002),
286 ! which is also in file "doc/exact_formula_to_scale_mu_from_a_frequency_f0_to_a_frequency_f1_from_Komatitsch_Tromp_GJI_2002.pdf":
287 !
288 ! mu(omega_c) = mu(omega_0) * (1 + 2 * ln(omega_c/omega_0) / (PI * Q_mu))
289 !
290 ! (the formula to scale Vs is the same except for the factor of 2, which needs to be removed because mu is related to Vs squared.
291 !
292 ! A similar expression can then be established for Q_Kappa, and conversion from Q_Kappa and Q_mu to Q_P and Q_S (if needed)
293 ! can be found for instance in file "Qkappa_Qmu_versus_Qp_Qs_relationship_in_2D_plane_strain.pdf"
294 ! in the "doc" directory of the code.
295 
296  use constants, only: one,two,pi,two_thirds,custom_real
297 
298  use specfem_par, only: axisym
299 
300  implicit none
301 
302 ! arguments
303  double precision, intent(inout) :: vp,vs
304 
305  double precision, intent(in) :: rho
306  double precision, intent(in) :: f0_attenuation
307 
308  integer,intent(in) :: N_SLS
309  real(kind=CUSTOM_REAL), dimension(N_SLS),intent(in) :: tau_epsilon_nu1,tau_epsilon_nu2
310  real(kind=CUSTOM_REAL), dimension(N_SLS),intent(in) :: inv_tau_sigma_nu1,inv_tau_sigma_nu2
311 
312 ! local variables
313  integer :: i_sls
314  double precision :: xtmp1_nu1,xtmp1_nu2,xtmp2_nu1,xtmp2_nu2,xtmp_ak_nu1,xtmp_ak_nu2
315  double precision :: factor_mu,factor_kappa
316  double precision :: kappa,mu,lambda
317 
318  mu = rho * vs*vs
319  lambda = rho * vp*vp - two * mu
320 
321  if (axisym) then ! CHECK kappa
322  kappa = lambda + two_thirds*mu
323  else
324  kappa = lambda + mu
325  endif
326 
327  xtmp1_nu1 = one
328  xtmp2_nu1 = one
329  xtmp1_nu2 = one
330  xtmp2_nu2 = one
331 
332  do i_sls = 1,n_sls
333 !! DK DK changed this to the pre-computed inverse xtmp_ak_nu2 = tau_epsilon_nu2(i_sls)/tau_sigma_nu2(i_sls) - ONE
334  xtmp_ak_nu2 = tau_epsilon_nu2(i_sls)*inv_tau_sigma_nu2(i_sls) - one
335  xtmp1_nu2 = xtmp1_nu2 + xtmp_ak_nu2/n_sls
336  xtmp2_nu2 = xtmp2_nu2 + xtmp_ak_nu2/(one + one/(two * pi * f0_attenuation / inv_tau_sigma_nu2(i_sls))**2)/n_sls
337 
338 !! DK DK changed this to the pre-computed inverse xtmp_ak_nu1 = tau_epsilon_nu1(i_sls)/tau_sigma_nu1(i_sls) - ONE
339  xtmp_ak_nu1 = tau_epsilon_nu1(i_sls)*inv_tau_sigma_nu1(i_sls) - one
340  xtmp1_nu1 = xtmp1_nu1 + xtmp_ak_nu1/n_sls
341  xtmp2_nu1 = xtmp2_nu1 + xtmp_ak_nu1/(one + one/(two * pi * f0_attenuation / inv_tau_sigma_nu1(i_sls))**2)/n_sls
342  enddo
343 
344  factor_mu = xtmp1_nu2/xtmp2_nu2
345  mu = mu * factor_mu
346 
347  factor_kappa = xtmp1_nu1/xtmp2_nu1
348  kappa = kappa * factor_kappa
349 
350  if (axisym) then ! CHECK kappa
351  lambda = kappa - two_thirds*mu
352  else
353  lambda = kappa - mu
354  endif
355 
356  ! returns shifted vp,vs
357  vp = dsqrt((lambda + two * mu) / rho)
358  vs = dsqrt(mu / rho)
359 
360  end subroutine shift_velocities_from_f0

538  subroutine decomposition_lu(a,i_min,n,indx,d)
539 
540  implicit none
541 
542  integer, intent(in) :: i_min,n
543  double precision, intent(out) :: d
544  integer, dimension(i_min:n), intent(inout) :: indx
545  double precision, dimension(i_min:n,i_min:n), intent(inout) :: a
546 
547  integer i,imax,j,k
548  double precision big,dum,somme,eps
549  double precision, dimension(i_min:n) :: vv
550 
551  imax = 0
552  d = 1.
553  eps = 1.e-20
554 
555  do i = i_min,n
556  big = 0.
557  do j = i_min,n
558  if (abs(a(i,j)) > big) then
559  big = abs(a(i,j))
560  endif
561  enddo
562  if (big == 0.) then
563  print *,'Singular matrix in routine decomposition_LU'
564  endif
565  vv(i) = 1./big
566  enddo
567 
568  do j = i_min,n
569  do i = i_min,j-1
570  somme = a(i,j)
571  do k = i_min,i-1
572  somme = somme - a(i,k)*a(k,j)
573  enddo
574  a(i,j) = somme
575  enddo
576 
577  big = 0.
578 
579  do i = j,n
580  somme = a(i,j)
581  do k = i_min,j-1
582  somme = somme - a(i,k)*a(k,j)
583  enddo
584  a(i,j) = somme
585  dum = vv(i)*abs(somme)
586  if (dum >= big) then
587  big = dum
588  imax = i
589  endif
590  enddo
591 
592  if (j /= imax) then
593  do k = i_min,n
594  dum = a(imax,k)
595  a(imax,k) = a(j,k)
596  a(j,k) = dum
597  enddo
598  d = -d
599  vv(imax) = vv(j)
600  endif
601 
602  indx(j) = imax
603  if (a(j,j) == 0.) then
604  a(j,j) = eps
605  endif
606  if (j /= n) then
607  dum = 1./a(j,j)
608  do i = j+1,n
609  a(i,j) = a(i,j)*dum
610  enddo
611  endif
612  enddo
613 
614  end subroutine decomposition_lu

682  subroutine lfit_zener(x,y,sig,ndat,poids,ia,covar,chisq,ma,Qref,point)
683 ! ma = nombre de variable diffusive
684 ! ndat = m = K nombre d'abcisse freq_k
685 
686  implicit none
687 
688  integer, intent(in) :: ndat,ma
689  logical, dimension(1:ma), intent(in) :: ia
690  double precision, intent(in) :: Qref

692  double precision, dimension(1:ndat), intent(in) :: x,y,sig
693  double precision, dimension(1:ma), intent(in) :: point

695  double precision, dimension(1:ma,1:ma), intent(out) :: covar
696 
697  integer i,j,k,l,mfit

699  double precision, dimension(1:ma) :: afunc
700  double precision, dimension(1:ma,1:1) :: beta
701 
702  mfit = 0
703 
704  do j = 1,ma
705  if (ia(j)) then
706  mfit = mfit + 1
707  endif
708  enddo
709  if (mfit == 0) then
710  print *,'lfit: no parameters to be fitted'
711  endif
712 
713  do j = 1,mfit
714  beta(j,1) = 0.
715  do k = 1,mfit
716  covar(j,k) = 0.
717  enddo
718  enddo
719 
720  do i = 1,ndat
721  call func_zener(x(i),afunc,ma,qref,point)
722  ym = y(i)
723  if (mfit < ma) then
724  do j = 1,ma
725  if (.not. ia(j)) then
726  ym = ym - poids(j) * afunc(j)
727  endif
728  enddo
729  endif
730  sig2i = 1. / (sig(i) * sig(i))
731  j = 0
732  do l= 1,ma
733  if (ia(l)) then
734  j = j+1
735  wt = afunc(l) * sig2i
736  k = count(ia(1:l))
737  covar(j,1:k) = covar(j,1:k) + wt * pack(afunc(1:l),ia(1:l))
738  beta(j,1) = beta(j,1) + ym * wt
739  endif
740  enddo
741  enddo
742 
743  do j = 2,mfit,1
744  do k = 1,j-1,1
745  covar(k,j) = covar(j,k)
746  enddo
747  enddo
748 
749  if (ma == 1) then
750  poids(1) = beta(1,1)/covar(1,1)
751  else if (ma > 1) then
752  call syst_lu(covar,1,mfit,beta,1)
753  poids(1:ma) = unpack(beta(1:ma,1),ia,poids(1:ma))
754  endif
755 
756  chisq = 0.
757  do i = 1,ndat
758  call func_zener(x(i),afunc,ma,qref,point)
759  chisq=chisq+((y(i)-dot_product(poids(1:ma),afunc(1:ma)))/sig(i))**2
760  enddo
761 
762  end subroutine lfit_zener
