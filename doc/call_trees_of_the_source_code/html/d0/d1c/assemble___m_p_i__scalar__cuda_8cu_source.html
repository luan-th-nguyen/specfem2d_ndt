45 // prepares a device array with with all inter-element edge-nodes -- this
46 // is followed by a memcpy and MPI operations
47 __global__ void prepare_boundary_potential_on_device(realw* d_potential_dot_dot_acoustic,
48  realw* d_send_potential_dot_dot_buffer,
49  const int ninterface_ac,
50  const int max_nibool_interfaces_ext_mesh,
51  const int* d_nibool_interfaces_ext_mesh,
52  const int* d_ibool_interfaces_ext_mesh,
53  const int* inum_inter_acoustic) {
54 
55  int id = threadIdx.x + blockIdx.x*blockDim.x + blockIdx.y*gridDim.x*blockDim.x;
56  int ientry,iglob,num_int;
57 
58  for(int iinterface=0; iinterface < ninterface_ac; iinterface++) {
59 
60  num_int=inum_inter_acoustic[iinterface]-1;
61 
62  if (id<d_nibool_interfaces_ext_mesh[num_int]) {
63 
64  // entry in interface array
65  ientry = id + max_nibool_interfaces_ext_mesh*num_int;
66  // global index in wavefield
67  iglob = d_ibool_interfaces_ext_mesh[ientry] - 1;
68 
69  d_send_potential_dot_dot_buffer[ientry] = d_potential_dot_dot_acoustic[iglob];
70  }
71  }
72 
73 }
