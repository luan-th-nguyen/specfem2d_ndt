37  PML_BOUNDARY_CONDITIONS,e1,e11,e13)
38 
39  ! updates memory variable in viscoelastic simulation
40 
41  ! compute forces for the elastic elements
42  use constants, only: custom_real,ngllx,ngllz,ndim
43 
44  use specfem_par, only: nglob,nspec,nspec_att,attenuation_viscoelastic,n_sls, &
45  ibool,xix,xiz,gammax,gammaz,hprime_xx,hprime_zz
46 
47  ! PML arrays
48  use specfem_par, only: ispec_is_pml
49 
50  implicit none
51 
52  real(kind=CUSTOM_REAL), dimension(NDIM,nglob),intent(in) :: displ_elastic,displ_elastic_old
53 
54  logical,dimension(nspec),intent(in) :: ispec_is_elastic
55 
56  ! CPML coefficients and memory variables
57  logical,intent(in) :: PML_BOUNDARY_CONDITIONS
58 
59  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec_ATT,N_SLS),intent(inout) :: e1,e11,e13
60 
61  ! local parameters
62  integer :: ispec
63  ! nsub1 denotes discrete time step n-1
64  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec) :: dux_dxl_n,dux_dzl_n,duz_dxl_n,duz_dzl_n
65  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec) :: dux_dxl_nsub1,dux_dzl_nsub1,duz_dxl_nsub1,duz_dzl_nsub1
66 
67  ! checks if anything to do
68  if (.not. attenuation_viscoelastic) return
69 
70  ! compute Grad(displ_elastic) at time step n for attenuation
71  call compute_gradient_attenuation(displ_elastic,dux_dxl_n,duz_dxl_n, &
72  dux_dzl_n,duz_dzl_n,xix,xiz,gammax,gammaz,ibool,ispec_is_elastic,hprime_xx,hprime_zz,nspec,nglob)
73 
74  ! compute Grad(disp_elastic_old) at time step n-1 for attenuation
75  call compute_gradient_attenuation(displ_elastic_old,dux_dxl_nsub1,duz_dxl_nsub1, &
76  dux_dzl_nsub1,duz_dzl_nsub1,xix,xiz,gammax,gammaz,ibool,ispec_is_elastic,hprime_xx,hprime_zz,nspec,nglob)
77 
78  ! loop over spectral elements
79  do ispec = 1,nspec
80 
81  ! attenuation is not implemented in acoustic (i.e. fluid) media for now, only in viscoelastic (i.e. solid) media
82  if (.not. ispec_is_elastic(ispec)) cycle
83 
84  if ((.not. pml_boundary_conditions) .or. (pml_boundary_conditions .and. (.not. ispec_is_pml(ispec)))) then
85  call compute_attenuation_viscoelastic_update(ispec,e1,e11,e13, &
86  dux_dxl_n,duz_dzl_n,duz_dxl_n,dux_dzl_n, &
87  dux_dxl_nsub1,duz_dzl_nsub1,duz_dxl_nsub1,dux_dzl_nsub1)
88  endif
89  enddo

91  end subroutine compute_attenuation_viscoelastic

96 
97  subroutine compute_attenuation_viscoelastic_update(ispec,e1,e11,e13, &
98  dux_dxl_n,duz_dzl_n,duz_dxl_n,dux_dzl_n, &
99  dux_dxl_nsub1,duz_dzl_nsub1,duz_dxl_nsub1,dux_dzl_nsub1)
100 
101  use constants, only: ngllx,ngllz,custom_real,two,alpha_lddrk,beta_lddrk,c_lddrk
102 
103  use specfem_par, only: nspec,nspec_att,n_sls, &
104  inv_tau_sigma_nu1,phi_nu1,inv_tau_sigma_nu2,phi_nu2, &
105  time_stepping_scheme,i_stage,deltat
106 
107  ! LDDRK & RK
108  use specfem_par, only: e1_lddrk,e11_lddrk,e13_lddrk, &
109  e1_initial_rk,e11_initial_rk,e13_initial_rk,e1_force_rk, e11_force_rk, e13_force_rk
110 
111  implicit none
112 
113 ! update the memory variables using a convolution or using a differential equation
114 ! (tests made by Ting Yu and also by Zhinan Xie, CNRS Marseille, France, show that it is better to leave it to .true.)
115  logical, parameter :: CONVOLUTION_MEMORY_VARIABLES = .true.
116 
117  integer,intent(in) :: ispec
118  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec_ATT,N_SLS),intent(inout) :: e1,e11,e13
119 
120  ! gradient of displacements (nsub1 denotes discrete time step n-1)
121  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec),intent(in) :: dux_dxl_n,duz_dzl_n,duz_dxl_n,dux_dzl_n
122  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ,nspec),intent(in) :: dux_dxl_nsub1,duz_dzl_nsub1,duz_dxl_nsub1,dux_dzl_nsub1
123 
124  ! local parameters
125  integer :: i,j
126  integer :: i_sls
127 
128  ! for attenuation
129  real(kind=CUSTOM_REAL) :: phinu1,phinu2,theta_n_u,theta_nsub1_u
130  double precision :: tauinvnu1,tauinvnu2
131  double precision :: coef0,coef1,coef2
132 
133  ! temporary RK4 variable
134  real(kind=CUSTOM_REAL) :: weight_rk
135 
136  do j = 1,ngllz
137  do i = 1,ngllx
138 
139  ! convention to indicate that Q = 9999 in that element i.e. that there is no viscoelasticity in that element
140  if (inv_tau_sigma_nu1(i,j,ispec,1) < 0.) cycle
141 
142  theta_n_u = dux_dxl_n(i,j,ispec) + duz_dzl_n(i,j,ispec)
143  theta_nsub1_u = dux_dxl_nsub1(i,j,ispec) + duz_dzl_nsub1(i,j,ispec)
144 
145  ! loop on all the standard linear solids
146  do i_sls = 1,n_sls
147  phinu1 = phi_nu1(i,j,ispec,i_sls)
148  tauinvnu1 = inv_tau_sigma_nu1(i,j,ispec,i_sls)
149  phinu2 = phi_nu2(i,j,ispec,i_sls)
150  tauinvnu2 = inv_tau_sigma_nu2(i,j,ispec,i_sls)
151 
152  ! update e1, e11, e13 in convolution formation with modified recursive convolution scheme on basis of
153  ! second-order accurate convolution term calculation from equation (21) of
154  ! Shumin Wang, Robert Lee, and Fernando L. Teixeira,
155  ! Anisotropic-medium PML for vector FETD with modified basis functions,
156  ! IEEE Transactions on Antennas and Propagation, vol. 54, no. 1, (2006)
157  select case (time_stepping_scheme)
158  case (1)
159  ! Newmark
160 
161 ! update the memory variables using a convolution or using a differential equation
162 ! From Zhinan Xie and Dimitri Komatitsch:
163 ! For cases in which a value of tau_sigma is small, then its inverse is large,
164 ! which may result in a in stiff ordinary differential equation to solve;
165 ! in such a case, resorting to the convolution formulation is better.
166  if (convolution_memory_variables) then
167  call compute_coef_convolution(tauinvnu1,deltat,coef0,coef1,coef2)
168 
169  e1(i,j,ispec,i_sls) = coef0 * e1(i,j,ispec,i_sls) + &
170  phinu1 * (coef1 * theta_n_u + coef2 * theta_nsub1_u)
171 
172  call compute_coef_convolution(tauinvnu2,deltat,coef0,coef1,coef2)
173 
174  e11(i,j,ispec,i_sls) = coef0 * e11(i,j,ispec,i_sls) + &
175  phinu2 * (coef1 * (dux_dxl_n(i,j,ispec)-theta_n_u/two) + &
176  coef2 * (dux_dxl_nsub1(i,j,ispec)-theta_nsub1_u/two))
177 
178  e13(i,j,ispec,i_sls) = coef0 * e13(i,j,ispec,i_sls) + &
179  phinu2 * (coef1 * (dux_dzl_n(i,j,ispec) + duz_dxl_n(i,j,ispec)) + &
180  coef2 * (dux_dzl_nsub1(i,j,ispec) + duz_dxl_nsub1(i,j,ispec)))
181  else
182  stop 'CONVOLUTION_MEMORY_VARIABLES == .false. is not accurate enough and has been discontinued for now'
183  e1(i,j,ispec,i_sls) = e1(i,j,ispec,i_sls) + deltat * &
184  (- e1(i,j,ispec,i_sls)*tauinvnu1 + phinu1 * theta_n_u)
185 
186  e11(i,j,ispec,i_sls) = e11(i,j,ispec,i_sls) + deltat * &
187  (- e11(i,j,ispec,i_sls)*tauinvnu2 + phinu2 * (dux_dxl_n(i,j,ispec)-theta_n_u/two))
188 
189  e13(i,j,ispec,i_sls) = e13(i,j,ispec,i_sls) + deltat * &
190  (- e13(i,j,ispec,i_sls)*tauinvnu2 + phinu2 * (dux_dzl_n(i,j,ispec) + duz_dxl_n(i,j,ispec)))
191  endif
192 
193  case (2)
194  ! LDDRK
195  ! update e1, e11, e13 in ADE formation with fourth-order LDDRK scheme
196  e1_lddrk(i,j,ispec,i_sls) = alpha_lddrk(i_stage) * e1_lddrk(i,j,ispec,i_sls) + &
197  deltat * (theta_n_u * phinu1 - e1(i,j,ispec,i_sls) * tauinvnu1)
198  e1(i,j,ispec,i_sls) = e1(i,j,ispec,i_sls) + beta_lddrk(i_stage) * e1_lddrk(i,j,ispec,i_sls)
199 
200  e11_lddrk(i,j,ispec,i_sls) = alpha_lddrk(i_stage) * e11_lddrk(i,j,ispec,i_sls) + &
201  deltat * ((dux_dxl_n(i,j,ispec)-theta_n_u/two) * phinu2) - &
202  deltat * (e11(i,j,ispec,i_sls) * tauinvnu2)
203  e11(i,j,ispec,i_sls) = e11(i,j,ispec,i_sls)+beta_lddrk(i_stage)*e11_lddrk(i,j,ispec,i_sls)
204 
205  e13_lddrk(i,j,ispec,i_sls) = alpha_lddrk(i_stage) * e13_lddrk(i,j,ispec,i_sls) + &
206  deltat * ((dux_dzl_n(i,j,ispec) + duz_dxl_n(i,j,ispec))*phinu2) - &
207  deltat * (e13(i,j,ispec,i_sls) * tauinvnu2)
208  e13(i,j,ispec,i_sls) = e13(i,j,ispec,i_sls)+beta_lddrk(i_stage) * e13_lddrk(i,j,ispec,i_sls)
209 
210  case (3)
211  ! Runge-Kutta
212  ! update e1, e11, e13 in ADE formation with classical fourth-order Runge-Kutta scheme
213  e1_force_rk(i,j,ispec,i_sls,i_stage) = deltat * (theta_n_u * phinu1 - e1(i,j,ispec,i_sls) * tauinvnu1)
214 
215  if (i_stage == 1 .or. i_stage == 2 .or. i_stage == 3) then
216  if (i_stage == 1) weight_rk = 0.5_custom_real
217  if (i_stage == 2) weight_rk = 0.5_custom_real
218  if (i_stage == 3) weight_rk = 1._custom_real
219 
220  if (i_stage == 1) e1_initial_rk(i,j,ispec,i_sls) = e1(i,j,ispec,i_sls)
221  e1(i,j,ispec,i_sls) = e1_initial_rk(i,j,ispec,i_sls) + weight_rk * e1_force_rk(i,j,ispec,i_sls,i_stage)
222  else if (i_stage == 4) then
223  e1(i,j,ispec,i_sls) = e1_initial_rk(i,j,ispec,i_sls) + 1._custom_real / 6._custom_real * &
224  (e1_force_rk(i,j,ispec,i_sls,1) + 2._custom_real * e1_force_rk(i,j,ispec,i_sls,2) + &
225  2._custom_real * e1_force_rk(i,j,ispec,i_sls,3) + e1_force_rk(i,j,ispec,i_sls,4))
226  endif
227 
228  e11_force_rk(i,j,ispec,i_sls,i_stage) = deltat * ((dux_dxl_n(i,j,ispec)-theta_n_u/two) * phinu2 - &
229  e11(i,j,ispec,i_sls) * tauinvnu2)
230 
231  if (i_stage == 1 .or. i_stage == 2 .or. i_stage == 3) then
232  if (i_stage == 1) weight_rk = 0.5_custom_real
233  if (i_stage == 2) weight_rk = 0.5_custom_real
234  if (i_stage == 3) weight_rk = 1._custom_real
235 
236  if (i_stage == 1) e11_initial_rk(i,j,ispec,i_sls) = e11(i,j,ispec,i_sls)
237  e11(i,j,ispec,i_sls) = e11_initial_rk(i,j,ispec,i_sls) + weight_rk * e11_force_rk(i,j,ispec,i_sls,i_stage)
238  else if (i_stage == 4) then
239  e11(i,j,ispec,i_sls) = e11_initial_rk(i,j,ispec,i_sls) + 1._custom_real / 6._custom_real * &
240  (e11_force_rk(i,j,ispec,i_sls,1) + 2._custom_real * e11_force_rk(i,j,ispec,i_sls,2) + &
241  2._custom_real * e11_force_rk(i,j,ispec,i_sls,3) + e11_force_rk(i,j,ispec,i_sls,4))
242  endif
243 
244  e13_force_rk(i,j,ispec,i_sls,i_stage) = deltat * ((dux_dzl_n(i,j,ispec) + duz_dxl_n(i,j,ispec))*phinu2 - &
245  e13(i,j,ispec,i_sls) * tauinvnu2)
246  if (i_stage == 1 .or. i_stage == 2 .or. i_stage == 3) then
247  if (i_stage == 1) weight_rk = 0.5_custom_real
248  if (i_stage == 2) weight_rk = 0.5_custom_real
249  if (i_stage == 3) weight_rk = 1._custom_real
250 
251  if (i_stage == 1) e13_initial_rk(i,j,ispec,i_sls) = e13(i,j,ispec,i_sls)
252  e13(i,j,ispec,i_sls) = e13_initial_rk(i,j,ispec,i_sls) + weight_rk * e13_force_rk(i,j,ispec,i_sls,i_stage)
253  else if (i_stage == 4) then
254  e13(i,j,ispec,i_sls) = e13_initial_rk(i,j,ispec,i_sls) + 1._custom_real / 6._custom_real * &
255  (e13_force_rk(i,j,ispec,i_sls,1) + 2._custom_real * e13_force_rk(i,j,ispec,i_sls,2) + &
256  2._custom_real * e13_force_rk(i,j,ispec,i_sls,3) + e13_force_rk(i,j,ispec,i_sls,4))
257  endif
258 
259  case default
260  stop 'Time stepping scheme not implemented yet in viscoelastic attenuation update'
261  end select
262 
263  enddo ! i_sls
264 
265  enddo
266  enddo
267 
268  end subroutine compute_attenuation_viscoelastic_update
