34  program convolve_source_time_function
35 
36 !
37 ! convolve seismograms computed for a Heaviside with given source time function
38 !
39 
40 ! we mimic a triangle of half duration equal to half_duration_triangle
41 ! using a Gaussian having a very close shape, as explained in Figure 4.2
42 ! of the manual
43 
44  implicit none
45 
46  include "constants.h"
47 
48  integer :: i,j,N_j,number_remove,nlines
49 
50  double precision :: alpha,dt,tau_j,source,exponentval,t1,t2,displ1,displ2,gamma,height,half_duration_triangle
51 
52  logical :: triangle
53 
54  double precision, dimension(:), allocatable :: timeval,sem,sem_fil
55 
56 ! read file with number of lines in input
57  open(unit=33,file='input_convolve_code.txt',status='old',action='read')
58  read(33,*) nlines
59  read(33,*) half_duration_triangle
60  read(33,*) triangle
61  close(33)
62 
63 ! allocate arrays
64  allocate(timeval(nlines),sem(nlines),sem_fil(nlines))
65 
66 ! read the input seismogram
67  do i = 1,nlines
68  read(5,*) timeval(i),sem(i)
69  enddo
70 
71 ! define a Gaussian with the right exponent to mimic a triangle of equivalent half duration
72  alpha = source_decay_mimic_triangle/half_duration_triangle
73 
74 ! compute the time step
75  dt = timeval(2) - timeval(1)
76 
77 ! number of integers for which the source wavelet is different from zero
78  if (triangle) then
79  n_j = ceiling(half_duration_triangle/dt)
80  else
81  n_j = ceiling(1.5d0*half_duration_triangle/dt)
82  endif
83 
84  do i = 1,nlines
85 
86  sem_fil(i) = 0.d0
87 
88  do j = -n_j,n_j
89 
90  if (i > j .and. i-j <= nlines) then
91 
92  tau_j = dble(j)*dt
93 
94 ! convolve with a triangle
95  if (triangle) then
96  height = 1.d0 / half_duration_triangle
97  if (abs(tau_j) > half_duration_triangle) then
98  source = 0.d0
99  else if (tau_j < 0.d0) then
100  t1 = - n_j * dt
101  displ1 = 0.d0
102  t2 = 0.d0
103  displ2 = height
104  gamma = (tau_j - t1) / (t2 - t1)
105  source= (1.d0 - gamma) * displ1 + gamma * displ2
106  else
107  t1 = 0.d0
108  displ1 = height
109  t2 = + n_j * dt
110  displ2 = 0.d0
111  gamma = (tau_j - t1) / (t2 - t1)
112  source= (1.d0 - gamma) * displ1 + gamma * displ2
113  endif
114 
115  else
116 
117 ! convolve with a Gaussian
118  exponentval = alpha**2 * tau_j**2
119  if (exponentval < 50.d0) then
120  source = alpha*exp(-exponentval)/sqrt(pi)
121  else
122  source = 0.d0
123  endif
124 
125  endif
126 
127  sem_fil(i) = sem_fil(i) + sem(i-j)*source*dt
128 
129  endif
130 
131  enddo
132  enddo
133 
134 ! compute number of samples to remove from end of seismograms
135  number_remove = n_j + 1
136  do i = 1,nlines - number_remove
137  write(*,*) sngl(timeval(i)),' ',sngl(sem_fil(i))
138  enddo
139 
140  end program convolve_source_time_function
