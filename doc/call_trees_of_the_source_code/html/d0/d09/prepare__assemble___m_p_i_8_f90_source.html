35 ! This file contains subroutines related to assembling (of the mass matrix, potential_dot_dot and
36 ! accel_elastic, accels_poroelastic, accelw_poroelastic).
37 ! These subroutines are for the most part not used in the sequential version.
38 !
39 
40 !-----------------------------------------------
41 ! Determines the points that are on the interfaces with other partitions, to help
42 ! build the communication buffers, and determines which elements are considered 'inner'
43 ! (no points in common with other partitions) and 'outer' (at least one point in common
44 ! with neighbouring partitions).
45 ! We have both acoustic and (poro)elastic buffers, for coupling between acoustic and (poro)elastic elements
46 ! led us to have two sets of communications.
47 !-----------------------------------------------
48  subroutine prepare_assemble_mpi()
49 
50  use constants, only: ngllx,ngllz
51 
52  use specfem_par, only: ibool, knods, ngnod, nglob, &
53  ispec_is_elastic, ispec_is_poroelastic, ispec_is_acoustic, ispec_is_gravitoacoustic
54 
55  use specfem_par, only: ninterface, my_nelmnts_neighbours, my_interfaces, &
56  nibool_interfaces_ext_mesh, ibool_interfaces_ext_mesh_init
57 
58  use specfem_par, only: nproc, &
59  ibool_interfaces_acoustic, ibool_interfaces_elastic, &
60  ibool_interfaces_poroelastic, &
61  nibool_interfaces_acoustic, nibool_interfaces_elastic, &
62  nibool_interfaces_poroelastic, &
63  inum_interfaces_acoustic, inum_interfaces_elastic, &
64  inum_interfaces_poroelastic, &
65  ninterface_acoustic, ninterface_elastic, ninterface_poroelastic
66 
67  implicit none
68 
69  ! local parameters
70  integer :: iinterface
71  integer :: ispec_interface
72 
73  logical, dimension(nglob) :: mask_ibool_acoustic
74  logical, dimension(nglob) :: mask_ibool_elastic
75  logical, dimension(nglob) :: mask_ibool_poroelastic
76  logical, dimension(nglob) :: mask_ibool_ext_mesh
77 
78  integer :: ixmin, ixmax, izmin, izmax, ix, iz
79  integer, dimension(ngnod) :: n
80  integer :: e1, e2, itype, ispec, k, sens, iglob
81  integer :: nglob_interface_acoustic
82  integer :: nglob_interface_elastic
83  integer :: nglob_interface_poroelastic
84  integer :: npoin_interface_ext_mesh
85 
86  ! checks if anything to do
87  if (nproc <= 1) return
88 
89  ! initializes
90  ! for all domains
91  nibool_interfaces_ext_mesh(:) = 0
92  ibool_interfaces_ext_mesh_init(:,:) = 0
93 
94  ! only specific domain interfaces
95  ibool_interfaces_acoustic(:,:) = 0
96  nibool_interfaces_acoustic(:) = 0
97 
98  ibool_interfaces_elastic(:,:) = 0
99  nibool_interfaces_elastic(:) = 0
100 
101  ibool_interfaces_poroelastic(:,:) = 0
102  nibool_interfaces_poroelastic(:) = 0
103 
104  do iinterface = 1, ninterface
105  ! initializes interface point counters
106  npoin_interface_ext_mesh = 0
107  mask_ibool_ext_mesh(:) = .false.
108 
109  nglob_interface_acoustic = 0
110  nglob_interface_elastic = 0
111  nglob_interface_poroelastic = 0
112 
113  mask_ibool_acoustic(:) = .false.
114  mask_ibool_elastic(:) = .false.
115  mask_ibool_poroelastic(:) = .false.
116 
117  do ispec_interface = 1, my_nelmnts_neighbours(iinterface)
118  ! element id
119  ispec = my_interfaces(1,ispec_interface,iinterface)
120 
121  ! type of interface: 1 = common point, 2 = common edge
122  itype = my_interfaces(2,ispec_interface,iinterface)
123 
124  ! element control node ids
125  do k = 1, ngnod
126  n(k) = knods(k,ispec)
127  enddo
128 
129  ! common node ids
130  e1 = my_interfaces(3,ispec_interface,iinterface)
131  e2 = my_interfaces(4,ispec_interface,iinterface)
132 
133  call get_edge(ngnod, n, itype, e1, e2, ixmin, ixmax, izmin, izmax, sens)
134 
135  ! sets interface points (all material domains)
136  do iz = izmin, izmax, sens
137  do ix = ixmin, ixmax, sens
138  ! global index
139  iglob = ibool(ix,iz,ispec)
140 
141  if (.not. mask_ibool_ext_mesh(iglob)) then
142  ! masks point as being accounted for
143  mask_ibool_ext_mesh(iglob) = .true.
144  ! adds point to interface
145  npoin_interface_ext_mesh = npoin_interface_ext_mesh + 1
146  ibool_interfaces_ext_mesh_init(npoin_interface_ext_mesh,iinterface) = iglob
147  endif
148  enddo
149  enddo
150 
151  ! sets interface points for specific domains
152  do iz = izmin, izmax, sens
153  do ix = ixmin, ixmax, sens
154  ! global index
155  iglob = ibool(ix,iz,ispec)
156 
157  ! checks to which material this common interface belongs
158  if (ispec_is_elastic(ispec)) then
159  ! elastic element
160  if (.not. mask_ibool_elastic(iglob)) then
161  mask_ibool_elastic(iglob) = .true.
162  nglob_interface_elastic = nglob_interface_elastic + 1
163  ibool_interfaces_elastic(nglob_interface_elastic,iinterface) = iglob
164  endif
165 
166  else if (ispec_is_poroelastic(ispec)) then
167  ! poroelastic element
168  if (.not. mask_ibool_poroelastic(iglob)) then
169  mask_ibool_poroelastic(iglob) = .true.
170  nglob_interface_poroelastic = nglob_interface_poroelastic + 1
171  ibool_interfaces_poroelastic(nglob_interface_poroelastic,iinterface) = iglob
172  endif
173 
174  else if (ispec_is_acoustic(ispec)) then
175  ! acoustic element
176  if (.not. mask_ibool_acoustic(iglob)) then
177  mask_ibool_acoustic(iglob) = .true.
178  nglob_interface_acoustic = nglob_interface_acoustic + 1
179  ibool_interfaces_acoustic(nglob_interface_acoustic,iinterface) = iglob
180  endif
181 
182  else if (ispec_is_gravitoacoustic(iglob)) then
183  ! gravitoacoustic element
184  stop 'Preparing assembly points with MPI not implemented yet for gravitoacoustic domains'
185 
186  else
187  stop 'Invalid element type found in prepare_assemble_MPI() routine'
188  endif
189 
190  enddo
191  enddo
192 
193  enddo
194 
195  ! stores total number of (global) points on this MPI interface
196  nibool_interfaces_ext_mesh(iinterface) = npoin_interface_ext_mesh
197 
198  ! stores counters for interface points
199  nibool_interfaces_acoustic(iinterface) = nglob_interface_acoustic
200  nibool_interfaces_elastic(iinterface) = nglob_interface_elastic
201  nibool_interfaces_poroelastic(iinterface) = nglob_interface_poroelastic
202 
203  enddo
204 
205  ! sets number of interfaces for each material domain
206  ninterface_acoustic = 0
207  ninterface_elastic = 0
208  ninterface_poroelastic = 0
209 
210  ! loops over all MPI interfaces
211  do iinterface = 1, ninterface
212  ! sets acoustic MPI interface (local) indices in range [1,ninterface_acoustic]
213  if (nibool_interfaces_acoustic(iinterface) > 0) then
214  ninterface_acoustic = ninterface_acoustic + 1
215  inum_interfaces_acoustic(ninterface_acoustic) = iinterface
216  endif
217  ! elastic
218  if (nibool_interfaces_elastic(iinterface) > 0) then
219  ninterface_elastic = ninterface_elastic + 1
220  inum_interfaces_elastic(ninterface_elastic) = iinterface
221  endif
222  ! poroelastic
223  if (nibool_interfaces_poroelastic(iinterface) > 0) then
224  ninterface_poroelastic = ninterface_poroelastic + 1
225  inum_interfaces_poroelastic(ninterface_poroelastic) = iinterface
226  endif
227  enddo
228 
229  end subroutine prepare_assemble_mpi
230 
231 
232 !-----------------------------------------------
233 ! Get the points (ixmin, ixmax, izmin and izmax) on an node/edge for one element.
234 ! 'sens' is used to have DO loops with increment equal to 'sens' (-/+1).
235 !-----------------------------------------------
236  subroutine get_edge ( ngnod, n, itype, e1, e2, ixmin, ixmax, izmin, izmax, sens )
237 
238  use constants, only: ngllx,ngllz
239 
240  implicit none
241 
242  integer, intent(in) :: ngnod
243  integer, dimension(ngnod), intent(in) :: n
244  integer, intent(in) :: itype, e1, e2
245  integer, intent(out) :: ixmin, ixmax, izmin, izmax
246  integer, intent(out) :: sens
247 
248  if (itype == 1) then
249 
250  ! common single point
251 
252  ! checks which corner point is given
253  if (e1 == n(1)) then
254  ixmin = 1
255  ixmax = 1
256  izmin = 1
257  izmax = 1
258  endif
259  if (e1 == n(2)) then
260  ixmin = ngllx
261  ixmax = ngllx
262  izmin = 1
263  izmax = 1
264  endif
265  if (e1 == n(3)) then
266  ixmin = ngllx
267  ixmax = ngllx
268  izmin = ngllz
269  izmax = ngllz
270  endif
271  if (e1 == n(4)) then
272  ixmin = 1
273  ixmax = 1
274  izmin = ngllz
275  izmax = ngllz
276  endif
277  sens = 1
278 
279  else if (itype == 2) then
280 
281  ! common edge
282 
283  ! checks which edge and corner points are given
284  if (e1 == n(1)) then
285  ixmin = 1
286  izmin = 1
287  if (e2 == n(2)) then
288  ixmax = ngllx
289  izmax = 1
290  sens = 1
291  endif
292  if (e2 == n(4)) then
293  ixmax = 1
294  izmax = ngllz
295  sens = 1
296  endif
297  endif
298  if (e1 == n(2)) then
299  ixmin = ngllx
300  izmin = 1
301  if (e2 == n(3)) then
302  ixmax = ngllx
303  izmax = ngllz
304  sens = 1
305  endif
306  if (e2 == n(1)) then
307  ixmax = 1
308  izmax = 1
309  sens = -1
310  endif
311  endif
312  if (e1 == n(3)) then
313  ixmin = ngllx
314  izmin = ngllz
315  if (e2 == n(4)) then
316  ixmax = 1
317  izmax = ngllz
318  sens = -1
319  endif
320  if (e2 == n(2)) then
321  ixmax = ngllx
322  izmax = 1
323  sens = -1
324  endif
325  endif
326  if (e1 == n(4)) then
327  ixmin = 1
328  izmin = ngllz
329  if (e2 == n(1)) then
330  ixmax = 1
331  izmax = 1
332  sens = -1
333  endif
334  if (e2 == n(3)) then
335  ixmax = ngllx
336  izmax = ngllz
337  sens = 1
338  endif
339  endif
340 
341  else
342 
343  stop 'ERROR get_edge unknown type'
344 
345  endif
346 
347  end subroutine get_edge
