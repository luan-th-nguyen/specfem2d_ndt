35  subroutine decompose_mesh()
36 
37  use constants, only: imain,max_neighbors,ncorners,max_nsize_shared
38 
39  use shared_parameters, only: nproc,add_periodic_conditions,periodic_horiz_dist, &
40  ngnod,nbmodels,num_material,partitioning_method,phi_read
41 
42  use part_unstruct_par, only: part,nelmnts,xadj_g,adjncy_g,elmnts,elmnts_bis,nb_edges, &
43  nnodes_elmnts,nnodes,nodes_elmnts,nodes_coords,ninterfaces
44 
45  use decompose_par
46  use compute_elements_load_par
47 
48  implicit none
49 
50  ! local parameters
51  integer :: i, iproc, ier
52 
53  ! allocates and initializes partitioning of elements
54  allocate(part(0:nelmnts-1),stat=ier)
55  if (ier /= 0) stop 'Error allocating partition array'
56  part(:) = -1
57 
58  ! connectivity
59  if (nproc > 1) then
60  allocate(xadj_g(0:nelmnts), &
61  adjncy_g(0:max_neighbors*nelmnts-1),stat=ier)
62  if (ier /= 0) stop 'Error allocating connectivity arrays'
63  xadj_g(:) = 0
64  adjncy_g(:) = -1
65  endif
66 
67  ! construction of the graph
68 
69  ! if ngnod == 9, we work on a subarray of elements that represents the elements with four nodes (four corners) only
70  ! because the adjacency of the mesh elements can be entirely determined from the knowledge of the four corners only
71  if (ngnod == 9) then
72  allocate(elmnts_bis(0:ncorners*nelmnts-1),stat=ier)
73  if (ier /= 0) stop 'Error allocating elmnts_bis array'
74 
75  do i = 0, nelmnts-1
76  elmnts_bis(i*ncorners:i*ncorners+ncorners-1) = elmnts(i*ngnod:i*ngnod+ncorners-1)
77  enddo
78 
79  if (nproc > 1) then
80 !! DK DK fixed problem in the previous implementation by Nicolas Le Goff:
81 !! DK DK (nxread+1)*(nzread+1) is OK for a regular internal mesh only, not for non structured external meshes
82 !! DK DK call mesh2dual_ncommonnodes(nelmnts, (nxread+1)*(nzread+1), &
83 !! DK DK elmnts_bis, xadj, adjncy, nnodes_elmnts, nodes_elmnts,1)
84 !! DK DK the subset of element corners is not renumbered therefore we must still use the nnodes computed for 9 nodes here
85  ! determines maximum neighbors based on 1 common node
86  call mesh2dual_ncommonnodes(elmnts_bis,1,xadj_g,adjncy_g)
87  endif
88  else
89  if (nproc > 1) then
90  ! determines maximum neighbors based on 1 common node
91  call mesh2dual_ncommonnodes(elmnts,1,xadj_g,adjncy_g)
92  endif
93  endif
94 
95 
96  if (nproc == 1) then
97  part(:) = 0 ! single process has rank 0
98  else
99  ! number of common edges
100  nb_edges = xadj_g(nelmnts)
101 
102  ! compute elements load for efficient partitioning (a PML element take more time to calculate than a fluid element for ex)
103  call compute_elements_load()
104 
105  ! partitioning
106  select case (partitioning_method)
107  case(1)
108  ! analytical
109  write(imain,*)
110  write(imain,*) 'Partitioning method: analytical'
111  write(imain,*)
112  do iproc = 0, nproc-2
113  part(iproc*floor(real(nelmnts)/real(nproc)):(iproc+1)*floor(real(nelmnts)/real(nproc))-1) = iproc
114  enddo
115  part(floor(real(nelmnts)/real(nproc))*(nproc-1):nelmnts-1) = nproc - 1
116 
117  case(2)
118  ! METIS
119  write(imain,*)
120  write(imain,*) 'Partitioning method: METIS'
121  write(imain,*)
122  call metis_partitioning()
123 
124  case(3)
125  ! SCOTCH
126  write(imain,*)
127  write(imain,*) 'Partitioning method: SCOTCH'
128  write(imain,*)
129  call scotch_partitioning()
130 
131  case default
132  stop 'Error invalid partitioning method value! must be 1, 2 or 3, please check your Par_file...'
133  end select
134 
135  endif
136 
137  ! fluid-solid edges: coupled elements are transferred to the same partition
138  if (ngnod == 9) then
139  call acoustic_elastic_repartitioning(elmnts_bis, nbmodels, phi_read, num_material, nproc)
140  else
141  call acoustic_elastic_repartitioning(elmnts, nbmodels, phi_read, num_material, nproc)
142  endif
143 
144  ! fluid-porous edges: coupled elements are transferred to the same partition
145  if (ngnod == 9) then
146  call acoustic_poro_repartitioning(elmnts_bis, nbmodels, phi_read, num_material, nproc)
147  else
148  call acoustic_poro_repartitioning(elmnts, nbmodels, phi_read, num_material, nproc)
149  endif
150 
151  ! porous-solid edges: coupled elements are transferred to the same partition
152  if (ngnod == 9) then
153  call poro_elastic_repartitioning(elmnts_bis, nbmodels, phi_read, num_material, nproc)
154  else
155  call poro_elastic_repartitioning(elmnts, nbmodels, phi_read, num_material, nproc)
156  endif
157 
158  ! periodic edges: coupled elements are transferred to the same partition
159  if (add_periodic_conditions .and. nproc > 1) then
160  if (ngnod == 9) then
161  call periodic_edges_repartitioning(elmnts_bis,nnodes,nodes_coords,periodic_horiz_dist)
162  else
163  call periodic_edges_repartitioning(elmnts,nnodes,nodes_coords,periodic_horiz_dist)
164  endif
165  endif
166 
167  ! local number of each element for each partition
168  call construct_glob2loc_elmnts(nproc)
169 
170  if (ngnod == 9) then
171  if (allocated(nnodes_elmnts) ) deallocate(nnodes_elmnts)
172  if (allocated(nodes_elmnts) ) deallocate(nodes_elmnts)
173  allocate(nnodes_elmnts(0:nnodes-1))
174  allocate(nodes_elmnts(0:max_nsize_shared*nnodes-1))
175  nnodes_elmnts(:) = 0
176  nodes_elmnts(:) = 0
177  do i = 0, ngnod*nelmnts-1
178  nodes_elmnts(elmnts(i)*max_nsize_shared+nnodes_elmnts(elmnts(i))) = i/ngnod
179  nnodes_elmnts(elmnts(i)) = nnodes_elmnts(elmnts(i)) + 1
180  enddo
181  else
182  if (nproc < 2) then
183  if (.not. allocated(nnodes_elmnts) ) allocate(nnodes_elmnts(0:nnodes-1))
184  if (.not. allocated(nodes_elmnts) ) allocate(nodes_elmnts(0:max_nsize_shared*nnodes-1))
185  nnodes_elmnts(:) = 0
186  nodes_elmnts(:) = 0
187  do i = 0, ngnod*nelmnts-1
188  nodes_elmnts(elmnts(i)*max_nsize_shared+nnodes_elmnts(elmnts(i))) = i/ngnod
189  nnodes_elmnts(elmnts(i)) = nnodes_elmnts(elmnts(i)) + 1
190  enddo
191  endif
192  endif
193 
194  ! local number of each node for each partition
195  call construct_glob2loc_nodes(nproc)
196 
197  ! construct the interfaces between partitions (used for MPI assembly)
198  if (nproc > 1) then
199  if (ngnod == 9) then
200  call construct_interfaces(nproc, elmnts_bis, &
201  nbmodels, phi_read, num_material)
202  else
203  call construct_interfaces(nproc, elmnts, &
204  nbmodels, phi_read, num_material)
205  endif
206  allocate(my_interfaces(0:ninterfaces-1))
207  allocate(my_nb_interfaces(0:ninterfaces-1))
208  else
209  ! dummy allocation
210  ninterfaces=0
211  allocate(my_interfaces(0:ninterfaces-1))
212  allocate(my_nb_interfaces(0:ninterfaces-1))
213  endif
214 
215  end subroutine decompose_mesh
