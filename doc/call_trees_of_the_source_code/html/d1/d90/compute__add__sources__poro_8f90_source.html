34 ! for poro solver
35 
36  subroutine compute_add_sources_poro(accels_poroelastic,accelw_poroelastic,it,i_stage)
37 
38  use constants, only: custom_real,ngllx,ngllz,ndim
39 
40  use specfem_par, only: myrank,ispec_is_poroelastic,nglob_poroelastic, &
41  nsources,source_time_function,sourcearrays, &
42  islice_selected_source,ispec_selected_source, &
43  ibool,porosity,tortuosity,density,kmato
44  implicit none
45 
46  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_poroelastic) :: accels_poroelastic,accelw_poroelastic
47 
48  integer :: it,i_stage
49 
50  !local variables
51  integer :: i_source,i,j,iglob,ispec
52  double precision :: phi,tort,rho_s,rho_f,rho_bar
53  real(kind=CUSTOM_REAL) :: fac_s,fac_w
54  real(kind=CUSTOM_REAL) :: stf_used
55  integer :: material
56 
57  do i_source = 1,nsources
58 
59  ! if this processor core carries the source
60  if (myrank == islice_selected_source(i_source)) then
61 
62  ! element containing source
63  ispec = ispec_selected_source(i_source)
64 
65  ! source element is poroelastic
66  if (ispec_is_poroelastic(ispec)) then
67 
68  material = kmato(ispec)
69  phi = porosity(material)
70  tort = tortuosity(material)
71  rho_s = density(1,material)
72  rho_f = density(2,material)
73 
74  rho_bar = (1.d0 - phi)*rho_s + phi*rho_f
75 
76  fac_s = real((1.d0 - phi/tort),kind=custom_real)
77  fac_w = real((1.d0 - rho_f/rho_bar),kind=custom_real)
78 
79  ! source time function
80  stf_used = source_time_function(i_source,it,i_stage)
81 
82  ! adds source contribution
83  ! note: we use sourcearrays for both, collocated force and moment tensor forces
84  ! (see setup in setup_souces_interpolation() routine), thus can write for both cases the same loop
85  do j = 1,ngllz
86  do i = 1,ngllx
87  iglob = ibool(i,j,ispec)
88 
89  ! solid contribution
90  accels_poroelastic(1,iglob) = accels_poroelastic(1,iglob) + &
91  fac_s * sourcearrays(i_source,1,i,j) * stf_used
92  accels_poroelastic(2,iglob) = accels_poroelastic(2,iglob) + &
93  fac_s * sourcearrays(i_source,2,i,j) * stf_used
94 
95  ! fluid contribution
96  accelw_poroelastic(1,iglob) = accelw_poroelastic(1,iglob) + &
97  fac_w * sourcearrays(i_source,1,i,j) * stf_used
98  accelw_poroelastic(2,iglob) = accelw_poroelastic(2,iglob) + &
99  fac_w * sourcearrays(i_source,2,i,j) * stf_used
100  enddo
101  enddo
102  endif
103  endif ! if this processor core carries the source and the source element is elastic
104  enddo ! do i_source= 1,NSOURCES
105 
106  end subroutine compute_add_sources_poro

112  subroutine compute_add_sources_poro_adjoint()
113 
114  use constants, only: custom_real,ngllx,ngllz
115 
116  use specfem_par, only: nrecloc,nstep,it, &
117  ispec_is_poroelastic,ispec_selected_rec_loc, &
118  ibool,initialfield,simulation_type, &
119  kmato,porosity,density, &
120  accels_poroelastic,accelw_poroelastic, &
121  source_adjoint,xir_store_loc,gammar_store_loc
122 
123  implicit none
124 
125  ! local parameters
126  integer :: irec_local,i,j,iglob,ispec
127  double precision :: phi,rho_s,rho_f,rho_bar
128 
129  ! checks if anything to do
130  if (initialfield) return
131 
132  ! only for adjoint/kernel simulations
133  if (.not. simulation_type == 3) return
134 
135  do irec_local = 1,nrecloc
136  ! element containing adjoint source
137  ispec = ispec_selected_rec_loc(irec_local)
138 
139  if (ispec_is_poroelastic(ispec)) then
140 
141  phi = porosity(kmato(ispec))
142  rho_s = density(1,kmato(ispec))
143  rho_f = density(2,kmato(ispec))
144 
145  rho_bar = (1.d0 - phi)*rho_s + phi*rho_f
146 
147  ! add source array
148  do j = 1,ngllz
149  do i = 1,ngllx
150  iglob = ibool(i,j,ispec)
151 
152  ! solid contribution
153  accels_poroelastic(1,iglob) = accels_poroelastic(1,iglob) + real(xir_store_loc(irec_local,i)*& gammar_store_loc(irec_local,j)*source_adjoint(irec_local,NSTEP-it+1,1),kind=custom_real)
154  accels_poroelastic(2,iglob) = accels_poroelastic(2,iglob) + real(xir_store_loc(irec_local,i)*& gammar_store_loc(irec_local,j)*source_adjoint(irec_local,NSTEP-it+1,2),kind=custom_real)
155 
156  ! fluid
157  accelw_poroelastic(1,iglob) = accelw_poroelastic(1,iglob) - &
158  real(rho_f/rho_bar * xir_store_loc(irec_local,i)*gammar_store_loc(irec_local,j)* & source_adjoint(irec_local,NSTEP-it+1,1),kind=custom_real)
159  accelw_poroelastic(2,iglob) = accelw_poroelastic(2,iglob) - &
160  real(rho_f/rho_bar * xir_store_loc(irec_local,i)*gammar_store_loc(irec_local,j)* & source_adjoint(irec_local,NSTEP-it+1,2),kind=custom_real)
161  enddo
162  enddo
163 
164  endif ! if element is poroelastic
165 
166  enddo ! irec_local = 1,nrecloc
167 
168  end subroutine compute_add_sources_poro_adjoint
