34  subroutine read_material_table()
35 
36 ! reads in material definitions in DATA/Par_file
37 
38  use constants, only: imain,tinyval,isotropic_material,anisotropic_material,poroelastic_material
39 
40  use shared_parameters, only: axisym,nbmodels,icodemat,cp,cs, &
41  aniso3,aniso4,aniso5,aniso6,aniso7,aniso8,aniso9,aniso10,aniso11,aniso12, &
42  comp_g,qkappa,qmu, &
43  rho_s_read,rho_f_read, &
44  phi_read,tortuosity_read, &
45  permxx_read,permxz_read,permzz_read,kappa_s_read,kappa_f_read,kappa_fr_read, &
46  eta_f_read,mu_fr_read
47 
48  implicit none
49 
50  ! local parameters
51  integer :: imaterial,i,icodematread,number_of_materials_defined_by_tomo_file
52  integer :: reread_nbmodels
53  double precision :: val0read,val1read,val2read,val3read,val4read, &
54  val5read,val6read,val7read,val8read,val9read,val10read,val11read,val12read
55 
56  integer,external :: err_occurred
57 
58 
59  ! re-read number of models to reposition read-header
60  call read_value_integer_p(reread_nbmodels, 'mesher.nbmodels')
61  if (err_occurred() /= 0) stop 'error reading parameter nbmodels in Par_file'
62 
63  ! check
64  if (reread_nbmodels /= nbmodels) stop 'Invalid reread value of nbmodels in reading material table'
65 
66  ! safety check
67  if (nbmodels <= 0) stop 'Non-positive number of materials not allowed!'
68 
69  ! allocates material tables
70  allocate(icodemat(nbmodels))
71  allocate(cp(nbmodels))
72  allocate(cs(nbmodels))
73 
74  allocate(aniso3(nbmodels))
75  allocate(aniso4(nbmodels))
76  allocate(aniso5(nbmodels))
77  allocate(aniso6(nbmodels))
78  allocate(aniso7(nbmodels))
79  allocate(aniso8(nbmodels))
80  allocate(aniso9(nbmodels))
81  allocate(aniso10(nbmodels))
82  allocate(aniso11(nbmodels))
83  allocate(aniso12(nbmodels))
84 
85  allocate(comp_g(nbmodels))
86  allocate(qkappa(nbmodels))
87  allocate(qmu(nbmodels))
88 
89  allocate(rho_s_read(nbmodels))
90  allocate(rho_f_read(nbmodels))
91 
92  allocate( phi_read(nbmodels), &
93  tortuosity_read(nbmodels), &
94  permxx_read(nbmodels), &
95  permxz_read(nbmodels), &
96  permzz_read(nbmodels), &
97  kappa_s_read(nbmodels), &
98  kappa_f_read(nbmodels), &
99  kappa_fr_read(nbmodels), &
100  eta_f_read(nbmodels), &
101  mu_fr_read(nbmodels))
102 
103  ! initializes material properties
104  icodemat(:) = 0
105 
106  cp(:) = 0.d0
107  cs(:) = 0.d0
108 
109  aniso3(:) = 0.d0
110  aniso4(:) = 0.d0
111  aniso5(:) = 0.d0
112  aniso6(:) = 0.d0
113  aniso7(:) = 0.d0
114  aniso8(:) = 0.d0
115  aniso9(:) = 0.d0
116  aniso10(:) = 0.d0
117  aniso11(:) = 0.d0
118 
119  comp_g(:) = 0.0d0
120  qkappa(:) = 9999.d0
121  qmu(:) = 9999.d0
122 
123  rho_s_read(:) = 0.d0
124  rho_f_read(:) = 0.d0
125 
126  phi_read(:) = 0.d0
127  tortuosity_read(:) = 0.d0
128  permxx_read(:) = 0.d0
129  permxz_read(:) = 0.d0
130  permzz_read(:) = 0.d0
131  kappa_s_read(:) = 0.d0
132  kappa_f_read(:) = 0.d0
133  kappa_fr_read(:) = 0.d0
134  eta_f_read(:) = 0.d0
135  mu_fr_read(:) = 0.d0
136 
137  number_of_materials_defined_by_tomo_file = 0
138 
139  ! reads in material definitions
140  do imaterial= 1,nbmodels
141 
142  call read_material_parameters_p(i,icodematread, &
143  val0read,val1read,val2read,val3read, &
144  val4read,val5read,val6read,val7read, &
145  val8read,val9read,val10read,val11read,val12read)
146 
147  ! checks material id
148  if (i < 1 .or. i > nbmodels) stop 'Wrong material number!'
149  icodemat(i) = icodematread
150 
151  ! sets material properties
152  if (icodemat(i) == isotropic_material) then
153  ! isotropic materials
154  rho_s_read(i) = val0read
155  cp(i) = val1read
156  cs(i) = val2read
157  comp_g(i) = val3read
158  qkappa(i) = val5read
159  qmu(i) = val6read
160 
161  ! for Cs we use a less restrictive test because acoustic media have Cs exactly equal to zero
162  if (rho_s_read(i) <= 0.00000001d0 .or. cp(i) <= 0.00000001d0 .or. cs(i) < 0.d0) then
163  write(*,*) 'rho,cp,cs=',rho_s_read(i),cp(i),cs(i)
164  stop 'negative value of velocity or density'
165  endif
166  if (qkappa(i) <= 0.00000001d0 .or. qmu(i) <= 0.00000001d0) then
167  stop 'non-positive value of QKappa or Qmu'
168  endif
169 
170  aniso3(i) = val3read
171  aniso4(i) = val4read
172  if (abs(cs(i)) > tinyval) then
173  phi_read(i) = 0.d0 ! elastic
174  else
175  phi_read(i) = 1.d0 ! acoustic
176  endif
177 
178  else if (icodemat(i) == anisotropic_material) then
179  ! anisotropic materials
180  rho_s_read(i) = val0read
181  aniso3(i) = val1read
182  aniso4(i) = val2read
183  aniso5(i) = val3read
184  aniso6(i) = val4read
185  aniso7(i) = val5read
186  aniso8(i) = val6read
187  aniso9(i) = val7read
188  aniso10(i) = val8read
189  aniso11(i) = val9read
190  aniso12(i) = val10read ! This value will be used only in AXISYM
191 
192  else if (icodemat(i) == poroelastic_material) then
193  ! poroelastic materials
194  rho_s_read(i) = val0read
195  rho_f_read(i) = val1read
196  phi_read(i) = val2read
197  tortuosity_read(i) = val3read
198  permxx_read(i) = val4read
199  permxz_read(i) = val5read
200  permzz_read(i) = val6read
201  kappa_s_read(i) = val7read
202  kappa_f_read(i) = val8read
203  kappa_fr_read(i) = val9read
204  eta_f_read(i) = val10read
205  mu_fr_read(i) = val11read
206  qmu(i) = val12read
207 
208  if (rho_s_read(i) <= 0.d0 .or. rho_f_read(i) <= 0.d0) &
209  stop 'non-positive value of density'
210  if (phi_read(i) <= 0.d0 .or. tortuosity_read(i) <= 0.d0) &
211  stop 'non-positive value of porosity or tortuosity'
212  if (kappa_s_read(i) <= 0.d0 .or. kappa_f_read(i) <= 0.d0 .or. kappa_fr_read(i) <= 0.d0 .or. mu_fr_read(i) <= 0.d0) &
213  stop 'non-positive value of modulus'
214  if (qmu(i) <= 0.00000001d0) &
215  stop 'non-positive value of Qmu'
216 
217  else if (icodemat(i) <= 0) then
218  ! tomographic material
219  number_of_materials_defined_by_tomo_file = number_of_materials_defined_by_tomo_file + 1
220 
221  if (number_of_materials_defined_by_tomo_file > 1) &
222  stop 'Just one material can be defined by a tomo file for now (we would need to write a nummaterial_velocity_file)'
223 
224  ! Assign dummy values for now (they will be read by the solver). Vs must be == 0 for acoustic media anyway
225  rho_s_read(i) = -1.0d0
226  cp(i) = -1.0d0
227  cs(i) = val2read
228  qkappa(i) = -1.0d0
229  qmu(i) = -1.0d0
230  aniso3(i) = -1.0d0
231  aniso4(i) = -1.0d0
232  if (abs(cs(i)) > tinyval) then
233  phi_read(i) = 0.d0 ! elastic
234  else
235  phi_read(i) = 1.d0 ! acoustic
236  endif
237 
238  else
239  ! default
240  stop 'Unknown material code'
241 
242  endif
243 
244  enddo ! nbmodels
245 
246  ! user output
247  write(imain,*) 'Materials:'
248  write(imain,*) ' Nb of solid, fluid or porous materials = ',nbmodels
249  write(imain,*)
250 
251  do i = 1,nbmodels
252  if (i == 1) write(imain,*) '--------'
253  if (icodemat(i) == isotropic_material) then
254  write(imain,*) 'Material #',i,' isotropic'
255  write(imain,*) 'rho,cp,cs = ',rho_s_read(i),cp(i),cs(i)
256  write(imain,*) 'Qkappa, Qmu = ',qkappa(i),qmu(i)
257  if (cs(i) < tinyval) then
258  write(imain,*) 'Material is fluid'
259  else
260  write(imain,*) 'Material is solid'
261  endif
262  else if (icodemat(i) == anisotropic_material) then
263  write(imain,*) 'Material #',i,' anisotropic'
264  write(imain,*) 'rho,cp,cs = ',rho_s_read(i),cp(i),cs(i)
265  if (axisym) then
266  write(imain,*) 'c11,c13,c15,c33,c35,c55,c12,c23,c25,c22 = ', &
267  aniso3(i),aniso4(i),aniso5(i),aniso6(i),aniso7(i),aniso8(i), &
268  aniso9(i),aniso10(i),aniso11(i),aniso12(i)
269  else
270  write(imain,*) 'c11,c13,c15,c33,c35,c55,c12,c23,c25 = ', &
271  aniso3(i),aniso4(i),aniso5(i),aniso6(i),aniso7(i),aniso8(i), &
272  aniso9(i),aniso10(i),aniso11(i)
273  write(imain,*) 'QKappa,Qmu = ',qkappa(i),qmu(i)
274  endif
275  else if (icodemat(i) == poroelastic_material) then
276  write(imain,*) 'Material #',i,' isotropic'
277  write(imain,*) 'rho_s, kappa_s = ',rho_s_read(i),kappa_s_read(i)
278  write(imain,*) 'rho_f, kappa_f, eta_f = ',rho_f_read(i),kappa_f_read(i),eta_f_read(i)
279  write(imain,*) 'phi, tortuosity = ',phi_read(i),tortuosity_read(i)
280  write(imain,*) 'permxx, permxz, permzz = ',permxx_read(i),permxz_read(i),permzz_read(i)
281  write(imain,*) 'kappa_fr, mu_fr, Qmu = ',kappa_fr_read(i),mu_fr_read(i),qmu(i)
282  write(imain,*) 'Material is porous'
283  else if (icodemat(i) <= 0) then
284  write(imain,*) 'Material #',i,' will be read in an external tomography file (TOMOGRAPHY_FILE in Par_file)'
285  else
286  stop 'Unknown material code'
287  endif
288  write(imain,*) '--------'
289  call flush_imain()
290  enddo
291 
292  end subroutine read_material_table
