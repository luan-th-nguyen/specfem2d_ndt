34  subroutine compute_vector_whole_medium(field_acoustic,field_gravitoacoustic, &
35  field_gravito,field_elastic,fields_poroelastic)
36 
37 ! compute Grad(potential) in acoustic elements
38 ! and combine with existing velocity vector field in elastic elements
39 
40  use constants, only: custom_real,ngllx,ngllz,ndim
41  use specfem_par, only: nspec,ibool, &
42  nglob_acoustic,nglob_elastic,nglob_gravitoacoustic,nglob_poroelastic
43 
44  use specfem_par_movie, only: vector_field_display
45 
46  implicit none
47 
48  real(kind=CUSTOM_REAL), dimension(nglob_acoustic) :: field_acoustic
49  real(kind=CUSTOM_REAL), dimension(nglob_gravitoacoustic) :: field_gravitoacoustic
50  real(kind=CUSTOM_REAL), dimension(nglob_gravitoacoustic) :: field_gravito
51  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_elastic) :: field_elastic
52  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_poroelastic) :: fields_poroelastic
53  ! vector field in an element
54  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ) :: vector_field_element
55 
56  ! local parameters
57  integer :: i,j,ispec,iglob
58 
59  ! loop over spectral elements
60  do ispec = 1,nspec
61  ! computes vector field in this element
62  call compute_vector_one_element(field_acoustic,field_gravitoacoustic, &
63  field_gravito,field_elastic,fields_poroelastic,ispec,vector_field_element)
64 
65  ! stores the result on global nodes
66  do j = 1,ngllz
67  do i = 1,ngllx
68  iglob = ibool(i,j,ispec)
69  vector_field_display(:,iglob) = vector_field_element(:,i,j)
70  enddo
71  enddo
72  enddo
73 
74  end subroutine compute_vector_whole_medium

80  subroutine compute_vector_one_element(field_acoustic,field_gravitoacoustic, &
81  field_gravito,field_elastic,fields_poroelastic,ispec,vector_field_element)
82 
83 ! compute Grad(potential) if acoustic element or copy existing vector if elastic element
84 
85  use constants, only: custom_real,ngllx,ngllz,ndim
86 
87  use specfem_par, only: nglob_acoustic,nglob_elastic,nglob_gravitoacoustic,nglob_poroelastic, &
88  assign_external_model,density,kmato,gravityext,rhoext, &
89  hprimebar_xx,hprime_xx,hprime_zz, &
90  xix,xiz,gammax,gammaz,ibool, &
91  ispec_is_elastic,ispec_is_poroelastic,ispec_is_acoustic,ispec_is_gravitoacoustic, &
92  axisym,is_on_the_axis, &
93  p_sv
94 
95  implicit none
96 
97  real(kind=CUSTOM_REAL), dimension(nglob_acoustic) :: field_acoustic
98  real(kind=CUSTOM_REAL), dimension(nglob_gravitoacoustic) :: field_gravitoacoustic
99  real(kind=CUSTOM_REAL), dimension(nglob_gravitoacoustic) :: field_gravito
100  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_elastic) :: field_elastic
101  real(kind=CUSTOM_REAL), dimension(NDIM,nglob_poroelastic) :: fields_poroelastic
102 
103  integer,intent(in) :: ispec
104 
105  ! vector field in an element
106  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ),intent(out) :: vector_field_element
107 
108  ! local variables
109  integer i,j,k,iglob
110  ! Jacobian matrix and determinant
111  double precision :: xixl,xizl,gammaxl,gammazl
112  double precision :: gravityl,hp1,hp2
113  double precision :: rhol
114  double precision :: tempx1l,tempx2l
115 
116  ! initializes
117  vector_field_element(:,:,:) = 0._custom_real
118 
119  ! determines vector field
120  if (ispec_is_elastic(ispec)) then
121  ! elastic element
122  ! simple copy of existing vector if elastic element
123  do j = 1,ngllz
124  do i = 1,ngllx
125  iglob = ibool(i,j,ispec)
126  if (p_sv) then
127  ! P_SV case
128  vector_field_element(1,i,j) = field_elastic(1,iglob)
129  vector_field_element(2,i,j) = field_elastic(2,iglob)
130  else
131  ! SH case
132  vector_field_element(1,i,j) = field_elastic(1,iglob)
133  endif
134  enddo
135  enddo
136 
137  else if (ispec_is_poroelastic(ispec)) then
138  ! poro-elastic element
139  ! simple copy of existing vector if poroelastic element
140  do j = 1,ngllz
141  do i = 1,ngllx
142  iglob = ibool(i,j,ispec)
143  vector_field_element(1,i,j) = fields_poroelastic(1,iglob)
144  vector_field_element(2,i,j) = fields_poroelastic(2,iglob)
145  enddo
146  enddo
147 
148  else if (ispec_is_acoustic(ispec)) then
149  ! acoustic element
150 
151  ! compute gradient of potential to calculate vector if acoustic element
152  ! we then need to divide by density because the potential is a potential of (density * displacement)
153  rhol = density(1,kmato(ispec))
154 
155  ! double loop over GLL points to compute and store gradients
156  do j = 1,ngllz
157  do i = 1,ngllx
158  ! derivative along x
159  tempx1l = 0._custom_real
160  if (axisym) then
161  if (is_on_the_axis(ispec)) then
162  do k = 1,ngllx
163  hp1 = hprimebar_xx(i,k)
164  iglob = ibool(k,j,ispec)
165  tempx1l = tempx1l + field_acoustic(iglob)*hp1
166  enddo
167  else
168  !AXISYM but not on the axis
169  do k = 1,ngllx
170  hp1 = hprime_xx(i,k)
171  iglob = ibool(k,j,ispec)
172  tempx1l = tempx1l + field_acoustic(iglob)*hp1
173  enddo
174  endif
175  else
176  !not AXISYM
177  do k = 1,ngllx
178  hp1 = hprime_xx(i,k)
179  iglob = ibool(k,j,ispec)
180  tempx1l = tempx1l + field_acoustic(iglob)*hp1
181  enddo
182  endif
183 
184  ! derivative along z
185  tempx2l = 0._custom_real
186  do k = 1,ngllz
187  hp2 = hprime_zz(j,k)
188  iglob = ibool(i,k,ispec)
189  tempx2l = tempx2l + field_acoustic(iglob)*hp2
190  enddo
191 
192  xixl = xix(i,j,ispec)
193  xizl = xiz(i,j,ispec)
194  gammaxl = gammax(i,j,ispec)
195  gammazl = gammaz(i,j,ispec)
196 
197  if (assign_external_model) rhol = rhoext(i,j,ispec)
198 
199  ! derivatives of potential
200  vector_field_element(1,i,j) = real((tempx1l*xixl + tempx2l*gammaxl) / rhol,kind=CUSTOM_REAL) ! u_x = 1/rho dChi/dx
201  vector_field_element(2,i,j) = real((tempx1l*xizl + tempx2l*gammazl) / rhol,kind=CUSTOM_REAL) ! u_z = 1/rho dChi/dz
202  enddo
203  enddo
204 
205  else if (ispec_is_gravitoacoustic(ispec)) then
206  ! gravito-acoustic element
207 
208  ! compute gradient of potential to calculate vector if gravitoacoustic element
209  ! we then need to divide by density because the potential is a potential of (density * displacement)
210  rhol = density(1,kmato(ispec))
211 
212  ! double loop over GLL points to compute and store gradients
213  do j = 1,ngllz
214  do i = 1,ngllx
215  ! derivative along x
216  tempx1l = 0._custom_real
217  do k = 1,ngllx
218  hp1 = hprime_xx(i,k)
219  iglob = ibool(k,j,ispec)
220  tempx1l = tempx1l + field_gravitoacoustic(iglob)*hp1
221  enddo
222 
223  ! derivative along z
224  tempx2l = 0._custom_real
225  do k = 1,ngllz
226  hp2 = hprime_zz(j,k)
227  iglob = ibool(i,k,ispec)
228  tempx2l = tempx2l + field_gravitoacoustic(iglob)*hp2
229  enddo
230 
231  xixl = xix(i,j,ispec)
232  xizl = xiz(i,j,ispec)
233  gammaxl = gammax(i,j,ispec)
234  gammazl = gammaz(i,j,ispec)
235 
236  if (assign_external_model) then
237  rhol = rhoext(i,j,ispec)
238  gravityl = gravityext(i,j,ispec)
239  endif
240 
241  ! derivatives of potential
242  vector_field_element(1,i,j) = real((tempx1l*xixl + tempx2l*gammaxl) / rhol,kind=custom_real)
243  vector_field_element(2,i,j) = real((tempx1l*xizl + tempx2l*gammazl) / rhol,kind=custom_real)
244 
245  ! add the gravito potential along the z component
246  iglob = ibool(i,j,ispec)
247  ! remove gravito contribution
248  ! sign gravito correction
249  vector_field_element(2,i,j) = vector_field_element(2,i,j) - real((field_gravito(iglob)*gravityl) / rhol,kind=custom_real)
250 
251  enddo
252  enddo
253 
254  endif ! end of test if acoustic or elastic element
255 
256  end subroutine compute_vector_one_element
