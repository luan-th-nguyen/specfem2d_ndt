34  subroutine pml_init()
35 
36 ! Rough strategy to select the PML damping parameters:
37 
38 ! In order to achieve absorbing for gradient incident wave for zPML:
39 ! we need to set K>1 to bending the wavefront aside from its tangential propagation direction.
40 ! Then a bigger damping factor d and alpha could help damping out the
41 ! bended wave along its propagation close to normal direction.
42 
43 ! Keeping in mind, with K>1, we decrease the absorbing efficiency a little
44 ! bit in normal direction.
45 ! Though not exact, you can thought like this: for example with K_x>1, you
46 ! scale the element size in x direction with a big factor.
47 ! Then the resolution decrease in PML but not in interior, thus you can
48 ! see reflections due to different resolutions in two sides of PML interface.
49 ! However if the resolution is already very accurate, that is you have a
50 ! relatively fine grid, then the above effect will be small.
51 
52 
53  use constants, only: imain,ngllx,ngllz,iright,ileft,ibottom,itop,cpml_x_only,cpml_z_only,cpml_xz
54 
55  use specfem_par, only: myrank,simulation_type,save_forward,nspec,nglob,ibool, &
56  anyabs,nelemabs,codeabs,numabs, &
57  nglob_interface,read_external_mesh
58 
59  ! PML arrays
60  use specfem_par, only: nspec_pml,ispec_is_pml,spec_to_pml,region_cpml,which_pml_elem, &
61  mask_ibool_pml,nelem_pml_thickness,pml_interior_interface
62 
63  implicit none
64 
65  ! local parameters
66  integer, dimension(nglob) :: icorner_iglob
67  integer :: nspec_PML_tot,ibound,ispecabs,ncorner,i_coef,i,j,k,ispec,iglob
68 
69  nspec_pml = 0
70 
71  ! detection of PML elements
72  if (.not. read_external_mesh) then
73 
74  ! ibound is the side we are looking (bottom, right, top or left)
75  do ibound=1,4
76  icorner_iglob = 0
77  ncorner=0
78 
79  if (anyabs) then
80  ! mark any elements on the boundary as PML and list their corners
81  do ispecabs = 1,nelemabs
82  ispec = numabs(ispecabs)
83  !array to know which PML it is
84  which_pml_elem(ibound,ispec)=codeabs(ibound,ispecabs)
85  if (codeabs(ibound,ispecabs)) then ! we are on the good absorbing boundary
86  do j = 1,ngllz,ngllz-1; do i = 1,ngllx,ngllx-1
87  iglob=ibool(i,j,ispec)
88  k=1
89  do while(k <= ncorner .and. icorner_iglob(k) /= iglob)
90  k=k+1
91  enddo
92  ncorner=ncorner+1
93  icorner_iglob(ncorner) = iglob
94  enddo; enddo
95  endif ! we are on the good absorbing boundary
96  enddo
97  endif
98 
99  !find elements stuck to boundary elements to define the 4 elements PML thickness
100  !we take 4 elements for the PML thickness
101  do i_coef= 2,nelem_pml_thickness
102 
103  do ispec= 1,nspec
104  if (.not. which_pml_elem(ibound,ispec)) then
105  do j = 1,ngllz,ngllz-1; do i = 1,ngllx,ngllx-1
106  iglob=ibool(i,j,ispec)
107  do k = 1,ncorner
108  if (iglob == icorner_iglob(k)) which_pml_elem(ibound,ispec) = .true.
109  enddo
110  enddo; enddo
111  endif
112  enddo
113 
114  ! list every corner of each PML element detected
115  ncorner=0
116  icorner_iglob=0
117  nspec_pml=0
118  do ispec= 1,nspec
119  if (which_pml_elem(ibound,ispec)) then
120  ispec_is_pml(ispec) = .true.
121  do j = 1,ngllz,ngllz-1; do i = 1,ngllx,ngllx-1
122  iglob=ibool(i,j,ispec)
123  k=1
124  do while(k <= ncorner .and. icorner_iglob(k) /= iglob)
125  k=k+1
126  enddo
127  ncorner=ncorner+1
128  icorner_iglob(ncorner) = iglob
129  enddo; enddo
130  nspec_pml=nspec_pml+1
131  endif
132  enddo
133 
134  enddo !end nelem_thickness loop
135 
136  if (simulation_type == 3 .or. (simulation_type == 1 .and. save_forward)) then
137 
138  do i_coef=nelem_pml_thickness,nelem_pml_thickness+1
139  do ispec= 1,nspec
140  if (.not. which_pml_elem(ibound,ispec)) then
141  do j = 1,ngllz,ngllz-1; do i = 1,ngllx,ngllx-1
142  iglob=ibool(i,j,ispec)
143  do k = 1,ncorner
144  if (iglob == icorner_iglob(k)) pml_interior_interface(ibound,ispec) = .true.
145  enddo
146  enddo; enddo
147  endif
148  enddo
149  enddo !end nelem_thickness loop
150 
151  endif !end of SIMULATION_TYPE == 3
152 
153  enddo ! end loop on the four boundaries
154 
155  if (simulation_type == 3 .or. (simulation_type == 1 .and. save_forward)) then
156  nglob_interface = 0
157  do ispec = 1,nspec
158  if (pml_interior_interface(ibottom,ispec) .and. (.not. pml_interior_interface(iright,ispec)) .and. &
159  (.not. pml_interior_interface(ileft,ispec)) .and. (.not. which_pml_elem(iright,ispec)) .and. &
160  (.not. which_pml_elem(ileft,ispec))) then
161  nglob_interface = nglob_interface + 5
162  else if (pml_interior_interface(itop,ispec) .and. (.not. pml_interior_interface(iright,ispec)) .and. &
163  (.not. pml_interior_interface(ileft,ispec)) .and. (.not. which_pml_elem(iright,ispec)) .and. &
164  (.not. which_pml_elem(ileft,ispec))) then
165  nglob_interface = nglob_interface + 5
166  else if (pml_interior_interface(iright,ispec) .and. (.not. pml_interior_interface(ibottom,ispec)) .and. &
167  (.not. pml_interior_interface(itop,ispec)) .and. (.not. which_pml_elem(ibottom,ispec)) .and. &
168  (.not. which_pml_elem(itop,ispec))) then
169  nglob_interface = nglob_interface + 5
170  else if (pml_interior_interface(ileft,ispec) .and. (.not. pml_interior_interface(ibottom,ispec)) .and. &
171  (.not. pml_interior_interface(itop,ispec)) .and. (.not. which_pml_elem(ibottom,ispec)) .and. &
172  (.not. which_pml_elem(itop,ispec))) then
173  nglob_interface = nglob_interface + 5
174  else if (pml_interior_interface(ileft,ispec) .and. pml_interior_interface(ibottom,ispec)) then
175  nglob_interface = nglob_interface + 10
176  else if (pml_interior_interface(iright,ispec) .and. pml_interior_interface(ibottom,ispec)) then
177  nglob_interface = nglob_interface + 10
178  else if (pml_interior_interface(ileft,ispec) .and. pml_interior_interface(itop,ispec)) then
179  nglob_interface = nglob_interface + 10
180  else if (pml_interior_interface(iright,ispec) .and. pml_interior_interface(itop,ispec)) then
181  nglob_interface = nglob_interface + 10
182  endif
183  enddo
184  endif
185 
186  do ispec= 1,nspec
187  if (ispec_is_pml(ispec)) then
188 ! element is in the left cpml layer
189  if ((which_pml_elem(ileft,ispec) .eqv. .true.) .and. (which_pml_elem(iright,ispec) .eqv. .false.) .and. &
190  (which_pml_elem(itop,ispec) .eqv. .false.) .and. (which_pml_elem(ibottom,ispec) .eqv. .false.)) then
191  region_cpml(ispec) = cpml_x_only
192 ! element is in the right cpml layer
193  else if ((which_pml_elem(ileft,ispec) .eqv. .false.) .and. (which_pml_elem(iright,ispec) .eqv. .true.) .and. &
194  (which_pml_elem(itop,ispec) .eqv. .false.) .and. (which_pml_elem(ibottom,ispec) .eqv. .false.)) then
195  region_cpml(ispec) = cpml_x_only
196 ! element is in the top cpml layer
197  else if ((which_pml_elem(ileft,ispec) .eqv. .false.) .and. (which_pml_elem(iright,ispec) .eqv. .false.) .and. &
198  (which_pml_elem(itop,ispec) .eqv. .true. ) .and. (which_pml_elem(ibottom,ispec) .eqv. .false.)) then
199  region_cpml(ispec) = cpml_z_only
200 ! element is in the bottom cpml layer
201  else if ((which_pml_elem(ileft,ispec) .eqv. .false.) .and. (which_pml_elem(iright,ispec) .eqv. .false.) .and. &
202  (which_pml_elem(itop,ispec) .eqv. .false.) .and. (which_pml_elem(ibottom,ispec) .eqv. .true. )) then
203  region_cpml(ispec) = cpml_z_only
204 ! element is in the left-top cpml corner
205  else if ((which_pml_elem(ileft,ispec) .eqv. .true. ) .and. (which_pml_elem(iright,ispec) .eqv. .false.) .and. &
206  (which_pml_elem(itop,ispec) .eqv. .true. ) .and. (which_pml_elem(ibottom,ispec) .eqv. .false.)) then
207  region_cpml(ispec) = cpml_xz
208 ! element is in the right-top cpml corner
209  else if ((which_pml_elem(ileft,ispec) .eqv. .false. ) .and. (which_pml_elem(iright,ispec) .eqv. .true. ) .and. &
210  (which_pml_elem(itop,ispec) .eqv. .true. ) .and. (which_pml_elem(ibottom,ispec) .eqv. .false.)) then
211  region_cpml(ispec) = cpml_xz
212 ! element is in the left-bottom cpml corner
213  else if ((which_pml_elem(ileft,ispec) .eqv. .true. ) .and. (which_pml_elem(iright,ispec) .eqv. .false.) .and. &
214  (which_pml_elem(itop,ispec) .eqv. .false. ) .and. (which_pml_elem(ibottom,ispec) .eqv. .true. )) then
215  region_cpml(ispec) = cpml_xz
216 ! element is in the right-bottom cpml corner
217  else if ((which_pml_elem(ileft,ispec) .eqv. .false. ) .and. (which_pml_elem(iright,ispec) .eqv. .true.) .and. &
218  (which_pml_elem(itop,ispec) .eqv. .false. ) .and. (which_pml_elem(ibottom,ispec) .eqv. .true.)) then
219  region_cpml(ispec) = cpml_xz
220  else
221  region_cpml(ispec) = 0
222  endif
223  endif
224  enddo
225 
226  !construction of table to use less memory for absorbing coefficients
227  spec_to_pml=0
228  nspec_pml=0
229  do ispec= 1,nspec
230  if (ispec_is_pml(ispec)) then
231  nspec_pml = nspec_pml+1
232  spec_to_pml(ispec) = nspec_pml
233  endif
234  enddo
235 
236  endif !end of detection of element inside PML layer for inner mesher
237 
238 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
239  if (read_external_mesh) then
240 
241  if (.not. allocated(mask_ibool_pml)) allocate(mask_ibool_pml(nglob))
242 
243  ispec_is_pml(:) = .false.
244  which_pml_elem(:,:) = .false.
245  nspec_pml = 0
246  spec_to_pml=0
247  mask_ibool_pml(:) = .false.
248  do ispec= 1,nspec
249  if (region_cpml(ispec) /= 0) then
250  nspec_pml = nspec_pml + 1
251  ispec_is_pml(ispec) = .true.
252  spec_to_pml(ispec) = nspec_pml
253  endif
254 
255  if (simulation_type == 3 .or. (simulation_type == 1 .and. save_forward)) then
256  if (region_cpml(ispec) == 0) then
257  do i = 1, ngllx; do j = 1, ngllz
258  iglob = ibool(i,j,ispec)
259  mask_ibool_pml(iglob) = .true.
260  enddo; enddo
261  endif
262  endif
263  enddo
264 
265  nglob_interface = 0
266  if (simulation_type == 3 .or. (simulation_type == 1 .and. save_forward)) then
267  do ispec= 1,nspec
268  if (region_cpml(ispec) /= 0) then
269  do i = 1, ngllx; do j = 1, ngllz
270  iglob = ibool(i,j,ispec)
271  if (mask_ibool_pml(iglob)) nglob_interface = nglob_interface + 1
272  enddo; enddo
273  endif
274  enddo
275  endif
276 
277  endif
278 
279  ! outputs total
280  call sum_all_i(nspec_pml,nspec_pml_tot)
281  if (myrank == 0) then
282  write(imain,*) "Total number of PML spectral elements: ", nspec_pml_tot
283  write(imain,*)
284  call flush_imain()
285  endif
286 
287  end subroutine pml_init

417  subroutine define_pml_coefficients()
418 
419  use constants, only: pi,ngllx,ngllz,cpml_x_only,cpml_z_only,cpml_xz
420 
421  use specfem_par, only: f0_source,ispec_is_elastic,ispec_is_acoustic, &
422  nsources,ispec_selected_source, &
423  nspec,kmato,density,poroelastcoef,ibool,coord
424 
425  ! PML arrays
426  use specfem_par, only: ispec_is_pml,spec_to_pml,region_cpml, &
427  k_x_store,k_z_store,d_x_store,d_z_store,alpha_x_store,alpha_z_store

429 
430  implicit none
431 
432 !ZN an automatic adjustment of PML parameter for elongated model.
433 !ZN The goal here is to improve the absorbing efficiency of PML for waves with large incident angle,
434 !ZN while keep the layer thickness of PML constant.
435 !ZN The idea behind this optimization is illustrated at the top the this file.
436 
437 !ZN we start this simple automatic adjustment only when you allow to do it, that is
438 !ZN The flag PML_parameter_adjustment_for_elongated_model = .true.
439 !ZN Since an bigger K only works when the mesh resolution in PML are fine enough,
440 !ZN we require an fine mesh to do PML_parameter_adjustment, that is the maximum elment size in PML
