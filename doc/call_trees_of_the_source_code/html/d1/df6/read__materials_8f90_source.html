34  subroutine read_materials(f0)
35 
36 ! reads properties of a 2D isotropic or anisotropic linear elastic element
37 
38  use constants, only: iin,imain,zero,four_thirds,two_thirds,half,tinyval
39 
40  use specfem_par, only: axisym,density,porosity,tortuosity,anisotropy,permeability,poroelastcoef, &
41  numat,myrank,qkappa_attenuation,qmu_attenuation, &
42  freq0,q0,attenuation_poro_fluid_part,assign_external_model,tomo_material,myrank
43 
44  implicit none
45 
46  ! frequency used for poroelastic velocities
47  double precision,intent(in) :: f0
48 
49  ! local parameters
50  double precision :: lambdaplus2mu,kappa,compaction_grad
51  double precision :: young,poisson,cp,cs,mu,two_mu,lambda,QKappa,Qmu
52  double precision :: lambdaplus2mu_s,lambdaplus2mu_fr,kappa_s,kappa_f,kappa_fr
53  double precision :: young_s,poisson_s,density_mat(2),phi,tortuosity_mat
54  double precision :: cpIsquare,cpIIsquare,cssquare,mu_s,mu_fr,eta_f,lambda_s,lambda_fr
55  double precision :: c11,c13,c15,c33,c35,c55,c12,c23,c25,c22
56  double precision :: val0,val1,val2,val3,val4,val5,val6,val7,val8,val9,val10,val11,val12
57 
58  double precision :: D_biot,H_biot,C_biot,M_biot
59 
60  double precision :: w_c
61  integer :: imat,n,indic
62 
63  !
64  !---- loop over the different material sets
65  !
66  density(:,:) = zero
67  porosity(:) = zero
68  tortuosity(:) = zero
69  anisotropy(:,:) = zero
70  permeability(:,:) = zero
71  poroelastcoef(:,:,:) = zero
72 
73  qkappa_attenuation(:) = 9999.
74  qmu_attenuation(:) = 9999.
75 
76  ! Index of the material that will be defined by an external tomo file if needed (TOMOGRAPHY_FILE)
77  tomo_material = 0
78 
79  ! user output
80  ! number of materials
81  if (myrank == 0) write(imain,100) numat
82 
83  ! skips material sets header
84  do imat = 1,numat
85 
86  read(iin) n,indic,val0,val1,val2,val3,val4,val5,val6,val7,val8,val9,val10,val11,val12
87 
88  if (n < 1 .or. n > numat) call exit_mpi(myrank,'Wrong material set number')
89 
90  !---- isotropic material, P and S velocities given, allows for declaration of elastic/acoustic material
91  !---- elastic (cs /= 0) and acoustic (cs = 0)
92  if (indic == 1) then
93  density_mat(1) = val0
94 
95  ! P and S velocity
96  cp = val1
97  cs = val2
98  compaction_grad = val3
99  ! QKappa and Qmu values
100  qkappa = val5
101  qmu = val6
102  if (qkappa <= 0.0000001 .or. qmu <= 0.0000001) &
103  stop 'negative or null values of Q attenuation factor not allowed; set them equal to 9999 to indicate no attenuation'
104 
105  ! Lame parameters
106  lambdaplus2mu = density_mat(1)*cp*cp
107  mu = density_mat(1)*cs*cs
108  two_mu = 2.d0*mu
109  lambda = lambdaplus2mu - two_mu
110 
111  ! bulk modulus Kappa
112 
113  if (axisym) then ! CHECK kappa
114  kappa = lambda + two_thirds * mu
115  else
116  kappa = lambda + mu
117  endif
118 
119  ! Young modulus
120  young = 9.d0*kappa*mu/(3.d0*kappa + mu)
121 
122  ! Poisson's ratio
123  poisson = half * (3.d0*kappa-two_mu)/(3.d0*kappa+mu)
124 
125  ! Poisson's ratio must be between -1 and +1/2
126  if (poisson < -1.d0 .or. poisson > 0.5d0) call exit_mpi(myrank,'Poisson''s ratio out of range')
127 
128  !---- anisotropic material, c11, c13, c33 and c44 given in Pascal
129  else if (indic == 2) then
130 
131  density_mat(1) = val0
132 
133  ! Anisotropy parameters
134  c11 = val1
135  c13 = val2
136  c15 = val3
137  c33 = val4
138  c35 = val5
139  c55 = val6
140  c12 = val7
141  c23 = val8
142  c25 = val9
143  c22 = val10 ! This value is used for AXISYM only
144 
145  ! P and S velocity
146  cp = sqrt(c33/density_mat(1))
147  cs = sqrt(c55/density_mat(1))
148 
149  ! Lame parameters
150  lambdaplus2mu = density_mat(1)*cp*cp
151  mu = density_mat(1)*cs*cs
152  two_mu = 2.d0*mu
153  lambda = lambdaplus2mu - two_mu
154 
155  ! bulk modulus Kappa
156 
157  if (axisym) then ! CHECK kappa
158  kappa = lambda + two_thirds * mu
159  else
160  kappa = lambda + mu
161  endif
162 
163  ! Young modulus
164  young = 9.d0*kappa*mu/(3.d0*kappa + mu)
165 
166  ! Poisson's ratio
167  poisson = half * (3.d0*kappa-two_mu)/(3.d0*kappa+mu)
168 
169  !---- isotropic material, moduli are given, allows for declaration of poroelastic material
170  !---- poroelastic (0 < phi < 1)
171  else if (indic == 3) then
172  ! Qmu values
173  qmu = val12
174 
175  density_mat(1) = val0
176  density_mat(2) = val1
177 
178  phi = val2
179  tortuosity_mat = val3
180 
181  ! Solid properties
182  kappa_s = val7
183  mu_s = val11
184  ! Fluid properties
185  kappa_f = val8
186  eta_f = val10
187  ! Frame properties
188  kappa_fr = val9
189  mu_fr = val11
190 
191  ! Lame parameters for the solid phase and the frame
192  !if (AXISYM) then ! ABAB !! Warning !! This is false for plane strain (look for: bulk modulus plane strain) Check Kappa
193  lambdaplus2mu_s = kappa_s + four_thirds*mu_s
194  lambda_s = lambdaplus2mu_s - 2.d0*mu_s
195  lambdaplus2mu_fr = kappa_fr + four_thirds*mu_fr
196  lambda_fr = lambdaplus2mu_fr - 2.d0*mu_fr
197  !else ! Correct lines:
198  ! lambdaplus2mu_s = kappa_s - mu_s
199  ! lambda_s = lambdaplus2mu_s - 2.d0*mu_s
200  ! lambdaplus2mu_fr = kappa_fr - mu_fr
201  ! lambda_fr = lambdaplus2mu_fr - 2.d0*mu_fr
202  !endif
203 
204  ! Biot coefficients for the input phi
205  call get_poroelastic_biot_coeff(phi,kappa_s,kappa_f,kappa_fr,mu_fr,d_biot,h_biot,c_biot,m_biot)
206 
207  call get_poroelastic_velocities(cpisquare,cpiisquare,cssquare, &
208  h_biot,c_biot,m_biot,mu_fr,phi, &
209  tortuosity_mat,density_mat(1),density_mat(2),eta_f, &
210  val4,f0,freq0,q0,w_c,attenuation_poro_fluid_part)
211 
212  porosity(n) = val2
213  tortuosity(n) = val3
214 
215  permeability(1,n) = val4
216  permeability(2,n) = val5
217  permeability(3,n) = val6
218 
219  ! Young modulus for the solid phase
220  young_s = 9.d0*kappa_s*mu_s/(3.d0*kappa_s + mu_s)
221 
222  ! Poisson's ratio for the solid phase
223  poisson_s = half * (3.d0*kappa_s- 2.d0*mu_s)/(3.d0*kappa_s+mu_s)
224 
225  ! Poisson's ratio must be between -1 and +1/2
226  if (poisson_s < -1.d0 .or. poisson_s > 0.5d0) stop 'Poisson''s ratio for the solid phase out of range'
227 
228  else if (indic <= 0) then
229  assign_external_model = .true.
230  tomo_material = n
231  mu = val2 ! for acoustic medium vs must be 0 anyway
232 
233  else
234  ! should not occur
235  call exit_mpi(myrank,'wrong model flag read')
236  endif
237 
238  !
239  !---- set elastic coefficients and density
240  !
241  if (indic == 1) then
242  density(1,n) = density_mat(1)
243  poroelastcoef(1,1,n) = lambda
244  poroelastcoef(2,1,n) = mu
245  poroelastcoef(3,1,n) = lambdaplus2mu
246  poroelastcoef(4,1,n) = compaction_grad
247  qkappa_attenuation(n) = qkappa
248  qmu_attenuation(n) = qmu
249  if (mu > tinyval) then
250  porosity(n) = 0.d0
251  else
252  porosity(n) = 1.d0
253  endif
254 
255  else if (indic == 2) then
256  density(1,n) = density_mat(1)
257  ! dummy poroelastcoef values, trick to avoid floating invalid
258  poroelastcoef(1,1,n) = lambda
259  poroelastcoef(2,1,n) = mu
260  poroelastcoef(3,1,n) = lambdaplus2mu
261  poroelastcoef(4,1,n) = zero
262  anisotropy(1,n) = c11
263  anisotropy(2,n) = c13
264  anisotropy(3,n) = c15
265  anisotropy(4,n) = c33
266  anisotropy(5,n) = c35
267  anisotropy(6,n) = c55
268  anisotropy(7,n) = c12
269  anisotropy(8,n) = c23
270  anisotropy(9,n) = c25
271  anisotropy(10,n) = c22 ! This value is used for AXISYM only
272  porosity(n) = 0.d0
273 
274  else if (indic == 3) then
275  density(1,n) = density_mat(1)
276  density(2,n) = density_mat(2)
277  poroelastcoef(1,1,n) = lambda_s
278  poroelastcoef(2,1,n) = mu_s ! = mu_fr
279  poroelastcoef(3,1,n) = lambdaplus2mu_s
280  poroelastcoef(4,1,n) = zero
281 
282  poroelastcoef(1,2,n) = kappa_f
283  poroelastcoef(2,2,n) = eta_f
284  poroelastcoef(3,2,n) = zero
285  poroelastcoef(4,2,n) = zero
286 
287  poroelastcoef(1,3,n) = lambda_fr
288  poroelastcoef(2,3,n) = mu_fr
289  poroelastcoef(3,3,n) = lambdaplus2mu_fr
290  poroelastcoef(4,3,n) = zero
291 
292  else if (indic <= 0) then
293  ! assign dummy values for now (for acoustic medium vs must be 0 anyway), these values will be read in read_external_model
294  density(1,n) = -1.0d0
295  poroelastcoef(1,1,n) = -1.0d0
296  poroelastcoef(2,1,n) = mu
297  poroelastcoef(3,1,n) = -1.0d0
298  poroelastcoef(4,1,n) = zero
299  qkappa_attenuation(n) = 9999.
300  qmu_attenuation(n) = 9999.
301  if (mu > tinyval) then
302  porosity(n) = 0.d0
303  else
304  porosity(n) = 1.d0
305  endif
306 
307  else
308  ! should not occur
309  call exit_mpi(myrank,'wrong model flag read')
310  endif
311 
312  !
313  !---- check what has been read
314  !
315  if (myrank == 0) then
316  ! user output
317  if (indic == 1) then
318  ! material can be acoustic (fluid) or elastic (solid)
319  if (poroelastcoef(2,1,n) > tinyval) then
320  ! elastic
321  write(imain,200) n,cp,cs,density_mat(1),poisson,lambda,mu,kappa,young,qkappa,qmu
322  if (poisson < 0.d0) then
323  write(imain,*)
324  write(imain,*) 'Materials with a negative Poisson''s ratio can exist,'
325  write(imain,*) 'see e.g. R. Lakes, "Science" vol. 235, p. 1038-1040 (1987),'
326  write(imain,*) 'but are extremely rare.'
327  write(imain,*) 'Hope you know what you are doing...'
328  write(imain,*)
329  endif
330  else
331  ! acoustic
332  write(imain,300) n,cp,density_mat(1),kappa,qkappa,qmu
333  endif
334 
335  else if (indic == 2) then
336  ! elastic (anisotropic)
337  if (axisym) then
338  write(imain,450) n,density_mat(1),c11,c13,c15,c33,c35,c55,c12,c23,c25,c22
339  else
340  write(imain,400) n,density_mat(1),c11,c13,c15,c33,c35,c55,c12,c23,c25
341  endif
342 
343  else if (indic == 3) then
344  ! material is poroelastic (solid/fluid)
345  write(imain,500) n,sqrt(cpisquare),sqrt(cpiisquare),sqrt(cssquare)
346  write(imain,600) density_mat(1),poisson_s,lambda_s,mu_s,kappa_s,young_s
347  if (poisson_s < 0.d0) then
348  write(imain,*)
349  write(imain,*) 'Materials with a negative Poisson''s ratio can exist,'
350  write(imain,*) 'see e.g. R. Lakes, "Science" vol. 235, p. 1038-1040 (1987),'
351  write(imain,*) 'but are extremely rare.'
352  write(imain,*) 'Hope you know what you are doing...'
353  write(imain,*)
354  endif
355  write(imain,700) density_mat(2),kappa_f,eta_f
356  write(imain,800) lambda_fr,mu_fr,kappa_fr,porosity(n),tortuosity(n), &
357  permeability(1,n),permeability(2,n),permeability(3,n),qmu
358  write(imain,900) d_biot,h_biot,c_biot,m_biot,w_c
359 
360  else if (indic <= 0) then
361  write(imain,1000) n
362  endif
363  call flush_imain()
364  endif
365 
366  enddo

369  !---- formats
370  !
371 100 format(//,' M a t e r i a l s e t s : ', &
372  ' 2 D (p o r o) e l a s t i c i t y', &
373  /1x,54('='),//5x,'Number of material sets . . . . . . (numat) =',i6)
374

381  'Mass density. . . . . . . . . . (density) =',1pe15.8,/5x, &
382  'Poisson''s ratio. . . . . . . . .(poisson) =',1pe15.8,/5x, &
383  'First Lame parameter Lambda. . . (lambda) =',1pe15.8,/5x, &
384  'Second Lame parameter Mu. . . . . . .(mu) =',1pe15.8,/5x, &

386  'Young''s modulus E. . . . . . . . .(young) =',1pe15.8,/5x, &
387  'QKappa_attenuation . . . . . . .(QKappa) =',1pe15.8,/5x, &
388  'Qmu_attenuation . . . . . . . . . . (Qmu) =',1pe15.8)
389 
390 300 format(//5x,'-------------------------------',/5x, &
391  '-- Acoustic (fluid) material --',/5x, &
392  '-------------------------------',/5x, &
393  'Material set number. . . . . . . . (jmat) =',i6,/5x, &
