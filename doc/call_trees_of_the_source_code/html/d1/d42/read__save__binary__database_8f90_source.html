34  subroutine save_binary_database()
35 
36  use constants
37  use specfem_par
38  use specfem_par_gpu
39 
40  implicit none
41 
42  integer ier
43  character(len=MAX_STRING_LEN) :: outputname
44 
45 
46  ! saves data in a binary file
47  write(outputname,'(a,i6.6,a)') 'proc',myrank,'_data.bin'
48  open(unit = 2040, file = trim(output_files)//outputname,status='unknown',action='write',form='unformatted', iostat=ier)
49  if (ier /= 0) stop 'Error writing data file to disk'
50 
51  write(2040) nglob,elastic_simulation,poroelastic_simulation, &
52  acoustic_simulation,gravitoacoustic_simulation,coupled_acoustic_elastic, &
53  any_acoustic,any_elastic,any_poroelastic,any_gravitoacoustic
54 
55  write(2040) coord,jacobian,xix,xiz,gammax,gammaz, &
56  this_ibool_is_a_periodic_edge,ibool,ispec_is_inner
57 
58  if (gpu_mode) then
59  write (2040) abs_boundary_ij,abs_boundary_normal,abs_boundary_jacobian1dw, &
60  free_ac_ispec,cote_abs,free_surface_ij,any_anisotropy
61  endif
62 
63  if (any_acoustic) then
64  write(2040) rmass_inverse_acoustic,num_phase_ispec_acoustic
65  write(2040) phase_ispec_inner_acoustic,nspec_inner_acoustic,nspec_outer_acoustic,ispec_is_acoustic
66  endif
67 
68  if (any_elastic) then
69  write(2040) rmass_inverse_elastic,num_phase_ispec_elastic
70  write(2040) phase_ispec_inner_elastic,nspec_inner_elastic,nspec_outer_elastic,ispec_is_elastic
71  endif
72 
73  if (coupled_acoustic_elastic) then
74  write(2040) fluid_solid_acoustic_ispec,fluid_solid_acoustic_iedge,fluid_solid_elastic_ispec,fluid_solid_elastic_iedge, &
75  ivalue_inverse,jvalue_inverse,ivalue,jvalue
76  if (gpu_mode) write(2040) coupling_ac_el_ispec,coupling_ac_el_ij,coupling_ac_el_normal,coupling_ac_el_jacobian1dw
77  endif
78 
79  if (nproc > 1) then
80  write(2040) ninterface_acoustic,ninterface_elastic,inum_interfaces_acoustic,inum_interfaces_elastic, &
81  nibool_interfaces_acoustic,nibool_interfaces_elastic,nibool_interfaces_ext_mesh, &
82  ibool_interfaces_acoustic,ibool_interfaces_elastic,ibool_interfaces_ext_mesh
83  endif
84 
85  close(2040)
86 
87  ! saves all data regarding sources in a binary file
88  write(outputname,'(a,i6.6,a)') 'proc',myrank,'_sources_info.bin'
89  open(unit = 2040, file = trim(output_files)//outputname,status='unknown',action='write',form='unformatted', iostat=ier)
90  if (ier /= 0) stop 'Error writing sources info file to disk'
91 
92  write(2040) source_time_function,nsources_local,sourcearrays,islice_selected_source,ispec_selected_source
93 
94  close(2040)
95 
96  ! saves all data regarding receivers in a binary file
97  write(outputname,'(a,i6.6,a)') 'proc',myrank,'_receivers_info.bin'
98  open(unit = 2040, file = trim(output_files)//outputname,status='unknown',action='write',form='unformatted', iostat=ier)
99  if (ier /= 0) stop 'Error writing receivers info data file to disk'
100 
101  write(2040) nrecloc,nrec
102  write(2040) recloc,ispec_selected_rec_loc,cosrot_irec,sinrot_irec,xir_store_loc,gammar_store_loc,st_xval,st_zval, &
103  station_name,network_name,islice_selected_rec
104 
105  close(2040)
106 
107  end subroutine save_binary_database

113  subroutine read_binary_database_part1()
114 
115  use constants
116  use specfem_par
117  use specfem_par_gpu
118  use specfem_par_movie
119 
120  implicit none
121 
122  integer ier
123  character(len=MAX_STRING_LEN) :: outputname
124 
125  ! read setup data from a binary file that need to be known in advance (for allocations purpose)
126  write(outputname,'(a,i6.6,a)') trim(output_files)//'proc',myrank,'_data.bin'
127  open(unit=2040,file=outputname,status='old',form='unformatted',iostat=ier)
128  if (ier /= 0) call exit_mpi(myrank,'Error opening model file proc**_data.bin')
129 
130  read(2040) nglob,elastic_simulation,poroelastic_simulation, &
131  acoustic_simulation,gravitoacoustic_simulation,coupled_acoustic_elastic, &
132  any_acoustic,any_elastic,any_poroelastic,any_gravitoacoustic
133 
134  if (any_poroelastic .or. any_gravitoacoustic) &
135  call exit_mpi(myrank,'currently cannot have database mode if gravitoacoustic/poroelastic simulation')
136  if (pml_boundary_conditions) &
137  call exit_mpi(myrank,'currently cannot have database mode with PML')
138  if (axisym) &
139  call exit_mpi(myrank,'currently cannot have database mode with AXISYM')
140  if (output_postscript_snapshot) &
141  call exit_mpi(myrank,'currently cannot have database mode with postscript')
142  if (gpu_mode) &
143  call exit_mpi(myrank,'currently cannot have database mode with GPU_MODE')
144 
145  end subroutine read_binary_database_part1

176

178  if (ier /= 0 ) stop 'Error allocating array phase_ispec_inner_acoustic'
179

185  if (ier /= 0 ) stop 'Error allocating dummy array phase_ispec_inner_acoustic'
186  endif
187 
188  if (any_elastic) then
189  read(2040) rmass_inverse_elastic,num_phase_ispec_elastic
190 
191  allocate( phase_ispec_inner_elastic(num_phase_ispec_elastic,2),stat=ier)
192  if (ier /= 0 ) stop 'Error allocating array phase_ispec_inner_elastic'
193 
194  read(2040) phase_ispec_inner_elastic,nspec_inner_elastic,nspec_outer_elastic,ispec_is_elastic
195  else
196  ! allocates dummy array
197  num_phase_ispec_elastic = 0
198  allocate( phase_ispec_inner_elastic(num_phase_ispec_elastic,2),stat=ier)
199  if (ier /= 0 ) stop 'Error allocating dummy array phase_ispec_inner_elastic'
200  endif
201 
202  if (coupled_acoustic_elastic) then
203  read(2040) fluid_solid_acoustic_ispec,fluid_solid_acoustic_iedge,fluid_solid_elastic_ispec,fluid_solid_elastic_iedge, &
204  ivalue_inverse,jvalue_inverse,ivalue,jvalue

206  endif
207

209  read(2040) ninterface_acoustic,ninterface_elastic,inum_interfaces_acoustic,inum_interfaces_elastic, &
210  nibool_interfaces_acoustic,nibool_interfaces_elastic,nibool_interfaces_ext_mesh, &
211  ibool_interfaces_acoustic,ibool_interfaces_elastic,ibool_interfaces_ext_mesh
212 
213  max_nibool_interfaces_ext_mesh = maxval(nibool_interfaces_ext_mesh(:))
214  max_ibool_interfaces_size_ac = maxval(nibool_interfaces_acoustic(:))
215  max_ibool_interfaces_size_el = ndim*maxval(nibool_interfaces_elastic(:))
216 
217  ! allocations
218  if (acoustic_simulation) then
219  allocate(request_send_recv_acoustic(ninterface_acoustic*2),stat=ier)
220  if (ier /= 0) stop 'error in allocation of array request_send_recv_acoustic'
221  allocate(buffer_send_faces_vector_ac(max_ibool_interfaces_size_ac,ninterface_acoustic),stat=ier)
222  if (ier /= 0) stop 'error in allocation of array buffer_send_faces_vector_ac'
223  allocate(buffer_recv_faces_vector_ac(max_ibool_interfaces_size_ac,ninterface_acoustic),stat=ier)
224  if (ier /= 0) stop 'error in allocation of array buffer_recv_faces_vector_ac'
225  endif
226 
227  if (elastic_simulation) then
228  allocate(request_send_recv_elastic(ninterface_elastic*2),stat=ier)
229  if (ier /= 0) stop 'error in allocation of array request_send_recv_elastic'
230  allocate(buffer_send_faces_vector_el(max_ibool_interfaces_size_el,ninterface_elastic),stat=ier)
231  if (ier /= 0) stop 'error in allocation of array buffer_send_faces_vector_el'
232  allocate(buffer_recv_faces_vector_el(max_ibool_interfaces_size_el,ninterface_elastic),stat=ier)
233  if (ier /= 0) stop 'error in allocation of array buffer_recv_faces_vector_el'
234  endif
235 
236  endif
237 
238  close(2040)
239 
240  allocate(vector_field_display(ndim,nglob))
241  vector_field_display(:,:) = 0.d0
242 
243  end subroutine read_binary_database_part2
