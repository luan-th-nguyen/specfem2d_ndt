43  subroutine create_mask_noise()
44

46 ! **** USERS need to modify this subroutine **** to suit their own needs
47

49

51

56 
57  ! chooses noise distribution
58  select case (noise_dist_type)

62  call create_mask_noise_uniform()
63  end select

71  subroutine create_mask_noise_uniform()
72 
73 ! uniform noise distribution
74 
75  use constants, only: custom_real
76  use specfem_par, only: nglob,coord
77  use specfem_par_noise, only: mask_noise
78 
79  implicit none
80 
81  !local
82  integer :: iglob
83  real(kind=CUSTOM_REAL) :: xx,zz
84 
85  !specify distribution of noise sources as a function of xx, zz
86  do iglob = 1,nglob
87  xx = coord(1,iglob)
88  zz = coord(2,iglob)
89 
90  !below, the noise is assumed to be uniform; users are free to
91  !to change this expression to one involving xx, zz
92  mask_noise(iglob) = 1.0
93 
94  enddo
95 
96  end subroutine create_mask_noise_uniform

101 
102  subroutine create_mask_noise_nonuniform()
103 
104 ! example for a non-uniform noise distribution
105 
106  use constants, only: custom_real,pi
107  use specfem_par, only: nglob,coord
108  use specfem_par_noise, only: mask_noise
109 
110  implicit none
111 
112  !local
113  integer :: iglob
114  real(kind=CUSTOM_REAL) :: xx,zz,aa,phi
115 
116  real(kind=CUSTOM_REAL),parameter :: POS_X = 105.e3
117  real(kind=CUSTOM_REAL),parameter :: POS_Z = 92.e3
118 
119  real(kind=CUSTOM_REAL),parameter :: RAD_X = 28.e3
120  real(kind=CUSTOM_REAL),parameter :: RAD_Z = 20.e3
121 
122  aa = pi/8.0
123 
124  ! specify distribution of noise sources as a function of xx, zz
125  do iglob = 1,nglob
126  xx = coord(1,iglob)
127  zz = coord(2,iglob)
128 
129  ! distance function
130  phi = (((xx-pos_x)*cos(aa)+(zz-pos_z)*sin(aa))/rad_x)**2 + &
131  (((xx-pos_x)*sin(aa)-(zz-pos_z)*cos(aa))/rad_z)**2
132 
133  ! weighting
134  if (phi < 0.9) then
135  mask_noise(iglob) = 1.0_custom_real
136  else if (phi < 1.1) then
137  mask_noise(iglob) = 0.5*(1.0 + cos(5.0*pi*(phi/1.1 - 0.8)))
138  else
139  mask_noise(iglob) = 0.0_custom_real
140  endif
141  enddo
142 
143  end subroutine create_mask_noise_nonuniform

238  subroutine compute_source_array_noise()
239 
240 ! reads in time series based on noise spectrum and construct noise "source" array
241 
242 ! ---------------------------------------------------------------------------------
243 ! A note about time functions for noise simulations
244 !
245 ! In noise forward modeling and inversion, "time function" is used to refer
246 ! to the source time function that drives the first noise simulation.
247 !
248 ! Typically, the time function must be computed based on the frequency
249 ! characterstics of the desired noise sources. For example, if you wanted to
250 ! generate synthetic correlograms representative of the noise field in a
251 ! particular geographic region, you would have to use a time function created
252 ! by inverse Fourier transforming a model of the noise spectrum for that
253 ! region.
254 !
255 ! FOR A REALISTIC MODEL OF THE SEISMIC NOISE FIELD,
256 ! THE VARIABE "noise_source_time_function_type" (SET IN setup/constants.h) SHOULD BE SET TO 0
257 ! AND A TIME function FOR THE DESIRED NOISE SPECTRUM SHOULD BE
258 ! ***SUPPLIED BY THE USER*** IN THE FILE "NOISE_TOMOGRAPHY/S_squared"
259 !
260 ! The alternative--setting "time_function_type" to a value other than
261 ! 0--results in an idealized time function that does not represent a physically
262 ! realizable noise spectrum but which nevertheless may be useful for
263 ! performing tests or illustrating certain theoretical concepts.
264 !
265 ! ----------------------------------------------------------------------------------
266 
267  use constants, only: custom_real,ngllx,ngllz,nglj,imain,noise_source_time_function_type,output_files
268 
269  use specfem_par, only: axisym,is_on_the_axis,xiglj,p_sv,nstep,deltat, &
270  xigll,zigll,myrank
271 
272  use specfem_par_noise
273 
274  implicit none
275 
276  !local
277  integer :: it,i,j,ier
278  real(kind=CUSTOM_REAL) :: t
279  double precision, dimension(NGLLX) :: hxi, hpxi
280  double precision, dimension(NGLLZ) :: hgamma, hpgamma
281  real(kind=CUSTOM_REAL) :: factor_noise, a_val, t0
282 
283  time_function_noise(:) = 0._custom_real
284 
285  t0 = ((nstep-1)/2.0_custom_real) * deltat
286 
287  a_val = 0.6_custom_real
288  factor_noise = 1.e3_custom_real
289 
290  if (noise_source_time_function_type == 0) then
291  !read in time function from file S_squared
292  ! user output
293  if (myrank == 0) then
294  write(imain,*) ' reading noise source from file ','NOISE_TOMOGRAPHY/S_squared'
295  endif
296  open(unit=500,file='NOISE_TOMOGRAPHY/S_squared',status='old',iostat=ier)
297  if (ier /= 0) stop 'Error opening file NOISE_TOMOGRAPHY/S_squared'
298  do it = 1,nstep
299  read(500,*) time_function_noise(it)
300  enddo
301  close(500)
302 
303  else if (noise_source_time_function_type == 1) then
304  !Ricker (second derivative of a Gaussian) time function
305  ! user output
306  if (myrank == 0) then
307  write(imain,*) ' Ricker (second derivative) noise source'
308  endif
309  do it = 1,nstep
310  t = it*deltat
311  time_function_noise(it) = - factor_noise * 2.*a_val * (1. - 2.*a_val*(t-t0)**2.) * &
312  exp(-a_val*(t-t0)**2.)
313  enddo
314 
315  else if (noise_source_time_function_type == 2) then
316  !first derivative of a Gaussian time function
317  ! user output
318  if (myrank == 0) then
319  write(imain,*) ' Ricker (first derivative) noise source'
320  endif
321  do it = 1,nstep
322  t = it*deltat
323  time_function_noise(it) = - factor_noise * (2.*a_val*(t-t0)) * exp(-a_val*(t-t0)**2.)
324  enddo
325 
326 
327  else if (noise_source_time_function_type == 3) then
328  !Gaussian time function
329  ! user output
330  if (myrank == 0) then
331  write(imain,*) ' Gaussian noise source'
332  endif
333  do it = 1,nstep
334  t = it*deltat
335  time_function_noise(it) = factor_noise * exp(-a_val*(t-t0)**2.)
336  enddo
337 
338 
339  else if (noise_source_time_function_type == 4) then
340  !reproduce time function from Figure 2a of Tromp et al. 2010
341  ! user output
342  if (myrank == 0) then
343  write(imain,*) ' Figure 2a noise source'
344  endif
345  do it = 1,nstep
346  t = it*deltat
347  time_function_noise(it) = factor_noise * &
348  4.*a_val**2. * (3. - 12.*a_val*(t-t0)**2. + 4.*a_val**2.*(t-t0)**4.) * &
349  exp(-a_val*(t-t0)**2.)
350  enddo
351 
352  else
353  call exit_mpi(myrank,'Bad noise_source_time_function_type in compute_source_array_noise.')
354  endif
355 
356  ! saves source time function
357  open(500,file=trim(output_files)//'plot_source_time_function_noise.txt',status='unknown',iostat=ier)
358  if (ier /= 0) stop 'Error opening noise source time function text-file'
359  do it = 1,nstep
360  t = it*deltat
361  write(500,*) (t-t0),time_function_noise(it)
362  enddo
363  close(500)
364 
365  !interpolate over GLL points
366  source_array_noise(:,:,:,:) = 0._custom_real
367 
368  if (axisym) then
369  if (is_on_the_axis(ispec_noise)) then
370  call lagrange_any(xi_noise,nglj,xiglj,hxi,hpxi)
371  else
372  call lagrange_any(xi_noise,ngllx,xigll,hxi,hpxi)
373  endif
374  else
375  call lagrange_any(xi_noise,ngllx,xigll,hxi,hpxi)
376  endif
377 
378  call lagrange_any(gamma_noise,ngllz,zigll,hgamma,hpgamma)
379 
380  ! master station for noise source: located in element ispec_noise
381  if (p_sv) then
382  ! P-SV simulation
383  do j = 1,ngllz
384  do i = 1,ngllx
385  ! iglob = ibool(i,j,ispec_noise)
386  source_array_noise(1,i,j,:) = time_function_noise(:) * hxi(i) * hgamma(j)
387  source_array_noise(2,i,j,:) = time_function_noise(:) * hxi(i) * hgamma(j)
388  enddo
389  enddo
390  else
391  ! SH (membrane) simulation
392  do j = 1,ngllz
393  do i = 1,ngllx
394  ! iglob = ibool(i,j,ispec_noise)
395  source_array_noise(1,i,j,:) = time_function_noise(:) * hxi(i) * hgamma(j)
396  enddo
397  enddo
398  endif
399 
400  end subroutine compute_source_array_noise

406  subroutine add_point_source_noise()
407 
408 ! inject the "source" that drives the "generating wavefield"
409 
410  use constants, only: ngllx,ngllz
411 
412  use specfem_par, only: p_sv,it,ibool,accel_elastic
413 
414  use specfem_par_noise, only: ispec_noise,angle_noise,source_array_noise
415 
416  implicit none
417 
418  !local
419  integer :: i,j,iglob
420 
421  if (p_sv) then
422  ! P-SV calculation
423  do j = 1,ngllz
424  do i = 1,ngllx
425  iglob = ibool(i,j,ispec_noise)
426  accel_elastic(1,iglob) = accel_elastic(1,iglob) + sin(angle_noise)*source_array_noise(1,i,j,it)
427  accel_elastic(2,iglob) = accel_elastic(2,iglob) - cos(angle_noise)*source_array_noise(2,i,j,it)
428  enddo
429  enddo
430  else
431  ! SH (membrane) calculation
432  do j = 1,ngllz
433  do i = 1,ngllx
434  iglob = ibool(i,j,ispec_noise)
435  accel_elastic(1,iglob) = accel_elastic(1,iglob) - source_array_noise(1,i,j,it)
436  enddo
437  enddo
438  endif
439 
440  end subroutine add_point_source_noise

446  subroutine add_surface_movie_noise(accel_elastic)
447 
448 ! read in and inject the "source" that drives the "enemble forward wavefield"
449 ! (recall that the ensemble forward wavefield has a spatially distributed source)
450 
451  use constants, only: custom_real,ngllx,ngllz,ndim,noise_save_everywhere,output_files
452 
453  use specfem_par, only: p_sv,it,nstep,nspec,nglob,ibool,jacobian,wxgll,wzgll,myrank
454 
455  use specfem_par_noise
456 
457  implicit none
458 
459  real(kind=CUSTOM_REAL), dimension(NDIM,nglob) :: accel_elastic
460 
461  !local
462  integer :: ier, i, j, ispec, iglob
463 
464  ! checks if anything to do
465  if (noise_tomography /= 2 .and. noise_tomography /= 3) return
466 
467  ! checks if reconstruction by file storage
468  if (noise_tomography == 3 .and. noise_save_everywhere) return
469 
470  ! opens noise source file
471  if (it == 1) then
472  open(unit=501,file=trim(output_files)//'noise_eta.bin',access='direct', &
473  recl=nglob*custom_real,action='read',iostat=ier)
474  if (ier /= 0) call exit_mpi(myrank,'Error retrieving generating wavefield.')
475  endif
476 
477  ! generating wavefield
478  if (noise_tomography == 2) then
479  ! reads backward
480  ! for both SH- or P-SV calculation (only vertical component as noise for now...)
481  read(unit=501,rec=nstep-it+1) surface_movie_y_or_z_noise
482  else if (noise_tomography == 3) then
483  ! for reconstructed/backward wavefield
484  ! reads forward again
485  ! for both SH- or P-SV calculation (only vertical component...)
486  read(unit=501,rec=it) surface_movie_y_or_z_noise
487  endif
488 
489  ! close file at simulation end
490  if (it == nstep) then
491  close(501)
492  endif
493 
494  do ispec = 1, nspec
495  do j = 1, ngllz
496  do i = 1, ngllx
497  iglob = ibool(i,j,ispec)
498  if (p_sv) then
499  ! P-SV calculation
500  !accel_elastic(1,iglob) = accel_elastic(1,iglob) + surface_movie_x_noise(iglob) * &
501  ! mask_noise(iglob) * wxgll(i)*wzgll(j)*jacobian(i,j,ispec)
502  ! only vertical for now...
503  accel_elastic(2,iglob) = accel_elastic(2,iglob) + surface_movie_y_or_z_noise(iglob) * &
504  mask_noise(iglob) * wxgll(i)*wzgll(j)*jacobian(i,j,ispec)
505 
506  else
507  ! SH (membrane) calculation
508  accel_elastic(1,iglob) = accel_elastic(1,iglob) + surface_movie_y_or_z_noise(iglob) * &
509  mask_noise(iglob) * wxgll(i)*wzgll(j)*jacobian(i,j,ispec)
510  endif
511  enddo
512  enddo
513  enddo
514 
515  end subroutine add_surface_movie_noise
