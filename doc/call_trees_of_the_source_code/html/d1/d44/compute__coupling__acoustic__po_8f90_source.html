34 ! for acoustic solver
35 
36  subroutine compute_coupling_acoustic_po()
37 
38  use constants, only: custom_real,ngllx,ngllz,nglj,cpml_x_only,cpml_z_only,iright,ileft,ibottom,itop,one
39 
40  use specfem_par, only: num_fluid_poro_edges,ibool,wxgll,wzgll,xix,xiz, &
41  gammax,gammaz,jacobian,ivalue,jvalue,ivalue_inverse,jvalue_inverse, &
42  fluid_poro_acoustic_ispec,fluid_poro_acoustic_iedge, &
43  fluid_poro_poroelastic_ispec,fluid_poro_poroelastic_iedge, &
44  displs_poroelastic,displw_poroelastic, &
45  accels_poroelastic_adj_coupling,accelw_poroelastic_adj_coupling, &
46  potential_dot_dot_acoustic,simulation_type
47 
48  implicit none
49 
50  !local variable
51  integer :: inum,ispec_acoustic,iedge_acoustic,ispec_poroelastic,iedge_poroelastic, &
52  ipoin1D,i,j,iglob
53  real(kind=CUSTOM_REAL) :: displ_x,displ_z,displw_x,displw_z,displ_n, &
54  xxi,zxi,xgamma,zgamma,jacobian1D,nx,nz,weight
55 
56 
57  ! loop on all the coupling edges
58  do inum = 1,num_fluid_poro_edges
59 
60  ! get the edge of the acoustic element
61  ispec_acoustic = fluid_poro_acoustic_ispec(inum)
62  iedge_acoustic = fluid_poro_acoustic_iedge(inum)
63 
64  ! get the corresponding edge of the poroelastic element
65  ispec_poroelastic = fluid_poro_poroelastic_ispec(inum)
66  iedge_poroelastic = fluid_poro_poroelastic_iedge(inum)
67 
68  ! implement 1D coupling along the edge
69  do ipoin1d = 1,ngllx
70  ! get point values for the poroelastic side, which matches our side in the inverse direction
71  i = ivalue_inverse(ipoin1d,iedge_poroelastic)
72  j = jvalue_inverse(ipoin1d,iedge_poroelastic)
73  iglob = ibool(i,j,ispec_poroelastic)
74 
75  displ_x = displs_poroelastic(1,iglob)
76  displ_z = displs_poroelastic(2,iglob)
77 
78  displw_x = displw_poroelastic(1,iglob)
79  displw_z = displw_poroelastic(2,iglob)
80 
81  if (simulation_type == 3) then
82  ! new definition of adjoint displacement and adjoint potential
83  displ_x = accels_poroelastic_adj_coupling(1,iglob)
84  displ_z = accels_poroelastic_adj_coupling(2,iglob)
85 
86  displw_x = accelw_poroelastic_adj_coupling(1,iglob)
87  displw_z = accelw_poroelastic_adj_coupling(2,iglob)
88  endif
89 
90  ! get point values for the acoustic side
91  i = ivalue(ipoin1d,iedge_acoustic)
92  j = jvalue(ipoin1d,iedge_acoustic)
93  iglob = ibool(i,j,ispec_acoustic)
94 
95  ! compute the 1D Jacobian and the normal to the edge: for their expression see for instance
96  ! O. C. Zienkiewicz and R. L. Taylor, The Finite Element Method for Solid and Structural Mechanics,
97  ! Sixth Edition, electronic version, www.amazon.com, p. 204 and Figure 7.7(a),
98  ! or Y. K. Cheung, S. H. Lo and A. Y. T. Leung, Finite Element Implementation,
99  ! Blackwell Science, page 110, equation (4.60).
100  if (iedge_acoustic == itop) then
101  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
102  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
103  jacobian1d = sqrt(xxi**2 + zxi**2)
104  nx = - zxi / jacobian1d
105  nz = + xxi / jacobian1d
106  weight = jacobian1d * wxgll(i)
107  else if (iedge_acoustic == ibottom) then
108  xxi = + gammaz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
109  zxi = - gammax(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
110  jacobian1d = sqrt(xxi**2 + zxi**2)
111  nx = + zxi / jacobian1d
112  nz = - xxi / jacobian1d
113  weight = jacobian1d * wxgll(i)
114  else if (iedge_acoustic == ileft) then
115  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
116  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
117  jacobian1d = sqrt(xgamma**2 + zgamma**2)
118  nx = - zgamma / jacobian1d
119  nz = + xgamma / jacobian1d
120  weight = jacobian1d * wzgll(j)
121  else if (iedge_acoustic == iright) then
122  xgamma = - xiz(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
123  zgamma = + xix(i,j,ispec_acoustic) * jacobian(i,j,ispec_acoustic)
124  jacobian1d = sqrt(xgamma**2 + zgamma**2)
125  nx = + zgamma / jacobian1d
126  nz = - xgamma / jacobian1d
127  weight = jacobian1d * wzgll(j)
128  endif
129 
130  ! compute dot product [u_s + u_w]*n
131  displ_n = (displ_x + displw_x)*nx + (displ_z + displw_z)*nz
132  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) + weight*displ_n
133  enddo
134  enddo
135 
136  end subroutine compute_coupling_acoustic_po
