34 ! note: the filename ending is .F90 to have pre-compilation with pragmas
35 ! (like #ifndef USE_MPI) working properly
36 
37  subroutine read_interfaces_file()
38 
39  use constants, only: imain,iin_interfaces,dont_ignore_junk,hugeval,mygroup,max_string_len,in_data_files
40 
41  use shared_parameters, only: interfacesfile,nx_param, &
42  nz_layer,number_of_layers, &
43  max_npoints_interface,number_of_interfaces, &
44  nxread,nzread,number_of_simultaneous_runs
45 
46  implicit none
47 
48  ! local parameters
49  integer :: ier,interface_current,ipoint_current,ilayer
50  integer :: npoints_interface_bottom
51  double precision :: xinterface_dummy,zinterface_dummy,xinterface_dummy_previous
52  character(len=MAX_STRING_LEN) :: interfaces_filename,path_to_add
53 
54  interfaces_filename = trim(in_data_files)//trim(interfacesfile)
55 
56  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
57  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
58  interfaces_filename = path_to_add(1:len_trim(path_to_add))//interfaces_filename(1:len_trim(interfaces_filename))
59  endif
60 
61  ! user output
62  write(imain,*)
63  write(imain,*) 'Reading interface data from file: '//interfaces_filename
64  !'Reading interface data from file: '//trim(IN_DATA_FILES)//interfacesfile(1:len_trim(interfacesfile))
65  call flush_imain()
66 
67  ! get interface data from external file to count the spectral elements along Z
68  open(unit=iin_interfaces,file=interfaces_filename,status='old',iostat=ier)
69  if (ier /= 0) then
70  print *,'Error opening file: '//interfaces_filename
71  call exit_mpi(0,'Error read interface file in meshfem2D')
72  endif
73 
74  max_npoints_interface = -1
75 
76  ! read number of interfaces
77  call read_value_integer(iin_interfaces,dont_ignore_junk,number_of_interfaces)
78  if (number_of_interfaces < 2) stop 'not enough interfaces (minimum is 2)'
79 
80  ! loop on all the interfaces
81  do interface_current = 1,number_of_interfaces
82 
83  call read_value_integer(iin_interfaces,dont_ignore_junk,npoints_interface_bottom)
84 
85  if (npoints_interface_bottom < 2) stop 'not enough interface points (minimum is 2)'
86 
87  max_npoints_interface = max(npoints_interface_bottom,max_npoints_interface)
88 
89  ! user output
90  write(imain,*) 'Reading ',npoints_interface_bottom,' points for interface ',interface_current
91 
92  ! loop on all the points describing this interface
93  xinterface_dummy_previous = -hugeval
94  do ipoint_current = 1,npoints_interface_bottom
95 
96  call read_two_interface_points(iin_interfaces,dont_ignore_junk,xinterface_dummy,zinterface_dummy)
97 
98  if (ipoint_current > 1 .and. xinterface_dummy <= xinterface_dummy_previous) &
99  stop 'interface points must be sorted in increasing X'
100 
101  xinterface_dummy_previous = xinterface_dummy
102  enddo
103  enddo
104 
105  ! define number of layers
106  number_of_layers = number_of_interfaces - 1
107 
108  allocate(nz_layer(number_of_layers),stat=ier)
109  if (ier /= 0) stop 'Error allocating array nz_layer'
110 
111  ! user output
112  write(imain,*)
113  write(imain,*) 'Total number of layers in z direction = ', number_of_layers
114  call flush_imain()
115 
116  ! loop on all the layers
117  do ilayer = 1,number_of_layers
118 
119  ! read number of spectral elements in vertical direction in this layer
120  call read_value_integer(iin_interfaces,dont_ignore_junk,nz_layer(ilayer))
121 
122  if (nz_layer(ilayer) < 1) stop 'not enough spectral elements along Z in layer (minimum is 1)'
123 
124  ! user output
125  write(imain,*) 'There are ',nz_layer(ilayer),' spectral elements along Z in layer ',ilayer
126  enddo
127 
128  close(iin_interfaces)
129 
130  ! compute total number of spectral elements in vertical direction
131  nzread = sum(nz_layer)
132 
133  ! sets number of elements along X for internal mesher
134  nxread = nx_param
135 
136  ! user output
137  write(imain,*)
138  write(imain,*) 'Total number of spectral elements along X = ',nxread
139  write(imain,*) 'Total number of spectral elements along Z = ',nzread
140  write(imain,*)
141  call flush_imain()
142 
143  end subroutine read_interfaces_file
