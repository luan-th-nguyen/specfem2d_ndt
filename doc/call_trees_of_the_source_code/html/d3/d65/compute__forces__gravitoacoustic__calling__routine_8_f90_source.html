34  subroutine compute_forces_gravitoacoustic_main()
35 
36  use constants, only: zero
37  use specfem_par
38 
39  implicit none
40 
41  ! local parameters
42  integer :: iglob
43 
44  ! main solver for the gravitoacoustic elements
45 
46 ! *********************************************************
47 ! only SIMULATION_TYPE == 1, time_stepping_scheme == 1, and no PML or STACEY yet
48 ! NO MIX OF ACOUSTIC AND GRAVITOACOUTIC ELEMENTS
49 ! NO COUPLING TO ELASTIC AND POROELASTIC SIDES
50 ! *********************************************************
51 
52  if ((any_gravitoacoustic)) then
53  if (time_stepping_scheme == 1) then
54  ! Newmark time scheme
55  !! DK DK this should be vectorized
56  potential_gravitoacoustic = potential_gravitoacoustic + deltat*potential_dot_gravitoacoustic + &
57  deltatsquareover2*potential_dot_dot_gravitoacoustic
58  potential_dot_gravitoacoustic = potential_dot_gravitoacoustic + &
59  deltatover2*potential_dot_dot_gravitoacoustic
60  potential_gravito = potential_gravito + deltat*potential_dot_gravito + &
61  deltatsquareover2*potential_dot_dot_gravito
62  potential_dot_gravito = potential_dot_gravito + deltatover2*potential_dot_dot_gravito
63  else
64  stop 'Only time_stepping_scheme=1 for gravitoacoustic'
65  endif
66 
67  potential_dot_dot_gravitoacoustic = zero
68  potential_dot_dot_gravito = zero
69 
70 ! Impose displacements from boundary forcing here
71 ! because at this step the displacement (potentials) values
72 ! are already equal to value at n+1
73 ! equivalent to free surface condition
74 ! the contour integral u.n is computed after compute_forces_gravitoacoustic
75 ! *********************************************************
76 ! ** impose displacement from acoustic forcing at a rigid boundary
77 ! ** force potential_dot_dot_gravito by displacement
78 ! *********************************************************
79  if (acoustic_forcing) then
80  call add_acoustic_forcing_at_rigid_boundary_gravitoacoustic()
81  endif ! end ACOUSTIC_FORCING !
82 
83 ! free surface for a gravitoacoustic medium
84 !!! to be coded !!!
85 ! call enforce_acoustic_free_surface(potential_dot_dot_gravitoacoustic,potential_dot_gravitoacoustic, &
86 ! potential_gravitoacoustic)
87 
88 ! if (SIMULATION_TYPE == 3) then ! Adjoint calculation
89 ! call enforce_acoustic_free_surface(b_potential_dot_dot_gravitoacoustic,b_potential_dot_gravitoacoustic, &
90 ! b_potential_gravitoacoustic)
91 ! endif
92 
93 ! *********************************************************
94 ! ************* compute forces for the gravitoacoustic elements
95 ! *********************************************************
96 
97  call compute_forces_gravitoacoustic(potential_dot_dot_gravitoacoustic,potential_dot_gravitoacoustic, &
98  potential_gravitoacoustic, potential_dot_dot_gravito, &
99  potential_gravito,.false.)
100 
101  ! debugging
102  if ((mod(it,100) == 0)) then
103  iglob=iglobzero
104  write(*,*)it, & ! Nsql,gravityl,
105  maxval(potential_dot_dot_gravito),potential_dot_dot_gravito(iglob), &
106  maxval(potential_dot_dot_gravitoacoustic),potential_dot_dot_gravitoacoustic(iglob)
107  endif
108  endif ! end of test if any gravitoacoustic element
109 
110 ! *********************************************************
111 ! ************* add coupling with the elastic side
112 ! *********************************************************
113 
114 ! *********************************************************
115 ! ************* add coupling with the poroelastic side
116 ! *********************************************************
117 
118 ! ************************************************************************************
119 ! ************************************ add force source
120 ! ************************************************************************************
121 
122 ! assembling potential_dot_dot for gravitoacoustic elements
123 !#ifdef USE_MPI
124 ! if (NPROC > 1 .and. any_acoustic .and. ninterface_acoustic > 0) then
125 ! call assemble_MPI_scalar_ac_blocking(potential_dot_dot_gravitoacoustic)
126 !
127 ! endif
128 !
129 !#endif
130 
131 ! ************************************************************************************
132 ! ************* multiply by the inverse of the mass matrix and update velocity
133 ! ************************************************************************************
134 
135  if ((any_gravitoacoustic)) then
136  if (time_stepping_scheme == 1) then
137  !! DK DK this should be vectorized
138  potential_dot_dot_gravitoacoustic(:) = potential_dot_dot_gravitoacoustic(:) * rmass_inverse_gravitoacoustic(:)
139  potential_dot_gravitoacoustic(:) = potential_dot_gravitoacoustic(:) + &
140  deltatover2 * potential_dot_dot_gravitoacoustic(:)
141 
142 !! line below already done in compute_forces_gravitoacoustic, because necessary
143 !! for the computation of potential_dot_dot_gravitoacoustic
144 ! potential_dot_dot_gravito = potential_dot_dot_gravito * rmass_inverse_gravito
145  potential_dot_gravito(:) = potential_dot_gravito(:) + deltatover2 * potential_dot_dot_gravito(:)
146  else
147  stop 'Only time_stepping_scheme = 1 implemented for gravitoacoustic case'
148  endif
149 
150 ! free surface for an acoustic medium
151 ! call enforce_acoustic_free_surface(potential_dot_dot_gravitoacoustic,potential_dot_gravitoacoustic, &
152 ! potential_gravitoacoustic)
153 !
154 ! if (SIMULATION_TYPE == 3) then
155 ! call enforce_acoustic_free_surface(b_potential_dot_dot_gravitoacoustic,b_potential_dot_gravitoacoustic, &
156 ! b_potential_gravitoacoustic)
157 ! endif
158 !
159  ! update the potential field (use a new array here) for coupling terms
160 ! potential_gravitoacoustic_adj_coupling = potential_gravitoacoustic &
161 ! + deltat*potential_dot_gravitoacoustic &
162 ! + deltatsquareover2*potential_dot_dot_gravitoacoustic
163 
164  endif ! of if (any_gravitoacoustic)
165 
166  end subroutine compute_forces_gravitoacoustic_main
