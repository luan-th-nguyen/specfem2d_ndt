34  !-----------------------------------------------
35  ! rotate_mesh_for_plane_wave.
36  ! rotate mesh elements to make sure topological absorbing surface
37  ! is aligned with physical absorbing surface, since this is the surface
38  ! that we use to impose the plane wave and Bielak boundary conditions.
39  !-----------------------------------------------
40 
41  subroutine rotate_mesh_for_plane_wave(ngnod)
42 
43  use part_unstruct_par, only: elmnts,nelmnts,nelemabs,abs_surface
44 
45  implicit none
46 
47  integer, intent(in) :: ngnod
48 
49  ! local parameters
50  integer :: i,j,ispec,i1,i2,inode,iswap
51  logical :: found_this_point
52  integer, dimension(:,:), allocatable :: ibool,ibool_rotated
53 !! DK DK beware here, "ibool" applies to the mesh corners (4 or 9 points) only,
54 !! DK DK not to the GLL points because there are no GLL points in the Gmsh mesh files
55  integer :: index_rotation1,index_rotation2,index_rotation3,index_rotation4, &
56  index_rotation5,index_rotation6,index_rotation7,index_rotation8,index_edge
57 
58  allocate(ibool(ngnod,nelmnts))
59  allocate(ibool_rotated(ngnod,nelmnts))
60 
61  ! At the end of the loop, thank to ibool we can access to the global number of
62  ! each node from the ispec of the element to which it belongs and from its
63  ! geometrical position :
64  ! 4 . . 7 . . 3
65  ! . .
66  ! . .
67  ! 8 9 6
68  ! . .
69  ! . .
70  ! 1 . . 5 . . 2
71  ! --> we just create a copy in an easier format for ease of use in this routine
72  do ispec = 1, nelmnts
73  if (ngnod == 4) then
74  ibool(1,ispec) = elmnts((ispec-1)*ngnod)
75  ibool(2,ispec) = elmnts((ispec-1)*ngnod+1)
76  ibool(3,ispec) = elmnts((ispec-1)*ngnod+2)
77  ibool(4,ispec) = elmnts((ispec-1)*ngnod+3)
78  else if (ngnod == 9) then
79  ibool(1,ispec) = elmnts((ispec-1)*ngnod)
80  ibool(2,ispec) = elmnts((ispec-1)*ngnod+1)
81  ibool(3,ispec) = elmnts((ispec-1)*ngnod+2)
82  ibool(4,ispec) = elmnts((ispec-1)*ngnod+3)
83  ibool(5,ispec) = elmnts((ispec-1)*ngnod+4)
84  ibool(6,ispec) = elmnts((ispec-1)*ngnod+5)
85  ibool(7,ispec) = elmnts((ispec-1)*ngnod+6)
86  ibool(8,ispec) = elmnts((ispec-1)*ngnod+7)
87  ibool(9,ispec) = elmnts((ispec-1)*ngnod+8)
88  else
89  stop 'error, ngnod should be either 4 or 9 for external meshes'
90  endif
91  enddo
92 
93  do j = 1, 4
94  if (j == 1) then
95  index_edge=3
96  ibool_rotated(:,:) = ibool(:,:)
97  else if (j == 2) then
98  index_edge=1
99  ibool(:,:) = ibool_rotated(:,:)
100  else if (j == 3) then
101  index_edge=4
102  ibool(:,:) = ibool_rotated(:,:)
103  else if (j == 4) then
104  index_edge=2
105  ibool(:,:) = ibool_rotated(:,:)
106  else
107  stop 'j should be >= 1 and <= 4'
108  endif
109 
110  if (index_edge == 1) then
111  ! bottom edge
112  index_rotation1 = 1
113  index_rotation2 = 2
114  index_rotation3 = 2
115  index_rotation4 = 3
116  index_rotation5 = 3
117  index_rotation6 = 4
118  index_rotation7 = 1
119  index_rotation8 = 4
120  ! index_rotation9 does not need to exist because the center rotates on itself
121  else if (index_edge == 2) then
122  ! right edge
123  index_rotation1 = 2
124  index_rotation2 = 3
125  index_rotation3 = 3
126  index_rotation4 = 4
127  index_rotation5 = 1
128  index_rotation6 = 4
129  index_rotation7 = 1
130  index_rotation8 = 2
131  ! index_rotation9 does not need to exist because the center rotates on itself
132  else if (index_edge == 3) then
133  ! top edge
134  index_rotation1 = 3
135  index_rotation2 = 4
136  index_rotation3 = 1
137  index_rotation4 = 4
138  index_rotation5 = 1
139  index_rotation6 = 2
140  index_rotation7 = 2
141  index_rotation8 = 3
142  ! index_rotation9 does not need to exist because the center rotates on itself
143  else if (index_edge == 4) then
144  ! left edge
145  index_rotation1 = 1
146  index_rotation2 = 4
147  index_rotation3 = 1
148  index_rotation4 = 2
149  index_rotation5 = 2
150  index_rotation6 = 3
151  index_rotation7 = 3
152  index_rotation8 = 4
153  ! index_rotation9 does not need to exist because the center rotates on itself
154  else
155  stop 'The edge on which abs_nodes is located should be defined'
156  endif
157 
158  do i = 1,nelemabs
159  if (index_edge == abs_surface(5,i)) then
160  ispec = abs_surface(1,i) + 1 !!!! be careful: ispec from abs_surface(1,i) start at zero
161  found_this_point = .false.
162  do inode = 1,ngnod
163  if (ibool(inode,ispec) == abs_surface(3,i)) then
164  i1 = inode
165  found_this_point = .true.
166  exit
167  endif
168  enddo
169 
170  if (.not. found_this_point) stop 'point not found'
171 
172  found_this_point = .false.
173  do inode = 1,4
174  if (ibool(inode,ispec) == abs_surface(4,i)) then
175  i2 = inode
176  found_this_point = .true.
177  exit
178  endif
179  enddo
180  if (.not. found_this_point) stop 'point not found'
181 
182  ! swap points if needed for clarity, to avoid testing both cases each time below
183  if (i1 > i2) then
184  iswap = i1
185  i1 = i2
186  i2 = iswap
187  endif
188 
189  ! test orientation
190  if (i1 == index_rotation1 .and. i2 == index_rotation2) then
191  ! print *,'orientation of element ',i,' is already good'
192 
193  else if (i1 == index_rotation3 .and. i2 == index_rotation4) then
194  !for this one, remember that we have swapped, thus 41 is 14
195  ! print *,'element ',i,' must be rotated 3 times'
196  ibool_rotated(4,ispec) = ibool(1,ispec)
197  ibool_rotated(1,ispec) = ibool(2,ispec)
198  ibool_rotated(2,ispec) = ibool(3,ispec)
199  ibool_rotated(3,ispec) = ibool(4,ispec)
200  if (ngnod == 9) then
201  ibool_rotated(8,ispec) = ibool(5,ispec)
202  ibool_rotated(5,ispec) = ibool(6,ispec)
203  ibool_rotated(6,ispec) = ibool(7,ispec)
204  ibool_rotated(7,ispec) = ibool(8,ispec)
205  ! 9th point is at the element center and thus never changes when we rotate an element
206  endif
207 
208  else if (i1 == index_rotation5 .and. i2 == index_rotation6) then
209  ! print *,'element ',i,ispec,' must be rotated 2 times top'
210  ibool_rotated(3,ispec) = ibool(1,ispec)
211  ibool_rotated(4,ispec) = ibool(2,ispec)
212  ibool_rotated(1,ispec) = ibool(3,ispec)
213  ibool_rotated(2,ispec) = ibool(4,ispec)
214  if (ngnod == 9) then
215  ibool_rotated(7,ispec) = ibool(5,ispec)
216  ibool_rotated(8,ispec) = ibool(6,ispec)
217  ibool_rotated(5,ispec) = ibool(7,ispec)
218  ibool_rotated(6,ispec) = ibool(8,ispec)
219  ! 9th point is at the element center and thus never changes when we rotate an element
220  endif
221 
222  else if (i1 == index_rotation7 .and. i2 == index_rotation8) then
223  ! print *,'element ',i,' must be rotated 1 time'
224  ibool_rotated(2,ispec) = ibool(1,ispec)
225  ibool_rotated(3,ispec) = ibool(2,ispec)
226  ibool_rotated(4,ispec) = ibool(3,ispec)
227  ibool_rotated(1,ispec) = ibool(4,ispec)
228  if (ngnod == 9) then
229  ibool_rotated(6,ispec) = ibool(5,ispec)
230  ibool_rotated(7,ispec) = ibool(6,ispec)
231  ibool_rotated(8,ispec) = ibool(7,ispec)
232  ibool_rotated(5,ispec) = ibool(8,ispec)
233  ! 9th point is at the element center and thus never changes when we rotate an element
234  endif
235  else
236  stop 'problem in an element'
237  endif
238  endif
239  enddo
240  enddo
241 
242 ! here we put the result back in the not-so-easy to use format at the end of the routine
243  do ispec = 1, nelmnts
244  if (ngnod == 4) then
245  elmnts((ispec-1)*ngnod) = ibool_rotated(1,ispec)
246  elmnts((ispec-1)*ngnod+1) = ibool_rotated(2,ispec)
247  elmnts((ispec-1)*ngnod+2) = ibool_rotated(3,ispec)
248  elmnts((ispec-1)*ngnod+3) = ibool_rotated(4,ispec)
249  else if (ngnod == 9) then
250  elmnts((ispec-1)*ngnod) = ibool_rotated(1,ispec)
251  elmnts((ispec-1)*ngnod+1) = ibool_rotated(2,ispec)
252  elmnts((ispec-1)*ngnod+2) = ibool_rotated(3,ispec)
253  elmnts((ispec-1)*ngnod+3) = ibool_rotated(4,ispec)
254  elmnts((ispec-1)*ngnod+4) = ibool_rotated(5,ispec)
255  elmnts((ispec-1)*ngnod+5) = ibool_rotated(6,ispec)
256  elmnts((ispec-1)*ngnod+6) = ibool_rotated(7,ispec)
257  elmnts((ispec-1)*ngnod+7) = ibool_rotated(8,ispec)
258  elmnts((ispec-1)*ngnod+8) = ibool_rotated(9,ispec)
259  else
260  stop 'error, ngnod should be either 4 or 9 for external meshes'
261  endif
262  enddo
263 
264  end subroutine rotate_mesh_for_plane_wave

270  subroutine rotate_mesh_for_axisym(ngnod)
271 
272 ! This routine is almost the same than the one above with index_edge = 4 (left side of the model)
273 ! It allows us to know that if i == 1 (first GLL point) we are on the axis.
274 ! From axisymmetric elements mesh file with just know the ispec of the elements on the axis and the ibool of
275 ! the two points that describe it :
276 ! Ex :
277 !
278 ! x=0
279 ! ...
280 ! |
281 ! O ibool = 235 ---> We want that index to be saved in ibool(1,ispec) or ibool(4,ispec)
282 ! |
283 ! |
284 ! | ispec = 15
285 ! |
286 ! |
287 ! O ibool = 423 ---> We want that index to be saved in ibool(4,ispec) or ibool(1,ispec)
288 ! |
289 ! ...
290 !
291 ! Indeed when we are using external mesher for axisymmetric simulations we do not control how the elements
292 ! will be orientated. We could have code everything taking that into account but we have preferred
293 ! to rotate the mesh. After that for each element on the axis we have got:
294 !
295 ! r=0
296 ! |
297 ! 4 . . 7 . . 3
298 ! . .
299 ! . .
300 ! 8 9 6
301 ! . .
302 ! . .
303 ! 1 . . 5 . . 2
304 ! |
305 ! r=0
306 
307  use part_unstruct_par, only: elmnts,nelmnts,nelem_on_the_axis,ispec_of_axial_elements, &
308  inode1_axial_elements,inode2_axial_elements
309 
310  implicit none
311 
312  integer, intent(in) :: ngnod
313  integer :: i,ispec,i1,i2,inode,iswap
314  logical :: found_this_point
315  integer, dimension(:,:), allocatable :: ibool,ibool_rotated
316  !! DK DK be careful here, "ibool" applies to the mesh corners (4 or 9 points) only,
317 
318  allocate(ibool(ngnod,nelmnts))
319  allocate(ibool_rotated(ngnod,nelmnts))
320 
321  do ispec = 1, nelmnts ! Loop on the elements
322  ! At the end of the loop, thank to ibool we can access to the global number of
323  ! each node from the ispec of the element to which it belongs and from its
324  ! geometrical position :
325  ! 4 . . 7 . . 3
326  ! . .
327  ! . .
328  ! 8 9 6
329  ! . .
330  ! . .
331  ! 1 . . 5 . . 2
332  ! --> we just create a copy in an easier format for ease of use in this routine
333  if (ngnod == 4) then
334  ibool(1,ispec) = elmnts((ispec-1)*ngnod) ! Have to be zero if ispec is on the axis
335  ibool(2,ispec) = elmnts((ispec-1)*ngnod+1)
336  ibool(3,ispec) = elmnts((ispec-1)*ngnod+2)
337  ibool(4,ispec) = elmnts((ispec-1)*ngnod+3) ! Have to be zero if ispec is on the axis
338  else if (ngnod == 9) then
339  ibool(1,ispec) = elmnts((ispec-1)*ngnod) ! Have to be zero if ispec is on the axis
340  ibool(2,ispec) = elmnts((ispec-1)*ngnod+1)
341  ibool(3,ispec) = elmnts((ispec-1)*ngnod+2)
342  ibool(4,ispec) = elmnts((ispec-1)*ngnod+3) ! Have to be zero if ispec is on the axis
343  ibool(5,ispec) = elmnts((ispec-1)*ngnod+4)
344  ibool(6,ispec) = elmnts((ispec-1)*ngnod+5)
345  ibool(7,ispec) = elmnts((ispec-1)*ngnod+6)
346  ibool(8,ispec) = elmnts((ispec-1)*ngnod+7)
347  ibool(9,ispec) = elmnts((ispec-1)*ngnod+8)
348  else
349  stop 'rotate_mesh_for_axisym: error, ngnod should be either 4 or 9 for external meshes'
350  endif
351  enddo
352 
353  ibool_rotated(:,:) = ibool(:,:) ! We make a copy of ibool in ibool_rotated
354 
355  ! print *," Loop on the elements on the axis... : "
356  do i = 1,nelem_on_the_axis ! Loop on the elements on the axis (read from the axisym file)
357  ispec = ispec_of_axial_elements(i) + 1 ! ispec_of_axial_elements starts from 0
358  found_this_point = .false.
359  ! print *," Loop on the control points and look for ", inode1_axial_elements(i)
360  do inode = 1,4 ! loop on the corners of axial element ispec_of_axial_elements(i) to see if we find inode1_axial_elements(i)
361  if (ibool(inode,ispec) == inode1_axial_elements(i)) then
362  i1 = inode
363  found_this_point = .true.
364  exit
365  endif
366  enddo
367  if (.not. found_this_point) stop 'rotate_mesh_for_axisym: point not found 1'
368  found_this_point = .false.
369  do inode = 1,4 ! loop on the corners of axial element ispec_of_axial_elements(i) to see if we find inode2_axial_elements(i)
370  if (ibool(inode,ispec) == inode2_axial_elements(i)) then
371  i2 = inode
372  found_this_point = .true.
373  exit
374  endif
375  enddo
376  if (.not. found_this_point) stop 'rotate_mesh_for_axisym: point not found 2'
377  if (i1 > i2) then
378  ! swap points if needed for clarity, to avoid testing both cases each time below
379  ! Otherwise we would have done : if ((i1 == 1 .and. i2 == 4) .or. (i1 == 4 .and. i2 == 1)) then
380  ! print *,'orientation of element ',i,' is already good'
381  ! ...
382  iswap = i1
383  i1 = i2
384  i2 = iswap
385  endif
386 
387  ! print *,"point 1 (with coord 0):",inode1_axial_elements(i)
388  ! print *,"point 2 (with coord 0):",inode2_axial_elements(i)
389  ! print *,"ibool of this element :"
390  ! print *," ibool(1,ispec) :",ibool(1,ispec)
391  ! print *," ibool(2,ispec) :",ibool(2,ispec)
392  ! print *," ibool(3,ispec) :",ibool(3,ispec)
393  ! print *," ibool(4,ispec) :",ibool(4,ispec)
394  ! print *," i1 -->",i1
395  ! print *," i2 -->",i2
396 
397  ! test orientation
398  if (i1 == 1 .and. i2 == 4) then ! Orientation of this element is already good'
399  ! print *,'orientation of element ',i,' is already good'
400  else if (i1 == 1 .and. i2 == 2) then ! Remember that we have swapped, thus no need to test i1 == 1 and i2 == 1
401  ! Element must be rotated 3 times
402  ! print *,'element ',i,' must be rotated 3 times'
403  ibool_rotated(4,ispec) = ibool(1,ispec)
404  ibool_rotated(1,ispec) = ibool(2,ispec)
405  ibool_rotated(2,ispec) = ibool(3,ispec)
406  ibool_rotated(3,ispec) = ibool(4,ispec)
407  if (ngnod == 9) then
408  ibool_rotated(8,ispec) = ibool(5,ispec)
409  ibool_rotated(5,ispec) = ibool(6,ispec)
410  ibool_rotated(6,ispec) = ibool(7,ispec)
411  ibool_rotated(7,ispec) = ibool(8,ispec)
412  ! 9th point is at the element center and thus never changes when we rotate an element
413  endif
414  else if (i1 == 2 .and. i2 == 3) then ! Element must be rotated 2 times
415  ! print *,'element ',i,ispec,' must be rotated 2 times top'
416  ibool_rotated(3,ispec) = ibool(1,ispec)
417  ibool_rotated(4,ispec) = ibool(2,ispec)
418  ibool_rotated(1,ispec) = ibool(3,ispec)
419  ibool_rotated(2,ispec) = ibool(4,ispec)
420  if (ngnod == 9) then
421  ibool_rotated(7,ispec) = ibool(5,ispec)
422  ibool_rotated(8,ispec) = ibool(6,ispec)
423  ibool_rotated(5,ispec) = ibool(7,ispec)
424  ibool_rotated(6,ispec) = ibool(8,ispec)
425  ! 9th point is at the element center and thus never changes when we rotate an element
426  endif
427  else if (i1 == 3 .and. i2 == 4) then ! Element must be rotated 1 time
428  ! print *,'element ',i,' must be rotated 1 time'
429  ibool_rotated(2,ispec) = ibool(1,ispec)
430  ibool_rotated(3,ispec) = ibool(2,ispec)
431  ibool_rotated(4,ispec) = ibool(3,ispec)
432  ibool_rotated(1,ispec) = ibool(4,ispec)
433  if (ngnod == 9) then
434  ibool_rotated(6,ispec) = ibool(5,ispec)
435  ibool_rotated(7,ispec) = ibool(6,ispec)
436  ibool_rotated(8,ispec) = ibool(7,ispec)
437  ibool_rotated(5,ispec) = ibool(8,ispec)
438  ! 9th point is at the element center and thus never changes when we rotate an element
439  endif
440  else
441  stop 'rotate_mesh_for_axisym: problem in an element'
442  endif
443  enddo
444 
445  ! Here we put the result back in the not-so-easy to use format at the end of the routine
446  do ispec = 1, nelmnts
447  if (ngnod == 4) then
448  elmnts((ispec-1)*ngnod) = ibool_rotated(1,ispec)
449  elmnts((ispec-1)*ngnod+1) = ibool_rotated(2,ispec)
450  elmnts((ispec-1)*ngnod+2) = ibool_rotated(3,ispec)
451  elmnts((ispec-1)*ngnod+3) = ibool_rotated(4,ispec)
452  else if (ngnod == 9) then
453  elmnts((ispec-1)*ngnod) = ibool_rotated(1,ispec)
454  elmnts((ispec-1)*ngnod+1) = ibool_rotated(2,ispec)
455  elmnts((ispec-1)*ngnod+2) = ibool_rotated(3,ispec)
456  elmnts((ispec-1)*ngnod+3) = ibool_rotated(4,ispec)
457  elmnts((ispec-1)*ngnod+4) = ibool_rotated(5,ispec)
458  elmnts((ispec-1)*ngnod+5) = ibool_rotated(6,ispec)
459  elmnts((ispec-1)*ngnod+6) = ibool_rotated(7,ispec)
460  elmnts((ispec-1)*ngnod+7) = ibool_rotated(8,ispec)
461  elmnts((ispec-1)*ngnod+8) = ibool_rotated(9,ispec)
462  else
463  stop 'rotate_mesh_for_axisym: error, ngnod should be either 4 or 9 for external meshes'
464  endif
465  enddo
466 
467  end subroutine rotate_mesh_for_axisym

472 
473  subroutine rotate_mesh_for_acoustic_forcing(ngnod)
474 
475 ! This routine is almost the same than the one at the top but for the acoustic forced elements
476 ! Ex :
477 !
478 ! x=0
479 ! ...
480 ! |
481 ! O ibool = 235 ---> We want that index to be saved in ibool(1,ispec) or ibool(4,ispec)
482 ! |
483 ! |
484 ! | ispec = 15
485 ! |
486 ! |
487 ! O ibool = 423 ---> We want that index to be saved in ibool(4,ispec) or ibool(1,ispec)
488 ! |
489 ! ...
490 !
491 ! Indeed when we are using external mesher we do not control how the elements
492 ! will be orientated. We could have code everything taking that into account but we have preferred
493 ! to rotate the mesh. After that for each element on the axis we have got:
494 !
495 ! r=0
496 ! |
497 ! 4 . . 7 . . 3
498 ! . .
499 ! . .
500 ! 8 9 6
501 ! . .
502 ! . .
503 ! 1 . . 5 . . 2
504 ! |
505 ! r=0
506 
507  use part_unstruct_par, only: elmnts,nelmnts,nelemacforcing,acforcing_surface
508 
509  implicit none
510 
511  integer, intent(in) :: ngnod
512 
513  ! local parameters
514  integer :: i,j,ispec,i1,i2,inode,iswap
515  logical :: found_this_point
516  integer, dimension(:,:), allocatable :: ibool,ibool_rotated
517 !! DK DK be careful here, "ibool" applies to the mesh corners (4 or 9 points) only,
518 !! DK DK not to the GLL points because there are no GLL points in the Gmsh mesh files
519  integer :: index_rotation1,index_rotation2,index_rotation3,index_rotation4, &
520  index_rotation5,index_rotation6,index_rotation7,index_rotation8,index_edge
521 
522  allocate(ibool(ngnod,nelmnts))
523  allocate(ibool_rotated(ngnod,nelmnts))
524 
525  ! At the end of the loop, thank to ibool we can access to the global number of
526  ! each node from the ispec of the element to which it belongs and from its
527  ! geometrical position :
528  ! 4 . . 7 . . 3
529  ! . .
