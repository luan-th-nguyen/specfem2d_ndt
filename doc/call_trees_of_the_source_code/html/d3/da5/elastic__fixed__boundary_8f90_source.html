34 ! From array 'surface' (element, type : node/edge, node(s) ) that describes the
35 ! elastic fixed surface, determines the points (ixmin, ixmax, izmin and izmax) on the surface for each element.
36 ! We chose to have ixmin <= ixmax and izmin <= izmax, so as to be able to have DO loops on it with an increment of +1.
37 
38  subroutine construct_elastic_fixed_surface ()
39 
40  use specfem_par, only: ngnod, knods, nelem_elastic_fixed_surface, elastic_fixed_edges, elastic_fixed_surface
41 
42  implicit none
43 
44  integer :: i, k
45  integer :: ixmin, ixmax
46  integer :: izmin, izmax
47  integer, dimension(ngnod) :: n
48  integer :: e1, e2
49  integer :: type_elastic
50 
51  do i = 1, nelem_elastic_fixed_surface
52  elastic_fixed_surface(1,i) = elastic_fixed_edges(1,i) ! Here we do a copy
53  type_elastic = elastic_fixed_edges(2,i)
54  e1 = elastic_fixed_edges(3,i)
55  e2 = elastic_fixed_edges(4,i)
56  do k = 1, ngnod
57  n(k) = knods(k,elastic_fixed_surface(1,i))
58  enddo
59 
60  call get_elastic_fixed_edge ( ngnod, n, type_elastic, e1, e2, ixmin, ixmax, izmin, izmax )
61 
62  elastic_fixed_surface(2,i) = ixmin
63  elastic_fixed_surface(3,i) = ixmax
64  elastic_fixed_surface(4,i) = izmin
65  elastic_fixed_surface(5,i) = izmax
66 
67  enddo
68 
69  end subroutine construct_elastic_fixed_surface
