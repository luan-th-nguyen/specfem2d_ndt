34 ! Recompute 2D jacobian at a given point in a 4-node or 9-node element.
35 ! Compute also the global coordinates of the point defined by: (xi,gamma,ispec).
36 
37  subroutine recompute_jacobian(xi,gamma,x,z,xix,xiz,gammax,gammaz,jacobian,coorg,knods,ispec,ngnod,nspec,npgeo, &
38  stop_if_negative_jacobian)
39 
40  use constants, only: ndim,zero
41 
42  implicit none
43 
44  integer ispec,ngnod,nspec,npgeo
45  double precision x,z,xix,xiz,gammax,gammaz
46  double precision xi,gamma,jacobian
47 
48  integer knods(ngnod,nspec)
49  double precision coorg(ndim,npgeo)
50 
51 ! 2D shape functions and their derivatives at receiver
52  double precision shape2D(ngnod)
53  double precision dershape2D(ndim,ngnod)
54 
55  double precision xxi,zxi,xgamma,zgamma,xelm,zelm
56 
57  integer ia,nnum
58 
59  logical stop_if_negative_jacobian
60 
61 ! only one problematic element is output to OpenDX for now in case of elements with a negative Jacobian
62  integer, parameter :: ntotspecAVS_DX = 1
63 
64 ! recompute jacobian for any (xi,gamma) point, not necessarily a GLL point
65 
66 ! create the 2D shape functions and then the Jacobian
67  call define_shape_functions(shape2d,dershape2d,xi,gamma,ngnod)
68 
69 ! compute coordinates and jacobian matrix
70  x = zero
71  z = zero
72 
73  xxi = zero
74  zxi = zero
75  xgamma = zero
76  zgamma = zero
77 
78  do ia = 1,ngnod
79 
80  nnum = knods(ia,ispec)
81 
82  xelm = coorg(1,nnum)
83  zelm = coorg(2,nnum)
84 
85  x = x + shape2d(ia)*xelm
86  z = z + shape2d(ia)*zelm
87 
88  xxi = xxi + dershape2d(1,ia)*xelm
89  zxi = zxi + dershape2d(1,ia)*zelm
90  xgamma = xgamma + dershape2d(2,ia)*xelm
91  zgamma = zgamma + dershape2d(2,ia)*zelm
92 
93  enddo
94 
95  jacobian = xxi*zgamma - xgamma*zxi
96 
97 ! the Jacobian is negative, so far this means that there is an error in the mesh
98 ! therefore print the coordinates of the mesh points of this element
99 ! and also create an OpenDX file to visualize it
100  if (jacobian <= zero .and. stop_if_negative_jacobian) then
101 
102 ! print the coordinates of the mesh points of this element
103  print *, 'ispec = ', ispec
104  print *, 'ngnod = ', ngnod
105  do ia = 1,ngnod
106  nnum = knods(ia,ispec)
107  xelm = coorg(1,nnum)
108  zelm = coorg(2,nnum)
109  print *,'node ', ia,' x,y = ',xelm,zelm
110  enddo
111 
112 ! create an OpenDX file to visualize this element
113  open(unit=11,file='DX_first_element_with_negative_jacobian.dx',status='unknown')
114 
115 ! output the points (the mesh is flat therefore the third coordinate is zero)
116  write(11,*) 'object 1 class array type float rank 1 shape 3 items ',ngnod,' data follows'
117  do ia = 1,ngnod
118  nnum = knods(ia,ispec)
119  xelm = coorg(1,nnum)
120  zelm = coorg(2,nnum)
121  write(11,*) xelm,zelm,' 0'
122  enddo
123 
124 ! output the element (use its four corners only for now)
125  write(11,*) 'object 2 class array type int rank 1 shape 4 items ',ntotspecavs_dx,' data follows'
126 ! point order in OpenDX is 1,4,2,3 *not* 1,2,3,4 as in AVS
127  write(11,*) '0 3 1 2'
128 
129 ! output element data
130  write(11,*) 'attribute "element type" string "quads"'
131  write(11,*) 'attribute "ref" string "positions"'
132  write(11,*) 'object 3 class array type float rank 0 items ',ntotspecavs_dx,' data follows'
133 
134 ! output dummy data value
135  write(11,*) '1'
136 
137 ! define OpenDX field
138  write(11,*) 'attribute "dep" string "connections"'
139  write(11,*) 'object "irregular positions irregular connections" class field'
140  write(11,*) 'component "positions" value 1'
141  write(11,*) 'component "connections" value 2'
142  write(11,*) 'component "data" value 3'
143  write(11,*) 'end'
144 
145 ! close OpenDX file
146  close(11)
147 
148  stop 'negative 2D Jacobian, element saved in DX_first_element_with_negative_jacobian.dx'
149  endif
150 
151 ! invert the relation
152  xix = zgamma / jacobian
153  gammax = - zxi / jacobian
154  xiz = - xgamma / jacobian
155  gammaz = xxi / jacobian
156 
157  end subroutine recompute_jacobian
