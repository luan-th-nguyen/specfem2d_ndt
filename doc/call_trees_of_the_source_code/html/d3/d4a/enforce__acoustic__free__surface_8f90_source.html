34  subroutine enforce_acoustic_free_surface(potential_dot_dot_acoustic,potential_dot_acoustic,potential_acoustic)
35 
36 ! free surface for an acoustic medium
37 ! if acoustic, the free surface condition is a Dirichlet condition for the potential,
38 ! not Neumann, in order to impose zero pressure at the surface
39 
40  use constants, only: custom_real,zero
41 
42  use specfem_par, only: acoustic_surface,ibool,nelem_acoustic_surface,nglob,this_ibool_is_a_periodic_edge,coord
43 
44  implicit none
45 
46 !! DK DK added this to exclude the bottom surface of the mesh for Laurent Guillon in order not to create
47 !! DK DK a Dirichlet condition for the potential, i.e. in order to create a Neumann condition for the potential
48 !! DK DK i.e. in order to create a rigid bottom surface instead of a free surface; I test the Z coordinate of the mesh point
49  logical, parameter :: ENFORCE_RIGID_SURFACE_BOTTOM = .false.
50 ! this should be a bit bigger than the Z coordinate of the bottom; since it is Z = 0 in Laurent Guillon's test
51 ! we use this small value (its actual value does not matter as long as it is smaller than the Z coordinate of the top surface)
52  double precision, parameter :: Zlimit = 0.0000001d0
53 
54  real(kind=CUSTOM_REAL), dimension(nglob) :: potential_dot_dot_acoustic,potential_dot_acoustic,potential_acoustic
55 
56 !---
57 !--- local variables
58 !---
59 
60  integer :: ispec_acoustic_surface,ispec,i,j,iglob
61 
62  ! checks if anything to do
63  if (nelem_acoustic_surface == 0) return
64 
65  do ispec_acoustic_surface = 1, nelem_acoustic_surface
66 
67  ispec = acoustic_surface(1,ispec_acoustic_surface)
68 
69  do j = acoustic_surface(4,ispec_acoustic_surface), acoustic_surface(5,ispec_acoustic_surface)
70  do i = acoustic_surface(2,ispec_acoustic_surface), acoustic_surface(3,ispec_acoustic_surface)
71  iglob = ibool(i,j,ispec)
72  ! make sure that an acoustic free surface is not enforced on periodic edges
73  if (.not. this_ibool_is_a_periodic_edge(iglob)) then
74 !! DK DK added this to exclude the bottom surface of the mesh for Laurent Guillon in order not to create
75 !! DK DK a Dirichlet condition for the potential, i.e. in order to create a Neumann condition for the potential
76 !! DK DK i.e. in order to create a rigid bottom surface instead of a free surface; I test the Z coordinate of the mesh point
77  if (.not. enforce_rigid_surface_bottom .or. coord(2,iglob) >= zlimit) then
78  potential_acoustic(iglob) = zero
79  potential_dot_acoustic(iglob) = zero
80  potential_dot_dot_acoustic(iglob) = zero
81  endif
82  endif
83  enddo
84  enddo
85 
86  enddo
87 
88  end subroutine enforce_acoustic_free_surface
